112         if (node.token == null) {
113             throw SqlException.position(node.position)

150      * Writes IR of the filter described by the given expression tree to memory.
151      *
152      * @param node       filter expression tree's root node.
153      * @param scalar     set use only scalar instruction set execution hint in the returned options.
154      * @param debug      set enable debug flag in the returned options.
155      * @param nullChecks a flag for JIT, allowing or disallowing generation of null check
156      * @return JIT compiler options stored in a single int in the following way:
157      * <ul>
158      * <li>1 LSB - debug flag</li>
159      * <li>2-3 LSBs - filter's arithmetic type size (widest type size): 0 - 1B, 1 - 2B, 2 - 4B, 3 - 8B</li>
160      * <li>4-5 LSBs - filter's execution hint: 0 - scalar, 1 - single size (SIMD-friendly), 2 - mixed sizes</li>
161      * <li>6 LSB - flag to include null checks for column values into compiled filter</li>
162      * </ul>
163      * <p>
164      * Examples:
165      * <ul>
166      * <li>00000000 00000000 00000000 00010100 - 4B, mixed types, debug off, null checks disabled</li>
167      * <li>00000000 00000000 00000000 00100111 - 8B, scalar, debug on, null checks enabled</li>
168      * </ul>
169      * @throws SqlException thrown when IR serialization failed.
170      */
171     public int serialize(ExpressionNode node, boolean scalar, boolean debug, boolean nullChecks) throws SqlException {
172         traverseAlgo.traverse(node, this);
173         putOperator(RET);
174 
175         TypesObserver typesObserver = predicateContext.globalTypesObserver;
176         int options = debug ? 1 : 0;
177         int typeSize = typesObserver.maxSize();
178         if (typeSize > 0) {
179             // typeSize is 2^n, so number of trailing zeros is equal to log2
180             int log2 = Integer.numberOfTrailingZeros(typeSize);
181             options = options | (log2 << 1);
182         }
183         if (!scalar && !forceScalarMode) {
184             int executionHint = typesObserver.hasMixedSizes() ? 2 : 1;
185             options = options | (executionHint << 3);
186         }
187 
188         options = options | ((nullChecks ? 1 : 0) << 5);
189 
190         return options;
191     }

193     @Override
194     public void visit(ExpressionNode node) throws SqlException {
195         int argCount = node.paramCount;
196         if (argCount == 0) {
197             switch (node.type) {
198                 case ExpressionNode.LITERAL:
199                     serializeColumn(node.position, node.token);
200                     break;
201                 case ExpressionNode.BIND_VARIABLE:
202                     serializeBindVariable(node);
203                     break;
204                 case ExpressionNode.CONSTANT:
205                     // Write stub values to be backfilled later
206                     serializeConstantStub(node);
207                     break;
208                 default:
209                     throw SqlException.position(node.position)
210                             .put("unsupported token: ")
211                             .put(node.token);
212             }
213         } else {
214             serializeOperator(node.position, node.token, argCount);
215         }
216 
217         boolean predicateLeft = predicateContext.onNodeVisited(node);
218 
219         if (predicateLeft) {
220             // We're out of a predicate
221 
222             // Force scalar mode if the predicate had byte or short arithmetic operations.
223             // That's because SIMD mode uses byte/short-sized overflows for arithmetic
224             // calculations instead of implicit upcast to int done by *.sql.Function classes.
225             forceScalarMode |=
226                     predicateContext.hasArithmeticOperations && predicateContext.localTypesObserver.maxSize() <= 2;
227 
228             // Then backfill constants and symbol bind variables and clean up
229             try {
230                 backfillNodes.forEach(backfillNodeConsumer);
231                 backfillNodes.clear();
232             } catch (SqlWrapperException e) {
233                 throw e.wrappedException;
234             }
235         }
236     }

404     private Function getBindVariableFunction(int position, CharSequence token) throws SqlException {
405         Function varFunction;
406 
407         if (token.charAt(0) == ':') {
408             // name bind variable case
409             Function bindFunction = getBindVariableService().getFunction(token);
410             if (bindFunction == null) {
411                 throw SqlException.position(position).put("failed to find function for bind variable: ").put(token);
412             }
413             varFunction = new NamedParameterLinkFunction(Chars.toString(token), bindFunction.getType());
414         } else {
415             // indexed bind variable case
416             try {
417                 final int variableIndex = Numbers.parseInt(token, 1, token.length());
418                 if (variableIndex < 1) {
419                     throw SqlException.$(position, "invalid bind variable index [value=").put(variableIndex).put(']');
420                 }
421                 Function bindFunction = getBindVariableService().getFunction(variableIndex - 1);
422                 if (bindFunction == null) {
423                     throw SqlException.position(position).put("failed to find function for bind variable: ").put(token);
424                 }
425                 varFunction = new IndexedParameterLinkFunction(
426                         variableIndex - 1,
427                         bindFunction.getType(),
428                         position);
429 
430             } catch (NumericException e) {
431                 throw SqlException.$(position, "invalid bind variable index [value=").put(token).put(']');
432             }
433         }

530     private void serializeColumn(int position, final CharSequence token) throws SqlException {
531         if (!predicateContext.isActive()) {
532             throw SqlException.position(position)
533                     .put("non-boolean column outside of predicate: ")
534                     .put(token);
535         }
536 
537         final int index = metadata.getColumnIndexQuiet(token);
538         if (index == -1) {
539             throw SqlException.invalidColumn(position, token);
540         }
541 
542         final int columnType = metadata.getColumnType(index);
543         final int columnTypeTag = ColumnType.tagOf(columnType);
544         int typeCode = columnTypeCode(columnTypeTag);
545         if (typeCode == UNDEFINED_CODE) {
546             throw SqlException.position(position)
547                     .put("unsupported column type: ")
548                     .put(ColumnType.nameOf(columnTypeTag));
549         }
550 
551         // In case of a top level boolean column, expand it to "boolean_column = true" expression.
552         if (predicateContext.singleBooleanColumn && columnTypeTag == ColumnType.BOOLEAN) {
553             // "true" constant
554             putOperand(IMM, I1_TYPE, 1);
555             // column
556             putOperand(MEM, typeCode, index);
557             // =
558             putOperator(EQ);
559             return;
560         }
561         putOperand(MEM, typeCode, index);
562     }

564     private void serializeConstant(long offset, int position, final CharSequence token, boolean negated) throws SqlException {
565         final int len = token.length();
566         final int typeCode = predicateContext.localTypesObserver.constantTypeCode();
567         if (typeCode == UNDEFINED_CODE) {
568             throw SqlException.position(position).put("all constants expression: ").put(token);
569         }
570 
571         if (SqlKeywords.isNullKeyword(token)) {
572             boolean geoHashExpression = PredicateType.GEO_HASH == predicateContext.type;
573             serializeNull(offset, position, typeCode, geoHashExpression);
574             return;
575         }
576 
577         if (PredicateType.SYMBOL == predicateContext.type) {
578             serializeSymbolConstant(offset, position, token);
579             return;
580         }
581 
582         if (Chars.isQuoted(token)) {
583             if (PredicateType.CHAR != predicateContext.type) {
584                 throw SqlException.position(position).put("char constant in non-char expression: ").put(token);
585             }
586             if (len == 3) {
587                 // this is 'x' - char
588                 putOperand(offset, IMM, I2_TYPE, token.charAt(1));
589                 return;
590             }
591             throw SqlException.position(position).put("unsupported string constant: ").put(token);
592         }
593 
594         if (SqlKeywords.isTrueKeyword(token)) {
595             if (PredicateType.BOOLEAN != predicateContext.type) {
596                 throw SqlException.position(position).put("boolean constant in non-boolean expression: ").put(token);
597             }
598 
599             putOperand(offset, IMM, I1_TYPE, 1);
600             return;
601         }
602 
603         if (SqlKeywords.isFalseKeyword(token)) {
604             if (PredicateType.BOOLEAN != predicateContext.type) {
605                 throw SqlException.position(position).put("boolean constant in non-boolean expression: ").put(token);
606             }
607             putOperand(offset, IMM, I1_TYPE, 0);
608             return;
609         }
610 
611         if (len > 1 && token.charAt(0) == '#') {
612             if (PredicateType.GEO_HASH != predicateContext.type) {
613                 throw SqlException.position(position).put("geo hash constant in non-geo hash expression: ").put(token);
614             }
615             ConstantFunction geoConstant = GeoHashUtil.parseGeoHashConstant(position, token, len);
616             if (geoConstant != null) {
