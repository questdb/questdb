73     private void bumpCountAt(int line, byte bytePosition, int increment) {
74         if (bytePosition > 0) {
75             final long pos = matrix + ((long) line * matrixRowSize + bytePosition * Integer.BYTES);
76             Unsafe.getUnsafe().putInt(pos, Unsafe.getUnsafe().getInt(pos) + increment);
77         }

80     byte scan(long address, long hi) throws TextException {
81         int lineCount = 0;
82         boolean quotes = false;
83         long cursor = address;
84         boolean delayedClosingQuote = false;
85 
86         // bit field that has bit set for bytes
87         // that occurred in this text
88         long byteBitFieldLo = 0;
89         long byteBitFieldHi = 0;
90 
91         int lineLen = 0;
92 
93         Vect.memset(matrix, matrixSize, 0);
94         while (cursor < hi && lineCount < lineCountLimit) {
95             byte b = Unsafe.getUnsafe().getByte(cursor++);
96 
97             if (delayedClosingQuote) {
98                 delayedClosingQuote = false;
99                 // this is double quote '""' situation
100                 if (b == '"') {
101                     lineLen++;
102                     continue;
103                 } else {
104                     // last quote was genuine closing one
105                     quotes = false;
106                 }
107             }
108 
109             // ignore everything that is in quotes
110             if (quotes) {
111                 if (b == '"') {
112                     delayedClosingQuote = true;
113                 }
114                 lineLen++;
115                 continue;
116             }
117 
118             switch (b) {
119                 case '\n':
120                 case '\r':
121                     // if line doesn't have content we just ignore
122                     // line end, thus skipping empty lines as well
123                     if (lineLen > 0) {
124                         // we are going to be storing this line length in 'A' byte
125                         // we don't collect frequencies on printable characters when trying to determine
126                         // column delimiter, therefore 'A' is free to use.
127                         bumpCountAt(lineCount, (byte) 'A', lineLen);
128                         byteBitFieldHi = byteBitFieldHi | (1L << (Byte.MAX_VALUE - 'A'));
129                         lineCount++;
130                         lineLen = 0;
131                     }
132                     continue;
133                 case '"':
134                     lineLen++;
135                     quotes = true;
136                     continue;
137                 default:
138                     lineLen++;
139                     if (potentialDelimiterBytes[b & 0xff] == 1) {
140                         break;
141                     }
142                     continue;
143             }
144 
145             bumpCountAt(lineCount, b, 1);
146 
147             if (b < 64) {
148                 byteBitFieldLo = byteBitFieldLo | (1L << b);
149             } else {
150                 byteBitFieldHi = byteBitFieldHi | (1L << (Byte.MAX_VALUE - b));
151             }
152         }
153 
154         // calculate standard deviation for every byte in the matrix
155         byte delimiter = Byte.MIN_VALUE;
156 
157         if (lineCount < 2) {
158             LOG.info().$("not enough lines [table=").$(tableName).$(']').$();
159             throw NotEnoughLinesException.$("not enough lines [table=").put(tableName).put(']');
160         }
161 
162         double lastDelimiterStdDev = Double.POSITIVE_INFINITY;
163         byte lastDelimiterPriority = Byte.MIN_VALUE;
164         double lastDelimiterMean = 0;
165         double lineLengthStdDev = 0;
166 
167         for (int i = 0, n = Byte.MAX_VALUE + 1; i < n; i++) {
168             boolean set;
169 
170             if (i < 64) {
171                 set = (byteBitFieldLo & (1L << i)) != 0;
172             } else {
173                 set = (byteBitFieldHi & (1L << (Byte.MAX_VALUE - i))) != 0;
174             }
175 
176             if (set) {
177                 long offset = i * 4L;
178 
179                 // calculate mean
180                 long sum = 0;
181                 for (int l = 0; l < lineCount; l++) {
182                     sum += Unsafe.getUnsafe().getInt(matrix + offset);
183                     offset += matrixRowSize;
184                 }
185 
186                 offset = i * 4L;
187                 final double mean = (double) sum / lineCount;
188                 if (mean > 0) {
189                     double squareSum = 0.0;
190                     for (int l = 0; l < lineCount; l++) {
191                         double x = Unsafe.getUnsafe().getInt(matrix + offset) - mean;
192                         squareSum += x * x;
193                         offset += matrixRowSize;
194                     }
195 
196                     final double stdDev = Math.sqrt(squareSum / lineCount);
197                     if (i == 'A') {
198                         lineLengthStdDev = stdDev;
199                         continue;
200                     }
201 
202                     final byte thisPriority = getPriority((byte) i);
203 
204                     // when stddev of this is less than last - use this
205                     // when stddev of this is the same as last then
206                     //    choose on priority (higher is better)
207                     //    when priority is the same choose on mean (higher is better)
208                     if (stdDev < lastDelimiterStdDev
209                             || (
210                             (Math.abs(stdDev - lastDelimiterStdDev) < DOUBLE_TOLERANCE)
211                                     &&
212                                     (lastDelimiterPriority < thisPriority || lastDelimiterPriority == thisPriority && lastDelimiterMean > mean)
213                     )) {
214                         lastDelimiterStdDev = stdDev;
215                         lastDelimiterPriority = thisPriority;
216                         lastDelimiterMean = mean;
217                         delimiter = (byte) i;
218                     }
219                 }
220             }
221         }
222 
223         // exclude '.' as delimiter
224         if (delimiter != '.' && lastDelimiterStdDev < maxRequiredDelimiterStdDev) {
225             LOG.info()
226                     .$("scan result [table=`").$(tableName)
227                     .$("`, delimiter='").$((char) delimiter)
228                     .$("', priority=").$(lastDelimiterPriority)
229                     .$(", mean=").$(lastDelimiterMean)
230                     .$(", stddev=").$(lastDelimiterStdDev)
231                     .$(']').$();
232 
233             return delimiter;
234         }
235 
236         // it appears we could not establish delimiter
237         // we could treat input as single column of data if line length change stays within tolerance
238         if (lineLengthStdDev < maxRequiredLineLengthStdDev) {
239             return ',';
240         }
241 
242         LOG.info()
243                 .$("min deviation is too high [delimiterStdDev=").$(lastDelimiterStdDev)
244                 .$(", delimiterMaxStdDev=").$(maxRequiredDelimiterStdDev)
245                 .$(", lineLengthStdDev=").$(lineLengthStdDev)
246                 .$(", lineLengthMaxStdDev=").$(maxRequiredLineLengthStdDev)
247                 .$(", lineCountLimit=").$(lineCountLimit)
248                 .$(", lineCount=").$(lineCount)
249                 .$(']').$();
250 
251         throw TextException.$("min deviation is too high [delimiterStdDev=")
252                 .put(lastDelimiterStdDev)
253                 .put(", delimiterMaxStdDev=").put(maxRequiredDelimiterStdDev)
254                 .put(", lineLengthStdDev=").put(lineLengthStdDev)
255                 .put(", lineLengthMaxStdDev=").put(maxRequiredLineLengthStdDev)
256                 .put(", lineCountLimit=").put(lineCountLimit)
257                 .put(", lineCount=").put(lineCount)
258                 .put(']');
259     }
