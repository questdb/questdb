82     public void evaluateResults(long lineCount, long errorCount) {
83         // try to calculate types counting all rows
84         // if all types come up as strings, reduce lineCount by one and retry
85         // if some fields come up as non-string after subtracting row - we have a header
86         if ((calcTypes(lineCount - errorCount, true) && !calcTypes(lineCount - errorCount - 1, false)) || forceHeader) {
87             // copy headers
88             header = true;
89         } else {
90             LOG.info()
91                     .$("no header [table=").$(tableName)
92                     .$(", lineCount=").$(lineCount)
93                     .$(", errorCount=").$(errorCount)
94                     .$(", forceHeader=").$(forceHeader)
95                     .$(']').$();
96         }
97 
98         for (int i = 0; i < fieldCount; i++) {
99             if (!header || columnNames.getQuick(i).length() == 0) {
100                 tempSink.clear();
101                 tempSink.put('f').put(i);
102 
103                 if (header) {
104                     for (int attempt = 0; attempt < 20; attempt++) {
105                         if (!columnNames.contains(tempSink)) {
106                             break;
107                         }
108 
109                         tempSink.put('_');
110                     }
111 
112                     if (columnNames.contains(tempSink)) {
113                         throw TextException.$("Failed to generate unique name for column [no=").put(i).put("]");
114                     }
115                 }
116 
117                 columnNames.setQuick(i, tempSink.toString());
118             }
119 
120             if (!uniqueColumnNames.add(columnNames.getQuick(i))) {
121                 throw TextException.$("duplicate column name found [no=").put(i).put(",name=").put(columnNames.get(i)).put(']');
122             }
123         }
124 
125         // override calculated types with user-supplied information
126         if (schemaColumns.size() > 0) {
127             for (int i = 0, k = columnNames.size(); i < k; i++) {
128                 TypeAdapter type = schemaColumns.get(columnNames.getQuick(i));
129                 if (type != null) {
130                     columnTypes.setQuick(i, type);
131                 }
132             }
133         }
134     }

153     @Override
154     public void onFields(long line, ObjList<DirectByteCharSequence> values, int fieldCount) {
155         // keep first line in case it's a header
156         if (line == 0) {
157             seedFields(fieldCount);
158             stashPossibleHeader(values, fieldCount);
159         }
160 
161         int count = typeManager.getProbeCount();
162         for (int i = 0; i < fieldCount; i++) {
163             DirectByteCharSequence cs = values.getQuick(i);
164             if (cs.length() == 0) {
165                 _blanks.increment(i);
166             }
167             int offset = i * count;
168             for (int k = 0; k < count; k++) {
169                 final TypeAdapter probe = typeManager.getProbe(k);
170                 if (probe.probe(cs)) {
171                     _histogram.increment(k + offset);
172                 }
173             }
174         }
175     }

177     /**
178      * Histogram contains counts for every probe that validates field. It is possible for multiple probes to validate same field.
179      * It can happen because of two reasons.
180      * <p>
181      * probes are compatible, for example INT is compatible with DOUBLE in a sense that DOUBLE probe will positively
182      * validate every INT. If this the case we will use order of probes as priority. First probe wins
183      * <p>
184      * it is possible to have mixed types in same column, in which case column has to become string.
185      * to establish if we have mixed column we check if probe count + blank values add up to total number of rows.
186      */
187     private boolean calcTypes(long count, boolean setDefault) {
188         boolean allStrings = true;
189         int probeCount = typeManager.getProbeCount();
190         for (int i = 0; i < fieldCount; i++) {
191             int offset = i * probeCount;
192             int blanks = _blanks.getQuick(i);
193             boolean unprobed = true;
194 
195             for (int k = 0; k < probeCount; k++) {
196                 if (_histogram.getQuick(k + offset) + blanks == count && blanks < count) {
197                     unprobed = false;
198                     columnTypes.setQuick(i, typeManager.getProbe(k));
199                     if (allStrings && typeManager.getProbe(k).getType() != ColumnType.CHAR) {
200                         allStrings = false;
201                     }
202                     break;
203                 }
204             }
205 
206             if (setDefault && unprobed) {
207                 columnTypes.setQuick(i, typeManager.getTypeAdapter(ColumnType.STRING));
208             }
209         }
210 
211         return allStrings;
212     }

218     private String normalise(CharSequence seq) {
219         boolean capNext = false;
220         tempSink.clear();
221         for (int i = 0, l = seq.length(); i < l; i++) {
222             char c = seq.charAt(i);
223             switch (c) {
224                 case ' ':
225                 case '?':
226                 case '.':
227                 case ',':
228                 case '\'':
229                 case '\"':
230                 case '\\':
231                 case '/':
232                 case ':':
233                 case ')':
234                 case '(':
235                 case '+':
236                 case '-':
237                 case '*':
238                 case '%':
239                 case '~':
240                 case '\u0000':
241                 case '\u0001':
242                 case '\u0002':
243                 case '\u0003':
244                 case '\u0004':
245                 case '\u0005':
246                 case '\u0006':
247                 case '\u0007':
248                 case '\u0008':
249                 case '\u0009':
250                 case '\u000B':
251                 case '\u000c':
252                 case '\r':
253                 case '\n':
254                 case '\u000e':
255                 case '\u000f':
256                 case '\u007f':
257                     capNext = true;
258                 case 0xfeff: // UTF-8 BOM (Byte Order Mark) can appear at the beginning of a character stream
259                     break;
260                 default:
261                     if (tempSink.length() == 0 && Character.isDigit(c)) {
262                         tempSink.put('_');
263                     }
264 
265                     if (capNext) {
266                         tempSink.put(Character.toUpperCase(c));
267                         capNext = false;
268                     } else {
269                         tempSink.put(c);
270                     }
271                     break;
272             }
273         }
274         return Chars.toString(tempSink);
275     }

284     private void stashPossibleHeader(ObjList<DirectByteCharSequence> values, int hi) {
285         for (int i = 0; i < hi; i++) {
286             DirectByteCharSequence value = values.getQuick(i);
287             utf8Sink.clear();
288             if (Chars.utf8Decode(value.getLo(), value.getHi(), utf8Sink)) {
289                 columnNames.setQuick(i, normalise(utf8Sink));
290             } else {
291                 LOG.info().$("utf8 error [table=").$(tableName).$(", line=0, col=").$(i).$(']').$();
292             }
293         }
