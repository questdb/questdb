122     private void updateStatus(
123             byte phase,
124             byte status,
125             final CharSequence msg,
126             long rowsHandled,
127             long rowsImported,
128             long errors
129     ) {
130         if (writer != null) {
131             stringSink.clear();
132             Numbers.appendHex(stringSink, task.getImportId(), true);
133             try {
134                 TableWriter.Row row = writer.newRow(clock.getTicks());
135                 row.putStr(1, stringSink);
136                 row.putSym(2, task.getTableName());
137                 row.putSym(3, task.getFileName());
138                 row.putSym(4, TextImportTask.getPhaseName(phase));
139                 row.putSym(5, TextImportTask.getStatusName(status));
140                 row.putStr(6, msg);
141                 row.putLong(7, rowsHandled);
142                 row.putLong(8, rowsImported);
143                 row.putLong(9, errors);
144                 row.append();
145                 writer.commit();
146             } catch (Throwable th) {
147                 LOG.error()
148                         .$("could not update status table [importId=").$hexPadded(task.getImportId())
149                         .$(", statusTableName=").$(statusTableToken)
150                         .$(", tableName=").$(task.getTableName())
151                         .$(", fileName=").$(task.getFileName())
152                         .$(", phase=").$(getPhaseName(phase))
153                         .$(", status=").$(getStatusName(phase))
154                         .$(", msg=").$(msg)
155                         .$(", rowsHandled=").$(rowsHandled)
156                         .$(", rowsImported=").$(rowsImported)
157                         .$(", errors=").$(errors)
158                         .$(", error=`").$(th).$('`')
159                         .I$();
160                 writer = Misc.free(writer);
161             }
162 
163             // if we closed the writer, we need to reopen it again
164             if (writer == null) {
165                 try {
166                     writer = engine.getWriter(AllowAllCairoSecurityContext.INSTANCE, statusTableToken, "QuestDB system");

170                             .$(", error=`").$(e).$('`')

172                 }
173             }
174         }
175     }

177     private boolean useParallelImport() {
178         TableToken tableToken = engine.getTableTokenIfExists(task.getTableName());
179         if (engine.getStatus(sqlExecutionContext.getCairoSecurityContext(), path, tableToken) != TableUtils.TABLE_EXISTS) {
180             return task.getPartitionBy() >= 0 && task.getPartitionBy() != PartitionBy.NONE;
181         }
182         try (TableReader reader = engine.getReader(sqlExecutionContext.getCairoSecurityContext(), tableToken)) {
183             return PartitionBy.isPartitioned(reader.getPartitionedBy());
184         }
185     }

187     void enforceLogRetention() {
188         if (writer != null) {
189             if (logRetentionDays < 1) {
190                 writer.truncate();
191                 return;
192             }
193             if (writer.getPartitionCount() > 0) {
194                 partitionsToRemove.clear();
195                 for (int i = writer.getPartitionCount() - logRetentionDays - 1; i > -1; i--) {
196                     partitionsToRemove.add(writer.getPartitionTimestamp(i));
197                 }
198 
199                 for (int i = 0, sz = partitionsToRemove.size(); i < sz; i++) {
200                     writer.removePartition(partitionsToRemove.getQuick(i));
201                 }
202             }
203         }
204     }

206     @Override
207     protected boolean runSerially() {
208         long cursor = requestSubSeq.next();
209         if (cursor > -1) {
210             task = requestQueue.get(cursor);
211             try {
212                 if (useParallelImport()) {
213                     parallelImporter.of(
214                             task.getTableName(),
215                             task.getFileName(),
216                             task.getImportId(),
217                             task.getPartitionBy(),
218                             task.getDelimiter(),
219                             task.getTimestampColumnName(),
220                             task.getTimestampFormat(),
221                             task.isHeaderFlag(),
222                             textImportExecutionContext.getCircuitBreaker(),
223                             task.getAtomicity()
224                     );
225                     parallelImporter.setStatusReporter(updateStatusRef);
226                     parallelImporter.process();
227                 } else {
228                     serialImporter.of(
229                             task.getTableName(),
230                             task.getFileName(),
231                             task.getImportId(),
232                             task.getDelimiter(),
233                             task.getTimestampColumnName(),
234                             task.getTimestampFormat(),
235                             task.isHeaderFlag(),
236                             textImportExecutionContext.getCircuitBreaker(),
237                             task.getAtomicity()
238                     );
239                     serialImporter.setStatusReporter(updateStatusRef);
240                     serialImporter.process();
241                 }
242             } catch (TextImportException e) {
243                 updateStatus(
244                         TextImportTask.NO_PHASE,
245                         e.isCancelled() ? TextImportTask.STATUS_CANCELLED : TextImportTask.STATUS_FAILED,
246                         e.getMessage(),
247                         0,
248                         0,
249                         0
250                 );
251             } finally {
252                 requestSubSeq.done(cursor);
253                 textImportExecutionContext.resetActiveImportId();
254             }
255             enforceLogRetention();
256             return true;
257         }
258         return false;
259     }
