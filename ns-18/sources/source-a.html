94     public void process() throws TextImportException {
95         LOG.info()
96                 .$("started [importId=").$hexPadded(importId)
97                 .$(", file=`").$(inputFilePath).$('`').I$();
98 
99         final long startMs = getCurrentTimeMs();
100 
101         updateImportStatus(TextImportTask.STATUS_STARTED, Numbers.LONG_NaN, Numbers.LONG_NaN, 0);
102         setupTextLoaderFromModel();
103 
104         final int sqlCopyBufferSize = cairoEngine.getConfiguration().getSqlCopyBufferSize();
105         final long buf = Unsafe.malloc(sqlCopyBufferSize, MemoryTag.NATIVE_IMPORT);
106         int fd = -1;
107         try {
108             fd = TableUtils.openRO(ff, inputFilePath, LOG);
109             long fileLen = ff.length(fd);
110             long n = ff.read(fd, buf, sqlCopyBufferSize, 0);
111             if (n > 0) {
112                 if (columnDelimiter > 0) {
113                     textLoader.configureColumnDelimiter(columnDelimiter);
114                 }
115                 textLoader.setForceHeaders(forceHeader);
116                 textLoader.setSkipLinesWithExtraValues(false);
117                 textLoader.parse(buf, buf + n, securityContext);
118                 textLoader.setState(TextLoader.LOAD_DATA);
119                 int read;
120                 while (n < fileLen) {
121                     if (circuitBreaker.checkIfTripped()) {
122                         TextImportException ex = TextImportException.instance(TextImportTask.NO_PHASE, "import was cancelled");
123                         ex.setCancelled(true);
124                         throw ex;
125                     }
126                     read = (int) ff.read(fd, buf, sqlCopyBufferSize, n);
127                     if (read < 1) {
128                         throw TextImportException.instance(TextImportTask.NO_PHASE, "could not read file [errno=").put(ff.errno()).put(']');
129                     }
130                     textLoader.parse(buf, buf + read, securityContext);
131                     n += read;
132                 }
133                 textLoader.wrapUp();
134 
135                 long errorCount = textLoader.getErrorLineCount();
136                 LongList columnErrorCounts = textLoader.getColumnErrorCounts();
137                 for (int i = 0, size = columnErrorCounts.size(); i < size; i++) {
138                     errorCount += columnErrorCounts.get(i);
139                 }
140                 updateImportStatus(TextImportTask.STATUS_FINISHED, textLoader.getParsedLineCount(), textLoader.getWrittenLineCount(), errorCount);
141 
142                 long endMs = getCurrentTimeMs();
143                 LOG.info()
144                         .$("import complete [importId=").$hexPadded(importId)
145                         .$(", file=`").$(inputFilePath).$('`')
146                         .$("', time=").$((endMs - startMs) / 1000).$('s')
147                         .I$();
148             }
149         } catch (TextException e) {
150             throw TextImportException.instance(TextImportTask.NO_PHASE, e.getFlyweightMessage());
151         } catch (CairoException e) {
152             throw TextImportException.instance(TextImportTask.NO_PHASE, e.getFlyweightMessage(), e.getErrno());
153         } finally {
154             ff.close(fd);
155             textLoader.clear();
156             Unsafe.free(buf, sqlCopyBufferSize, MemoryTag.NATIVE_IMPORT);
157         }
158     }
