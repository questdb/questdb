57 /**
58  * Class is responsible for importing of large unordered import files into partitioned tables.
59  * It does the following (in parallel) :
60  * - splits the file into N-chunks, scans in parallel and finds correct line start for each chunk
61  * - scans each chunk and extracts timestamps and line offsets to per-partition index files
62  * (index files are stored as $inputWorkDir/$inputFileName/$partitionName/$workerId_$chunkNumber)
63  * then it sorts each file by timestamp value
64  * - merges all partition index chunks into one index file per partition (index.m)
65  * - loads partitions into separate tables using merged indexes (one table per worker)
66  * - scans all symbol columns to build per-column global symbol table
67  * - remaps all symbol values
68  * - moves and attaches partitions from temp tables to target table
69  * - removes temp tables and index files
70  */

379         final long startMs = getCurrentTimeMs();
380 
381         int fd = -1;
382         try {
383             try {
384                 updateImportStatus(TextImportTask.STATUS_STARTED, Numbers.LONG_NaN, Numbers.LONG_NaN, 0);
385 
386                 try {
387                     fd = TableUtils.openRO(ff, inputFilePath, LOG);
388                 } catch (CairoException e) {
389                     throw TextImportException.instance(TextImportTask.PHASE_SETUP, e.getFlyweightMessage(), e.getErrno());
390                 }
391 
392                 long length = ff.length(fd);
393                 if (length < 1) {
394                     throw TextImportException.instance(TextImportTask.PHASE_SETUP, "ignored empty input file [file='").put(inputFilePath).put(']');
395                 }
396 
397                 try (TableWriter writer = parseStructure(fd)) {
398                     phaseBoundaryCheck(length);
399                     phaseIndexing();
400                     phasePartitionImport();
401                     phaseSymbolTableMerge(writer);
402                     phaseUpdateSymbolKeys(writer);
403                     phaseBuildSymbolIndex(writer);
404                     try {
405                         movePartitions();
406                         attachPartitions(writer);
407                     } catch (Throwable t) {

410                     }

417                         removeWorkDir();
418                     }
419                 }

421             } catch (CairoException e) {
422                 throw TextImportException.instance(TextImportTask.PHASE_CLEANUP, e.getFlyweightMessage(), e.getErrno());
423             } catch (TextException e) {
424                 throw TextImportException.instance(TextImportTask.PHASE_CLEANUP, e.getFlyweightMessage());
425             } finally {

427             }

1136     //load balances existing partitions between given number of workers using partition sizes
1137     //returns number of tasks
1138     static int assignPartitions(ObjList<PartitionInfo> partitions, int workerCount) {
1139         partitions.sort((p1, p2) -> Long.compare(p2.bytes, p1.bytes));
1140         long[] workerSums = new long[workerCount];
1141 
1142         for (int i = 0, n = partitions.size(); i < n; i++) {
1143             int minIdx = -1;
1144             long minSum = Long.MAX_VALUE;
1145 
1146             for (int j = 0; j < workerCount; j++) {
1147                 if (workerSums[j] == 0) {
1148                     minIdx = j;
1149                     break;
1150                 } else if (workerSums[j] < minSum) {
1151                     minSum = workerSums[j];
1152                     minIdx = j;
1153                 }
1154             }
1155 
1156             workerSums[minIdx] += partitions.getQuick(i).bytes;
1157             partitions.getQuick(i).taskId = minIdx;
1158         }
1159 
1160         partitions.sort((p1, p2) -> {
1161             long workerIdDiff = p1.taskId - p2.taskId;
1162             if (workerIdDiff != 0) {
1163                 return (int) workerIdDiff;
1164             }
1165 
1166             return Long.compare(p1.key, p2.key);
1167         });
1168 
1169         int taskIds = 0;
1170         for (int i = 0, n = workerSums.length; i < n; i++) {
1171             if (workerSums[i] != 0) {
1172                 taskIds++;
1173             }
1174         }
1175 
1176         return taskIds;
1177     }
