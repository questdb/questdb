117     @Override
118     public void onEvent(int code, CharSequence tag, int position) throws JsonException {
119         switch (code) {
120             case JsonLexer.EVT_ARRAY_START:
121                 if (state != S_NEED_ARRAY) {
122                     throw JsonException.$(position, "Unexpected array");
123                 }
124                 state = S_NEED_OBJECT;
125                 break;
126             case JsonLexer.EVT_OBJ_START:
127                 if (state != S_NEED_OBJECT) {
128                     throw JsonException.$(position, "Unexpected object");
129                 }
130                 state = S_NEED_PROPERTY;
131                 break;
132             case JsonLexer.EVT_NAME:
133                 this.propertyIndex = propertyNameMap.get(tag);
134                 if (this.propertyIndex == -1) {
135                     LOG.info().$("unknown [table=").$(tableName).$(", tag=").$(tag).$(']').$();
136                 }
137                 break;
138             case JsonLexer.EVT_VALUE:
139                 switch (propertyIndex) {
140                     case P_NAME:
141                         name = copy(tag);
142                         break;
143                     case P_TYPE:
144                         type = ColumnType.typeOf(tag);
145                         if (type == -1) {
146                             throw JsonException.$(position, "Invalid type");
147                         }
148                         break;
149                     case P_PATTERN:
150                         pattern = copy(tag);
151                         break;
152                     case P_LOCALE:
153                         locale = copy(tag);
154                         localePosition = position;
155                         break;
156                     case P_UTF8:
157                         utf8 = SqlKeywords.isTrueKeyword(tag);
158                         break;
159                     case P_INDEX:
160                         index = SqlKeywords.isTrueKeyword(tag);
161                         break;
162                     default:
163                         LOG.info().$("ignoring [table=").$(tableName).$(", value=").$(tag).$(']').$();
164                         break;
165                 }
166                 break;
167             case JsonLexer.EVT_OBJ_END:
168                 state = S_NEED_OBJECT;
169                 createImportedType(position);
170                 break;
171             case JsonLexer.EVT_ARRAY_VALUE:
172                 throw JsonException.$(position, "Must be an object");
173             default:
174                 break;
175         }
176     }

204     private CharSequence copy(CharSequence tag) {
205         final int l = tag.length() * 2;
206         final long n = bufSize + l;
207         if (n > bufCapacity) {
208             long ptr = Unsafe.malloc(n * 2, MemoryTag.NATIVE_TEXT_PARSER_RSS);
209             Vect.memcpy(ptr, buf, bufSize);
210             if (bufCapacity > 0) {
211                 Unsafe.free(buf, bufCapacity, MemoryTag.NATIVE_TEXT_PARSER_RSS);
212             }
213             buf = ptr;
214             bufCapacity = n * 2;
215         }
216 
217         strcpyw(tag, l / 2, buf + bufSize);
218         CharSequence cs = csPool.next().of(bufSize, l / 2);
219         bufSize += l;
220         return cs;
221     }

223     private void createImportedType(int position) throws JsonException {
224         checkInputs(position, name, type);
225 
226         columnNames.add(name);
227 
228         switch (ColumnType.tagOf(type)) {
229             case ColumnType.DATE:
230                 DateLocale dateLocale = locale == null ? this.dateLocale : dateLocaleFactory.getLocale(locale);
231 
232                 if (dateLocale == null) {
233                     throw JsonException.$(localePosition, "Invalid date locale");
234                 }
235 
236                 // date pattern is required
237                 if (pattern == null) {
238                     throw JsonException.$(0, "DATE format pattern is required");
239                 }
240                 columnTypes.add(typeManager.nextDateAdapter().of(dateFormatFactory.get(pattern), dateLocale));
241                 break;
242             case ColumnType.TIMESTAMP:
243                 DateLocale timestampLocale =
244                         locale == null ?
245                                 this.dateLocale
246                                 : dateLocaleFactory.getLocale(locale);
247                 if (timestampLocale == null) {
248                     throw JsonException.$(localePosition, "Invalid timestamp locale");
249                 }
250 
251                 // timestamp pattern is required
252                 if (pattern == null) {
253                     throw JsonException.$(0, "TIMESTAMP format pattern is required");
254                 }
255                 columnTypes.add(typeManager.nextTimestampAdapter(utf8, timestampFormatFactory.get(pattern), timestampLocale));
256                 break;
257             case ColumnType.SYMBOL:
258                 columnTypes.add(typeManager.nextSymbolAdapter(index));
259                 break;
260             default:
261                 columnTypes.add(typeManager.getTypeAdapter(type));
262                 break;
263         }
264         // prepare for next iteration
265         clearStage();
266     }
