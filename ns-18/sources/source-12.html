273         } catch (TextImportException e) {

389         public void run(long fileBufPtr, long fileBufSize) throws TextException {
390             long offset = chunkStart;
391 
392             //output vars
393             long quotes = 0;
394             long[] nlCount = new long[2];
395             long[] nlFirst = new long[]{-1, -1};
396 
397             long read;
398             long ptr;
399             long hi;
400 
401             int fd = TableUtils.openRO(ff, path, LOG);
402             ff.fadvise(fd, chunkStart, chunkEnd - chunkStart, Files.POSIX_FADV_SEQUENTIAL);
403             try {
404 
405                 do {
406                     long leftToRead = Math.min(chunkEnd - offset, fileBufSize);
407                     read = (int) ff.read(fd, fileBufPtr, leftToRead, offset);
408                     if (read < 1) {
409                         break;
410                     }
411                     hi = fileBufPtr + read;
412                     ptr = fileBufPtr;
413 
414                     while (ptr < hi) {
415                         final byte c = Unsafe.getUnsafe().getByte(ptr++);
416                         if (c == '"') {
417                             quotes++;
418                         } else if (c == '\n') {
419                             nlCount[(int) (quotes & 1)]++;
420                             if (nlFirst[(int) (quotes & 1)] == -1) {
421                                 nlFirst[(int) (quotes & 1)] = offset + (ptr - fileBufPtr);
422                             }
423                         }
424                     }
425 
426                     offset += read;
427                 } while (offset < chunkEnd);
428 
429                 if (read < 0 || offset < chunkEnd) {
430                     throw TextException
431                             .$("could not read import file [path='").put(path)
432                             .put("', offset=").put(offset)
433                             .put(", errno=").put(ff.errno())
434                             .put(']');
435                 }
436             } finally {
437                 ff.close(fd);
438             }
439 
440             this.quoteCount = quotes;
441             this.newLineCountEven = nlCount[0];
442             this.newLineCountOdd = nlCount[1];
443             this.newLineOffsetEven = nlFirst[0];
444             this.newLineOffsetOdd = nlFirst[1];
445         }
446     }

629             path.concat(columnName).put(TableUtils.FILE_SUFFIX_D);
630 
631             long columnMemory = 0;
632             long columnMemorySize = 0;
633             long remapTableMemory = 0;
634             long remapTableMemorySize = 0;
635             int columnFd = -1;

637             try {

649                     long columnMemSize = partitionSize * Integer.BYTES;
650                     long remapMemSize = (long) symbolCount * Integer.BYTES;
651                     ColumnUtils.symbolColumnUpdateKeys(columnMemory, columnMemSize, remapTableMemory, remapMemSize);
652                 }

655                 ff.close(remapFd);
656                 if (columnMemory > 0) {

658                 }
659                 if (remapTableMemory > 0) {

918         private void consumeIOURing(
919                 FilesFacade ff,
920                 long sqeMin,
921                 AbstractTextLexer lexer,
922                 long fileBufAddr,
923                 LongList offsets,
924                 IOURing ring,
925                 int cc,
926                 Path tmpPath
927         ) {
928             int submitted = ring.submit();
929             assert submitted == cc;
930 
931             long nextCqe = sqeMin;
932             int writtenMax = 0;
933             // consume submitted tasks
934             for (int i = 0; i < submitted; i++) {
935 
936                 while (!ring.nextCqe()) {
937                     Os.pause();
938                 }
939 
940                 if (ring.getCqeRes() < 0) {
941                     throw TextException
942                             .$("could not read from file [path='").put(tmpPath)

945                             .put("]");
946                 }

949                     // only parse lines in order of submissions
950                     nextCqe++;
951                     parseLinesAndWrite(lexer, fileBufAddr, offsets, writtenMax);
952                     writtenMax++;

954             }
955

957             for (int i = writtenMax; i < submitted; i++) {

1012         private void importPartitionDataURing(
1013                 final IOURingFacade rf,
1014                 AbstractTextLexer lexer,
1015                 long address,
1016                 long size,
1017                 long fileBufAddr,
1018                 long fileBufSize,
1019                 DirectCharSink utf8Sink,
1020                 Path tmpPath
1021         ) {
1022             final CairoConfiguration configuration = cairoEngine.getConfiguration();
1023             final FilesFacade ff = configuration.getFilesFacade();
1024 
1025             offsets.clear();
1026             lexer.setupBeforeExactLines(onFieldsPartitioned);
1027 
1028             int fd = -1;
1029             try {
1030                 tmpPath.of(configuration.getSqlCopyInputRoot()).concat(inputFileName).$();
1031                 utf8Sink.clear();
1032                 fd = TableUtils.openRO(ff, tmpPath, LOG);
1033 
1034                 final long len = ff.length(fd);
1035                 if (len == -1) {
1036                     throw CairoException.critical(ff.errno())
1037                             .put("could not get length of file [path=").put(tmpPath)
1038                             .put(']');
1039                 }
1040 
1041                 ff.fadvise(fd, 0, len, Files.POSIX_FADV_RANDOM);
1042 
1043                 final long MASK = ~((255L) << 56 | (255L) << 48);
1044                 final long count = size / (2 * Long.BYTES);
1045 
1046                 int ringCapacity = 32;
1047                 long sqeMin = 0;
1048                 long sqeMax = -1;
1049                 try (IOURing ring = rf.newInstance(ringCapacity)) {
1050                     long addr = fileBufAddr;
1051                     long lim = fileBufAddr + fileBufSize;
1052                     int cc = 0;
1053                     int bytesToRead;
1054                     int additionalLines;
1055                     for (long i = 0; i < count; i++) {
1056                         throwIfCancelled();
1057 
1058                         final long lengthAndOffset = Unsafe.getUnsafe().getLong(address + i * 2L * Long.BYTES + Long.BYTES);
1059                         final int lineLength = (int) (lengthAndOffset >>> 48);
1060                         // the offset is used by the callback to report errors
1061                         offset = lengthAndOffset & MASK;
1062                         bytesToRead = lineLength;
1063 
1064                         // schedule reads until we either run out of ring capacity or
1065                         // our read buffer size
1066 
1067                         if (cc == ringCapacity || (cc > 0 && addr + lineLength > lim)) {
1068                             // we are out of ring capacity or our buffer is exhausted
1069                             consumeIOURing(ff, sqeMin, lexer, fileBufAddr, offsets, ring, cc, tmpPath);
1070 
1071                             cc = 0;
1072                             addr = fileBufAddr;
1073                             offsets.clear();
1074                             sqeMin = sqeMax + 1;
1075                         }
1076                         if (addr + lineLength > lim) {
1077                             throw TextException.$("buffer overflow [path='").put(tmpPath)
1078                                     .put("', lineLength=").put(lineLength)
1079                                     .put(", fileBufSize=").put(fileBufSize)
1080                                     .put("]");
1081                         }
1082 
1083                         // try to coalesce ahead lines into the same read, if they're sequential
1084                         additionalLines = 0;
1085                         for (long j = i + 1; j < count; j++) {
1086                             long nextLengthAndOffset = Unsafe.getUnsafe().getLong(address + j * 2L * Long.BYTES + Long.BYTES);
1087                             int nextLineLength = (int) (nextLengthAndOffset >>> 48);
1088                             long nextOffset = nextLengthAndOffset & MASK;
1089 
1090                             // line indexing stops on first EOL char, e.g. \r, but it could be followed by \n
1091                             long diff = nextOffset - offset - bytesToRead;
1092                             long nextBytesToRead = diff + nextLineLength;
1093                             if (diff > -1 && diff < 2 && addr + bytesToRead + nextBytesToRead <= lim) {
1094                                 bytesToRead += nextBytesToRead;
1095                                 additionalLines++;
1096                             } else {
1097                                 break;
1098                             }
1099                         }
1100                         i += additionalLines;
1101 
1102                         sqeMax = ring.enqueueRead(fd, offset, addr, bytesToRead);
1103                         if (sqeMax == -1) {
1104                             throw TextException.$("io_uring error [path='").put(tmpPath)
1105                                     .put("', cqeRes=").put(-ring.getCqeRes())
1106                                     .put("]");
1107                         }
1108 
1109                         offsets.add(addr - fileBufAddr, bytesToRead);
1110 
1111                         cc++;
1112                         addr += bytesToRead;
1113                     } // for
1114 
1115                     // check if something is enqueued
1116                     if (cc > 0) {
1117                         consumeIOURing(ff, sqeMin, lexer, fileBufAddr, offsets, ring, cc, tmpPath);
1118                     }
1119                 }
1120 
1121             } finally {
1122                 ff.close(fd);
1123             }
1124         }
