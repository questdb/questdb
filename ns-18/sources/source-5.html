205             long chunkLo,
206             long chunkHi,
207             long lineNumber,
208             LongList partitionKeysAndSizes,
209             long fileBufAddr,
210             long fileBufSize
211     ) {
212         assert chunkHi > 0;
213         assert chunkLo >= 0 && chunkLo < chunkHi;
214 
215         openInputFile();
216 
217         this.offset = chunkLo;
218         long read;
219 
220         this.lastLineStart = offset;
221         this.lineNumber = lineNumber;
222 
223         try {
224             do {
225                 if (circuitBreaker != null && circuitBreaker.checkIfTripped()) {
226                     this.cancelled = true;
227                     throw TextException.$("Cancelled");
228                 }
229                 long leftToRead = Math.min(chunkHi - offset, fileBufSize);
230                 read = (int) ff.read(fd, fileBufAddr, leftToRead, offset);
231                 if (read < 1) {
232                     break;
233                 }
234                 parse(fileBufAddr, fileBufAddr + read);
235                 offset += read;
236             } while (offset < chunkHi);
237 
238             if (read < 0 || offset < chunkHi) {
239                 throw TextException

241                         .put("', offset=").put(offset)

243                         .put(']');
244             } else {
245                 parseLast();
246             }

249             sortAndCloseOutputFiles();
250         } finally {
251             closeOutputFiles();//close without sorting if there's an error
252             closeSortBuffer();

264     public void indexLine(long ptr, long lo) throws TextException {
265         //this is fine because Long.MIN_VALUE is treated as null and would be rejected by partitioned tables
266         if (timestampValue == Long.MIN_VALUE) {
267             return;
268         }
269 
270         long lineStartOffset = lastLineStart;
271         long length = offset + ptr - lo - lastLineStart;
272         if (length >= (1L << 16)) {
273             LOG.error().$("row exceeds maximum line length (65k) for parallel import [line=").$(lineNumber)
274                     .$(", length=").$(length).I$();
275             errorCount++;
276             return;
277         }
278 
279         // second long stores:
280         //  length as 16 bits unsigned number followed by
281         //  offset as 48-bits unsigned number
282         //  allowing for importing 256TB big files with rows up to 65kB long
283         long lengthAndOffset = (length << 48 | lineStartOffset);
284         long partitionKey = partitionFloorMethod.floor(timestampValue);
285         long mapKey = partitionKey / Timestamps.HOUR_MICROS; //remove trailing zeros to avoid excessive collisions in hashmap
286 
287         final IndexOutputFile target;
288         int keyIndex = outputFileLookupMap.keyIndex(mapKey);
289         if (keyIndex > -1) {
290             target = prepareTargetFile(partitionKey);
291             outputFileDenseList.add(target);
292             outputFileLookupMap.putAt(keyIndex, mapKey, target);
293         } else {
294             target = outputFileLookupMap.valueAt(keyIndex);
295         }
296 
297         if (target.indexChunkSize == maxIndexChunkSize) {
298             target.nextChunk(ff, getPartitionIndexPrefix(partitionKey));
299         }
300 
301         target.putEntry(timestampValue, lengthAndOffset, length);
302     }

477     private void parse(long lo, long hi) {
478         this.fieldHi = useFieldRollBuf ? fieldRollBufCur : (this.fieldLo = lo);
479         long ptr = lo;
480 
481         while (ptr < hi) {
482             final byte c = Unsafe.getUnsafe().getByte(ptr++);
483 
484             if (rollBufferUnusable) {
485                 eol(ptr, c);
486                 continue;
487             }
488 
489             if (useFieldRollBuf) {
490                 putToRollBuf(c);
491                 if (rollBufferUnusable) {
492                     continue;
493                 }
494             }
495 
496             this.fieldHi++;
497 
498             if (delayedOutQuote && c != '"') {
499                 inQuote = delayedOutQuote = false;
500             }
501 
502             if (c == columnDelimiter) {
503                 onColumnDelimiter(lo, ptr);
504             } else if (c == '"') {
505                 checkEol(lo);
506                 onQuote();
507             } else if (c == '\n' || c == '\r') {
508                 onLineEnd(ptr, lo);
509             } else {
510                 checkEol(lo);
511             }
512         }
513 
514         if (useFieldRollBuf) {
515             return;
516         }
517 
518         if (eol) {
519             this.fieldLo = 0;
520         } else if (fieldIndex == timestampIndex) {
521             rollField(hi);
522         }
523     }

562     private void rollField(long hi) {
563         // lastLineStart is an offset from 'lo'
564         // 'lo' is the address of incoming buffer
565         int length = (int) (hi - fieldLo);
566         if (length > 0 && fitsInBuffer(length)) {
567             assert fieldLo + length <= hi;
568             Vect.memcpy(fieldRollBufPtr, fieldLo, length);
569             fieldRollBufCur = fieldRollBufPtr + length;
570             shift(fieldLo - fieldRollBufPtr);
571             useFieldRollBuf = true;
572         }
