272     public void wrapUp() throws TextException {
273         switch (state) {
274             case LOAD_JSON_METADATA:
275                 try {
276                     jsonLexer.parseLast();
277                 } catch (JsonException e) {
278                     throw TextException.$(e.getFlyweightMessage());
279                 }
280                 break;
281             case ANALYZE_STRUCTURE:
282             case LOAD_DATA:
283                 // when file is empty lexer could be null because we
284                 // didn't get to find out the delimiter
285                 if (lexer != null) {
286                     lexer.parseLast();
287                 }
288                 textWriter.commit();
289                 break;
290             default:
291                 break;
292         }
293     }

307     private void parseStructure(long lo, long hi, CairoSecurityContext cairoSecurityContext) throws TextException {
308         if (columnDelimiter > 0) {
309             setDelimiter(columnDelimiter);
310         } else {
311             setDelimiter(textDelimiterScanner.scan(lo, hi));
312         }
313 
314         if (timestampColumn != null && timestampAdapter != null) {
315             textMetadataParser.getColumnNames().add(timestampColumn);
316             textMetadataParser.getColumnTypes().add(timestampAdapter);
317         }
318 
319         textMetadataDetector.of(
320                 getTableName(),
321                 textMetadataParser.getColumnNames(),
322                 textMetadataParser.getColumnTypes(),
323                 forceHeaders
324         );
325         parse(lo, hi, textAnalysisMaxLines, textMetadataDetector);
326         textMetadataDetector.evaluateResults(getParsedLineCount(), getErrorLineCount());
327         restart(textMetadataDetector.isHeader());
328 
329         prepareTable(
330                 cairoSecurityContext,
331                 textMetadataDetector.getColumnNames(),
332                 textMetadataDetector.getColumnTypes(),
333                 path,
334                 typeManager,
335                 timestampAdapter
336         );
337         parse(lo, hi, Integer.MAX_VALUE);
338         state = LOAD_DATA;
339     }
