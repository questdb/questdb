244     private boolean growRollBuf(int requiredLength, boolean updateFields) {
245         if (requiredLength > lineRollBufLimit) {
246             LOG.info()
247                     .$("too long [table=").$(tableName)
248                     .$(", line=").$(lineCount)
249                     .$(", requiredLen=").$(requiredLength)
250                     .$(", rollLimit=").$(lineRollBufLimit)
251                     .$(']').$();
252             errorCount++;
253             rollBufferUnusable = true;
254             return false;
255         }
256 
257         final int len = Math.min(lineRollBufLimit, requiredLength << 1);
258         LOG.info().$("resizing ").$(lineRollBufSize).$(" -> ").$(len).$(" [table=").$(tableName).$(']').$();
259         long p = Unsafe.malloc(len, MemoryTag.NATIVE_TEXT_PARSER_RSS);
260         long l = lineRollBufCur - lineRollBufPtr;
261         if (l > 0) {
262             Vect.memcpy(p, lineRollBufPtr, l);
263         }
264         Unsafe.free(lineRollBufPtr, lineRollBufSize, MemoryTag.NATIVE_TEXT_PARSER_RSS);
265         if (updateFields) {
266             shift(lineRollBufPtr - p);
267         }
268         lineRollBufCur = p + l;
269         lineRollBufPtr = p;
270         lineRollBufSize = len;
271         return true;
272     }

330     private void rollLine(long lo, long hi) {
331         // lastLineStart is an offset from 'lo'
332         // 'lo' is the address of incoming buffer
333         int l = (int) (hi - lo - lastLineStart);
334         if (l < lineRollBufSize || growRollBuf(l, false)) {
335             assert lo + lastLineStart + l <= hi;
336             Vect.memcpy(lineRollBufPtr, lo + lastLineStart, l);
337             lineRollBufCur = lineRollBufPtr + l;
338             shift(lo + lastLineStart - lineRollBufPtr);
339         }
340     }
