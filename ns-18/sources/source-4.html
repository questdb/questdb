234     private boolean onField(long line, DirectByteCharSequence dbcs, TableWriter.Row w, int i) {
235         try {
236             final int tableIndex = remapIndex.size() > 0 ? remapIndex.get(i) : i;
237             types.getQuick(i).write(w, tableIndex, dbcs);
238         } catch (Exception ignore) {
239             logError(line, i, dbcs);
240             switch (atomicity) {
241                 case Atomicity.SKIP_ALL:
242                     writer.rollback();
243                     throw CairoException.nonCritical().put("bad syntax [line=").put(line).put(", col=").put(i).put(']');
244                 case Atomicity.SKIP_ROW:
245                     w.cancel();
246                     return true;
247                 default:
248                     // SKIP column
249                     break;
250             }
251         }
252         return false;
253     }

255     private TableWriter openWriterAndOverrideImportTypes(
256             TableToken tableToken,
257             ObjList<CharSequence> names,
258             ObjList<TypeAdapter> detectedTypes,
259             CairoSecurityContext cairoSecurityContext,
260             TypeManager typeManager
261     ) {
262         TableWriter writer = engine.getWriter(cairoSecurityContext, tableToken, WRITER_LOCK_REASON);
263         RecordMetadata metadata = writer.getMetadata();
264 
265         // now, compare column count.
266         // Cannot continue if different
267 
268         if (metadata.getColumnCount() < detectedTypes.size()) {
269             writer.close();
270             throw CairoException.nonCritical()
271                     .put("column count mismatch [textColumnCount=").put(detectedTypes.size())
272                     .put(", tableColumnCount=").put(metadata.getColumnCount())
273                     .put(", table=").put(tableName)
274                     .put(']');
275         }
276 
277         this.types = detectedTypes;
278 
279         // now overwrite detected types with actual table column types
280         remapIndex.ensureCapacity(this.types.size());
281         for (int i = 0, n = this.types.size(); i < n; i++) {
282 
283             final int columnIndex = metadata.getColumnIndexQuiet(names.getQuick(i));
284             final int idx = (columnIndex > -1 && columnIndex != i) ? columnIndex : i; // check for strict match ?
285             remapIndex.set(i, idx);
286 
287             final int columnType = metadata.getColumnType(idx);
288             final TypeAdapter detectedAdapter = this.types.getQuick(i);
289             final int detectedType = detectedAdapter.getType();
290             if (detectedType != columnType) {
291                 // when DATE type is mis-detected as STRING we
292                 // would not have either date format nor locale to
293                 // use when populating this field
294                 switch (ColumnType.tagOf(columnType)) {
295                     case ColumnType.DATE:
296                         logTypeError(i);
297                         this.types.setQuick(i, BadDateAdapter.INSTANCE);
298                         break;
299                     case ColumnType.TIMESTAMP:
300                         if (detectedAdapter instanceof TimestampCompatibleAdapter) {
301                             this.types.setQuick(i, otherToTimestampAdapterPool.next().of((TimestampCompatibleAdapter) detectedAdapter));
302                         } else {
303                             logTypeError(i);
304                             this.types.setQuick(i, BadTimestampAdapter.INSTANCE);
305                         }
306                         break;
307                     case ColumnType.BINARY:
308                         writer.close();
309                         throw CairoException.nonCritical().put("cannot import text into BINARY column [index=").put(i).put(']');
310                     default:
311                         this.types.setQuick(i, typeManager.getTypeAdapter(columnType));
312                         break;
313                 }
314             }
315         }
316         return writer;
317     }

319     void prepareTable(
320             CairoSecurityContext cairoSecurityContext,
321             ObjList<CharSequence> names,
322             ObjList<TypeAdapter> detectedTypes,
323             Path path,
324             TypeManager typeManager,
325             TimestampAdapter timestampAdapter
326     ) throws TextException {
327         assert writer == null;
328 
329         if (detectedTypes.size() == 0) {
330             throw CairoException.nonCritical().put("cannot determine text structure");
331         }
332 
333         boolean canUpdateMetadata = true;
334         TableToken tableToken = engine.getTableTokenIfExists(tableName);
335 
336         switch (engine.getStatus(cairoSecurityContext, path, tableToken)) {
337             case TableUtils.TABLE_DOES_NOT_EXIST:
338                 tableToken = createTable(names, detectedTypes, cairoSecurityContext, path);
339                 writer = engine.getWriter(cairoSecurityContext, tableToken, WRITER_LOCK_REASON);
340                 designatedTimestampColumnName = writer.getDesignatedTimestampColumnName();
341                 designatedTimestampIndex = writer.getMetadata().getTimestampIndex();
342                 partitionBy = writer.getPartitionBy();
343                 break;
344             case TableUtils.TABLE_EXISTS:
345                 if (overwrite) {
346                     engine.drop(cairoSecurityContext, path, tableToken);
347                     tableToken = createTable(names, detectedTypes, cairoSecurityContext, path);
348                     writer = engine.getWriter(cairoSecurityContext, tableToken, WRITER_LOCK_REASON);
349                 } else {
350                     canUpdateMetadata = false;
351                     writer = openWriterAndOverrideImportTypes(tableToken, names, detectedTypes, cairoSecurityContext, typeManager);
352                     designatedTimestampColumnName = writer.getDesignatedTimestampColumnName();
353                     designatedTimestampIndex = writer.getMetadata().getTimestampIndex();
354                     if (importedTimestampColumnName != null &&
355                             !Chars.equalsNc(importedTimestampColumnName, designatedTimestampColumnName)) {
356                         warnings |= TextLoadWarning.TIMESTAMP_MISMATCH;
357                     }
358                     if (PartitionBy.isPartitioned(partitionBy) && partitionBy != writer.getPartitionBy()) {
359                         warnings |= TextLoadWarning.PARTITION_TYPE_MISMATCH;
360                     }
361                     partitionBy = writer.getPartitionBy();
362                     tableStructureAdapter.of(names, detectedTypes);
363                 }
364                 break;
365             default:
366                 throw CairoException.nonCritical().put("name is reserved [table=").put(tableName).put(']');
367         }
368         if (canUpdateMetadata) {
369             if (PartitionBy.isPartitioned(partitionBy)) {
370                 if (o3MaxLag > -1) {
371                     writer.setMetaO3MaxLag(o3MaxLag);
372                     LOG.info().$("updating metadata attribute o3MaxLag to ").$(o3MaxLag).$(", table=").utf8(tableName).$();
373                 }
374                 if (maxUncommittedRows > -1) {
375                     writer.setMetaMaxUncommittedRows(maxUncommittedRows);
376                     LOG.info().$("updating metadata attribute maxUncommittedRows to ").$(maxUncommittedRows).$(", table=").utf8(tableName).$();
377                 }
378             }
379         } else {
380             LOG.info().$("cannot update metadata attributes o3MaxLag and maxUncommittedRows when the table exists and parameter overwrite is false").$();
381         }
382         _size = writer.size();
383         columnErrorCounts.seed(writer.getMetadata().getColumnCount(), 0);
384 
385         if (timestampIndex != NO_INDEX) {
386             if (timestampAdapter != null) {
387                 this.timestampAdapter = timestampAdapter;
388             } else if (ColumnType.isTimestamp(types.getQuick(timestampIndex).getType())) {
389                 this.timestampAdapter = (TimestampAdapter) types.getQuick(timestampIndex);
390             }
391         }
392     }

469             this.names = names;

472             if (importedTimestampColumnName == null && designatedTimestampColumnName == null) {
