152     public boolean lock(TableToken tableToken) {
153         Entry<T> e = getEntry(tableToken);
154         final long thread = Thread.currentThread().getId();
155         if (Unsafe.cas(e, LOCK_OWNER, UNLOCKED, thread) || e.lockOwner == thread) {
156             do {
157                 for (int i = 0; i < ENTRY_SIZE; i++) {
158                     if (Unsafe.cas(e.allocations, i, UNALLOCATED, thread)) {
159                         closeTenant(thread, e, i, PoolListener.EV_LOCK_CLOSE, PoolConstants.CR_NAME_LOCK);
160                     } else if (Unsafe.arrayGetVolatile(e.allocations, i) == thread) {
161                         // same thread, don't need to order reads
162                         if (e.getTenant(i) != null) {
163                             // this thread has busy reader, it should close first
164                             e.lockOwner = -1L;
165                             return false;
166                         }
167                     } else {
168                         LOG.info().$("could not lock, busy [table=`").utf8(tableToken.getDirName()).$("`, at=").$(e.index).$(':').$(i).$(", owner=").$(e.allocations[i]).$(", thread=").$(thread).$(']').$();
169                         e.lockOwner = -1L;
170                         return false;
171                     }
172                 }
173 
174                 // try to prevent new entries from being created
175                 if (e.next == null) {
176                     if (Unsafe.getUnsafe().compareAndSwapInt(e, NEXT_STATUS, NEXT_OPEN, NEXT_LOCKED)) {
177                         break;
178                     } else if (e.nextStatus == NEXT_ALLOCATED) {
179                         // now we must wait until another thread that executes a get() call
180                         // assigns the newly created next entry
181                         while (e.next == null) {
182                             Os.pause();
183                         }
184                     }
185                 }
186 
187                 e = e.next;
188             } while (e != null);
189         } else {
190             LOG.error().$("' already locked [table=`").utf8(tableToken.getDirName()).$("`, owner=").$(e.lockOwner).$(']').$();
191             notifyListener(thread, tableToken, PoolListener.EV_LOCK_BUSY, -1, -1);
192             return false;
193         }
194         notifyListener(thread, tableToken, PoolListener.EV_LOCK_SUCCESS, -1, -1);
195         LOG.debug().$("locked [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).$(']').$();
196         return true;
197     }

200         Entry<T> e = entries.get(tableToken.getDirName());
201         long thread = Thread.currentThread().getId();
202         if (e == null) {
203             LOG.info().$("not found, cannot unlock [table=`").$(tableToken).$("`]").$();
204             notifyListener(thread, tableToken, PoolListener.EV_NOT_LOCKED, -1, -1);
205             return;
206         }
207 
208         if (e.lockOwner == thread) {
209             entries.remove(tableToken.getDirName());
210             while (e != null) {
211                 e = e.next;
212             }
213         } else {
214             notifyListener(thread, tableToken, PoolListener.EV_NOT_LOCK_OWNER);
215             throw CairoException.nonCritical().put("Not the lock owner of ").put(tableToken.getDirName());
216         }
217 
218         notifyListener(thread, tableToken, PoolListener.EV_UNLOCKED, -1, -1);
219         LOG.debug().$("unlocked [table=`").utf8(tableToken.getDirName()).$("`]").$();

294     protected boolean releaseAll(long deadline) {
295         long thread = Thread.currentThread().getId();
296         boolean removed = false;
297         int casFailures = 0;
298         int closeReason = deadline < Long.MAX_VALUE ? PoolConstants.CR_IDLE : PoolConstants.CR_POOL_CLOSE;
299 
300         TableToken leftBehind = null;
301         for (Entry<T> e : entries.values()) {
302             do {
303                 for (int i = 0; i < ENTRY_SIZE; i++) {
304                     T r;
305                     if (deadline > Unsafe.arrayGetVolatile(e.releaseOrAcquireTimes, i) && (r = e.getTenant(i)) != null) {
306                         if (Unsafe.cas(e.allocations, i, UNALLOCATED, thread)) {
307                             // check if deadline violation still holds
308                             if (deadline > e.releaseOrAcquireTimes[i]) {
309                                 removed = true;
310                                 closeTenant(thread, e, i, PoolListener.EV_EXPIRE, closeReason);
311                             }
312                             Unsafe.arrayPutOrdered(e.allocations, i, UNALLOCATED);
313                         } else {
314                             casFailures++;
315                             if (deadline == Long.MAX_VALUE) {
316                                 r.goodbye();
317                                 Unsafe.arrayPutOrdered(e.allocations, i, UNALLOCATED);
318                                 LOG.info().$("shutting down. '").utf8(r.getTableToken().getDirName()).$("' is left behind").$();
319                                 leftBehind = r.getTableToken();
320                             }
321                         }
322                     }
323                 }
324                 // this does not release the next
325                 e = e.next;
326             } while (e != null);
327         }
328 
329         if (leftBehind != null) {
330             // This code branch should be in tests only.
331             // Release the item, to not block the pool, but throw an exception to fail the test
332             throw CairoException.nonCritical().put('\'').put(leftBehind.getDirName()).put("' is left behind on pool shutdown");
333         }
334 
335         // when we are timing out entries the result is "true" if there was any work done
336         // when we're closing pool, the result is true when pool is empty
337         if (closeReason == PoolConstants.CR_IDLE) {
338             return removed;
339         } else {
340             return casFailures == 0;
341         }
342     }

348         }

358

360             e.releaseOrAcquireTimes[index] = clock.getTicks();
361             Unsafe.arrayPutOrdered(e.allocations, index, UNALLOCATED);
362             final boolean closed = isClosed();
363 
364             // When pool is closed we will race against release thread
365             // to release our entry. No need to bother releasing map entry, pool is going down.
366             return !closed || !Unsafe.cas(e.allocations, index, UNALLOCATED, owner);
367         }
368 
369         throw CairoException.critical(0).put("double close [table=").put(tableToken.getDirName()).put(", index=").put(index).put(']');
370     }
