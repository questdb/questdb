47 /**
48  * This class maintains cache of open writers to avoid OS overhead of
49  * opening and closing files. While doing so it abides by the same
50  * rule as non-pooled writers: there can only be one TableWriter instance
51  * for any given table name.
52  * <p>
53  * This implementation is thread-safe. Writer allocated by one thread
54  * cannot be used by any other threads until it is released. This factory
55  * will be returning NULL when writer is already in use and cached
56  * instance of writer otherwise. Writers are released back to pool via
57  * standard writer.close() call.
58  * <p>
59  * Writers that have been idle for some time can be expunged from pool
60  * by calling Job.run() method asynchronously. Pool implementation is
61  * guaranteeing thread-safety of this method at all times.
62  * <p>
63  * This factory can be closed via close() call. This method is also
64  * thread-safe and is guarantying that all open writers will be eventually
65  * closed.
66  */
67 public class WriterPool extends AbstractPool {

167      * Locks writer. Locking operation is always non-blocking. Lock is usually successful
168      * when writer is in pool or owned by calling thread, in which case
169      * writer instance is closed. Lock will also succeed when writer does not exist.
170      * This will prevent from writer being created before it is unlocked.
171      * <p>
172      * Lock fails immediately with {@link EntryUnavailableException} when writer is used by another thread and with
173      * {@link PoolClosedException} when pool is closed.
174      * </p>
175      * <p>
176      * Lock is beneficial before table directory is renamed or deleted.
177      * </p>
178      *
179      * @param tableToken table name
180      * @param lockReason description of where or why lock is held
181      * @return lock reason as String
182      */
183     public String lock(TableToken tableToken, String lockReason) {
184         checkClosed();
185 
186         long thread = Thread.currentThread().getId();
187 
188         Entry e = entries.get(tableToken.getDirName());
189         if (e == null) {
190             // We are racing to create new writer!
191             e = new Entry(clock.getTicks());
192             Entry other = entries.putIfAbsent(tableToken.getDirName(), e);
193             if (other == null) {
194                 if (lockAndNotify(thread, e, tableToken, lockReason)) {
195                     return OWNERSHIP_REASON_NONE;
196                 } else {
197                     entries.remove(tableToken.getDirName());
198                     return reinterpretOwnershipReason(e.ownershipReason);
199                 }
200             } else {
201                 e = other;
202             }
203         }
204 
205         // try to change owner
206         if ((Unsafe.cas(e, ENTRY_OWNER, UNALLOCATED, thread) /*|| (e.owner == thread)*/)) {
207             closeWriter(thread, e, PoolListener.EV_LOCK_CLOSE, PoolConstants.CR_NAME_LOCK);
208             if (lockAndNotify(thread, e, tableToken, lockReason)) {
209                 return OWNERSHIP_REASON_NONE;
210             }
211             return reinterpretOwnershipReason(e.ownershipReason);
212         }
213 
214         LOG.error().$("could not lock, busy [table=`").utf8(tableToken.getDirName()).$("`, owner=").$(e.owner).$(", thread=").$(thread).$(']').$();
215         notifyListener(thread, tableToken, PoolListener.EV_LOCK_BUSY);
216         return reinterpretOwnershipReason(e.ownershipReason);
217     }

227     public void unlock(TableToken tableToken, @Nullable TableWriter writer, boolean newTable) {
228         long thread = Thread.currentThread().getId();
229 
230         Entry e = entries.get(tableToken.getDirName());
231         if (e == null) {
232             notifyListener(thread, tableToken, PoolListener.EV_NOT_LOCKED);
233             return;
234         }
235 
236         if (e.owner == thread) {
237             if (e.writer != null) {
238                 // When entry is locked, writer must be null, however if writer is not null,
239                 // calling thread must be trying to unlock writer that hasn't been locked.
240                 // This qualifies for "illegal state".
241                 notifyListener(thread, tableToken, PoolListener.EV_NOT_LOCKED);
242                 throw CairoException.critical(0).put("Writer ").put(tableToken.getDirName()).put(" is not locked");
243             }
244 
245             if (newTable) {
246                 // Note that the TableUtils.createTable method will create files, but on some OS's these files
247                 // will not immediately become visible on all threads, only in this thread will they definitely
248                 // be visible. To prevent spurious file system errors (or even allowing the same table to be
249                 // created twice), we cache the writer in the WriterPool whose access via the engine is thread safe.
250                 assert writer == null && e.lockFd != -1;
251                 LOG.info().$("created [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).$(']').$();
252                 writer = new TableWriter(configuration, tableToken, messageBus, null, false, e, root, metrics);
253             }
254 
255             if (writer == null) {
256                 // unlock must remove entry because pool does not deal with null writer
257 
258                 if (e.lockFd != -1) {
259                     Path path = Path.getThreadLocal(root).concat(tableToken.getDirName());
260                     TableUtils.lockName(path);
261                     if (!ff.closeRemove(e.lockFd, path)) {
262                         LOG.error().$("could not remove [file=").$(path).$(']').$();
263                     }
264                 }
265                 entries.remove(tableToken.getDirName());
266             } else {
267                 e.writer = writer;
268                 writer.setLifecycleManager(e);
269                 writer.transferLock(e.lockFd);
270                 e.lockFd = -1;
271                 e.ownershipReason = OWNERSHIP_REASON_NONE;
272                 Unsafe.getUnsafe().storeFence();
273                 Unsafe.getUnsafe().putOrderedLong(e, ENTRY_OWNER, UNALLOCATED);
274             }
275             notifyListener(thread, tableToken, PoolListener.EV_UNLOCKED);
276             LOG.debug().$("unlocked [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).I$();
277         } else {
278             notifyListener(thread, tableToken, PoolListener.EV_NOT_LOCK_OWNER);
279             throw CairoException.critical(0).put("Not lock owner of ").put(tableToken.getDirName());
280         }
281     }

283     private void addCommandToWriterQueue(Entry e, AsyncWriterCommand asyncWriterCommand, long thread) {
284         TableWriter writer;
285         while ((writer = e.writer) == null && e.owner != UNALLOCATED) {
286             Os.pause();
287         }
288         if (writer == null) {
289             // Retry from very beginning
290             throw EntryUnavailableException.instance("please retry");
291         }
292         // Mark command as being executed asynchronously and publish it.
293         asyncWriterCommand.startAsync();
294         writer.publishAsyncWriterCommand(asyncWriterCommand);
295 
296         // Make sure writer does not go to the pool with command in the queue
297         // Wait until writer is either in the pool or out
298         while (e.owner == QUEUE_PROCESSING_OWNER) {
299             Os.pause();
300         }
301 
302         // If the writer is suddenly in the pool, lock it and call tick to process command queue
303         if (Unsafe.cas(e, ENTRY_OWNER, UNALLOCATED, thread)) {
304             // Writer became available straight after setting items in the queue.
305             // Don't leave it unprocessed
306             try {
307                 writer.tick(true);
308             } finally {
309                 Unsafe.cas(e, ENTRY_OWNER, thread, UNALLOCATED);
310             }
311         }
312     }

376     private TableWriter getWriterEntry(
377             TableToken tableToken,
378             String lockReason,
379             @Nullable AsyncWriterCommand asyncWriterCommand,
380             boolean logBusy
381     ) {
382         assert null != lockReason;
383         checkClosed();
384 
385         long thread = Thread.currentThread().getId();
386 
387         while (true) {
388             Entry e = entries.get(tableToken.getDirName());
389             if (e == null) {
390                 // We are racing to create new writer!
391                 e = new Entry(clock.getTicks());
392                 Entry other = entries.putIfAbsent(tableToken.getDirName(), e);
393                 if (other == null) {
394                     // race won
395                     return createWriter(tableToken, e, thread, lockReason);
396                 } else {
397                     e = other;
398                 }
399             }
400 
401             long owner = e.owner;
402             // try to change owner
403             if (Unsafe.cas(e, ENTRY_OWNER, UNALLOCATED, thread)) {
404                 // in an extreme race condition it is possible that e.writer will be null
405                 // in this case behaviour should be identical to entry missing entirely
406                 if (e.writer == null) {
407                     return createWriter(tableToken, e, thread, lockReason);
408                 }
409                 return checkClosedAndGetWriter(tableToken, e, lockReason);
410             } else {
411                 if (owner < 0) {
412                     // writer is about to be released from the pool by release method.
413                     // try again, it should become available soon.
414                     Os.pause();
415                     continue;
416                 }
417                 if (owner == thread) {
418                     if (e.lockFd != -1L) {
419                         throw EntryLockedException.instance(reinterpretOwnershipReason(e.ownershipReason));
420                     }
421 
422                     if (e.ex != null) {
423                         notifyListener(thread, tableToken, PoolListener.EV_EX_RESEND);
424                         // this writer failed to allocate by this very thread
425                         // ensure consistent response
426                         entries.remove(tableToken.getDirName());
427                         throw e.ex;
428                     }
429                 }
430                 if (asyncWriterCommand != null) {
431                     addCommandToWriterQueue(e, asyncWriterCommand, thread);
432                     return null;
433                 }
434 
435                 String reason = reinterpretOwnershipReason(e.ownershipReason);
436 
437                 if (logBusy) {
438                     LOG.info().$("busy [table=`").utf8(tableToken.getDirName())
439                             .$("`, owner=").$(owner)
440                             .$(", thread=").$(thread)
441                             .$(", reason=").$(reason)
442                             .I$();
443                 }
444                 throw EntryUnavailableException.instance(reason);
445             }
446         }
447     }

449     private boolean lockAndNotify(long thread, Entry e, TableToken tableToken, String lockReason) {
450         assertLockReasonIsNone(lockReason);
451         Path path = Path.getThreadLocal(root).concat(tableToken.getDirName());
452         TableUtils.lockName(path);
453         e.lockFd = TableUtils.lock(ff, path);
454         if (e.lockFd == -1) {
455             LOG.error().$("could not lock [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).$(']').$();
456             e.ownershipReason = OWNERSHIP_REASON_MISSING;
457             e.owner = UNALLOCATED;
458             return false;
459         }
460         LOG.debug().$("locked [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).$(']').$();
461         notifyListener(thread, tableToken, PoolListener.EV_LOCK_SUCCESS);

472     private String reinterpretOwnershipReason(String providedReason) {
473         // we cannot always guarantee that ownership reason is set
474         // allocating writer and setting "reason" are non-atomic
475         // therefore we could be in a situation where we can be confident writer is locked
476         // but reason has not yet caught up. In this case we do not really know the reason
477         // but not to confuse the caller, we have to provide a non-null value

482     private boolean returnToPool(Entry e) {
483         final long thread = Thread.currentThread().getId();
484         final TableToken tableToken = e.writer.getTableToken();
485         try {
486             e.writer.rollback();
487 
488             if (e.owner != UNALLOCATED) {
489                 e.owner = QUEUE_PROCESSING_OWNER;
490             }
491             // We can apply structure changes with ALTER TABLE and do UPDATE(s) before the writer returned to the pool
492             e.writer.tick(true);
493         } catch (Throwable ex) {
494             // We are here because of a systemic issues of some kind
495             // one of the known issues is "disk is full" so we could not roll back properly.
496             // In this case we just close TableWriter
497             entries.remove(tableToken.getDirName());
498             closeWriter(thread, e, PoolListener.EV_LOCK_CLOSE, PoolConstants.CR_DISTRESSED);
499             return true;
500         }
501 
502         if (e.owner != UNALLOCATED) {
503             LOG.info().$("<< [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(thread).$(']').$();
504 
505             e.ownershipReason = OWNERSHIP_REASON_NONE;
506             e.lastReleaseTime = configuration.getMicrosecondClock().getTicks();
507             Unsafe.getUnsafe().storeFence();
508             Unsafe.getUnsafe().putOrderedLong(e, ENTRY_OWNER, UNALLOCATED);
509 
510             if (isClosed()) {
511                 // when pool is closed it could be busy releasing writer
512                 // to avoid race condition try to grab the writer before declaring it a
513                 // free agent
514                 if (Unsafe.cas(e, ENTRY_OWNER, UNALLOCATED, thread)) {
515                     e.writer = null;
516                     notifyListener(thread, tableToken, PoolListener.EV_OUT_OF_POOL_CLOSE);
517                     return false;
518                 }
519             }
520 
521             notifyListener(thread, tableToken, PoolListener.EV_RETURN);
522         } else {
523             LOG.critical().$("orphaned [table=`").utf8(tableToken.getDirName()).$("`]").$();
524             notifyListener(thread, tableToken, PoolListener.EV_UNEXPECTED_CLOSE);
525         }
526         return true;
527     }

555     @Override
556     protected boolean releaseAll(long deadline) {
557         long thread = Thread.currentThread().getId();
558         boolean removed = false;
559         final int reason;
560 
561         if (deadline == Long.MAX_VALUE) {
562             reason = PoolConstants.CR_POOL_CLOSE;
563         } else {
564             reason = PoolConstants.CR_IDLE;
565         }
566 
567         Iterator<Entry> iterator = entries.values().iterator();
568         while (iterator.hasNext()) {
569             Entry e = iterator.next();
570             // lastReleaseTime is volatile, which makes
571             // order of conditions important
572             if ((deadline > e.lastReleaseTime && e.owner == UNALLOCATED)) {
573                 // looks like this writer is unallocated and can be released
574                 // Lock with negative 3-based owner thread id to indicate it's that next
575                 // allocating thread can wait until the entry is released.
576                 // Avoid negative thread id clashing with UNALLOCATED and QUEUE_PROCESSING values
577                 if (Unsafe.cas(e, ENTRY_OWNER, UNALLOCATED, -thread - 3)) {
578                     // lock successful
579                     closeWriter(thread, e, PoolListener.EV_EXPIRE, reason);
580                     iterator.remove();
581                     removed = true;
582                 }
583             } else if (e.lockFd != -1 && deadline == Long.MAX_VALUE) {
584                 // do not release locks unless pool is shutting down, which is
585                 // indicated via deadline to be Long.MAX_VALUE
586                 if (ff.close(e.lockFd)) {
587                     e.lockFd = -1;
588                     iterator.remove();
589                     removed = true;
590                 }
591             } else if (e.ex != null) {
592                 LOG.info().$("purging entry for failed to allocate writer").$();
593                 iterator.remove();
594                 removed = true;
595             }
596         }
597         return removed;
598     }
