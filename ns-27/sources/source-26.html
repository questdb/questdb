37     public static final SymbolConstant NULL = new SymbolConstant(null, VALUE_IS_NULL);

42     public SymbolConstant(CharSequence value, int index) {
43         if (value == null) {
44             this.value = null;
45             this.index = SymbolTable.VALUE_IS_NULL;
46         } else {
47             if (Chars.startsWith(value, '\'')) {
48                 this.value = Chars.toString(value, 1, value.length() - 1);
49             } else {
50                 this.value = Chars.toString(value);
51             }
52             this.index = index;
53         }
54     }
55 
56     public static SymbolConstant newInstance(CharSequence value) {

60

64 
65         if (SqlKeywords.isFalseKeyword(value)) {
66             return FALSE;
67         }
68 
69         return new SymbolConstant(Chars.toString(value), 0);
70     }
