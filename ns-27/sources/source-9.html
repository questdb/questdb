38     public static ConstantFunction getGeoHashConstant(long hash, int bits) {
39         final int type = ColumnType.getGeoHashTypeWithBits(bits);
40         return getGeoHashConstantWithType(hash, type);
41     }

43     @NotNull
44     public static ConstantFunction getGeoHashConstantWithType(long hash, int type) {
45         switch (ColumnType.tagOf(type)) {
46             case ColumnType.GEOBYTE:
47                 return new GeoByteConstant((byte) hash, type);
48             case ColumnType.GEOSHORT:
49                 return new GeoShortConstant((short) hash, type);
50             case ColumnType.GEOINT:
51                 return new GeoIntConstant((int) hash, type);
52             default:
53                 return new GeoLongConstant(hash, type);
54         }
55     }
56 
57     public static ConstantFunction getNullConstant(int columnType) {
58         int typeTag = ColumnType.tagOf(columnType);
59         if (typeTag >= ColumnType.GEOBYTE &&
60                 typeTag <= ColumnType.GEOLONG) {
61             int bits = ColumnType.getGeoHashBits(columnType);
62             if (bits != 0) {
63                 return geoNullConstants.get(bits);
64             }
65         }
66         return nullConstants.getQuick(typeTag);
67     }
68 
69     public static TypeConstant getTypeConstant(int columnType) {
70         // GEOHASH takes a different path, no need to extract tag
71         return typeConstants.getQuick(columnType);
72     }
