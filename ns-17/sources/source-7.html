42 public class PGJobContext implements Closeable {
43 
44     private final SqlCompiler compiler;
45     private final AssociativeCache<TypesAndSelect> typesAndSelectCache;
46     private final WeakSelfReturningObjectPool<TypesAndSelect> typesAndSelectPool;
47     private final AssociativeCache<TypesAndUpdate> typesAndUpdateCache;
48     private final WeakSelfReturningObjectPool<TypesAndUpdate> typesAndUpdatePool;
49 
50     public PGJobContext(
51             PGWireConfiguration configuration,
52             CairoEngine engine,
53             FunctionFactoryCache functionFactoryCache,
54             DatabaseSnapshotAgent snapshotAgent
55     ) {
56         this.compiler = new SqlCompiler(engine, functionFactoryCache, snapshotAgent);
57 
58         final Metrics metrics = engine.getMetrics();
59 
60         final boolean enableSelectCache = configuration.isSelectCacheEnabled();
61         final int blockCount = enableSelectCache ? configuration.getSelectCacheBlockCount() : 1;
62         final int rowCount = enableSelectCache ? configuration.getSelectCacheRowCount() : 1;
63         typesAndSelectCache = new AssociativeCache<>(blockCount, rowCount, metrics.pgWire().cachedSelectsGauge());
64         typesAndSelectPool = new WeakSelfReturningObjectPool<>(TypesAndSelect::new, blockCount * rowCount);
65 
66         final boolean enabledUpdateCache = configuration.isUpdateCacheEnabled();
67         final int updateBlockCount = enabledUpdateCache ? configuration.getUpdateCacheBlockCount() : 1; // 8
68         final int updateRowCount = enabledUpdateCache ? configuration.getUpdateCacheRowCount() : 1; // 8
69         typesAndUpdateCache = new AssociativeCache<>(updateBlockCount, updateRowCount, metrics.pgWire().cachedUpdatesGauge());
70         typesAndUpdatePool = new WeakSelfReturningObjectPool<>(parent -> new TypesAndUpdate(parent, engine), updateBlockCount * updateRowCount);
71     }
72 
73     @Override
74     public void close() {
75         Misc.free(compiler);
76         Misc.free(typesAndSelectCache);
77         Misc.free(typesAndUpdateCache);
78     }
79 
80     public void flushQueryCache() {

83     }
