51     private static final Log LOG = LogFactory.getLog(PGWireServer.class);

53     private final IODispatcher<PGConnectionContext> dispatcher;
54     private final Metrics metrics;
55     private final WorkerPool workerPool;
56 
57     public PGWireServer(
58             PGWireConfiguration configuration,
59             CairoEngine engine,
60             WorkerPool workerPool,
61             FunctionFactoryCache functionFactoryCache,
62             DatabaseSnapshotAgent snapshotAgent,
63             PGConnectionContextFactory contextFactory
64     ) {
65         this.dispatcher = IODispatchers.create(
66                 configuration.getDispatcherConfiguration(),
67                 contextFactory
68         );
69         this.metrics = engine.getMetrics();
70         this.workerPool = workerPool;
71 
72         workerPool.assign(dispatcher);
73 
74         for (int i = 0, n = workerPool.getWorkerCount(); i < n; i++) {
75             final PGJobContext jobContext = new PGJobContext(configuration, engine, functionFactoryCache, snapshotAgent);
76 
77             final SCSequence queryCacheEventSubSeq = new SCSequence();
78             final FanOut queryCacheEventFanOut = engine.getMessageBus().getQueryCacheEventFanOut();
79             queryCacheEventFanOut.and(queryCacheEventSubSeq);
80 
81             workerPool.assign(i, new Job() {
82                 private final IORequestProcessor<PGConnectionContext> processor = (operation, context) -> {
83                     try {
84                         jobContext.handleClientOperation(context, operation);
85                         context.getDispatcher().registerChannel(context, IOOperation.READ);
86                         return true;
87                     } catch (PeerIsSlowToWriteException e) {
88                         context.getDispatcher().registerChannel(context, IOOperation.READ);
89                     } catch (PeerIsSlowToReadException e) {
90                         context.getDispatcher().registerChannel(context, IOOperation.WRITE);
91                     } catch (QueryPausedException e) {
92                         context.setSuspendEvent(e.getEvent());
93                         context.getDispatcher().registerChannel(context, IOOperation.WRITE);
94                     } catch (PeerDisconnectedException e) {
95                         context.getDispatcher().disconnect(
96                                 context,
97                                 operation == IOOperation.READ
98                                         ? DISCONNECT_REASON_PEER_DISCONNECT_AT_RECV
99                                         : DISCONNECT_REASON_PEER_DISCONNECT_AT_SEND
100                         );
101                     } catch (BadProtocolException e) {
102                         context.getDispatcher().disconnect(context, DISCONNECT_REASON_PROTOCOL_VIOLATION);
103                     } catch (Throwable e) { // must remain last in catch list!
104                         LOG.critical().$("internal error [ex=").$(e).$(']').$();
105                         // This is a critical error, so we treat it as an unhandled one.
106                         metrics.health().incrementUnhandledErrors();
107                         context.getDispatcher().disconnect(context, DISCONNECT_REASON_SERVER_ERROR);
108                     }
109                     return false;
110                 };
111 
112                 @Override
113                 public boolean run(int workerId, @NotNull RunStatus runStatus) {
114                     long seq = queryCacheEventSubSeq.next();
115                     if (seq > -1) {
116                         // Queue is not empty, so flush query cache.
117                         LOG.info().$("flushing PG Wire query cache [worker=").$(workerId).$(']').$();
118                         jobContext.flushQueryCache();
119                         queryCacheEventSubSeq.done(seq);
120                     }
121                     return dispatcher.processIOQueue(processor);
122                 }
123             });
124 
125             // http context factory has thread local pools
126             // therefore we need each thread to clean their thread locals individually
127             workerPool.assignThreadLocalCleaner(i, contextFactory::freeThreadLocal);
128             workerPool.freeOnExit((QuietCloseable) () -> {
129                 Misc.free(jobContext);
130                 engine.getMessageBus().getQueryCacheEventFanOut().remove(queryCacheEventSubSeq);
131                 queryCacheEventSubSeq.clear();
132             });
133         }
134     }
135
