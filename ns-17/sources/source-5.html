33 public class PGBasicAuthenticator implements PGAuthenticator {
34     private final DirectByteCharSequence dbcs = new DirectByteCharSequence();
35     private final String password;
36     private final CairoSecurityContext securityContext;
37     private final String username;
38 
39     public PGBasicAuthenticator(String username, String password, boolean readOnlyContext) {
40         this.username = username;
41         this.password = password;
42         this.securityContext = readOnlyContext ? new CairoSecurityContextImpl(false) : AllowAllCairoSecurityContext.INSTANCE;
43     }
44 
45     @Override
46     public CairoSecurityContext authenticate(
47             CharSequence username,
48             long msg,
49             long msgLimit
50     ) throws BadProtocolException, AuthenticationException {
51         if (Chars.equals(this.username, username)) {
52             // check 'p' message
53             // +1 is 'type' byte that message length does not account for
54             long hi = PGConnectionContext.getStringLength(msg, msgLimit, "bad password length");
55             dbcs.of(msg, hi);
56             // check password
57             if (Chars.equals(this.password, dbcs)) {
58                 return securityContext;
59             }
60         }
61         // -1 position here means we will not send it to postgresql
62         throw AuthenticationException.INSTANCE;
63     }
