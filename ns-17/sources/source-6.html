366     public void handleClientOperation(
367             @Transient SqlCompiler compiler,
368             @Transient AssociativeCache<TypesAndSelect> selectAndTypesCache,
369             @Transient WeakSelfReturningObjectPool<TypesAndSelect> selectAndTypesPool,
370             @Transient AssociativeCache<TypesAndUpdate> typesAndUpdateCache,
371             @Transient WeakSelfReturningObjectPool<TypesAndUpdate> typesAndUpdatePool,
372             int operation
373     ) throws PeerDisconnectedException, PeerIsSlowToReadException, PeerIsSlowToWriteException, QueryPausedException, BadProtocolException {
374 
375         this.typesAndSelectCache = selectAndTypesCache;
376         this.typesAndSelectPool = selectAndTypesPool;
377         this.typesAndUpdateCache = typesAndUpdateCache;
378         this.typesAndUpdatePool = typesAndUpdatePool;
379 
380         try {
381             if (isPausedQuery) {
382                 isPausedQuery = false;
383                 if (resumeProcessor != null) {
384                     resumeProcessor.resume(true);
385                 }
386             } else if (bufferRemainingSize > 0) {
387                 doSend(bufferRemainingOffset, bufferRemainingSize);
388                 if (resumeProcessor != null) {
389                     resumeProcessor.resume(false);
390                 }
391             }
392 
393             boolean keepReceiving = true;
394             OUTER:
395             do {
396                 if (operation == IOOperation.READ) {
397                     if (recv() == 0) {
398                         keepReceiving = false;
399                     }
400                 }
401 
402                 // we do not pre-compute length because 'parse' will mutate 'recvBufferReadOffset'
403                 if (keepReceiving) {
404                     do {
405                         // Parse will update the value of recvBufferOffset upon completion of
406                         // logical block. We cannot count on return value because 'parse' may try to
407                         // respond to client and fail with exception. When it does fail we would have
408                         // to retry 'send' but not parse the same input again
409 
410                         long readOffsetBeforeParse = recvBufferReadOffset;
411                         totalReceived += (recvBufferWriteOffset - recvBufferReadOffset);
412                         parse(
413                                 recvBuffer + recvBufferReadOffset,
414                                 (int) (recvBufferWriteOffset - recvBufferReadOffset),
415                                 compiler
416                         );
417 
418                         // nothing changed?
419                         if (readOffsetBeforeParse == recvBufferReadOffset) {
420                             // shift to start
421                             if (readOffsetBeforeParse > 0) {
422                                 shiftReceiveBuffer(readOffsetBeforeParse);
423                             }
424                             continue OUTER;
425                         }
426                     } while (recvBufferReadOffset < recvBufferWriteOffset);
427                     clearRecvBuffer();
428                 }
429             } while (keepReceiving && operation == IOOperation.READ);
430         } catch (SqlException e) {
431             reportNonCriticalError(e.getPosition(), e.getFlyweightMessage());
432         } catch (ImplicitCastException e) {
433             reportNonCriticalError(-1, e.getFlyweightMessage());
434         } catch (CairoException e) {
435             if (e.isInterruption()) {
436                 reportQueryCancelled(e.getFlyweightMessage());
437             } else {
438                 reportError(e);
439             }
440         } catch (AuthenticationException e) {
441             prepareNonCriticalError(-1, e.getMessage());
442             sendAndReset();
443             clearRecvBuffer();
444         }
445     }
