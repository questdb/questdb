38     private MPSequence(int cycle, WaitStrategy waitStrategy) {
39         super(cycle, waitStrategy);
40         this.cycle = cycle;
41     }
42 
43     @Override
44     public long next() {
45         // reading cache before value is essential because algo relies on barrier inserted by value read.
46         long cached = cache;
47         long current = value;
48         long next = current + 1;
49         long lo = next - cycle;
50 
51         if (lo <= cached) {
52             return casValue(current, next) ? next : -2;
53         }
54 
55         final long avail = barrier.availableIndex(lo);
56 
57         if (avail > cached) {
58             setCacheFenced(avail);
59             if (lo <= avail) {
60                 return casValue(current, next) ? next : -2;
61             }
62         }
63         return -1;
64     }
