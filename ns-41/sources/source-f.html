29 import java.io.Closeable;
30 
31 public class RingQueue<T> implements Closeable {
32     private final T[] buf;
33     private final int mask;
34     private final long memory;
35     private final int memoryTag;
36     private long memorySize;
37 
38     @SuppressWarnings("unchecked")
39     public RingQueue(ObjectFactory<T> factory, int cycle) {
40 
41         // zero queue is allowed for testing
42         assert cycle == 0 || Numbers.isPow2(cycle);
43 
44         this.mask = cycle - 1;
45         this.buf = (T[]) new Object[cycle];
46 
47         for (int i = 0; i < cycle; i++) {
48             buf[i] = factory.newInstance();
49         }
50 
51         // heap based queue
52         this.memory = 0;
53         this.memorySize = 0;
54         this.memoryTag = 0;
55     }
56 
57     @SuppressWarnings("unchecked")
58     public RingQueue(DirectObjectFactory<T> factory, long slotSize, int cycle, int memoryTag) {
59         this.mask = cycle - 1;
60         this.buf = (T[]) new Object[cycle];
61 
62         this.memorySize = slotSize * cycle;
63         this.memoryTag = memoryTag;
64         this.memory = Unsafe.calloc(memorySize, memoryTag);
65         long p = memory;
66         for (int i = 0; i < cycle; i++) {
67             // intention is that whatever comes out of the factory it should work with the
68             // memory allocated by the queue for this slot and should not reallocate ever
69             buf[i] = factory.newInstance(p, slotSize);
70             p += slotSize;
71         }
72     }
73 
74     @Override
75     public void close() {
76         for (int i = 0, n = buf.length; i < n; i++) {
77             Misc.freeIfCloseable(buf[i]);
78         }
79         if (memorySize > 0) {
80             Unsafe.free(memory, memorySize, memoryTag);
81             this.memorySize = 0;
82         }
83     }
84 
85     public T get(long cursor) {
86         return buf[(int) (cursor & mask)];
87     }
88 
89     public int getCycle() {
90         return buf.length;
91     }
92 }
