173     public void start(@Nullable Log log) {
174         if (!closed.get() && running.compareAndSet(false, true)) {
175 
176             // very common cleaner
177             // it is setup from start() to make sure it is called last
178             // some other thread local cleaners are liable to access thread local Path instances
179             setupPathCleaner();
180 
181             for (int i = 0; i < workerCount; i++) {
182                 final int index = i;
183                 Worker worker = new Worker(
184                         workerJobs.getQuick(i),
185                         halted,
186                         workerAffinity[i],
187                         log,
188                         (ex) -> {
189                             Misc.freeObjListAndClear(threadLocalCleaners.getQuick(index));
190                             if (log != null) {
191                                 log.info().$("cleaned worker [name=").$(poolName)
192                                         .$(", worker=").$(index)
193                                         .$(", total=").$(workerCount)
194                                         .I$();

200                         yieldThreshold,
