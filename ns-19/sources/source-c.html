140     public void parseConfiguration(JsonLexer jsonLexer, String confRoot, String configFileName) throws JsonException {
141         clear();
142         jsonLexer.clear();
143 
144         final JsonParser parser = this::onJsonEvent;
145 
146         try (InputStream stream = openStream(confRoot, configFileName)) {
147             // here is where using direct memory is very disadvantageous
148             // we will copy buffer twice to parse json, but luckily contents should be small,
149             // and we should be parsing this only once on startup
150             byte[] heapBuffer = new byte[4096];
151             long memBuffer = Unsafe.malloc(heapBuffer.length, MemoryTag.NATIVE_TEXT_PARSER_RSS);
152             try {
153                 int len;
154                 while ((len = stream.read(heapBuffer)) > 0) {
155                     // copy to mem buffer
156                     for (int i = 0; i < len; i++) {
157                         Unsafe.getUnsafe().putByte(memBuffer + i, heapBuffer[i]);
158                     }
159                     jsonLexer.parse(memBuffer, memBuffer + len, parser);
160                 }
161                 jsonLexer.clear();
162             } finally {
163                 Unsafe.free(memBuffer, heapBuffer.length, MemoryTag.NATIVE_TEXT_PARSER_RSS);
164             }
165         } catch (IOException e) {
166             throw JsonException.$(0, "could not read input format config [confRoot=").put(confRoot)
167                     .put(", configFileName=").put(configFileName)
168                     .put(']');
169         }
170     }

172     private void onJsonEvent(int code, CharSequence tag, int position) throws JsonException {
173         switch (code) {
174             case JsonLexer.EVT_OBJ_START:
175                 switch (jsonState) {
176                     case STATE_EXPECT_TOP:
177                         // this is top level object
178                         // lets dive in
179                         jsonState = STATE_EXPECT_FIRST_LEVEL_NAME;
180                         break;
181                     case STATE_EXPECT_DATE_FORMAT_VALUE:
182                     case STATE_EXPECT_TIMESTAMP_FORMAT_VALUE:
183                         throw JsonException.$(position, "format value expected (obj)");
184                     case STATE_EXPECT_DATE_LOCALE_VALUE:
185                     case STATE_EXPECT_TIMESTAMP_LOCALE_VALUE:
186                         throw JsonException.$(position, "locale value expected (obj)");
187                     case STATE_EXPECT_DATE_FORMAT_ENTRY:
188                         jsonDateFormat = null;
189                         jsonDateLocale = null;
190                         break;
191                     case STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY:
192                         jsonTimestampFormat = null;
193                         jsonTimestampLocale = null;
194                         break;
195                     default:
196                         throw JsonException.$(position, "array expected (obj)");
197                 }
198                 break;
199             case JsonLexer.EVT_OBJ_END:
200                 switch (jsonState) {
201                     case STATE_EXPECT_DATE_FORMAT_ENTRY: // we just closed a date object
202                         if (jsonDateFormat == null) {
203                             throw JsonException.$(position, "date format is missing");
204                         }
205                         dateFormats.add(jsonDateFormat);
206                         dateLocales.add(jsonDateLocale == null ? dateLocale : jsonDateLocale);
207                         dateUtf8Flags.add(jsonDateUtf8 ? 1 : 0);
208                         break;
209                     case STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY:
210                         if (jsonTimestampFormat == null) {
211                             throw JsonException.$(position, "timestamp format is missing");
212                         }
213 
214                         timestampFormats.add(jsonTimestampFormat);
215                         timestampLocales.add(jsonTimestampLocale == null ? dateLocale : jsonTimestampLocale);
216                         timestampUtf8Flags.add(jsonTimestampUtf8 ? 1 : 0);
217                         break;
218                     default:
219                         // the only time we get here would be when
220                         // main object is closed.
221                         // other end_of_object cannot get there unless we
222                         // allow to enter these objects in the first place
223                         break;
224                 }
225                 break;
226             case JsonLexer.EVT_ARRAY_END:
227                 jsonState = STATE_EXPECT_FIRST_LEVEL_NAME;
228                 break;
229             case JsonLexer.EVT_NAME:
230                 switch (jsonState) {
231                     case STATE_EXPECT_FIRST_LEVEL_NAME:
232                         if (Chars.equals(tag, "date")) {
233                             jsonState = STATE_EXPECT_DATE_FORMAT_ARRAY; // expect array with date formats
234                         } else if (Chars.equals(tag, "timestamp")) {
235                             jsonState = STATE_EXPECT_TIMESTAMP_FORMAT_ARRAY; // expect array with timestamp formats
236                         } else {
237                             // unknown tag name?
238                             throw JsonException.$(position, "'date' and/or 'timestamp' expected");
239                         }
240                         break;
241                     case STATE_EXPECT_DATE_FORMAT_ENTRY:
242                         processEntry(tag, position, STATE_EXPECT_DATE_FORMAT_VALUE, STATE_EXPECT_DATE_LOCALE_VALUE, STATE_EXPECT_DATE_UTF8_VALUE);
243                         break;
244                     default:
245                         processEntry(tag, position, STATE_EXPECT_TIMESTAMP_FORMAT_VALUE, STATE_EXPECT_TIMESTAMP_LOCALE_VALUE, STATE_EXPECT_TIMESTAMP_UTF8_VALUE);
246                         break;
247                 }
248                 break;
249             case JsonLexer.EVT_VALUE:
250                 switch (jsonState) {
251                     case STATE_EXPECT_DATE_FORMAT_VALUE:
252                         // date format
253                         assert jsonDateFormat == null;
254                         if (Chars.equals("null", tag)) {
255                             throw JsonException.$(position, "null format");
256                         }
257                         jsonDateFormat = dateFormatFactory.get(tag);
258                         jsonState = STATE_EXPECT_DATE_FORMAT_ENTRY;
259                         break;
260                     case STATE_EXPECT_DATE_LOCALE_VALUE: // date locale
261                         assert jsonDateLocale == null;
262                         jsonDateLocale = dateLocaleFactory.getLocale(tag);
263                         if (jsonDateLocale == null) {
264                             throw JsonException.$(position, "invalid [locale=").put(tag).put(']');
265                         }
266                         jsonState = STATE_EXPECT_DATE_FORMAT_ENTRY;
267                         break;
268                     case STATE_EXPECT_TIMESTAMP_FORMAT_VALUE: // timestamp format
269                         assert jsonTimestampFormat == null;
270                         if (Chars.equals("null", tag)) {
271                             throw JsonException.$(position, "null format");
272                         }
273                         jsonTimestampFormat = timestampFormatFactory.get(tag);
274                         jsonState = STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY;
275                         break;
276                     case STATE_EXPECT_TIMESTAMP_LOCALE_VALUE:
277                         assert jsonTimestampLocale == null;
278                         jsonTimestampLocale = dateLocaleFactory.getLocale(tag);
279                         if (jsonTimestampLocale == null) {
280                             throw JsonException.$(position, "invalid [locale=").put(tag).put(']');
281                         }
282                         jsonState = STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY;
283                         break;
284                     case STATE_EXPECT_TIMESTAMP_UTF8_VALUE:
285                         jsonTimestampUtf8 = Chars.equalsLowerCaseAscii(tag, "true");
286                         jsonState = STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY;
287                         break;
288                     case STATE_EXPECT_DATE_UTF8_VALUE:
289                         jsonDateUtf8 = Chars.equalsLowerCaseAscii(tag, "true");
290                         jsonState = STATE_EXPECT_DATE_FORMAT_ENTRY;
291                         break;
292                     default:
293                         // we are picking up values from attributes we don't expect
294                         throw JsonException.$(position, "array expected (value)");
295                 }
296                 break;
297             case JsonLexer.EVT_ARRAY_START:
298                 switch (jsonState) {
299                     case STATE_EXPECT_DATE_FORMAT_ARRAY: // we are working on dates
300                         jsonState = STATE_EXPECT_DATE_FORMAT_ENTRY;
301                         break;
302                     case STATE_EXPECT_TIMESTAMP_FORMAT_ARRAY: // we are working on timestamps
303                         jsonState = STATE_EXPECT_TIMESTAMP_FORMAT_ENTRY;
304                         break;
305                     case STATE_EXPECT_DATE_FORMAT_VALUE:
306                     case STATE_EXPECT_TIMESTAMP_FORMAT_VALUE:
307                         throw JsonException.$(position, "format value expected (array)");
308                     default:
309                         throw JsonException.$(position, "locale value expected (array)");
310                 }
311                 break;
312             default:
313                 break;
314         }
315     }

317     private InputStream openStream(String confRoot, String configFileName) throws IOException, JsonException {
318         // First, check the user-provided file.
319         if (confRoot != null) {
320             final File configFile = new File(confRoot, configFileName);
321             if (configFile.exists()) {
322                 LOG.info().$("loading input format config [file=").$(configFile.getAbsolutePath()).$(']').$();
323                 return new FileInputStream(configFile);
324             }
325         }
326         // Second, fall back to the default config.
327         final InputStream stream = this.getClass().getResourceAsStream(configFileName);
328         if (stream != null) {
329             LOG.info().$("loading input format config [resource=").$(configFileName).$(']').$();
330             return stream;
331         }
332         throw JsonException.$(0, "could not find input format config [confRoot=").put(confRoot)
333                 .put(", configFileName=").put(configFileName)
334                 .put(']');
335     }

337     private void processEntry(CharSequence tag, int position, int stateExpectFormatValue, int stateExpectLocaleValue, int stateExpectUtf8Value) throws JsonException {
338         if (Chars.equals(tag, "format")) {
339             jsonState = stateExpectFormatValue; // expect date format
340         } else if (Chars.equals(tag, "locale")) {
341             jsonState = stateExpectLocaleValue;
342         } else if (Chars.equals(tag, "utf8")) {
343             jsonState = stateExpectUtf8Value;
344         } else {
345             // unknown tag name?
346             throw JsonException.$(position, "unknown [tag=").put(tag).put(']');
347         }
348     }
