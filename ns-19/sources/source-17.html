49     public TypeManager(
50             TextConfiguration configuration,
51             DirectCharSink utf8Sink
52     ) {
53         this.dateAdapterPool = new ObjectPool<>(() -> new DateUtf8Adapter(utf8Sink), configuration.getDateAdapterPoolCapacity());
54         this.timestampUtf8AdapterPool = new ObjectPool<>(() -> new TimestampUtf8Adapter(utf8Sink), configuration.getTimestampAdapterPoolCapacity());
55         this.timestampAdapterPool = new ObjectPool<>(TimestampAdapter::new, configuration.getTimestampAdapterPoolCapacity());
56         this.inputFormatConfiguration = configuration.getInputFormatConfiguration();
57         this.stringAdapter = new StringAdapter(utf8Sink);
58         this.indexedSymbolAdapter = new SymbolAdapter(utf8Sink, true);
59         this.notIndexedSymbolAdapter = new SymbolAdapter(utf8Sink, false);
60         addDefaultProbes();
61 
62         final ObjList<DateFormat> dateFormats = inputFormatConfiguration.getDateFormats();
63         final ObjList<DateLocale> dateLocales = inputFormatConfiguration.getDateLocales();
64         final IntList dateUtf8Flags = inputFormatConfiguration.getDateUtf8Flags();
65         for (int i = 0, n = dateFormats.size(); i < n; i++) {
66             if (dateUtf8Flags.getQuick(i) == 1) {
67                 probes.add(new DateUtf8Adapter(utf8Sink).of(dateFormats.getQuick(i), dateLocales.getQuick(i)));
68             } else {
69                 probes.add(new DateAdapter().of(dateFormats.getQuick(i), dateLocales.getQuick(i)));
70             }
71         }
72 
73         final ObjList<DateFormat> timestampFormats = inputFormatConfiguration.getTimestampFormats();
74         final ObjList<DateLocale> timestampLocales = inputFormatConfiguration.getTimestampLocales();
75         final IntList timestampUtf8Flags = inputFormatConfiguration.getTimestampUtf8Flags();
76         for (int i = 0, n = timestampFormats.size(); i < n; i++) {
77             if (timestampUtf8Flags.getQuick(i) == 1) {
78                 probes.add(new TimestampUtf8Adapter(utf8Sink).of(timestampFormats.getQuick(i), timestampLocales.getQuick(i)));
79             } else {
80                 probes.add(new TimestampAdapter().of(timestampFormats.getQuick(i), timestampLocales.getQuick(i)));
81             }
82         }
83         this.probeCount = probes.size();
84     }

105     public TypeAdapter getTypeAdapter(int columnType) {
106         switch (ColumnType.tagOf(columnType)) {
107             case ColumnType.BYTE:
108                 return ByteAdapter.INSTANCE;
109             case ColumnType.SHORT:
110                 return ShortAdapter.INSTANCE;
111             case ColumnType.CHAR:
112                 return CharAdapter.INSTANCE;
113             case ColumnType.INT:
114                 return IntAdapter.INSTANCE;
115             case ColumnType.LONG:
116                 return LongAdapter.INSTANCE;
117             case ColumnType.BOOLEAN:
118                 return BooleanAdapter.INSTANCE;
119             case ColumnType.FLOAT:
120                 return FloatAdapter.INSTANCE;
121             case ColumnType.DOUBLE:
122                 return DoubleAdapter.INSTANCE;
123             case ColumnType.STRING:
124                 return stringAdapter;
125             case ColumnType.SYMBOL:
126                 return nextSymbolAdapter(false);
127             case ColumnType.LONG256:
128                 return Long256Adapter.INSTANCE;
129             case ColumnType.GEOBYTE:
130             case ColumnType.GEOSHORT:
131             case ColumnType.GEOINT:
132             case ColumnType.GEOLONG:
133                 GeoHashAdapter adapter = GeoHashAdapter.getInstance(columnType);
134                 if (adapter != null) {
135                     return adapter;
136                 }
137             case ColumnType.UUID:
138                 return UuidAdapter.INSTANCE;
139             default:
140                 throw CairoException.nonCritical().put("no adapter for type [id=").put(columnType).put(", name=").put(ColumnType.nameOf(columnType)).put(']');
141         }
142     }
