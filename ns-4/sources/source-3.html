44     private static boolean removedPartitionsIncludes(long ts, MemoryR txMem, int symbolsCount) {
45         long removedPartitionLo = MigrationActions.prefixedBlockOffset(
46                 MigrationActions.TX_OFFSET_MAP_WRITER_COUNT_505,
47                 symbolsCount + 1L,
48                 Integer.BYTES
49         );
50         long removedPartitionCount = txMem.getInt(removedPartitionLo);
51         long removedPartitionsHi = MigrationActions.prefixedBlockOffset(removedPartitionLo, Long.BYTES, removedPartitionCount);
52 
53         for (long offset = removedPartitionLo + Integer.BYTES; offset < removedPartitionsHi; offset += Long.BYTES) {
54             long removedPartition = txMem.getLong(offset);
55             if (removedPartition == ts) {
56                 return true;

58         }
59         return false;
