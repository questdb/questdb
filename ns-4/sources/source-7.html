41     static void migrate(MigrationContext migrationContext) {
42         //  META_COLUMN_DATA_SIZE = 16 -> 32;
43         //  TX_OFFSET_MAP_WRITER_COUNT = 72 -> 128
44 
45         final FilesFacade ff = migrationContext.getFf();
46         final Path path = migrationContext.getTablePath();
47         final int plen = path.length();
48 
49         path.concat(META_FILE_NAME).$();
50         if (!ff.exists(path)) {
51             MigrationActions.LOG.error().$("meta file does not exist, nothing to migrate [path=").$(path).I$();
52             return;
53         }
54 
55         // modify metadata
56         try (final MemoryMARW rwMem = migrationContext.getRwMemory()) {
57             final long thatMetaColumnDataSize = 16;
58             final long thisMetaColumnDataSize = 32;
59 
60             rwMem.of(ff, path, ff.getPageSize(), ff.length(path), MemoryTag.NATIVE_MIG_MMAP);
61 
62             // column count
63             final int columnCount = rwMem.getInt(TableUtils.META_OFFSET_COUNT);
64             long offset = TableUtils.META_OFFSET_COLUMN_TYPES;
65             // 32L here is TableUtils.META_COLUMN_DATA_SIZE at the time of writing this migration
66             long newNameOffset = offset + thisMetaColumnDataSize * columnCount;
67 
68             // the intent is to resize the _meta file and move the variable length (names) segment
69             // to do that we need to work out size of the variable length segment first
70             long oldNameOffset = offset + thatMetaColumnDataSize * columnCount;
71             long o = oldNameOffset;
72             for (int i = 0; i < columnCount; i++) {
73                 int len = rwMem.getStrLen(o);
74                 o += Vm.getStorageLength(len);
75             }
76 
77             final long nameSegmentLen = o - oldNameOffset;
78 
79             // resize the file
80             rwMem.extend(newNameOffset + nameSegmentLen);
81             // move name segment
82             Vect.memmove(rwMem.addressOf(newNameOffset), rwMem.addressOf(oldNameOffset), nameSegmentLen);
83 
84             // copy column information in reverse order
85             o = offset + thatMetaColumnDataSize * (columnCount - 1);
86             long o2 = offset + thisMetaColumnDataSize * (columnCount - 1);
87             final Rnd rnd = SharedRandom.getRandom(migrationContext.getConfiguration());
88             while (o >= offset) {
89                 rwMem.putInt(o2, rwMem.getInt(o)); // type
90                 rwMem.putLong(o2 + 4, rwMem.getInt(o + 4)); // flags
91                 rwMem.putInt(o2 + 12, rwMem.getInt(o + 12)); // index block capacity
92                 rwMem.putLong(o2 + 20, rnd.nextLong()); // column hash
93                 o -= thatMetaColumnDataSize;
94                 o2 -= thisMetaColumnDataSize;
95             }
96             rwMem.jumpTo(newNameOffset + nameSegmentLen);
97         }
98 
99         // update _txn file
100         path.trimTo(plen).concat(TXN_FILE_NAME).$();
101         if (!ff.exists(path)) {
102             MigrationActions.LOG.error().$("tx file does not exist, nothing to migrate [path=").$(path).I$();
103             return;
104         }
105         EngineMigration.backupFile(ff, path, migrationContext.getTablePath2(), TXN_FILE_NAME, 422);
106 
107         MigrationActions.LOG.debug().$("opening for rw [path=").$(path).I$();
108         try (MemoryMARW txMem = migrationContext.createRwMemoryOf(ff, path.$())) {
109 
110             // calculate size of the _txn file
111             final long thatTxOffsetMapWriterCount = 72;
112             final long thisTxOffsetMapWriterCount = 128;
113             final int longsPerAttachedPartition = 4;
114 
115             int symbolCount = txMem.getInt(thatTxOffsetMapWriterCount);
116             int partitionTableSize = txMem.getInt(thatTxOffsetMapWriterCount + 4 + symbolCount * 8L) * 8 * longsPerAttachedPartition;
117 
118             // resize existing file:
119             // thisTxOffsetMapWriterCount + symbolCount + symbolData + partitionTableEntryCount + partitionTableSize
120             long thatSize = thatTxOffsetMapWriterCount + 4 + symbolCount * 8L + 4L + partitionTableSize;
121             long thisSize = thisTxOffsetMapWriterCount + 4 + symbolCount * 8L + 4L + partitionTableSize;
122             txMem.extend(thisSize);
123             txMem.jumpTo(thisSize);
124             Vect.memmove(txMem.addressOf(thisTxOffsetMapWriterCount), txMem.addressOf(thatTxOffsetMapWriterCount), thatSize - thatTxOffsetMapWriterCount);
125 
126             // zero out reserved area
127             Vect.memset(txMem.addressOf(thatTxOffsetMapWriterCount), thisTxOffsetMapWriterCount - thatTxOffsetMapWriterCount, 0);
128         }
129     }
