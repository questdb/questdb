53     public static void migrateEngineTo(CairoEngine engine, int latestVersion, boolean force) {
54         final FilesFacade ff = engine.getConfiguration().getFilesFacade();
55         final CairoConfiguration configuration = engine.getConfiguration();
56         int tempMemSize = Long.BYTES;
57         long mem = Unsafe.malloc(tempMemSize, MemoryTag.NATIVE_MIG);
58 
59         try (
60                 MemoryARW virtualMem = Vm.getARWInstance(ff.getPageSize(), Integer.MAX_VALUE, MemoryTag.NATIVE_MIG_MMAP);
61                 Path path = new Path();
62                 MemoryMARW rwMemory = Vm.getMARWInstance()
63         ) {
64             MigrationContext context = new MigrationContext(engine, mem, tempMemSize, virtualMem, rwMemory);
65             path.of(configuration.getRoot());
66 
67             // check if all tables have been upgraded already
68             path.concat(TableUtils.UPGRADE_FILE_NAME).$();
69             final boolean existed = !force && ff.exists(path);
70             int upgradeFd = openFileRWOrFail(ff, path, configuration.getWriterFileOpenOpts());
71             LOG.debug()
72                     .$("open [fd=").$(upgradeFd)
73                     .$(", path=").$(path)
74                     .I$();
75             if (existed) {
76                 int currentVersion = TableUtils.readIntOrFail(
77                         ff,
78                         upgradeFd,
79                         0,
80                         mem,
81                         path
82                 );
83 
84                 if (currentVersion >= latestVersion) {
85                     LOG.info().$("table structures are up to date").$();
86                     ff.close(upgradeFd);
87                     upgradeFd = -1;
88                 }
89             }
90 
91             if (upgradeFd != -1) {
92                 try {
93                     LOG.info().$("upgrading database [version=").$(latestVersion).I$();
94                     if (upgradeTables(context, latestVersion)) {
95                         TableUtils.writeIntOrFail(
96                                 ff,
97                                 upgradeFd,
98                                 0,
99                                 latestVersion,
100                                 mem,
101                                 path
102                         );
103                     }
104                 } finally {
105                     Vm.bestEffortClose(
106                             ff,
107                             LOG,
108                             upgradeFd,
109                             Integer.BYTES
110                     );
111                 }
112             }
113         } finally {
114             Unsafe.free(mem, tempMemSize, MemoryTag.NATIVE_MIG);
115         }
116     }

122     private static boolean upgradeTables(MigrationContext context, int latestVersion) {
123         final FilesFacade ff = context.getFf();
124         final CharSequence root = context.getConfiguration().getRoot();
125         long mem = context.getTempMemory(8);
126         final AtomicBoolean updateSuccess = new AtomicBoolean(true);
127 
128         try (Path path = new Path(); Path copyPath = new Path()) {
129             path.of(root);
130             copyPath.of(root);
131             final int rootLen = path.length();
132 
133             ff.iterateDir(path.$(), (pUtf8NameZ, type) -> {
134                 if (ff.isDirOrSoftLinkDirNoDots(path, rootLen, pUtf8NameZ, type)) {
135                     copyPath.trimTo(rootLen);
136                     copyPath.concat(pUtf8NameZ);
137                     final int tablePlen = path.length();
138 
139                     if (ff.exists(path.concat(TableUtils.META_FILE_NAME).$())) {
140                         final int fdMeta = openFileRWOrFail(ff, path, context.getConfiguration().getWriterFileOpenOpts());
141                         try {
142                             int currentTableVersion = TableUtils.readIntOrFail(ff, fdMeta, META_OFFSET_VERSION, mem, path);
143                             if (currentTableVersion < latestVersion) {
144                                 LOG.info()
145                                         .$("upgrading [path=").utf8(copyPath.$())
146                                         .$(", fromVersion=").$(currentTableVersion)
147                                         .$(", toVersion=").$(latestVersion)
148                                         .I$();
149 
150                                 copyPath.trimTo(tablePlen);
151                                 backupFile(ff, path, copyPath, TableUtils.META_FILE_NAME, currentTableVersion);
152 
153                                 path.trimTo(tablePlen);
154                                 context.of(path, copyPath, fdMeta);
155 
156                                 for (int ver = currentTableVersion + 1; ver <= latestVersion; ver++) {
157                                     final MigrationAction migration = getMigrationToVersion(ver);
158                                     if (migration != null) {
159                                         try {
160                                             LOG.info().$("upgrading table [path=").utf8(path)
161                                                     .$(", toVersion=").$(ver)
162                                                     .I$();
163                                             migration.migrate(context);
164                                             path.trimTo(tablePlen);
165                                             copyPath.trimTo(tablePlen);
166                                         } catch (Throwable e) {
167                                             LOG.error().$("failed to upgrade table [path=").utf8(path.trimTo(tablePlen))
168                                                     .$(", e=").$(e)
169                                                     .I$();
170                                             throw e;
171                                         }
172                                     }
173 
174                                     path.trimTo(tablePlen).concat(TableUtils.META_FILE_NAME).$();
175                                     LOG.info().$("upgrading table _meta [path=").utf8(path).$(", toVersion=").$(ver).I$();
176                                     TableUtils.writeIntOrFail(ff, fdMeta, META_OFFSET_VERSION, ver, mem, path);
177                                     path.trimTo(tablePlen);
178                                 }
179                             }
180                         } finally {
181                             ff.close(fdMeta);
182                             path.trimTo(tablePlen);
183                             copyPath.trimTo(tablePlen);
184                         }
185                     }
186                 }
187             });
188             LOG.info().$("upgraded tables to ").$(latestVersion).$();
189         }
190         return updateSuccess.get();
191     }

193     static void backupFile(FilesFacade ff, Path src, Path toTemp, String backupName, int version) {
194         // make a copy
195         int copyPathLen = toTemp.length();
196         try {
197             toTemp.concat(backupName).put(".v").put(version);
198             int versionLen = toTemp.length();
199             for (int i = 1; ff.exists(toTemp.$()); i++) {
200                 // if backup file already exists
201                 // add .<num> at the end until file name is unique
202                 LOG.info().$("backup dest exists [to=").$(toTemp).I$();
203                 toTemp.trimTo(versionLen).put('.').put(i);
204             }
205 
206             LOG.info().$("backing up [file=").utf8(src).$(", to=").utf8(toTemp).I$();
207             if (ff.copy(src.$(), toTemp) < 0) {
208                 throw CairoException.critical(ff.errno()).put("Cannot backup transaction file [to=").put(toTemp).put(']');
209             }
210         } finally {
211             toTemp.trimTo(copyPathLen);
212         }
213     }
