124      * Reads 8 bytes from "top" file. Moved from TableUtils when refactored column tops into column version file
125      *
126      * @param ff                 files facade, - intermediary to intercept OS file system calls.
127      * @param path               path has to be set to location of "top" file, excluding file name. Zero terminated string.
128      * @param name               name of top file
129      * @param plen               path length to truncate "path" back to, path is reusable.
130      * @param failIfCouldNotRead if true the method will throw exception if top file cannot be read. Otherwise, 0.
131      * @return number of rows column doesn't have when column was added to table that already had data.
132      */
133     public static long readColumnTop(FilesFacade ff, Path path, CharSequence name, int plen, boolean failIfCouldNotRead) {
134         try {
135             if (ff.exists(topFile(path, name))) {
136                 final int fd = TableUtils.openRO(ff, path, LOG);
137                 try {
138                     long n;
139                     if ((n = ff.readNonNegativeLong(fd, 0)) < 0) {
140                         if (failIfCouldNotRead) {
141                             throw CairoException.critical(Os.errno())
142                                     .put("could not read top of column [file=").put(path)
143                                     .put(", read=").put(n).put(']');
144                         } else {
145                             LOG.error().$("could not read top of column [file=").$(path)
146                                     .$(", read=").$(n)
147                                     .$(", errno=").$(ff.errno())
148                                     .I$();
149                             return 0L;
150                         }
151                     }
152                     return n;
153                 } finally {
154                     ff.close(fd);
155                 }
156             }
157             return 0L;
158         } finally {
159             path.trimTo(plen);
160         }
161     }

184     private static void trimFile(FilesFacade ff, Path path, long size, long opts) {
185         final int fd = TableUtils.openFileRWOrFail(ff, path, opts);
186         if (!ff.truncate(fd, size)) {
187             // This should never happen on migration but better to be on safe side anyway
188             throw CairoException.critical(ff.errno()).put("Cannot trim to size [file=").put(path).put(']');
189         }
190         if (!ff.close(fd)) {
191             // This should never happen on migration but better to be on safe side anyway
192             throw CairoException.critical(ff.errno()).put("Cannot close [file=").put(path).put(']');
