114         txMemory.putInt(TX_OFFSET_COLUMN_VERSION_MIG, 0);
115         txMemory.putInt(TX_OFFSET_TRUNCATE_VERSION_MIG, 0);
116 
117         long pageAddress = txMemory.getPageAddress(0);

179     private static long readColumnTop(FilesFacade ff, Path path, CharSequence name, int plen) {
180         try {
181             if (ff.exists(topFile(path, name))) {
182                 final int fd = openRO(ff, path);
183                 try {
184                     long n;
185                     if ((n = ff.readNonNegativeLong(fd, 0)) < 0) {
186                         return 0L;
187                     }
188                     return n;
189                 } finally {
190                     ff.close(fd);
191                 }
192             }
193             return 0L;
194         } finally {

208     private static LongList readColumnTopsAllPartitions(int columnCount, int partitionBy, long partitionSizeOffset, int partitionTableSize, MemoryMARW txMemory, FilesFacade ff, Path path, int pathLen, ObjList<String> columnNames) {
209         LongList result = new LongList();
210         int partitionCount = partitionTableSize / 8 / 4;
211         long offset = partitionSizeOffset + 4;
212         long prevPartition = Long.MIN_VALUE;
213         long txSize = txMemory.size() - 4 * 8;
214         for (int partitionIndex = 0; partitionIndex < partitionCount; partitionIndex++) {
215             if (offset > txSize) {
216                 throw CairoException.critical(0).put("corrupt _txn file ").put(path.trimTo(pathLen).$()).put(", file is too small to read offset ").put(offset);
217             }
218             long partitionTs = txMemory.getLong(offset);
219             if (partitionTs <= prevPartition) {
220                 throw CairoException.critical(0).put("corrupt _txn file, partitions are not ordered at ").put(path.trimTo(pathLen).$());
221             }
222             long partitionNameTxn = txMemory.getLong(offset + PARTITION_NAME_TX_OFFSET_MIG * 8);
223             readColumnTopsForPartition(result, columnNames, columnCount, partitionBy, partitionTs, partitionNameTxn, ff, path, pathLen);
224             offset += 4 * 8;
225             prevPartition = partitionTs;
226         }
227         return result;
228     }

230     private static void readColumnTopsForPartition(LongList tops, ObjList<String> columnNames, int columnCount, int partitionBy, long partitionTimestamp, long partitionNameTxn, FilesFacade ff, Path path, int pathLen) {
231         tops.add(partitionTimestamp);
232 
233         path.trimTo(pathLen);
234         setPathForPartition(path, partitionBy, partitionTimestamp, partitionNameTxn);
235         int partitionPathLen = path.length();
236 
237         if (ff.exists(path.put(Files.SEPARATOR).$())) {
238             for (int i = 0; i < columnCount; i++) {
239                 path.trimTo(partitionPathLen);
240                 String columnName = columnNames.get(i);
241                 dFile(path, columnName);
242                 long columnTop = -1;
243                 if (ff.exists(path)) {
244                     columnTop = readColumnTop(ff, path.trimTo(partitionPathLen), columnName, partitionPathLen);
245                 }
246                 tops.add(columnTop);
247             }
248         } else {
249             // Sometimes _txn file does not match the table directories, e.g. snapshot is inconsistent.
250             // Consider that file presence is same as previous partition.
251             // Except if previous partition column existed but column top was not 0, make it 0
252             if (tops.size() > columnCount) {
253                 tops.add(tops, tops.size() - columnCount - 1, tops.size() - 1);
254                 for (int i = tops.size() - columnCount, n = tops.size(); i < n; i++) {
255                     if (tops.getQuick(i) > 0) {
256                         tops.setQuick(i, 0);
257                     }
258                 }
259             } else {
260                 for (int i = 0; i < columnCount; i++) {
261                     tops.add(-1L);
262                 }
263             }
264         }
265     }

278     private static long writeColumnVersion(Path tablePath, LongList columnTops, int columnCount, ObjList<String> columnNames, MemoryMARW cvMemory) {
279         int topStep = columnCount + 1;
280         LongList columnVersions = new LongList();
281         LongList maxPartitionIndexWithNoColumnList = new LongList();
282 
283         for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
284             // Column Tops list has long values as follows:
285             // Partition Timestamp
286             // Column top value per every column as follows:
287             // -1: column does not exist for the partition
288             //  0: column present with no column top
289             // >0: column preset and there is a column top
290 
291             int maxPartitionIndexWithNoColumn = -1;
292             for (int partitionIndex = 0; partitionIndex < columnTops.size(); partitionIndex += topStep) {
293                 long columnTop = columnTops.getQuick(partitionIndex + columnIndex + 1);
294                 if (columnTop < 0) {
295                     maxPartitionIndexWithNoColumn = partitionIndex;
296                 }
297             }
298 
299             if (maxPartitionIndexWithNoColumn != -1) {
300                 if (maxPartitionIndexWithNoColumn + topStep >= columnTops.size()) {
301                     throw CairoException.critical(0).put("Table ").put(tablePath).put(" column '").put(columnNames.getQuick(columnIndex)).put("' is not present in the last partition.");
302                 }
303                 long columnAddedPartitionTs = columnTops.getQuick(maxPartitionIndexWithNoColumn + topStep);
304                 columnVersions.add(CV_COL_TOP_DEFAULT_PARTITION_MIG, columnIndex, -1L, columnAddedPartitionTs);
305             }
306             maxPartitionIndexWithNoColumnList.add(maxPartitionIndexWithNoColumn);
307         }
308 
309         // Column Version file must be sorted by Partition Timestamp first and then by Column Index
310         // Go partition by partition to keep adding records sorted
311         for (int partitionIndex = 0; partitionIndex < columnTops.size(); partitionIndex += topStep) {
312             for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
313                 long maxPartitionIndexWithNoColumn = maxPartitionIndexWithNoColumnList.getQuick(columnIndex);
314                 long partitionTs = columnTops.getQuick(partitionIndex);
315                 long columnTop = columnTops.getQuick(partitionIndex + columnIndex + 1);
316 
317                 if (columnTop > 0 || (columnTop == 0 && partitionIndex < maxPartitionIndexWithNoColumn)) {
318                     columnVersions.add(partitionTs, columnIndex, -1, columnTop);
319                 }
320             }
321         }
322         // Flush column tops to the file.
323         for (int i = 0; i < columnVersions.size(); i++) {
324             cvMemory.putLong(Mig620.CV_HEADER_SIZE + i * 8L, columnVersions.getQuick(i));
325         }
326         int sizeByes = columnVersions.size() * 8;
327         cvMemory.jumpTo(sizeByes + Mig620.CV_HEADER_SIZE);
328         return sizeByes;
329     }
