124             LineTcpReceiverConfiguration config,
125             CairoEngine cairoEngine,
126             WorkerPoolManager workerPoolManager,
127             Metrics metrics
128     ) {
129         if (!config.isEnabled()) {
130             return null;
131         }
132 
133         // The ioPool is:
134         // - DEDICATED when PropertyKey.LINE_TCP_IO_WORKER_COUNT is > 0
135         // - DEDICATED (2 worker) when ^ ^ is not set and host has 8 < cpus < 17
136         // - DEDICATED (6 worker) when ^ ^ is not set and host has > 16 cpus
137         // - SHARED otherwise
138 
139         // The writerPool is:
140         // - DEDICATED when PropertyKey.LINE_TCP_WRITER_WORKER_COUNT is > 0
141         // - DEDICATED (1 worker) when ^ ^ is not set
142         // - SHARED otherwise
143 
144         final WorkerPool ioPool = workerPoolManager.getInstance(
145                 config.getIOWorkerPoolConfiguration(),
146                 metrics.health(),
147                 Requester.LINE_TCP_IO
148         );
149         final WorkerPool writerPool = workerPoolManager.getInstance(
150                 config.getWriterWorkerPoolConfiguration(),
151                 metrics.health(),
152                 Requester.LINE_TCP_WRITER
153         );
154         return new LineTcpReceiver(config, cairoEngine, ioPool, writerPool);
155     }
