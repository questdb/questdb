60     private boolean analyze(
61             AliasTranslator translator,
62             ExpressionNode node,
63             RecordMetadata metadata
64     ) throws SqlException {
65         switch (ops.get(node.token)) {
66             case OP_IN:
67                 analyzeIn(translator, node, metadata);
68                 return true;
69             case OP_EQUAL:
70                 analyzeEquals(translator, node, metadata);
71                 return true;
72             case OP_NOT_EQ:
73                 analyzeNotEquals(translator, node, metadata);
74                 return true;
75             case OP_NOT:
76                 if (isInKeyword(node.rhs.token)) {
77                     analyzeNotIn(translator, node.rhs, metadata);
78                 } else {
79                     giveUp();
80                 }
81                 return true;
82             default:
83                 return false;
84         }

96     private void analyzeIn(
97             AliasTranslator translator,
98             ExpressionNode node,
99             RecordMetadata metadata
100     ) throws SqlException {
101         if (node.paramCount < 2) {
102             return;
103         }
104 
105         ExpressionNode col = node.paramCount < 3 ? node.lhs : node.args.getLast();
106         if (col.type != ExpressionNode.LITERAL) {
107             return;
108         }
109 
110         CharSequence column = translator.translateAlias(col.token);
111         int columnIndex = metadata.getColumnIndexQuiet(column);
112         if (columnIndex == -1) {
113             throw SqlException.invalidColumn(node.position, node.token);
114         }
115 
116         int keyIndex = columnIndexesToListIndexes.keyIndex(columnIndex);
117         if (keyIndex > -1) {
118             return;
119         }
120 
121         if (node.rhs != null && node.rhs.type == ExpressionNode.QUERY) {
122             // give up in sym in (select ...) case
123             giveUp();
124             return;
125         }
126 
127         int listIndex = columnIndexesToListIndexes.valueAt(keyIndex);
128         symbolCounts.increment(listIndex, node.paramCount - 1);
129     }

229     /**
230      * Estimates symbol counts for simple filters like the following one:
231      * <code>
232      * sym1 in ('a','b','c') and sym2 = 'd'
233      * </code>
234      * <p>
235      * For more complex filters the estimates are guaranteed to be equal
236      * or great than the actual count of filtered symbol values.
237      */
238     IntList estimate(
239             AliasTranslator translator,
240             ExpressionNode node,
241             RecordMetadata metadata,
242             IntList columnIndexes
243     ) throws SqlException {
244         if (node == null) {
245             return null;
246         }
247 
248         clear();
249 
250         for (int i = 0, n = columnIndexes.size(); i < n; i++) {
251             columnIndexesToListIndexes.put(columnIndexes.getQuick(i), i);
252         }
253 
254         symbolCounts = new IntList(columnIndexes.size());
255         symbolCounts.setAll(columnIndexes.size(), 0);
256 
257         // pre-order iterative tree traversal
258         // see: http://en.wikipedia.org/wiki/Tree_traversal
259 
260         if (!analyze(translator, node, metadata)) {
261             while (!stack.isEmpty() || node != null) {
262                 if (node != null) {
263                     if (isAndKeyword(node.token)) {
264                         if (!analyze(translator, node.rhs, metadata)) {
265                             stack.push(node.rhs);
266                         }
267                         if (gaveUp) {
268                             break;
269                         }
270                         node = analyze(translator, node.lhs, metadata) ? null : node.lhs;
271                         if (gaveUp) {
272                             break;
273                         }
274                     } else if (isOrKeyword(node.token)) {
275                         giveUp();
276                         break;
277                     } else {
278                         node = stack.poll();
279                     }
280                 } else {
281                     node = stack.poll();
282                 }
283             }
284         }
285 
286         for (int i = 0, n = symbolCounts.size(); i < n; i++) {
287             if (symbolCounts.getQuick(i) == 0) {
288                 symbolCounts.setQuick(i, Integer.MAX_VALUE);
289             }
290         }
291         return symbolCounts;
292     }
