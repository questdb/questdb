43             // this is a validation request

45                 throw SqlException.$(node.position, "query is not allowed here");
46             }

48         }

53

56                 break;
57             case 1:
58                 node.rhs = stack.poll();

60             case 2:
61                 node.rhs = stack.poll();

63                 break;
64             default:
65                 for (int i = 0; i < node.paramCount; i++) {
66                     node.args.add(stack.poll());
67                 }
68                 break;
69         }

71     }
