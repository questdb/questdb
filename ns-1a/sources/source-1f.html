355     private ExecutionModel parseCopy(GenericLexer lexer) throws SqlException {
356         if (Chars.isBlank(configuration.getSqlCopyInputRoot())) {
357             throw SqlException.$(lexer.lastTokenPosition(), "COPY is disabled ['cairo.sql.copy.root' is not set?]");
358         }
359         ExpressionNode target = expectExpr(lexer);
360         CharSequence tok = tok(lexer, "'from' or 'to' or 'cancel'");
361 
362         if (isCancelKeyword(tok)) {
363             CopyModel model = copyModelPool.next();
364             model.setCancel(true);
365             model.setTarget(target);
366             return model;
367         }
368 
369         if (isFromKeyword(tok)) {
370             final ExpressionNode fileName = expectExpr(lexer);
371             if (fileName.token.length() < 3 && Chars.startsWith(fileName.token, '\'')) {
372                 throw SqlException.$(fileName.position, "file name expected");
373             }
374 
375             CopyModel model = copyModelPool.next();
376             model.setTarget(target);
377             model.setFileName(fileName);
378 
379             tok = optTok(lexer);
380             if (tok != null && isWithKeyword(tok)) {
381                 tok = tok(lexer, "copy option");
382                 while (tok != null && !isSemicolon(tok)) {
383                     if (isHeaderKeyword(tok)) {
384                         model.setHeader(isTrueKeyword(tok(lexer, "'true' or 'false'")));
385                         tok = optTok(lexer);
386                     } else if (isPartitionKeyword(tok)) {
387                         expectTok(lexer, "by");
388                         tok = tok(lexer, "year month day hour");
389                         int partitionBy = PartitionBy.fromString(tok);
390                         if (partitionBy == -1) {
391                             throw SqlException.$(lexer.getPosition(), "'NONE', 'HOUR', 'DAY', 'MONTH' or 'YEAR' expected");
392                         }
393                         model.setPartitionBy(partitionBy);
394                         tok = optTok(lexer);
395                     } else if (isTimestampKeyword(tok)) {
396                         tok = tok(lexer, "timestamp column name expected");
397                         CharSequence columnName = GenericLexer.immutableOf(GenericLexer.unquote(tok));
398                         if (!TableUtils.isValidColumnName(columnName, configuration.getMaxFileNameLength())) {
399                             throw SqlException.$(lexer.getPosition(), "timestamp column name contains invalid characters");
400                         }
401                         model.setTimestampColumnName(columnName);
402                         tok = optTok(lexer);
403                     } else if (isFormatKeyword(tok)) {
404                         tok = tok(lexer, "timestamp format expected");
405                         CharSequence format = GenericLexer.immutableOf(GenericLexer.unquote(tok));
406                         model.setTimestampFormat(format);
407                         tok = optTok(lexer);
408                     } else if (isOnKeyword(tok)) {
409                         expectTok(lexer, "error");
410                         tok = tok(lexer, "skip_column skip_row abort");
411                         if (Chars.equalsIgnoreCase(tok, "skip_column")) {
412                             model.setAtomicity(Atomicity.SKIP_COL);
413                         } else if (Chars.equalsIgnoreCase(tok, "skip_row")) {
414                             model.setAtomicity(Atomicity.SKIP_ROW);
415                         } else if (Chars.equalsIgnoreCase(tok, "abort")) {
416                             model.setAtomicity(Atomicity.SKIP_ALL);
417                         } else {
418                             throw SqlException.$(lexer.getPosition(), "invalid 'on error' copy option found");
419                         }
420                         tok = optTok(lexer);
421                     } else if (isDelimiterKeyword(tok)) {
422                         tok = tok(lexer, "timestamp character expected");
423                         CharSequence delimiter = GenericLexer.immutableOf(GenericLexer.unquote(tok));
424                         if (delimiter == null || delimiter.length() != 1) {
425                             throw SqlException.$(lexer.getPosition(), "delimiter is empty or contains more than 1 character");
426                         }
427                         char delimiterChar = delimiter.charAt(0);
428                         if (delimiterChar > 127) {
429                             throw SqlException.$(lexer.getPosition(), "delimiter is not an ascii character");
430                         }
431                         model.setDelimiter((byte) delimiterChar);
432                         tok = optTok(lexer);
433                     } else {
434                         throw SqlException.$(lexer.lastTokenPosition(), "unexpected option");
435                     }
436                 }
437             } else if (tok != null && !SqlKeywords.isSemicolon(tok)) {
438                 throw SqlException.$(lexer.lastTokenPosition(), "'with' expected");
439             }
440             return model;
441         }
442         throw SqlException.$(lexer.lastTokenPosition(), "'from' expected");
443     }

450     private ExecutionModel parseCreateTable(GenericLexer lexer, SqlExecutionContext executionContext) throws SqlException {
451         final CreateTableModel model = createTableModelPool.next();
452         final CharSequence tableName;
453         CharSequence tok = tok(lexer, "table name or 'if'");
454         if (SqlKeywords.isIfKeyword(tok)) {
455             if (SqlKeywords.isNotKeyword(tok(lexer, "'not'")) && SqlKeywords.isExistsKeyword(tok(lexer, "'exists'"))) {
456                 model.setIgnoreIfExists(true);
457                 tableName = tok(lexer, "table name");
458             } else {
459                 throw SqlException.$(lexer.lastTokenPosition(), "'if not exists' expected");
460             }
461         } else {
462             tableName = tok;
463         }
464 
465         model.setName(nextLiteral(GenericLexer.assertNoDotsAndSlashes(GenericLexer.unquote(tableName), lexer.lastTokenPosition()), lexer.lastTokenPosition()));
466 
467         tok = tok(lexer, "'(' or 'as'");
468 
469         if (Chars.equals(tok, '(')) {
470             tok = tok(lexer, "like");
471             if (isLikeKeyword(tok)) {
472                 parseLikeTableName(lexer, model);
473                 return model;
474             } else {
475                 lexer.unparseLast();
476                 parseCreateTableColumns(lexer, model);
477             }
478         } else if (isAsKeyword(tok)) {
479             parseCreateTableAsSelect(lexer, model, executionContext);
480         } else {
481             throw errUnexpected(lexer, tok);
482         }
483 
484         while ((tok = optTok(lexer)) != null && Chars.equals(tok, ',')) {
485             tok = tok(lexer, "'index' or 'cast'");
486             if (isIndexKeyword(tok)) {
487                 parseCreateTableIndexDef(lexer, model);
488             } else if (isCastKeyword(tok)) {
489                 parseCreateTableCastDef(lexer, model);
490             } else {
491                 throw errUnexpected(lexer, tok);
492             }
493         }
494 
495         ExpressionNode timestamp = parseTimestamp(lexer, tok);
496         if (timestamp != null) {
497             // ignore index, validate column
498             int timestampIdx = getCreateTableColumnIndex(model, timestamp.token, timestamp.position);
499             int timestampType = model.getColumnType(timestampIdx);
500             if (timestampType != ColumnType.TIMESTAMP && timestampType != -1) { //type can be -1 for create table as select because types aren't known yet
501                 throw SqlException.position(timestamp.position).put("TIMESTAMP column expected [actual=").put(ColumnType.nameOf(timestampType)).put(']');
502             }
503             model.setTimestamp(timestamp);
504             tok = optTok(lexer);
505         }
506         int maxUncommittedRows = configuration.getMaxUncommittedRows();
507         long o3MaxLag = configuration.getO3MaxLag();
508 
509         int walSetting = WAL_NOT_SET;
510 
511         ExpressionNode partitionBy = parseCreateTablePartition(lexer, tok);
512         if (partitionBy != null) {
513             if (model.getTimestamp() == null) {
514                 throw SqlException.$(partitionBy.position, "partitioning is possible only on tables with designated timestamps");
515             }
516             if (PartitionBy.fromString(partitionBy.token) == -1) {
517                 throw SqlException.$(partitionBy.position, "'NONE', 'HOUR', 'DAY', 'MONTH' or 'YEAR' expected");
518             }
519             model.setPartitionBy(partitionBy);
520             tok = optTok(lexer);
521 
522             if (tok != null) {
523                 if (isWalKeyword(tok)) {
524                     if (!PartitionBy.isPartitioned(model.getPartitionBy())) {
525                         throw SqlException.position(lexer.lastTokenPosition()).put("WAL Write Mode can only be used on partitioned tables");
526                     }
527                     walSetting = WAL_ENABLED;
528                     tok = optTok(lexer);
529                 } else if (isBypassKeyword(tok)) {
530                     tok = optTok(lexer);
531                     if (tok != null && isWalKeyword(tok)) {
532                         walSetting = WAL_DISABLED;
533                         tok = optTok(lexer);
534                     } else {
535                         throw SqlException.position(
536                                         tok == null ? lexer.getPosition() : lexer.lastTokenPosition()
537                                 ).put(" invalid syntax, should be BYPASS WAL but was BYPASS ")
538                                 .put(tok != null ? tok : "");
539                     }
540                 }
541             }
542 
543             if (tok != null && isWithKeyword(tok)) {
544                 ExpressionNode expr;
545                 while ((expr = expr(lexer, (QueryModel) null)) != null) {
546                     if (Chars.equals(expr.token, '=')) {

782     private QueryModel parseDml(
783             GenericLexer lexer,
784             @Nullable LowerCaseCharSequenceObjHashMap<WithClauseModel> withClauses,
785             int modelPosition
786     ) throws SqlException {
787         QueryModel model = null;
788         QueryModel prevModel = null;
789         while (true) {
790 
791             LowerCaseCharSequenceObjHashMap<WithClauseModel> parentWithClauses = prevModel != null ? prevModel.getWithClauses() : withClauses;
792             LowerCaseCharSequenceObjHashMap<WithClauseModel> topWithClauses = model == null ? topLevelWithModel : null;
793 
794             QueryModel unionModel = parseDml0(lexer, parentWithClauses, topWithClauses, modelPosition);
795             if (prevModel == null) {
796                 model = unionModel;
797                 prevModel = model;
798             } else {
799                 prevModel.setUnionModel(unionModel);
800                 prevModel = unionModel;
801             }
802 
803             CharSequence tok = optTok(lexer);
804             if (tok == null || Chars.equals(tok, ';') || setOperations.excludes(tok)) {
805                 lexer.unparseLast();
806                 return model;
807             }
808 
809             if (prevModel.getNestedModel() != null) {
810                 if (prevModel.getNestedModel().getOrderByPosition() > 0) {
811                     throw SqlException.$(prevModel.getNestedModel().getOrderByPosition(), "unexpected token 'order'");
812                 }
813                 if (prevModel.getNestedModel().getLimitPosition() > 0) {
814                     throw SqlException.$(prevModel.getNestedModel().getLimitPosition(), "unexpected token 'limit'");
815                 }
816             }
817 
818             if (isUnionKeyword(tok)) {
819                 tok = tok(lexer, "all or select");
820                 if (isAllKeyword(tok)) {
821                     prevModel.setSetOperationType(QueryModel.SET_OPERATION_UNION_ALL);
822                     modelPosition = lexer.getPosition();
823                 } else {
824                     prevModel.setSetOperationType(QueryModel.SET_OPERATION_UNION);
825                     lexer.unparseLast();
826                     modelPosition = lexer.lastTokenPosition();
827                 }
828                 continue;
829             }
830 
831             if (isExceptKeyword(tok)) {
832                 prevModel.setSetOperationType(QueryModel.SET_OPERATION_EXCEPT);
833                 continue;
834             }
835 
836             if (isIntersectKeyword(tok)) {
837                 prevModel.setSetOperationType(QueryModel.SET_OPERATION_INTERSECT);
838             }
839         }
840     }

913     private QueryModel parseDmlUpdate(GenericLexer lexer) throws SqlException {
914         // Update QueryModel structure is
915         // QueryModel with SET column expressions (updateQueryModel)
916         // |-- nested QueryModel of select-virtual or select-choose of data selected for update (fromModel)
917         //     |-- nested QueryModel with selected data (nestedModel)
918         //         |-- join QueryModels to represent FROM clause
919         CharSequence tok;
920         final int modelPosition = lexer.getPosition();
921 
922         QueryModel updateQueryModel = queryModelPool.next();
923         updateQueryModel.setModelType(ExecutionModel.UPDATE);
924         updateQueryModel.setModelPosition(modelPosition);
925         QueryModel fromModel = queryModelPool.next();
926         fromModel.setModelPosition(modelPosition);
927         updateQueryModel.setIsUpdate(true);
928         fromModel.setIsUpdate(true);
929         tok = tok(lexer, "UPDATE, WITH or table name expected");
930 
931         // [update]
932         if (isUpdateKeyword(tok)) {
933             // parse SET statements into updateQueryModel and rhs of SETs into fromModel to select
934             parseUpdateClause(lexer, updateQueryModel, fromModel);
935 
936             // create nestedModel QueryModel to source rowids for the update
937             QueryModel nestedModel = queryModelPool.next();
938             nestedModel.setTableNameExpr(fromModel.getTableNameExpr());
939             nestedModel.setAlias(updateQueryModel.getAlias());
940             nestedModel.setIsUpdate(true);
941 
942             // nest nestedModel inside fromModel
943             fromModel.setTableNameExpr(null);
944             fromModel.setNestedModel(nestedModel);
945 
946             // Add WITH clauses if they exist into fromModel
947             fromModel.getWithClauses().putAll(topLevelWithModel);
948 
949             tok = optTok(lexer);
950 
951             // [from]
952             if (tok != null && isFromKeyword(tok)) {
953                 tok = ","; // FROM in Postgres UPDATE statement means cross join
954                 int joinType;
955                 int i = 0;
956                 while (tok != null && (joinType = joinStartSet.get(tok)) != -1) {
957                     if (i++ == 1) {
958                         throw SqlException.$(lexer.lastTokenPosition(), "JOIN is not supported on UPDATE statement");
959                     }
960                     // expect multiple [[inner | outer | cross] join]
961                     nestedModel.addJoinModel(parseJoin(lexer, tok, joinType, topLevelWithModel));
962                     tok = optTok(lexer);
963                 }
964             } else if (tok != null && isSemicolon(tok)) {
965                 tok = null;
966             } else if (tok != null && !isWhereKeyword(tok)) {
967                 throw SqlException.$(lexer.lastTokenPosition(), "FROM, WHERE or EOF expected");
968             }
969 
970             // [where]
971             if (tok != null && isWhereKeyword(tok)) {
972                 ExpressionNode expr = expr(lexer, fromModel);
973                 if (expr != null) {
974                     nestedModel.setWhereClause(expr);
975                 } else {
976                     throw SqlException.$((lexer.lastTokenPosition()), "empty where clause");
977                 }
978             } else if (tok != null && !isSemicolon(tok)) {
979                 throw errUnexpected(lexer, tok);
980             }
981 
982             updateQueryModel.setNestedModel(fromModel);
983         }
984         return updateQueryModel;
985     }

1042     private void parseFromClause(GenericLexer lexer, QueryModel model, QueryModel masterModel) throws SqlException {
1043         CharSequence tok = expectTableNameOrSubQuery(lexer);
1044         // expect "(" in case of sub-query
1045 
1046         if (Chars.equals(tok, '(')) {
1047             QueryModel proposedNested = parseAsSubQueryAndExpectClosingBrace(lexer, masterModel.getWithClauses());
1048             tok = optTok(lexer);
1049 
1050             // do not collapse aliased sub-queries or those that have timestamp()
1051             // select * from (table) x
1052             if (tok == null || (tableAliasStop.contains(tok) && !SqlKeywords.isTimestampKeyword(tok))) {
1053                 final QueryModel target = proposedNested.getNestedModel();
1054                 // when * is artificial, there is no union, there is no "where" clause inside sub-query,
1055                 // e.g. there was no "select * from" we should collapse sub-query to a regular table
1056                 if (
1057                         proposedNested.isArtificialStar()
1058                                 && proposedNested.getUnionModel() == null
1059                                 && target.getWhereClause() == null
1060                                 && target.getOrderBy().size() == 0
1061                                 && target.getLatestBy().size() == 0
1062                                 && target.getNestedModel() == null
1063                                 && target.getSampleBy() == null
1064                                 && target.getGroupBy().size() == 0
1065                                 && proposedNested.getLimitLo() == null
1066                                 && proposedNested.getLimitHi() == null
1067                 ) {
1068                     model.setTableNameExpr(target.getTableNameExpr());
1069                     model.setAlias(target.getAlias());
1070                     model.setTimestamp(target.getTimestamp());
1071 
1072                     int n = target.getJoinModels().size();
1073                     for (int i = 1; i < n; i++) {
1074                         model.addJoinModel(target.getJoinModels().getQuick(i));
1075                     }
1076                     proposedNested = null;
1077                 } else {
1078                     lexer.unparseLast();
1079                 }
1080             } else {
1081                 lexer.unparseLast();
1082             }
1083 
1084             if (proposedNested != null) {
1085                 model.setNestedModel(proposedNested);
1086                 model.setNestedModelIsSubQuery(true);
1087                 tok = setModelAliasAndTimestamp(lexer, model);
1088             }
1089         } else {
1090             lexer.unparseLast();
1091             parseSelectFrom(lexer, model, masterModel.getWithClauses());
1092             tok = setModelAliasAndTimestamp(lexer, model);
1093 
1094             // expect [latest by] (deprecated syntax)
1095             if (tok != null && isLatestKeyword(tok)) {
1096                 parseLatestBy(lexer, model);
1097                 tok = optTok(lexer);
1098             }
1099         }
1100 
1101         // expect multiple [[inner | outer | cross] join]
1102         int joinType;
1103         while (tok != null && (joinType = joinStartSet.get(tok)) != -1) {
1104             model.addJoinModel(parseJoin(lexer, tok, joinType, masterModel.getWithClauses()));
1105             tok = optTok(lexer);
1106         }
1107 
1108         checkSupportedJoinType(lexer, tok);
1109 
1110         // expect [where]
1111 
1112         if (tok != null && isWhereKeyword(tok)) {
1113             if (model.getLatestByType() == QueryModel.LATEST_BY_NEW) {
1114                 throw SqlException.$((lexer.lastTokenPosition()), "unexpected where clause after 'latest on'");
1115             }
1116             ExpressionNode expr = expr(lexer, model);
1117             if (expr != null) {
1118                 model.setWhereClause(expr);
1119                 tok = optTok(lexer);
1120             } else {
1121                 throw SqlException.$((lexer.lastTokenPosition()), "empty where clause");
1122             }
1123         }
1124 
1125         // expect [latest by] (new syntax)
1126 
1127         if (tok != null && isLatestKeyword(tok)) {
1128             if (model.getLatestByType() == QueryModel.LATEST_BY_DEPRECATED) {
1129                 throw SqlException.$((lexer.lastTokenPosition()), "mix of new and deprecated 'latest by' syntax");
1130             }
1131             expectTok(lexer, "on");
1132             parseLatestByNew(lexer, model);
1133             tok = optTok(lexer);
