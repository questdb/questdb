271      * Allows processing of batches of sql statements (sql scripts) separated by ';' .
272      * Each query is processed in sequence and processing stops on first error and whole batch gets discarded.
273      * Noteworthy difference between this and 'normal' query is that all empty queries get ignored, e.g.
274      * <br>
275      * select 1;<br>
276      * ; ;/* comment \*\/;--comment\n; - these get ignored <br>
277      * update a set b=c  ; <br>
278      * <p>
279      * Useful PG doc link :
280      *
281      * @param query            - block of queries to process
282      * @param executionContext - SQL execution context
283      * @param batchCallback    - callback to perform actions prior to or after batch part compilation, e.g. clear caches or execute command
284      * @throws SqlException              - in case of syntax error
285      * @throws PeerDisconnectedException - when peer is disconnected
286      * @throws PeerIsSlowToReadException - when peer is too slow
287      * @throws QueryPausedException      - when query is paused
288      * @see <a href="https://www.postgresql.org/docs/current/protocol-flow.html#id-1.10.5.7.4">PostgreSQL documentation</a>
289      */
290     public void compileBatch(
291             @NotNull CharSequence query,
292             @NotNull SqlExecutionContext executionContext,
293             BatchCallback batchCallback
294     ) throws PeerIsSlowToReadException, PeerDisconnectedException, QueryPausedException, SqlException {
295 
296         LOG.info().$("batch [text=").$(query).I$();
297 
298         clear();
299         lexer.of(query);
300         isSingleQueryMode = false;
301 
302         if (batchCallback == null) {
303             batchCallback = EMPTY_CALLBACK;
304         }
305 
306         int position;
307 
308         while (lexer.hasNext()) {
309             // skip over empty statements that'd cause error in parser
310             position = getNextValidTokenPosition();
311             if (position == -1) {
312                 return;
313             }
314 
315             boolean recompileStale = true;
316             for (int retries = 0; recompileStale; retries++) {
317                 try {
318                     batchCallback.preCompile(this);
319                     clear(); // we don't use normal compile here because we can't reset existing lexer
320                     CompiledQuery current = compileInner(executionContext, query);
321                     // We've to move lexer because some query handlers don't consume all tokens (e.g. SET )
322                     // some code in postCompile might need full text of current query
323                     CharSequence currentQuery = query.subSequence(position, goToQueryEnd());
324                     batchCallback.postCompile(this, current, currentQuery);
325                     recompileStale = false;
326                 } catch (TableReferenceOutOfDateException e) {
327                     if (retries == TableReferenceOutOfDateException.MAX_RETRY_ATTEMPS) {
328                         throw e;
329                     }
330                     LOG.info().$(e.getFlyweightMessage()).$();
331                     // will recompile
332                     lexer.restart();
333                 }
334             }
335         }
336     }

414         CharSequence tok;
415         tok = expectToken(lexer, "'table' or 'system'");
416 
417         if (SqlKeywords.isTableKeyword(tok)) {
418             final int tableNamePosition = lexer.getPosition();
419             tok = GenericLexer.unquote(expectToken(lexer, "table name"));
420             TableToken tableToken = tableExistsOrFail(tableNamePosition, tok, executionContext);
421 
422             try (TableRecordMetadata tableMetadata = executionContext.getMetadata(tableToken)) {
423                 tok = expectToken(lexer, "'add', 'alter' or 'drop'");
424 
425                 if (SqlKeywords.isAddKeyword(tok)) {
426                     return alterTableAddColumn(tableNamePosition, tableToken, tableMetadata);
427                 } else if (SqlKeywords.isDropKeyword(tok)) {
428                     tok = expectToken(lexer, "'column' or 'partition'");
429                     if (SqlKeywords.isColumnKeyword(tok)) {
430                         return alterTableDropColumn(tableNamePosition, tableToken, tableMetadata);
431                     } else if (SqlKeywords.isPartitionKeyword(tok)) {
432                         return alterTableDropDetachOrAttachPartition(tableMetadata, tableToken, PartitionAction.DROP, executionContext);
433                     } else {
434                         throw SqlException.$(lexer.lastTokenPosition(), "'column' or 'partition' expected");
435                     }
436                 } else if (SqlKeywords.isRenameKeyword(tok)) {
437                     tok = expectToken(lexer, "'column'");
438                     if (SqlKeywords.isColumnKeyword(tok)) {
439                         return alterTableRenameColumn(tableNamePosition, tableToken, tableMetadata);
440                     } else {
441                         throw SqlException.$(lexer.lastTokenPosition(), "'column' expected");
442                     }
443                 } else if (SqlKeywords.isAttachKeyword(tok)) {
444                     tok = expectToken(lexer, "'partition'");
445                     if (SqlKeywords.isPartitionKeyword(tok)) {
446                         return alterTableDropDetachOrAttachPartition(tableMetadata, tableToken, PartitionAction.ATTACH, executionContext);
447                     } else {
448                         throw SqlException.$(lexer.lastTokenPosition(), "'partition' expected");
449                     }
450                 } else if (SqlKeywords.isDetachKeyword(tok)) {
451                     tok = expectToken(lexer, "'partition'");
452                     if (SqlKeywords.isPartitionKeyword(tok)) {
453                         return alterTableDropDetachOrAttachPartition(tableMetadata, tableToken, PartitionAction.DETACH, executionContext);
454                     } else {
455                         throw SqlException.$(lexer.lastTokenPosition(), "'partition' expected");
456                     }
457                 } else if (SqlKeywords.isAlterKeyword(tok)) {
458                     tok = expectToken(lexer, "'column'");
459                     if (SqlKeywords.isColumnKeyword(tok)) {
460                         final int columnNamePosition = lexer.getPosition();
461                         tok = expectToken(lexer, "column name");
462                         final CharSequence columnName = GenericLexer.immutableOf(tok);
463                         tok = expectToken(lexer, "'add index' or 'drop index' or 'cache' or 'nocache'");
464                         if (SqlKeywords.isAddKeyword(tok)) {
465                             expectKeyword(lexer, "index");
466                             tok = SqlUtil.fetchNext(lexer);
467                             int indexValueCapacity = -1;
468 
469                             if (tok != null && (!isSemicolon(tok))) {
470                                 if (!SqlKeywords.isCapacityKeyword(tok)) {
471                                     throw SqlException.$(lexer.lastTokenPosition(), "'capacity' expected");

473                                     tok = expectToken(lexer, "capacity value");
474                                     try {
475                                         indexValueCapacity = Numbers.parseInt(tok);

477                                             throw SqlException.$(lexer.lastTokenPosition(), "positive integer literal expected as index capacity");
478                                         }
479                                     } catch (NumericException e) {
480                                         throw SqlException.$(lexer.lastTokenPosition(), "positive integer literal expected as index capacity");
481                                     }
482                                 }
483                             }
484

486                         } else if (SqlKeywords.isDropKeyword(tok)) {

488                             expectKeyword(lexer, "index");

492                             }

497                             return alterTableColumnCacheFlag(tableNamePosition, tableToken, columnName, tableMetadata, false);
498                         } else {
499                             throw SqlException.$(lexer.lastTokenPosition(), "'add', 'drop', 'cache' or 'nocache' expected").put(" found '").put(tok).put('\'');
500                         }
501                     } else {
502                         throw SqlException.$(lexer.lastTokenPosition(), "'column' or 'partition' expected");
503                     }
504                 } else if (SqlKeywords.isSetKeyword(tok)) {
505                     tok = expectToken(lexer, "'param' or 'type'");
506                     if (SqlKeywords.isParamKeyword(tok)) {
507                         final int paramNamePosition = lexer.getPosition();
508                         tok = expectToken(lexer, "param name");
