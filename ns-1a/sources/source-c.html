46     public FunctionFactoryCache(CairoConfiguration configuration, Iterable<FunctionFactory> functionFactories) {
47         boolean enableTestFactories = configuration.enableTestFactories();
48         LOG.info().$("loading functions [test=").$(enableTestFactories).$(']').$();
49         for (FunctionFactory factory : functionFactories) {
50             if (!factory.getClass().getName().contains("test") || enableTestFactories) {
51                 try {
52                     final FunctionFactoryDescriptor descriptor = new FunctionFactoryDescriptor(factory);
53                     final String name = descriptor.getName();
54                     addFactoryToList(factories, descriptor);
55 
56                     // Add != counterparts to equality function factories
57                     if (factory.isBoolean()) {
58                         switch (name) {
59                             case "=":
60                                 addFactoryToList(factories, createNegatingFactory("!=", factory));
61                                 addFactoryToList(factories, createNegatingFactory("<>", factory));
62                                 if (descriptor.getArgTypeMask(0) != descriptor.getArgTypeMask(1)) {
63                                     FunctionFactory swappingFactory = createSwappingFactory("=", factory);
64                                     addFactoryToList(factories, swappingFactory);
65                                     addFactoryToList(factories, createNegatingFactory("!=", swappingFactory));
66                                     addFactoryToList(factories, createNegatingFactory("<>", swappingFactory));
67                                 }
68                                 break;
69                             case "<":
70                                 // `a < b` == `a >= b`
71                                 addFactoryToList(factories, createNegatingFactory(">=", factory));
72                                 FunctionFactory greaterThan = createSwappingFactory(">", factory);
73                                 // `a < b` == `b > a`
74                                 addFactoryToList(factories, greaterThan);
75                                 // `b > a` == !(`b <= a`)
76                                 addFactoryToList(factories, createNegatingFactory("<=", greaterThan));
77                                 break;
78                         }
79                     } else if (factory.isGroupBy()) {
80                         groupByFunctionNames.add(name);
81                     } else if (factory.isCursor()) {
82                         cursorFunctionNames.add(name);
83                     } else if (factory.isRuntimeConstant()) {
84                         runtimeConstantFunctionNames.add(name);
85                     }
86                 } catch (SqlException e) {
87                     LOG.error().$((Sinkable) e).$(" [signature=").$(factory.getSignature()).$(",class=").$(factory.getClass().getName()).$(']').$();
88                 }
89             }
90         }
91     }

114     public boolean isValidNoArgFunction(ExpressionNode node) {
115         final ObjList<FunctionFactoryDescriptor> overload = getOverloadList(node.token);
116         if (overload == null) {
117             return false;
118         }
119 
120         for (int i = 0, n = overload.size(); i < n; i++) {
121             FunctionFactoryDescriptor ffd = overload.getQuick(i);
122             if (ffd.getSigArgCount() == 0) {
123                 return true;
124             }
125         }
126

134     private void addFactoryToList(LowerCaseCharSequenceObjHashMap<ObjList<FunctionFactoryDescriptor>> list, FunctionFactoryDescriptor descriptor) {
135         String name = descriptor.getName();
136         int index = list.keyIndex(name);
137         ObjList<FunctionFactoryDescriptor> overload;
138         if (index < 0) {
139             overload = list.valueAtQuick(index);
140         } else {
141             overload = new ObjList<>(4);
142             list.putAt(index, name, overload);
143         }
144         overload.add(descriptor);
