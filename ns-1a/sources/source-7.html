56     public void executeDropIndex(CharSequence columnName, int columnIndex) {
57         int partitionBy = tableWriter.getPartitionBy();
58         int partitionCount = tableWriter.getPartitionCount();
59         try {
60             updateColumnIndexes.clear();
61             cleanupColumnVersions.clear();
62             rollbackColumnVersions.clear();
63             for (int pIndex = 0; pIndex < partitionCount; pIndex++) {
64                 long pTimestamp = tableWriter.getPartitionTimestamp(pIndex);
65                 long pVersion = tableWriter.getPartitionNameTxn(pIndex);
66                 long columnVersion = tableWriter.getColumnNameTxn(pTimestamp, columnIndex);
67                 long columnTop = tableWriter.getColumnTop(pTimestamp, columnIndex, -1L);
68 
69                 if (columnTop != -1L) {
70                     // bump up column version, metadata will be updated later
71                     tableWriter.upsertColumnVersion(pTimestamp, columnIndex, columnTop);
72                     final long columnDropIndexVersion = tableWriter.getColumnNameTxn(pTimestamp, columnIndex);
73 
74                     // create hard link to column data
75                     // src
76                     partitionDFile(path, rootLen, partitionBy, pTimestamp, pVersion, columnName, columnVersion);
77                     // hard link
78                     partitionDFile(other, rootLen, partitionBy, pTimestamp, pVersion, columnName, columnDropIndexVersion);
79                     if (-1 == ff.hardLink(path, other)) {
80                         throw CairoException.critical(ff.errno())
81                                 .put("cannot hardLink [src=").put(path)
82                                 .put(", hardLink=").put(other)
83                                 .put(']');
84                     }
85 
86                     // add to cleanup tasks, the index will be removed in due time
87                     updateColumnIndexes.add(columnIndex);
88                     cleanupColumnVersions.add(columnIndex, columnVersion, pTimestamp, pVersion);
89                     rollbackColumnVersions.add(columnIndex, columnDropIndexVersion, pTimestamp, pVersion);
90                 }
91             }
92         } catch (Throwable th) {
93             LOG.error().$("Could not DROP INDEX: ").$(th.getMessage()).$();
94             updateColumnIndexes.clear();
95             cleanupColumnVersions.clear();
96 
97             // cleanup successful links prior to the failed link operation
98             int limit = rollbackColumnVersions.size();
99             if (limit / 4 < partitionCount) {
100                 for (int i = 0; i < limit; i += 4) {
101                     final long columnDropIndexVersion = rollbackColumnVersions.getQuick(i + 1);
102                     final long pTimestamp = rollbackColumnVersions.getQuick(i + 2);
103                     final long partitionNameTxn = rollbackColumnVersions.getQuick(i + 3);
104                     partitionDFile(other, rootLen, partitionBy, pTimestamp, partitionNameTxn, columnName, columnDropIndexVersion);
105                     if (!ff.remove(other)) {
106                         LOG.info().$("Please remove this file \"").$(other).$('"').I$();
107                     }
108                 }
109             }
110             throw th;
111         } finally {
112             path.trimTo(rootLen);
113             other.trimTo(rootLen);
114         }
115     }
