78     public static int extractGeoHashSuffix(int position, CharSequence tok) throws SqlException {
79         assert tok.charAt(0) == '#'; // ^ ^
80         // EP has already checked that the 'd' in '/d', '/dd' are numeric [0..9]
81         int tokLen = tok.length();
82         if (tokLen > 1) {
83             if (tokLen >= 3 && tok.charAt(tokLen - 3) == '/') { // '/dd'
84                 short bits = (short) (10 * tok.charAt(tokLen - 2) + tok.charAt(tokLen - 1) - 528); // 10 * 48 + 48
85                 if (bits >= 1 && bits <= ColumnType.GEO_HASH_MAX_BITS_LENGTH) {
86                     return Numbers.encodeLowHighShorts((short) 3, bits);
87                 }
88                 throw SqlException.$(position, "invalid bits size for GEOHASH constant: ").put(tok);
89             }
90             if (tok.charAt(tokLen - 2) == '/') { // '/d'
91                 char du = tok.charAt(tokLen - 1);
92                 if (du >= '1' && du <= '9') {
93                     return Numbers.encodeLowHighShorts((short) 2, (short) (du - 48));
94                 }
95                 throw SqlException.$(position, "invalid bits size for GEOHASH constant: ").put(tok);
96             }
97         }
98         return Numbers.encodeLowHighShorts((short) 0, (short) (5 * Math.max(tokLen - 1, 0))); // - 1 to exclude '#'
99     }

108     public static boolean isGeoHashCharsConstant(CharSequence tok) {
109         assert tok.charAt(0) == '#'; // called by ExpressionParser where this has been checked.
110         // the EP will eagerly try to detect '/dd' following the geohash token, and if so
111         // it will create a FloatingSequencePair with '/' as separator. At this point

157     private int processLambdaQuery(GenericLexer lexer, ExpressionParserListener listener, int argStackDepth) throws SqlException {
158         // It is highly likely this expression parser will be re-entered when
159         // parsing sub-query. To prevent sub-query consuming operation stack we must add a
160         // control node, which would prevent such consumption
161 
162         // precedence must be max value to make sure control node isn't
163         // consumed as parameter to a greedy function
164         opStack.push(expressionNodePool.next().of(ExpressionNode.CONTROL, "|", Integer.MAX_VALUE, lexer.lastTokenPosition()));
165 
166         final int paramCountStackSize = copyToBackup(paramCountStack, backupParamCountStack);
167         final int argStackDepthStackSize = copyToBackup(argStackDepthStack, backupArgStackDepthStack);
168         final int castBraceCountStackSize = copyToBackup(castBraceCountStack, backupCastBraceCountStack);
169         final int caseBraceCountStackSize = copyToBackup(caseBraceCountStack, backupCaseBraceCountStack);
170 
171         int pos = lexer.lastTokenPosition();
172         // allow sub-query to parse "select" keyword
173         lexer.unparseLast();
174 
175         ExpressionNode node = expressionNodePool.next().of(ExpressionNode.QUERY, null, 0, pos);
176         // validate is Query is allowed
177         onNode(listener, node, argStackDepth);
178         // we can compile query if all is well
179         node.queryModel = sqlParser.parseAsSubQuery(lexer, null);
180         argStackDepth = onNode(listener, node, argStackDepth);
181 
182         // pop our control node if sub-query hasn't done it
183         ExpressionNode control = opStack.peek();
184         if (control != null && control.type == ExpressionNode.CONTROL && Chars.equals(control.token, '|')) {
185             opStack.pop();
186         }
187 
188         backupParamCountStack.copyTo(paramCountStack, paramCountStackSize);
189         backupArgStackDepthStack.copyTo(argStackDepthStack, argStackDepthStackSize);
190         backupCastBraceCountStack.copyTo(castBraceCountStack, castBraceCountStackSize);
191         backupCaseBraceCountStack.copyTo(caseBraceCountStack, caseBraceCountStackSize);
192         return argStackDepth;
193     }

197         try {
198             int paramCount = 0;
199             int braceCount = 0;
200             int bracketCount = 0;
201             int betweenCount = 0;
202             int betweenAndCount = 0;
203             int caseCount = 0;

205             int castAsCount = 0;
206             int castBraceCount = 0;

215             OUT:
216             while ((tok = SqlUtil.fetchNext(lexer)) != null) {
217                 thisChar = tok.charAt(0);
218                 prevBranch = thisBranch;
219                 boolean processDefaultBranch = false;
220                 final int lastPos = lexer.lastTokenPosition();
221                 switch (thisChar) {
222                     case '-':
223                     case '+':
224                         processDefaultBranch = true;
225                         if (prevBranch == BRANCH_CONSTANT) {
226                             if (lastPos > 0) {
227                                 char c = lexer.getContent().charAt(lastPos - 1);
228                                 if (c == 'e' || c == 'E') { // Incomplete scientific floating-point literal
229                                     ExpressionNode en = opStack.peek();
230                                     ((GenericLexer.FloatingSequence) en.token).setHi(lastPos + 1);
231                                     processDefaultBranch = false;
232                                 }
233                             }
234                         }
235                         break;
236                     case '.':
237                         // Check what is on stack. If we have 'a .b' we have to stop processing
238                         if (thisBranch == BRANCH_LITERAL || thisBranch == BRANCH_CONSTANT) {
239                             char c = lexer.getContent().charAt(lastPos - 1);
240                             if (GenericLexer.WHITESPACE_CH.contains(c)) {
241                                 lexer.unparseLast();
242                                 break OUT;
243                             }
244 
245                             if (Chars.isQuote(c)) {
246                                 ExpressionNode en = opStack.pop();
247                                 CharacterStoreEntry cse = characterStore.newEntry();
248                                 cse.put(GenericLexer.unquote(en.token)).put('.');
249                                 opStack.push(expressionNodePool.next().of(ExpressionNode.LITERAL, cse.toImmutable(), Integer.MIN_VALUE, en.position));
250                             } else {
251                                 // attach dot to existing literal or constant
252                                 ExpressionNode en = opStack.peek();
253                                 ((GenericLexer.FloatingSequence) en.token).setHi(lastPos + 1);
254                             }
255                         }
256                         if (prevBranch == BRANCH_DOT || prevBranch == BRANCH_DOT_DEREFERENCE) {
257                             throw SqlException.$(lastPos, "too many dots");
258                         }
259 
260                         if (prevBranch == BRANCH_RIGHT_PARENTHESIS) {
261                             thisBranch = BRANCH_DOT_DEREFERENCE;
262                         } else {
263                             thisBranch = BRANCH_DOT;
264                         }
265                         break;
266                     case ',':
267                         if (prevBranch == BRANCH_COMMA || prevBranch == BRANCH_LEFT_PARENTHESIS) {
268                             throw missingArgs(lastPos);
269                         }
270                         thisBranch = BRANCH_COMMA;
271 
272                         if (braceCount == 0) {
273                             // comma outside of braces
274                             lexer.unparseLast();
275                             break OUT;
276                         }
277 
278                         if (castBraceCount > 0 && castBraceCountStack.peek() == castBraceCount) {
279                             throw SqlException.$(lastPos, "',' is not expected here");
280                         }
281 
282                         // If the token is a function argument separator (e.g., a comma):
283                         // Until the token at the top of the stack is a left parenthesis,
284                         // pop operators off the stack onto the output queue. If no left
285                         // parentheses are encountered, either the separator was misplaced or
286                         // parentheses were mismatched.
287                         while ((node = opStack.pop()) != null && node.token.charAt(0) != '(') {
288                             argStackDepth = onNode(listener, node, argStackDepth);
289                         }
290 
291                         if (node != null) {
292                             opStack.push(node);
293                         }
294 
295                         paramCount++;
296                         break;
297 
298                     case '[':
299                         if (isTypeQualifier()) {
300                             ExpressionNode en = opStack.peek();
301                             ((GenericLexer.FloatingSequence) en.token).setHi(lastPos + 1);
302                         } else {
303                             thisBranch = BRANCH_LEFT_BRACKET;
304 
305                             // If the token is a left parenthesis, then push it onto the stack.
306                             paramCountStack.push(paramCount);
307                             paramCount = 0;
308 
309                             argStackDepthStack.push(argStackDepth);
310                             argStackDepth = 0;
311

313                             // the precedence of [ is fixed to 2

316                                 if ((2 > other.precedence)) {
317                                     argStackDepth = onNode(listener, other, argStackDepth);
318                                     opStack.pop();
319                                 } else {
320                                     break;

322                             }

325                             // consumed as parameter to a greedy function
326                             opStack.push(expressionNodePool.next().of(ExpressionNode.CONTROL, "[", Integer.MAX_VALUE, lastPos));
327                             bracketCount++;

330                             braceCount = 0;
331                         }

334

338                             ((GenericLexer.FloatingSequence) en.token).setHi(lastPos + 1);
339                         } else {

343                             }

348                             }

353                             while ((node = opStack.pop()) != null && (node.type != ExpressionNode.CONTROL || node.token.charAt(0) != '[')) {
354                                 argStackDepth = onNode(listener, node, argStackDepth);
355                             }

358                                 argStackDepth += argStackDepthStack.pop();
359                             }
360 
361                             if (paramCountStack.notEmpty()) {
362                                 paramCount = paramCountStack.pop();
363                             }
364 
365                             if (braceCountStack.notEmpty()) {
366                                 braceCount = braceCountStack.pop();
367                             }

370                                     ExpressionNode.ARRAY_ACCESS,
371                                     "[]",
372                                     2,
373                                     lastPos
374                             );

377                         }

384                             CharSequence geohashTok = GenericLexer.immutableOf(tok);

393                             if (tok != null && tok.charAt(0) != ')') {
394                                 opStack.push(expressionNodePool.next().of(
395                                         ExpressionNode.CONSTANT,
396                                         lexer.immutablePairOf(geohashTok, tok),
397                                         Integer.MIN_VALUE,
398                                         lastPos));
399                                 tok = SqlUtil.fetchNext(lexer);
400                                 if (tok == null || tok.charAt(0) != ')') {
401                                     throw SqlException.$(lexer.lastTokenPosition(), "invalid GEOHASH, missing ')'");
402                                 }
403                             } else {
404                                 throw SqlException.$(lexer.lastTokenPosition(), "invalid GEOHASH, invalid type precision");
405                             }
406                             thisBranch = BRANCH_GEOHASH;
407                         } else {
408                             processDefaultBranch = true;
409                         }
410                         break;
411 
412                     case '#':
413                         if (isGeoHashCharsConstant(tok)) { // e.g. #sp052w92p1p8
414                             thisBranch = BRANCH_CONSTANT;

416                             // optional / bits '/dd', '/d'
417                             CharSequence slash = SqlUtil.fetchNext(lexer);
418                             if (slash == null || slash.charAt(0) != '/') {

421                                         ExpressionNode.CONSTANT,
422                                         geohashTok, // standard token, no suffix '/d', '/dd'

426                             }

432                                     ExpressionNode.CONSTANT,
433                                     lexer.immutablePairOf(geohashTok, '/', tok), // token plus suffix '/d', '/dd', where d in [0..9]

440                             thisBranch = BRANCH_CONSTANT;

442                                     ExpressionNode.CONSTANT,
443                                     GenericLexer.immutableOf(tok), // geohash bit literals do not allow suffix syntax
444                                     Integer.MIN_VALUE,
445                                     lastPos));
446                             break;
447                         }

450                         break;

454                             throw SqlException.$(lastPos, "not a method call");

458                         // If the token is a left parenthesis, then push it onto the stack.
459                         paramCountStack.push(paramCount);
460                         paramCount = 0;
461

463                         argStackDepth = 0;
464

469                         // consumed as parameter to a greedy function

475                         }

477                         if (castBraceCountStack.size() > 0) {
478                             castBraceCount++;
479                         }
480 
481                         braceCount++;
482                         break;
483

486                             throw missingArgs(lastPos);
487                         }
488 
489                         if (braceCount == 0) {
490                             lexer.unparseLast();
491                             break OUT;
492                         }
493

495                         int localParamCount = (prevBranch == BRANCH_LEFT_PARENTHESIS ? 0 : paramCount + 1);
496                         final boolean thisWasCast;

501                             }

506                         } else {
507                             thisWasCast = false;
508                         }

511                             castBraceCount--;
512                         }
513

515                         // If the token is a right parenthesis:
516                         // Until the token at the top of the stack is a left parenthesis, pop operators off the stack onto the output queue.
517                         // Pop the left parenthesis from the stack, but not onto the output queue.
518                         //        If the token at the top of the stack is a function token, pop it onto the output queue.
519                         //        If the stack runs out without finding a left parenthesis, then there are mismatched parentheses.
520                         while ((node = opStack.pop()) != null && node.token.charAt(0) != '(') {
521                             // special case - (*) expression
522                             if (Chars.equals(node.token, '*') && argStackDepth == 0 && isCount()) {
523                                 argStackDepth = onNode(listener, node, 2);
524                             } else {
525                                 if (thisWasCast) {
526                                     if (prevBranch != BRANCH_GEOHASH) {
527                                         // validate type
528                                         final short columnTypeTag = ColumnType.tagOf(node.token);
529                                         if (((columnTypeTag < ColumnType.BOOLEAN || (columnTypeTag > ColumnType.LONG256 && columnTypeTag != ColumnType.UUID)) && !asPoppedNull) ||
530                                                 (columnTypeTag == ColumnType.GEOHASH && node.type == ExpressionNode.LITERAL)) {
531                                             throw SqlException.$(node.position, "unsupported cast");
532                                         }
533                                         node.type = ExpressionNode.CONSTANT;
534                                     }
535                                 }
536                                 argStackDepth = onNode(listener, node, argStackDepth);
537                             }
538                         }
539 
540                         if (argStackDepthStack.notEmpty()) {
541                             argStackDepth += argStackDepthStack.pop();
542                         }
543 
544                         // enable operation or literal absorb parameters
545                         if ((node = opStack.peek()) != null && (node.type == ExpressionNode.LITERAL || (node.type == ExpressionNode.SET_OPERATION))) {

547                                 node.paramCount = localParamCount + Math.max(0, node.paramCount - 1);
548                                 node.type = ExpressionNode.FUNCTION;

550                                 opStack.pop();
551                             }
552                         } else {

555                             if (localParamCount > 1

561                         }

565                         }
566

569                         }
570

572                     case 'c':
573                     case 'C':

581                                 break;
582                             }

589                                 opStack.push(expressionNodePool.next().of(ExpressionNode.LITERAL, "cast", Integer.MIN_VALUE, lastPos));
590                                 break;
591                             } else {
592                                 throw SqlException.$(lastPos, "'cast' is not allowed here");
593                             }

598                     case 'A':

601

603

605                                 int nodeCount = 0;
606                                 while ((node = opStack.pop()) != null && node.token.charAt(0) != '(') {
607                                     nodeCount++;

609                                     argStackDepth = onNode(listener, node, argStackDepth);
610                                 }

614                                 }
615 
616                                 if (node != null) {
617                                     opStack.push(node);
618                                 }
619

621                                 castAsCount++;

623                                 processDefaultBranch = true;
624                             }

638                             }

650                             }

661                                 } else {
662                                     throw SqlException.$(

665                                     );
666                                 }

675                                 } else {

677                                 }
678                                 processDefaultBranch = true;
679                             }

683                         }

685                     case 'b':
686                     case 'B':
687                         if (SqlKeywords.isBetweenKeyword(tok)) {
688                             thisBranch = BRANCH_BETWEEN_START;
689                             if (betweenCount > betweenAndCount) {
690                                 // Nested between are not supported
691                                 throw SqlException.$(lastPos, "between statements cannot be nested");
692                             }
693                             betweenCount++;
694                             betweenStartCaseCount = caseCount;
695                         }
696                         processDefaultBranch = true;
697                         break;
698                     case 's':
699                     case 'S':
700                         if (SqlKeywords.isSelectKeyword(tok)) {
701                             thisBranch = BRANCH_LAMBDA;
702                             if (betweenCount > 0) {
703                                 throw SqlException.$(lastPos, "constant expected");
704                             }
705                             argStackDepth = processLambdaQuery(lexer, listener, argStackDepth);
706                         } else {
707                             processDefaultBranch = true;
708                         }
709                         break;
710                     case '0':
711                     case '1':
712                     case '2':
713                     case '3':
714                     case '4':
715                     case '5':
716                     case '6':
717                     case '7':
718                     case '8':
719                     case '9':
720                     case '\'':
721                     case 'E':
722 
723                         if (prevBranch != BRANCH_DOT_DEREFERENCE) {
724                             // check if this is E'str'
725                             if (thisChar == 'E' && (tok.length() < 3 || tok.charAt(1) != '\'')) {
726                                 processDefaultBranch = true;
727                                 break;
728                             }
729 
730                             thisBranch = BRANCH_CONSTANT;

738                                     } else {
739                                         assert false;
740                                     }

742                             }

744                                 final ExpressionNode en = opStack.peek();

746                                     // check if this is '1.2' or '1. 2'
747                                     if (lastPos > 0 && lexer.getContent().charAt(lastPos - 1) == '.') {
748                                         if (en.token instanceof GenericLexer.FloatingSequence) {
749                                             ((GenericLexer.FloatingSequence) en.token).setHi(lexer.getTokenHi());

754                                             opStack.push(expressionNodePool.next().of(ExpressionNode.LITERAL, cse.toImmutable(), Integer.MIN_VALUE, en.position));
755                                         }
756                                         break;
757                                     }
758                                 } else {

760                                             expressionNodePool.next().of(
761                                                     ExpressionNode.CONSTANT,
762                                                     lexer.immutableBetween(lastPos - 1, lexer.getTokenHi()),
763                                                     0,
764                                                     lastPos
765                                             )
766                                     );
767                                     break;
768                                 }
769                             }
770                             if (prevBranch != BRANCH_DOT && nonLiteralBranches.excludes(prevBranch)) {
771                                 if (SqlKeywords.isQuote(tok)) {
772                                     throw SqlException.$(lastPos, "unclosed quoted string?");
773                                 }
774 
775                                 ExpressionNode constNode = expressionNodePool.next().of(ExpressionNode.CONSTANT,
776                                         GenericLexer.immutableOf(tok),
777                                         0,
778                                         lastPos
779                                 );
780 
781                                 if (prevBranch == BRANCH_TIMESTAMP_ZONE) {
782                                     argStackDepth = onNode(
783                                             listener,
784                                             constNode,
785                                             argStackDepth
786                                     );
787 
788                                     // replace const node with 'to_timezone' function node
789                                     constNode = expressionNodePool.next().of(ExpressionNode.FUNCTION,
790                                             "to_timezone",
791                                             Integer.MIN_VALUE,
792                                             lastPos
793                                     );
794                                     constNode.paramCount = 2;
795                                     // fall thru
796                                 }
797                                 opStack.push(constNode);
798                                 break;
799                             } else {
800                                 if (opStack.size() > 0 && prevBranch == BRANCH_LITERAL && thisChar == '\'') {
801                                     ExpressionNode prevNode = opStack.pop();
802                                     // This is postgres syntax to cast string literal to a type
803                                     // timestamp '2005-04-02 12:00:00-07'
804                                     // long '12321312'
805                                     // timestamp with time zone '2005-04-02 12:00:00-07'
806

808                                     final short columnType = ColumnType.tagOf(prevNode.token);
809                                     if (columnType < ColumnType.BOOLEAN || (columnType > ColumnType.LONG256 && columnType != ColumnType.UUID)) {
810                                         throw SqlException.$(prevNode.position, "invalid type");
811                                     } else {
812                                         ExpressionNode stringLiteral = expressionNodePool.next().of(ExpressionNode.CONSTANT, GenericLexer.immutableOf(tok), 0, lastPos);
813                                         onNode(listener, stringLiteral, 0);
814 
815                                         prevNode.type = ExpressionNode.CONSTANT;
816                                         onNode(listener, prevNode, 0);
817 
818                                         ExpressionNode cast = expressionNodePool.next().of(ExpressionNode.FUNCTION, "cast", 0, prevNode.position);
819                                         cast.paramCount = 2;
820 
821                                         onNode(listener, cast, argStackDepth + 2);
822                                         argStackDepth++;
823                                         break;
824                                     }
825                                 }
826 
827                                 if (opStack.size() > 1) {
828                                     throw SqlException.$(lastPos, "dangling expression");
829                                 }
830                                 lexer.unparseLast();
831                                 break OUT;
832                             }
833                         } else {
834                             throw SqlException.$(lastPos, "constant is not allowed here");
835                         }
836                     case 'N':
837                     case 'n':
838                         if (SqlKeywords.isNotKeyword(tok)) {
839                             ExpressionNode nn = opStack.peek();
840                             if (nn != null && nn.type == ExpressionNode.LITERAL) {
841                                 opStack.pop();
842 
843                                 node = expressionNodePool.next().of(
844                                         ExpressionNode.OPERATION,
845                                         GenericLexer.immutableOf(tok),
846                                         11,
847                                         lastPos
848                                 );
849                                 node.paramCount = 1;
850                                 opStack.push(node);
851                                 opStack.push(nn);
852                                 break;
853                             }
854                         }
855                     case 't':
856                     case 'T':
857                     case 'f':
858                     case 'F':
859                         if (SqlKeywords.isNanKeyword(tok)
860                                 || SqlKeywords.isNullKeyword(tok)
861                                 || SqlKeywords.isTrueKeyword(tok)
862                                 || SqlKeywords.isFalseKeyword(tok)
863                         ) {
864                             if (prevBranch != BRANCH_DOT_DEREFERENCE) {
865                                 thisBranch = BRANCH_CONSTANT;
866                                 // If the token is a number, then add it to the output queue.
867                                 opStack.push(expressionNodePool.next().of(ExpressionNode.CONSTANT, GenericLexer.immutableOf(tok), 0, lastPos));
868                             } else {
869                                 throw SqlException.$(lastPos, "constant is not allowed here");
870                             }
871                             break;
872                         }
873                         processDefaultBranch = true;
874                         break;
875                     case 'i':
876                     case 'I':
877                         if (SqlKeywords.isIsKeyword(tok)) {
878                             // replace:
879                             // <literal or constant> IS NULL     -> <literal or constant> = NULL
880                             // <literal or constant> IS NOT NULL -> <literal or constant> != NULL
881                             if (prevBranch == BRANCH_LITERAL || prevBranch == BRANCH_CONSTANT || prevBranch == BRANCH_RIGHT_PARENTHESIS) {
882                                 final CharSequence isTok = GenericLexer.immutableOf(tok);
883                                 tok = SqlUtil.fetchNext(lexer);
884                                 if (tok == null) {
885                                     throw SqlException.$(lastPos, "IS must be followed by [NOT] NULL");
886                                 }
887                                 if (SqlKeywords.isNotKeyword(tok)) {
888                                     final int notTokPosition = lexer.lastTokenPosition();
889                                     final CharSequence notTok = GenericLexer.immutableOf(tok);
890                                     tok = SqlUtil.fetchNext(lexer);
891                                     if (tok != null && SqlKeywords.isNullKeyword(tok)) {
892                                         lexer.backTo(notTokPosition + 3, notTok);
893                                         tok = "!=";
894                                     } else {
895                                         throw SqlException.$(lastPos, "IS NOT must be followed by NULL");
896                                     }
897                                 } else if (SqlKeywords.isNullKeyword(tok)) {
898                                     lexer.backTo(lastPos + 2, isTok);
899                                     tok = "=";
900                                 } else {
901                                     throw SqlException.$(lastPos, "IS must be followed by NULL");
902                                 }
903                             } else {
904                                 throw SqlException.$(lastPos, "IS [NOT] not allowed here");
905                             }
906                         }
907                         processDefaultBranch = true;
908                         break;
909                     case '*':
910                         // special case for tab.*
911                         if (prevBranch == BRANCH_DOT) {
912                             thisBranch = BRANCH_LITERAL;
913                             final ExpressionNode en = opStack.peek();
914                             if (en != null && en.type != ExpressionNode.CONTROL) {
915                                 // leverage the fact '*' is dedicated token and it returned from cache
916                                 // therefore lexer.tokenHi does not move when * follows dot without whitespace
917                                 // e.g. 'a.*'
918                                 if (GenericLexer.WHITESPACE_CH.excludes(lexer.getContent().charAt(lastPos - 1))) {
919                                     if (en.token instanceof GenericLexer.FloatingSequence) {
920                                         GenericLexer.FloatingSequence fs = (GenericLexer.FloatingSequence) en.token;
921                                         fs.setHi(lastPos + 1);
922                                     } else {
923                                         // "foo".* or 'foo'.*
924                                         // foo was unquoted, and we cannot simply move hi to include the *
925                                         opStack.pop();
926                                         CharacterStoreEntry cse = characterStore.newEntry();
927                                         cse.put(en.token).put('*');
928                                         opStack.push(expressionNodePool.next().of(ExpressionNode.LITERAL, cse.toImmutable(), Integer.MIN_VALUE, en.position));
929                                     }
930                                 } else {
931                                     // in this case we have whitespace, e.g. 'a. *'
932                                     processDefaultBranch = true;
933                                 }
934                             } else {
935                                 opStack.push(
936                                         expressionNodePool.next().of(
937                                                 ExpressionNode.CONSTANT,
938                                                 lexer.immutableBetween(lastPos - 1, lexer.getTokenHi()),
939                                                 0,
940                                                 lastPos
941                                         )
942                                 );
943                             }
944                             break;
945                         }
946                     default:
947                         processDefaultBranch = true;
948                         break;
949                 }
950 
951                 if (processDefaultBranch) {
952                     OperatorExpression op;
953                     if ((op = OperatorExpression.opMap.get(tok)) != null) {
954 
955                         thisBranch = BRANCH_OPERATOR;
956 
957                         // If the token is an operator, o1, then:
958                         // while there is an operator token, o2, at the top of the operator stack, and either
959                         // o1 is left-associative and its precedence is less than or equal to that of o2, or
960                         // o1 is right associative, and has precedence less than that of o2,
961                         //        then pop o2 off the operator stack, onto the output queue;
962                         // push o1 onto the operator stack.
963 
964                         int operatorType = op.type;
965 
966                         if (thisChar == '-' || thisChar == '~') {
967                             switch (prevBranch) {
968                                 case BRANCH_OPERATOR:
969                                 case BRANCH_LEFT_PARENTHESIS:
970                                 case BRANCH_COMMA:
971                                 case BRANCH_NONE:
972                                 case BRANCH_CASE_CONTROL:
973                                     // we have unary minus
974                                     operatorType = OperatorExpression.UNARY;
975                                     break;
976                                 default:
977                                     break;
978                             }
979                         }
980 
981                         ExpressionNode other;
982                         // UNARY operators must never pop BINARY ones regardless of precedence
983                         // this is to maintain correctness of -a^b
984                         while ((other = opStack.peek()) != null) {
985                             boolean greaterPrecedence = (op.leftAssociative && op.precedence >= other.precedence) || (!op.leftAssociative && op.precedence > other.precedence);
986                             if (greaterPrecedence &&
987                                     (operatorType != OperatorExpression.UNARY || (operatorType == OperatorExpression.UNARY && other.paramCount == 1))) {
988                                 argStackDepth = onNode(listener, other, argStackDepth);
989                                 opStack.pop();
990                             } else {
991                                 break;
992                             }
993                         }
994                         node = expressionNodePool.next().of(
995                                 op.type == OperatorExpression.SET ? ExpressionNode.SET_OPERATION : ExpressionNode.OPERATION,
996                                 op.token,
997                                 op.precedence,
998                                 lastPos
999                         );
1000                         if (operatorType == OperatorExpression.UNARY) {
1001                             node.paramCount = 1;
1002                         } else if (SqlKeywords.isBetweenKeyword(node.token)) {
1003                             node.paramCount = 3;
1004                         } else {
1005                             node.paramCount = 2;
1006                         }
1007                         opStack.push(node);
1008                     } else if (caseCount > 0 || nonLiteralBranches.excludes(thisBranch)) {
1009                         // here we handle literals, in case of "case" statement some of these literals
1010                         // are going to flush operation stack
1011                         if (Chars.toLowerCaseAscii(thisChar) == 'c' && SqlKeywords.isCaseKeyword(tok)) {
1012                             if (prevBranch != BRANCH_DOT_DEREFERENCE) {
1013                                 caseCount++;
1014                                 paramCountStack.push(paramCount);
1015                                 paramCount = 0;
1016 
1017                                 caseBraceCountStack.push(braceCount);
1018                                 braceCount = 0;
1019 
1020                                 argStackDepthStack.push(argStackDepth);
1021                                 argStackDepth = 0;
1022                                 opStack.push(expressionNodePool.next().of(ExpressionNode.FUNCTION, "case", Integer.MAX_VALUE, lastPos));
1023                                 thisBranch = BRANCH_CASE_START;
1024                                 continue;
1025                             } else {
1026                                 throw SqlException.$(lastPos, "'case' is not allowed here");
1027                             }
1028                         }
1029 
1030                         thisBranch = BRANCH_LITERAL;
1031 
1032                         if (caseCount > 0) {
1033                             switch (Chars.toLowerCaseAscii(thisChar)) {
1034                                 case 'e':
1035                                     if (SqlKeywords.isEndKeyword(tok)) {
1036                                         if (prevBranch == BRANCH_CASE_CONTROL) {
1037                                             throw missingArgs(lastPos);
1038                                         }
1039 
1040                                         if (paramCount == 0) {
1041                                             throw SqlException.$(lastPos, "'when' expected");
1042                                         }
1043 
1044                                         // If the token is a right parenthesis:
1045                                         // Until the token at the top of the stack is a left parenthesis, pop operators off the stack onto the output queue.
1046                                         // Pop the left parenthesis from the stack, but not onto the output queue.
1047                                         //        If the token at the top of the stack is a function token, pop it onto the output queue.
1048                                         //        If the stack runs out without finding a left parenthesis, then there are mismatched parentheses.
1049                                         while ((node = opStack.pop()) != null && !SqlKeywords.isCaseKeyword(node.token)) {
1050                                             argStackDepth = onNode(listener, node, argStackDepth);
1051                                         }
1052 
1053                                         // 'when/else' have been clearing argStackDepth to ensure
1054                                         // expressions between 'when' and 'when' do not pick up arguments outside of scope
1055                                         // now we need to restore stack depth before 'case' entry
1056                                         if (argStackDepthStack.notEmpty()) {
1057                                             argStackDepth += argStackDepthStack.pop();
1058                                         }
1059 
1060                                         if (caseBraceCountStack.notEmpty()) {
1061                                             braceCount = caseBraceCountStack.pop();
1062                                         }
1063 
1064                                         node.paramCount = paramCount;
1065                                         // we also add number of 'case' arguments to original stack depth
1066                                         argStackDepth = onNode(listener, node, argStackDepth + paramCount);
1067 
1068                                         // make sure we restore paramCount
1069                                         if (paramCountStack.notEmpty()) {
1070                                             paramCount = paramCountStack.pop();
1071                                         }
1072 
1073                                         caseCount--;
1074                                         continue;
1075                                     }
1076                                     // fall through
1077                                 case 'w':
1078                                 case 't':
1079                                     int keywordIndex = caseKeywords.get(tok);
1080                                     if (keywordIndex > -1) {
1081 
1082                                         if (prevBranch == BRANCH_CASE_CONTROL) {
1083                                             throw missingArgs(lastPos);
1084                                         }
1085 
1086                                         // we need to track argument consumption so that operators and functions
1087                                         // do no steal parameters outside of local 'case' scope
1088                                         int argCount = 0;
1089                                         while ((node = opStack.pop()) != null && !SqlKeywords.isCaseKeyword(node.token)) {
1090                                             argStackDepth = onNode(listener, node, argStackDepth);
1091                                             argCount++;
1092                                         }
1093 
1094                                         if (paramCount == 0) {
1095                                             if (argCount == 0) {
1096                                                 // this is 'case when', we will
1097                                                 // indicate that this is regular 'case' to the rewrite logic
1098                                                 onNode(listener, expressionNodePool.next().of(ExpressionNode.LITERAL, null, Integer.MIN_VALUE, -1), argStackDepth);
1099                                             }
1100                                             paramCount++;
1101                                         }
1102 
1103                                         switch (keywordIndex) {
1104                                             case 0: // when
1105                                             case 2: // else
1106                                                 if ((paramCount % 2) == 0) {
1107                                                     throw SqlException.$(lastPos, "'then' expected");
1108                                                 }
1109                                                 break;
1110                                             default: // then
1111                                                 if ((paramCount % 2) != 0) {
1112                                                     throw SqlException.$(lastPos, "'when' expected");
1113                                                 }
1114                                                 break;
1115                                         }
1116 
1117                                         if (node != null) {
1118                                             opStack.push(node);
1119                                         }
1120 
1121                                         argStackDepth = 0;
