73      */
74     public static CharSequence fetchNext(GenericLexer lexer) {
75         int blockCount = 0;
76         boolean lineComment = false;
77         while (lexer.hasNext()) {
78             CharSequence cs = lexer.next();
79 
80             if (lineComment) {
81                 if (Chars.equals(cs, '\n') || Chars.equals(cs, '\r')) {
82                     lineComment = false;
83                 }
84                 continue;
85             }
86 
87             if (Chars.equals("--", cs)) {
88                 lineComment = true;
89                 continue;
90             }
91 
92             if (Chars.equals("/*", cs)) {
93                 blockCount++;
94                 continue;
95             }
96 
97             if (Chars.equals("*/", cs) && blockCount > 0) {
98                 blockCount--;
99                 continue;
100             }
101 
102             if (blockCount == 0 && GenericLexer.WHITESPACE.excludes(cs)) {
103                 return cs;
104             }
105         }
106         return null;
107     }

531     static CharSequence createColumnAlias(
532             CharacterStore store,
533             CharSequence base,
534             int indexOfDot,
535             LowerCaseCharSequenceObjHashMap<QueryColumn> aliasToColumnMap,
536             boolean cleanColumnNames
537     ) {
538         final boolean disallowed = cleanColumnNames && disallowedAliases.contains(base);
539 
540         // short and sweet version
541         if (indexOfDot == -1 && !disallowed && aliasToColumnMap.excludes(base)) {
542             return base;
543         }
544 
545         final CharacterStoreEntry characterStoreEntry = store.newEntry();
546 
547         if (indexOfDot == -1) {
548             if (disallowed) {
549                 characterStoreEntry.put("column");
550             } else {
551                 characterStoreEntry.put(base);
552             }
553         } else {
554             if (indexOfDot + 1 == base.length()) {
555                 characterStoreEntry.put("column");
556             } else {
557                 characterStoreEntry.put(base, indexOfDot + 1, base.length());
558             }
559         }
560 
561         int len = characterStoreEntry.length();
562         int sequence = 0;
563         while (true) {
564             if (sequence > 0) {
565                 characterStoreEntry.trimTo(len);
566                 characterStoreEntry.put(sequence);
567             }
568             sequence++;
569             CharSequence alias = characterStoreEntry.toImmutable();
570             if (aliasToColumnMap.excludes(alias)) {
571                 return alias;
572             }
573         }

576     static long expectMicros(CharSequence tok, int position) throws SqlException {
577         int k = -1;
578 
579         final int len = tok.length();
580 
581         // look for end of digits
582         for (int i = 0; i < len; i++) {
583             char c = tok.charAt(i);
584             if (c < '0' || c > '9') {
585                 k = i;
586                 break;
587             }
588         }
589 
590         if (k == -1) {
591             throw SqlException.$(position + len, "expected interval qualifier in ").put(tok);
592         }
593 
594         try {
595             long interval = Numbers.parseLong(tok, 0, k);
596             int nChars = len - k;
597             if (nChars > 2) {
598                 throw SqlException.$(position + k, "expected 1/2 letter interval qualifier in ").put(tok);
599             }
600 
601             switch (tok.charAt(k)) {
602                 case 's':
603                     if (nChars == 1) {
604                         // seconds
605                         return interval * Timestamps.SECOND_MICROS;
606                     }
607                     break;
608                 case 'm':
609                     if (nChars == 1) {
610                         // minutes
611                         return interval * Timestamps.MINUTE_MICROS;
612                     } else {
613                         if (tok.charAt(k + 1) == 's') {
614                             // millis
615                             return interval * Timestamps.MILLI_MICROS;
616                         }
617                     }
618                     break;
619                 case 'h':
620                     if (nChars == 1) {
621                         // hours
622                         return interval * Timestamps.HOUR_MICROS;
623                     }
624                     break;
625                 case 'd':
626                     if (nChars == 1) {
627                         // days
628                         return interval * Timestamps.DAY_MICROS;
629                     }
630                     break;
631                 case 'u':
632                     if (nChars == 2 && tok.charAt(k + 1) == 's') {
633                         return interval;
634                     }
635                     break;
636                 default:
637                     break;
638             }
639         } catch (NumericException ex) {
640             // Ignored
641         }
642 
643         throw SqlException.$(position + len, "invalid interval qualifier ").put(tok);
644     }
