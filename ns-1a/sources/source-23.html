71     public long executeUpdate(SqlExecutionContext sqlExecutionContext, UpdateOperation op) throws TableReferenceOutOfDateException {
72 
73         TableToken tableToken = tableWriter.getTableToken();
74         LOG.info().$("updating [table=").$(tableToken).$(" instance=").$(op.getCorrelationId()).I$();
75 
76         try {
77             final int tableId = op.getTableId();
78             final long tableVersion = op.getTableVersion();
79             final RecordCursorFactory factory = op.getFactory();
80 
81             cleanupColumnVersions.clear();
82 
83             if (tableWriter.inTransaction()) {
84                 LOG.info().$("committing current transaction before UPDATE execution [table=").$(tableToken).$(" instance=").$(op.getCorrelationId()).I$();
85                 tableWriter.commit();
86             }
87 
88             final TableRecordMetadata tableMetadata = tableWriter.getMetadata();
89 
90             // Check that table structure hasn't changed between planning and executing the UPDATE
91             if (tableMetadata.getTableId() != tableId || tableWriter.getStructureVersion() != tableVersion) {
92                 throw TableReferenceOutOfDateException.of(tableToken, tableId, tableMetadata.getTableId(),
93                         tableVersion, tableWriter.getStructureVersion());
94             }
95 
96             // Select the rows to be updated
97             final RecordMetadata updateMetadata = factory.getMetadata();
98             final int affectedColumnCount = updateMetadata.getColumnCount();
99 
100             // Build index column map from table to update to values returned from the update statement row cursors
101             updateColumnIndexes.clear();
102             for (int i = 0; i < affectedColumnCount; i++) {
103                 CharSequence columnName = updateMetadata.getColumnName(i);
104                 int tableColumnIndex = tableMetadata.getColumnIndex(columnName);
105                 assert tableColumnIndex >= 0;
106                 updateColumnIndexes.add(tableColumnIndex);
107             }
108 
109             // Create update memory list of all columns to be updated
110             configureColumns(tableMetadata, affectedColumnCount);
111 
112             // Start execution frame by frame
113             // Partition to update
114             int partitionIndex = -1;
115             long rowsUpdated = 0;
116 
117             // Update may be queued and requester already disconnected, force check someone still waits for it
118             op.forceTestTimeout();
119             // Row by row updates for now
120             // This should happen parallel per file (partition and column)
121             try (RecordCursor recordCursor = factory.getCursor(sqlExecutionContext)) {
122                 Record masterRecord = recordCursor.getRecord();
123 
124                 long prevRow = 0;
125                 long minRow = -1L;
126                 long lastRowId = Long.MIN_VALUE;
127                 while (recordCursor.hasNext()) {
128                     long rowId = masterRecord.getUpdateRowId();
129 
130                     // Some joins expand results set and returns same row multiple times
131                     if (rowId == lastRowId) {
132                         continue;
133                     }
134                     if (rowId < lastRowId) {
135                         // We're assuming, but not enforcing the fact that
136                         // factory produces rows in incrementing order.
137                         throw CairoException.critical(0).put("Update statement generated invalid query plan. Rows are not returned in order.");
138                     }
139                     lastRowId = rowId;
140 
141                     final int rowPartitionIndex = Rows.toPartitionIndex(rowId);
142                     final long currentRow = Rows.toLocalRowID(rowId);
143 
144                     if (rowPartitionIndex != partitionIndex) {
145                         if (tableWriter.isPartitionReadOnly(rowPartitionIndex)) {
146                             throw CairoException.critical(0)
147                                     .put("cannot update read-only partition [table=").put(tableToken.getTableName())
148                                     .put(", partitionTimestamp=").ts(tableWriter.getPartitionTimestamp(rowPartitionIndex))
149                                     .put(']');
150                         }
151                         if (partitionIndex > -1) {
152                             LOG.info()
153                                     .$("updating partition [partitionIndex=").$(partitionIndex)
154                                     .$(", rowPartitionIndex=").$(rowPartitionIndex)
155                                     .$(", rowPartitionTs=").$ts(tableWriter.getPartitionTimestamp(rowPartitionIndex))
156                                     .$(", affectedColumnCount=").$(affectedColumnCount)
157                                     .$(", prevRow=").$(prevRow)
158                                     .$(", minRow=").$(minRow)
159                                     .I$();
160 
161                             copyColumns(
162                                     partitionIndex,
163                                     affectedColumnCount,
164                                     prevRow,
165                                     minRow
166                             );
167 
168                             updateEffectiveColumnTops(
169                                     tableWriter,
170                                     partitionIndex,
171                                     updateColumnIndexes,
172                                     affectedColumnCount,
173                                     minRow
174                             );
175 
176                             rebuildIndexes(tableWriter.getPartitionTimestamp(partitionIndex), tableMetadata, tableWriter);
177                         }
178 
179                         openColumns(srcColumns, rowPartitionIndex, false);
180                         openColumns(dstColumns, rowPartitionIndex, true);
181 
182                         partitionIndex = rowPartitionIndex;
183                         prevRow = 0;
184                         minRow = currentRow;
185                     }
186 
187                     appendRowUpdate(
188                             rowPartitionIndex,
189                             affectedColumnCount,
190                             prevRow,
191                             currentRow,
192                             masterRecord,
193                             minRow
194                     );
195 
196                     prevRow = currentRow + 1;
197                     rowsUpdated++;
198 
199                     op.testTimeout();
200                 }
201 
202                 if (partitionIndex > -1) {
203                     copyColumns(partitionIndex, affectedColumnCount, prevRow, minRow);
204 
205                     updateEffectiveColumnTops(
206                             tableWriter,
207                             partitionIndex,
208                             updateColumnIndexes,
209                             affectedColumnCount,
210                             minRow
211                     );
212 
213                     rebuildIndexes(tableWriter.getPartitionTimestamp(partitionIndex), tableMetadata, tableWriter);
214                 }
215             } finally {
216                 Misc.freeObjList(srcColumns);
217                 Misc.freeObjList(dstColumns);
218                 // todo: we are opening columns incrementally, e.g. if we don't have enough objects
219                 //   but here we're always clearing columns, making incremental "open" pointless
220                 //   perhaps we should keep N column object max to kick around ?
221                 srcColumns.clear();
222                 dstColumns.clear();
223             }
224 
225             if (partitionIndex > -1) {
226                 op.forceTestTimeout();
227                 tableWriter.commit();
228                 tableWriter.openLastPartition();
229                 purgeOldColumnVersions();
230             }
231 
232             LOG.info().$("update finished [table=").$(tableToken)
233                     .$(", instance=").$(op.getCorrelationId())
234                     .$(", updated=").$(rowsUpdated)
235                     .$(", txn=").$(tableWriter.getTxn())
236                     .I$();
237 
238             return rowsUpdated;
239         } catch (TableReferenceOutOfDateException e) {
240             throw e;
241         } catch (SqlException e) {
242             throw CairoException.critical(0).put("could not apply update on SPI side [e=").put((CharSequence) e).put(']');
243         } catch (Throwable th) {
244             LOG.error().$("could not update").$(th).$();
245             throw th;
246         } finally {
247             op.closeWriter();
248         }
249     }

258     private static void fillUpdatesGapWithNull(
259             int columnType,
260             long fromRow, // inclusive
261             long toRow, // exclusive
262             MemoryCMARW dstFixMem,
263             MemoryCMARW dstVarMem,
264             int shl
265     ) {
266         final short columnTag = ColumnType.tagOf(columnType);
267         switch (columnTag) {
268             case ColumnType.STRING:
269                 for (long row = fromRow; row < toRow; row++) {
270                     dstFixMem.putLong(dstVarMem.putNullStr());
271                 }
272                 break;
273             case ColumnType.BINARY:

275                     dstFixMem.putLong(dstVarMem.putNullBin());
276                 }
277                 break;
278             default:
279                 final long rowCount = toRow - fromRow;
280                 TableUtils.setNull(
281                         columnType,
282                         dstFixMem.appendAddressFor(rowCount << shl),
283                         rowCount
284                 );
285                 break;
286         }

443     private void copyColumn(
444             long prevRow,
445             long maxRow,
446             MemoryCMR srcFixMem,
447             MemoryCMR srcVarMem,
448             MemoryCMARW dstFixMem,
449             MemoryCMARW dstVarMem,
450             long newColumnTop,
451             long oldColumnTop,
452             int columnType
453     ) {
454         assert newColumnTop <= oldColumnTop || oldColumnTop < 0;
455 
456         final int shl = getFixedColumnSize(columnType);
457 
458         if (oldColumnTop == -1 && prevRow > 0) {
459             // Column did not exist at the partition
460             fillUpdatesGapWithNull(columnType, prevRow, maxRow, dstFixMem, dstVarMem, shl);
461         }
462 
463         if (oldColumnTop == 0) {
464             // Column fully exists in the partition
465             copyValues(
466                     prevRow,
467                     maxRow,
468                     srcFixMem,
469                     srcVarMem,
470                     dstFixMem,
471                     dstVarMem,
472                     columnType,
473                     shl
474             );
475         }
476 
477         if (oldColumnTop > 0) {
478             if (prevRow >= oldColumnTop) {
479                 copyValues(
480                         prevRow - oldColumnTop,
481                         maxRow - oldColumnTop,
482                         srcFixMem,
483                         srcVarMem, dstFixMem,
484                         dstVarMem,
485                         columnType,
486                         shl
487                 );
488             } else {
489                 // prevRow < oldColumnTop
490                 if (maxRow <= oldColumnTop) {
491                     if (prevRow > 0) {
492                         fillUpdatesGapWithNull(
493                                 columnType,
494                                 prevRow,
495                                 maxRow,
496                                 dstFixMem,
497                                 dstVarMem,
498                                 shl
499                         );
500                     }
501                 } else {
502                     // prevRow < oldColumnTop &&  oldColumnTop < maxRow
503                     if (prevRow > newColumnTop) {
504                         fillUpdatesGapWithNull(
505                                 columnType,
506                                 prevRow,
507                                 oldColumnTop,
508                                 dstFixMem,
509                                 dstVarMem,
510                                 shl
511                         );
512                     }
513                     copyValues(
514                             0,
515                             maxRow - oldColumnTop,
516                             srcFixMem,
517                             srcVarMem,
518                             dstFixMem,
519                             dstVarMem,
520                             columnType,
521                             shl
522                     );
523                 }
524             }
525         }
526     }

559     private void copyValues(
560             long fromRowId,
561             long toRowId,
562             MemoryCMR srcFixMem,
563             MemoryCMR srcVarMem,
564             MemoryCMARW dstFixMem,
565             MemoryCMARW dstVarMem,
566             int columnType,
567             int shl
568     ) {
569         long address = srcFixMem.addressOf(fromRowId << shl);
570         switch (ColumnType.tagOf(columnType)) {
571             case ColumnType.STRING:
572             case ColumnType.BINARY:
573                 long varStartOffset = srcFixMem.getLong(fromRowId * Long.BYTES);
574                 long varEndOffset = srcFixMem.getLong((toRowId) * Long.BYTES);
575                 long varAddress = srcVarMem.addressOf(varStartOffset);
576                 long copyToOffset = dstVarMem.getAppendOffset();
577                 dstVarMem.putBlockOfBytes(varAddress, varEndOffset - varStartOffset);
578                 dstFixMem.extend((toRowId + 1) << shl);
579                 Vect.shiftCopyFixedSizeColumnData(
580                         varStartOffset - copyToOffset,
581                         address + Long.BYTES,
582                         0,
583                         toRowId - fromRowId - 1,
584                         dstFixMem.getAppendAddress()
585                 );
586                 dstFixMem.jumpTo((toRowId + 1) << shl);
587                 break;
588             default:
589                 dstFixMem.putBlockOfBytes(address, (toRowId - fromRowId) << shl);
590                 break;
591         }
592     }

594     private void openColumns(ObjList<? extends MemoryCM> columns, int partitionIndex, boolean forWrite) {
595         long partitionTimestamp = tableWriter.getPartitionTimestamp(partitionIndex);
596         long partitionNameTxn = tableWriter.getPartitionNameTxn(partitionIndex);
597         RecordMetadata metadata = tableWriter.getMetadata();
598         try {
599             path.trimTo(rootLen);
600             TableUtils.setPathForPartition(path, tableWriter.getPartitionBy(), partitionTimestamp, false);
601             TableUtils.txnPartitionConditionally(path, partitionNameTxn);
602             int pathTrimToLen = path.length();
603             for (int i = 0, n = updateColumnIndexes.size(); i < n; i++) {
604                 int columnIndex = updateColumnIndexes.get(i);
605                 CharSequence name = metadata.getColumnName(columnIndex);
606                 int columnType = metadata.getColumnType(columnIndex);
607 
608                 final long columnTop = tableWriter.getColumnTop(partitionTimestamp, columnIndex, -1L);
609 
610                 if (forWrite) {
611                     long existingVersion = tableWriter.getColumnNameTxn(partitionTimestamp, columnIndex);
612                     tableWriter.upsertColumnVersion(partitionTimestamp, columnIndex, columnTop);
613                     if (columnTop > -1) {
614                         // columnTop == -1 means column did not exist at the partition
615                         cleanupColumnVersions.add(columnIndex, existingVersion, partitionTimestamp, partitionNameTxn);
616                     }
617                 }
618 
619                 long columnNameTxn = tableWriter.getColumnNameTxn(partitionTimestamp, columnIndex);
620                 if (isVariableLength(columnType)) {
621                     MemoryCMR colMemIndex = (MemoryCMR) columns.get(2 * i);
622                     colMemIndex.close();
623                     assert !colMemIndex.isOpen();
624                     MemoryCMR colMemVar = (MemoryCMR) columns.get(2 * i + 1);
625                     colMemVar.close();
626                     assert !colMemVar.isOpen();
627 
628                     if (forWrite || columnTop != -1) {
629                         colMemIndex.of(
630                                 ff,
631                                 iFile(path.trimTo(pathTrimToLen), name, columnNameTxn),
632                                 dataAppendPageSize,
633                                 -1,
634                                 MemoryTag.MMAP_UPDATE,
635                                 fileOpenOpts
636                         );
637                         colMemVar.of(
638                                 ff,
639                                 dFile(path.trimTo(pathTrimToLen), name, columnNameTxn),
