34     private final IntStack indexStack = new IntStack();

39     public void backup() {
40         int size = stack.size();
41         backupDepth.push(size);
42         for (int i = 0; i < size; i++) {
43             stackBackup.push(stack.poll());
44             indexStackBackup.push(indexStack.pop());
45         }
46     }

49         if (backupDepth.size() > 0) {
50             int size = backupDepth.pop();
51             for (int i = 0; i < size; i++) {
52                 stack.push(stackBackup.poll());
53                 indexStack.push(indexStackBackup.pop());
54             }
55         }
56     }

58     public void traverse(ExpressionNode node, Visitor visitor) throws SqlException {
59         backup();
60         try {
61             // post-order iterative tree traversal
62             // see http://en.wikipedia.org/wiki/Tree_traversal
63 
64             stack.clear();
65             indexStack.clear();
66 
67             ExpressionNode lastVisited = null;
68 
69             while (!stack.isEmpty() || node != null) {
70                 if (node != null) {
71                     if (!visitor.descend(node)) {
72                         lastVisited = node;
73                         node = null;
74                         continue;
75                     }
76                     stack.push(node);
77                     indexStack.push(0);
78                     node = node.rhs;
79                 } else {
80                     ExpressionNode peek = stack.peek();
81                     assert peek != null;
82                     if (peek.paramCount < 3) {
83                         if (peek.lhs != null && lastVisited != peek.lhs) {
84                             node = peek.lhs;
85                         } else {
86                             visitor.visit(peek);
87                             lastVisited = stack.poll();
88                             indexStack.pop();
89                         }
90                     } else {
91                         int index = indexStack.peek();
92                         if (index < peek.paramCount) {
93                             node = peek.args.getQuick(index);
94                             indexStack.update(index + 1);
95                         } else {
96                             visitor.visit(peek);
97                             lastVisited = stack.poll();
98                             indexStack.pop();
99                         }
100                     }
101                 }
102             }
103         } finally {
104             restore();
105         }
106     }

112          * this method returns false, the algorithm does not visit the current
