492                         && joinBarriers.excludes(parent.getJoinModels().get(literalCollectorBIndexes.get(0)).getJoinType())

607     private void assignFilters(QueryModel parent) throws SqlException {
608         tablesSoFar.clear();
609         postFilterRemoved.clear();
610         postFilterTableRefs.clear();
611 
612         literalCollector.withModel(parent);
613         ObjList<ExpressionNode> filterNodes = parent.getParsedWhere();
614         // collect table indexes from each part of global filter
615         int pc = filterNodes.size();
616         for (int i = 0; i < pc; i++) {
617             IntHashSet indexes = intHashSetPool.next();
618             literalCollector.resetCounts();
619             traversalAlgo.traverse(filterNodes.getQuick(i), literalCollector.to(indexes));
620             postFilterTableRefs.add(indexes);
621         }
622 
623         IntList ordered = parent.getOrderedJoinModels();
624         // match table references to set of table in join order
625         for (int i = 0, n = ordered.size(); i < n; i++) {
626             int index = ordered.getQuick(i);
627             tablesSoFar.add(index);
628 
629             for (int k = 0; k < pc; k++) {
630                 if (postFilterRemoved.contains(k)) {
631                     continue;
632                 }
633 
634                 final ExpressionNode node = filterNodes.getQuick(k);
635 
636                 IntHashSet refs = postFilterTableRefs.getQuick(k);
637                 int rs = refs.size();
638                 if (rs == 0) {
639                     // condition has no table references
640                     // must evaluate as constant
641                     postFilterRemoved.add(k);
642                     parent.setConstWhereClause(concatFilters(parent.getConstWhereClause(), node));
643                 } else if (rs == 1 && // single table reference and this table is not joined via OUTER or ASOF
644                         joinBarriers.excludes(parent.getJoinModels().getQuick(refs.get(0)).getJoinType())) {
645                     // get single table reference out of the way right away
646                     // we don't have to wait until "our" table comes along
647                     addWhereNode(parent, refs.get(0), node);
648                     postFilterRemoved.add(k);
649                 } else {
650                     boolean qualifies = true;
651                     // check if filter references table processed so far
652                     for (int y = 0; y < rs; y++) {
653                         if (tablesSoFar.excludes(refs.get(y))) {
654                             qualifies = false;
655                             break;
656                         }
657                     }
658                     if (qualifies) {
659                         postFilterRemoved.add(k);
660                         QueryModel m = parent.getJoinModels().getQuick(index);
661                         // it is possible that filter references only top query via alias
662                         // we will need to strip these aliases before assigning filter
663                         if (index == 0) {
664                             traversalAlgo.traverse(node, literalRewritingVisitor.of(m.getAliasToColumnNameMap()));
665                         }
666                         m.setPostJoinWhereClause(concatFilters(m.getPostJoinWhereClause(), node));
667                     }
668                 }
669             }
670         }
671         assert postFilterRemoved.size() == pc;

675     // select * from t1 union all select * from t2 order by x
676     // will have "order by" clause on the last model of the union linked list.
677     // Semantically, order by must be executed after union. To get there, we will
678     // create outer model(s) for the union block and move "order by" there.
679     private QueryModel bubbleUpOrderByAndLimitFromUnion(QueryModel model) throws SqlException {
680         QueryModel m = model.getUnionModel();
681         QueryModel nested = model.getNestedModel();
682         if (nested != null) {
683             QueryModel _n = bubbleUpOrderByAndLimitFromUnion(nested);
684             if (_n != nested) {
685                 model.setNestedModel(_n);
686             }
687         }
688 
689         if (m != null) {
690             // find order by clauses
691             if (m.getNestedModel() != null) {
692                 final QueryModel m1 = bubbleUpOrderByAndLimitFromUnion(m.getNestedModel());
693                 if (m1 != m) {
694                     m.setNestedModel(m1);
695                 }
696             }
697 
698             do {
699                 if (m.getUnionModel() == null) {
700                     // last model in the linked list
701                     QueryModel un = m.getNestedModel();
702                     int n = un.getOrderBy().size();
703                     // order by clause is on the nested model
704                     final ObjList<ExpressionNode> orderBy = un.getOrderBy();
705                     final IntList orderByDirection = un.getOrderByDirection();
706                     // limit is on the parent model
707                     final ExpressionNode limitLo = m.getLimitLo();
708                     final ExpressionNode limitHi = m.getLimitHi();
709 
710                     if (n > 0 || limitHi != null || limitLo != null) {
711                         // we have some order by clauses to move
712                         QueryModel _nested = queryModelPool.next();
713                         for (int i = 0; i < n; i++) {
714                             _nested.addOrderBy(orderBy.getQuick(i), orderByDirection.getQuick(i));
715                         }
716                         orderBy.clear();
717                         orderByDirection.clear();
718 
719                         m.setLimit(null, null);
720 
721                         _nested.setNestedModel(model);
722                         QueryModel _model = queryModelPool.next();
723                         _model.setNestedModel(_nested);
724                         SqlUtil.addSelectStar(_model, queryColumnPool, expressionNodePool);
725                         _model.setLimit(limitLo, limitHi);
726                         return _model;
727                     }
728                     break;
729                 }
730 
731                 m = m.getUnionModel();
732             } while (true);
733         }
734         return model;

878     // order hash is used to determine redundant order when parsing analytic function definition
879     private void createOrderHash(QueryModel model) {
880         LowerCaseCharSequenceIntHashMap hash = model.getOrderHash();
881         hash.clear();
882 
883         final ObjList<ExpressionNode> orderBy = model.getOrderBy();
884         final int n = orderBy.size();
885         final QueryModel nestedModel = model.getNestedModel();
886 
887         if (n > 0) {
888             final IntList orderByDirection = model.getOrderByDirection();
889             for (int i = 0; i < n; i++) {
890                 // if a column appears multiple times in the ORDER BY clause then we use only the first occurrence.
891                 // why? consider this clause: ORDER BY A DESC, A ASC. In this case we want to order to be "A DESC" only.
892                 // why? "A ASC" is a lower in priority, and it's applicable if and only if the items within first clause
893                 // are equal. but if the items are already equal then there is no point in ordering by the same column again.
894                 // unconditional put() would be a bug as a lower priority ordering would replace ordering with a higher priority.
895                 hash.putIfAbsent(orderBy.getQuick(i).token, orderByDirection.getQuick(i));
896             }
897         }
898 
899         if (nestedModel != null) {
900             createOrderHash(nestedModel);
901         }
902 
903         final ObjList<QueryModel> joinModels = model.getJoinModels();
904         for (int i = 1, z = joinModels.size(); i < z; i++) {
905             createOrderHash(joinModels.getQuick(i));
906         }
907 
908         final QueryModel union = model.getUnionModel();
909         if (union != null) {
910             createOrderHash(union);

1138

1140             QueryModel q = joinModels.getQuick(i);
