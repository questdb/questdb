198     /**
199      * Creates function instance. When node type is {@link ExpressionNode#LITERAL} a column or parameter
200      * function is returned. We will be using the supplied {@link #metadata} to resolve type of column. When node token
201      * begins with ':' parameter is looked up from the supplied bindVariableService.
202      * <p>
203      * When node type is {@link ExpressionNode#CONSTANT} a constant function is returned. Type of constant is
204      * inferred from value of node token.
205      * <p>
206      * When node type is {@link ExpressionNode#QUERY} a cursor function is returned. Cursor function can be wrapping
207      * stateful instance of {@link RecordCursorFactory} that has to be closed when disposed of.
208      * Such instances are added to the supplied list of {@link java.io.Closeable} items.
209      * <p>
210      * For any other node type a function instance is created using {@link FunctionFactory}
211      *
212      * @param node             expression node
213      * @param metadata         metadata for resolving types of columns.
214      * @param executionContext for resolving parameters, which are ':' prefixed literals and creating cursors
215      * @return function instance
216      * @throws SqlException when function cannot be created. Can be one of list but not limited to
217      *                      <ul>
218      *                      <li>column not found</li>
219      *                      <li>parameter not found</li>
220      *                      <li>unknown function name</li>
221      *                      <li>function argument mismatch</li>
222      *                      <li>sql compilation errors in case of lambda</li>
223      *                      </ul>
224      */
225     public Function parseFunction(
226             ExpressionNode node,
227             RecordMetadata metadata,
228             SqlExecutionContext executionContext
229     ) throws SqlException {
230 
231         this.sqlExecutionContext = executionContext;
232 
233         if (this.metadata != null) {
234             metadataStack.push(this.metadata);
235         }
236         try {
237             this.metadata = metadata;
238             try {
239                 traverseAlgo.traverse(node, this);
240             } catch (Exception e) {
241                 // release parsed functions
242                 for (int i = functionStack.size(); i > 0; i--) {
243                     Misc.free(functionStack.poll());
244                 }
245                 positionStack.clear();
246                 throw e;
247             }
248 
249             final Function function = functionStack.poll();
250             positionStack.pop();
251             assert positionStack.size() == functionStack.size();
252             if (function != null && function.isConstant() && (function instanceof ScalarFunction)) {
253                 return functionToConstant(function);
254             }
255             return function;
256         } finally {
257             if (metadataStack.size() == 0) {
258                 this.metadata = null;
259             } else {
260                 this.metadata = metadataStack.poll();
261             }
262         }
263     }

269     @Override
270     public void visit(ExpressionNode node) throws SqlException {
271         int argCount = node.paramCount;
272         if (argCount == 0) {
273             switch (node.type) {
274                 case ExpressionNode.LITERAL:
275                     functionStack.push(createColumn(node.position, node.token));
276                     break;
277                 case ExpressionNode.BIND_VARIABLE:
278                     functionStack.push(createBindVariable0(node.position, node.token));
279                     break;
280                 case ExpressionNode.MEMBER_ACCESS:
281                     functionStack.push(new StrConstant(node.token));
282                     break;
283                 case ExpressionNode.CONSTANT:
284                     functionStack.push(createConstant(node.position, node.token));
285                     break;
286                 case ExpressionNode.QUERY:
287                     functionStack.push(createCursorFunction(node));
288                     break;
289                 default:
290                     // lookup zero arg function from symbol table
291                     functionStack.push(createFunction(node, null, null));
292                     break;
293             }
294         } else {
295             mutableArgs.clear();
296             mutableArgs.setPos(argCount);
297             mutableArgPositions.clear();
298             mutableArgPositions.setPos(argCount);
299             for (int n = 0; n < argCount; n++) {
300                 final Function arg = functionStack.poll();
301                 final int pos = positionStack.pop();
302 
303                 mutableArgs.setQuick(n, arg);
304                 mutableArgPositions.setQuick(n, pos);
305 
306                 if (arg instanceof GroupByFunction) {
307                     Misc.freeObjList(mutableArgs);
308                     throw SqlException.position(pos).put("Aggregate function cannot be passed as an argument");
309                 }
310             }
311             functionStack.push(createFunction(node, mutableArgs, mutableArgPositions));
312         }
313         positionStack.push(node.position);
314     }

423     private Function createConstant(int position, final CharSequence tok) throws SqlException {
424 
425         final int len = tok.length();
426 
427         if (isNullKeyword(tok)) {
428             return NullConstant.NULL;
429         }
430 
431         if (Chars.isQuoted(tok)) {
432             if (len == 3) {
433                 // this is 'x' - char
434                 return CharConstant.newInstance(tok.charAt(1));
435             }
436 
437             if (len == 2) {
438                 // empty
439                 return StrConstant.EMPTY;
440             }
441             return new StrConstant(tok);
442         }
443 
444         // special case E'str'
445         // we treat it like normal string for now
446         if (len > 2 && tok.charAt(0) == 'E' && tok.charAt(1) == '\'') {
447             return new StrConstant(Chars.toString(tok, 2, len - 1));
448         }
449 
450         if (SqlKeywords.isTrueKeyword(tok)) {
451             return BooleanConstant.TRUE;
452         }
453 
454         if (SqlKeywords.isFalseKeyword(tok)) {
455             return BooleanConstant.FALSE;
456         }
457 
458         try {
459             return IntConstant.newInstance(Numbers.parseInt(tok));
460         } catch (NumericException ignore) {
461         }
462 
463         try {
464             return LongConstant.newInstance(Numbers.parseLong(tok));
465         } catch (NumericException ignore) {
466         }
467 
468         try {
469             return DoubleConstant.newInstance(Numbers.parseDouble(tok));
470         } catch (NumericException ignore) {
471         }
472 
473         try {
474             return FloatConstant.newInstance(Numbers.parseFloat(tok));
475         } catch (NumericException ignore) {
476         }
477 
478         // type constant for 'CAST' operation
479 
480         final short columnType = ColumnType.tagOf(tok);
481 
482         if (
483                 (columnType >= ColumnType.BOOLEAN && columnType <= ColumnType.BINARY)
484                         || columnType == ColumnType.REGCLASS
485                         || columnType == ColumnType.REGPROCEDURE
486                         || columnType == ColumnType.ARRAY_STRING
487                         || columnType == ColumnType.UUID
488         ) {
489             return Constants.getTypeConstant(columnType);
490         }
491 
492         // geohash type constant
493 
494         if (startsWithGeoHashKeyword(tok)) {
495             return GeoHashTypeConstant.getInstanceByPrecision(
496                     GeoHashUtil.parseGeoHashBits(position, 7, tok));
497         }
498 
499         if (len > 1 && tok.charAt(0) == '#') {
500             ConstantFunction geoConstant = GeoHashUtil.parseGeoHashConstant(position, tok, len);
501             if (geoConstant != null) {
502                 return geoConstant;
503             }
504         }
505 
506         // long256
507         if (Numbers.extractLong256(tok, len, long256Sink)) {
508             return new Long256Constant(long256Sink); // values are copied from this sink
509         }
510 
511         throw SqlException.position(position).put("invalid constant: ").put(tok);
512     }

515         assert node.queryModel != null;
516         // Disable async offload for in (select ...) sub-queries to avoid infinite loops
517         // due to nested reduce calls. See SqlCodeGenerator#testBug484() for the reproducer.
518         boolean currentFilterEnabled = sqlExecutionContext.isParallelFilterEnabled();

530     private Function createFunction(
531             ExpressionNode node,
532             @Transient ObjList<Function> args,
533             @Transient IntList argPositions
534     ) throws SqlException {
535         final ObjList<FunctionFactoryDescriptor> overload = functionFactoryCache.getOverloadList(node.token);
536         if (overload == null) {
537             throw invalidFunction(node, args);
538         }
539 
540         final int argCount = args == null ? 0 : args.size();
541         FunctionFactory candidate = null;
542         FunctionFactoryDescriptor candidateDescriptor = null;
543         boolean candidateSigVarArgConst = false;
544         int candidateSigArgCount = 0;
545         int candidateSigArgTypeSum = -1;
546         int bestMatch = MATCH_NO_MATCH;
547 
548         undefinedVariables.clear();
549 
550         // find all undefined args for the purpose of setting
551         // their types when we find suitable candidate function
552         for (int i = 0; i < argCount; i++) {
553             if (args.getQuick(i).isUndefined()) {
554                 undefinedVariables.add(i);
555             }
556         }
557 
558         for (int i = 0, n = overload.size(); i < n; i++) {
559             final FunctionFactoryDescriptor descriptor = overload.getQuick(i);
560             final FunctionFactory factory = descriptor.getFactory();
561             int sigArgCount = descriptor.getSigArgCount();
562 
563             final boolean sigVarArg;
564             final boolean sigVarArgConst;
565 
566 
567             if (sigArgCount > 0) {
568                 final int lastSigArgMask = descriptor.getArgTypeMask(sigArgCount - 1);
569                 sigVarArg = FunctionFactoryDescriptor.toType(lastSigArgMask) == ColumnType.VAR_ARG;
570                 sigVarArgConst = FunctionFactoryDescriptor.isConstant(lastSigArgMask);
571             } else {
572                 sigVarArg = false;
573                 sigVarArgConst = false;
574             }
575 
576             if (sigVarArg) {
577                 sigArgCount--;
578             }
579 
580             if (argCount == 0 && sigArgCount == 0) {
581                 // this is no-arg function, match right away
582                 return checkAndCreateFunction(factory, args, argPositions, node, configuration);
583             }
584 
585             // otherwise, is number of arguments the same?
586             if (candidateDescriptor == null) {
587                 candidateDescriptor = descriptor;
588             }
589             if (sigArgCount == argCount || (sigVarArg && argCount >= sigArgCount)) {
590                 int match = MATCH_NO_MATCH; // no match
591                 if (sigArgCount == 0) {
592                     match = MATCH_EXACT_MATCH;
593                 }
594 
595                 int sigArgTypeSum = 0;
596                 for (int k = 0; k < sigArgCount; k++) {
597                     final Function arg = args.getQuick(k);
598                     final int sigArgTypeMask = descriptor.getArgTypeMask(k);
599 
600                     if (FunctionFactoryDescriptor.isConstant(sigArgTypeMask) && !arg.isConstant()) {
601                         match = MATCH_NO_MATCH; // no match
602                         break;
603                     }
604 
605                     final boolean isArray = FunctionFactoryDescriptor.isArray(sigArgTypeMask);
606                     final boolean isScalar = arg instanceof ScalarFunction;
607                     if ((isArray && isScalar) || (!isArray && !isScalar)) {
608                         match = MATCH_NO_MATCH; // no match
609                         break;
610                     }
611 
612                     final short sigArgType = FunctionFactoryDescriptor.toType(sigArgTypeMask);
613                     final int argType = arg.getType();
614                     final short argTypeTag = ColumnType.tagOf(argType);
615                     final short sigArgTypeTag = ColumnType.tagOf(sigArgType);
616 
617                     if (sigArgTypeTag == argTypeTag || (sigArgTypeTag == ColumnType.GEOHASH && ColumnType.isGeoHash(argType))) {
618                         switch (match) {
619                             case MATCH_NO_MATCH: // was it no match
620                                 match = MATCH_EXACT_MATCH;
621                                 break;
622                             case MATCH_FUZZY_MATCH: // was it fuzzy match ?
623                                 match = MATCH_PARTIAL_MATCH; // this is mixed match, fuzzy and exact
624                                 break;
625                             default:
626                                 // don't change match otherwise
627                                 break;
628                         }
629                         continue;
630                     }
631 
632                     boolean overloadPossible = false;
633                     // we do not want to use any overload when checking the output of a cast() function.
634                     // the output must be the exact type as specified by a user. that's the whole point of casting. 
635                     // for all other functions else we want to explore possible casting opportunities
636                     //
637                     // output of a cast() function is always the 2nd argument in a function signature
638                     if (k != 1 || !Chars.equals("cast", node.token)) {
639                         int overloadDistance = ColumnType.overloadDistance(argTypeTag, sigArgType); // NULL to any is 0
640                         sigArgTypeSum += overloadDistance;
641                         // Overload with cast to higher precision
642                         overloadPossible = overloadDistance != ColumnType.NO_OVERLOAD;
643 
644                         // Overload when arg is double NaN to func which accepts INT, LONG
645                         overloadPossible |= argTypeTag == ColumnType.DOUBLE &&
646                                 arg.isConstant() &&
647                                 Double.isNaN(arg.getDouble(null)) &&
648                                 (sigArgTypeTag == ColumnType.LONG || sigArgTypeTag == ColumnType.INT);
649 
650                         // Implicit cast from CHAR to STRING
651                         overloadPossible |= argTypeTag == ColumnType.CHAR &&
652                                 sigArgTypeTag == ColumnType.STRING;
653 
654                         // Implicit cast from STRING to TIMESTAMP
655                         overloadPossible |= argTypeTag == ColumnType.STRING && arg.isConstant() &&
656                                 sigArgTypeTag == ColumnType.TIMESTAMP && !factory.isGroupBy();
657 
658                         // Implicit cast from STRING to GEOHASH
659                         overloadPossible |= argTypeTag == ColumnType.STRING &&
660                                 sigArgTypeTag == ColumnType.GEOHASH && !factory.isGroupBy();
661 
662                         // Implicit cast from SYMBOL to TIMESTAMP
663                         overloadPossible |= argTypeTag == ColumnType.SYMBOL && arg.isConstant() &&
664                                 sigArgTypeTag == ColumnType.TIMESTAMP && !factory.isGroupBy();
665 
666                         overloadPossible |= arg.isUndefined();
667                     }
668 
669                     // can we use overload mechanism?
670                     if (overloadPossible) {
671                         switch (match) {
672                             case MATCH_NO_MATCH: // no match?
673                                 if (argTypeTag == ColumnType.NULL) {
674                                     match = MATCH_PARTIAL_MATCH;
675                                 } else {
676                                     match = MATCH_FUZZY_MATCH; // upgrade to fuzzy match
677                                 }
678                                 break;
679                             case MATCH_EXACT_MATCH: // was it full match so far? ? oh, well, fuzzy now
680                                 match = MATCH_PARTIAL_MATCH; // downgrade
681                                 break;
682                             default:
683                                 break; // don't change match otherwise
684                         }
685                     } else {
686                         // types mismatch
687                         match = MATCH_NO_MATCH;
688                         break;
689                     }
690                 }
691 
692                 if (match == MATCH_NO_MATCH) {
693                     continue;
694                 }
695 
696                 if (match == MATCH_EXACT_MATCH || match >= bestMatch) {
697                     // exact match may be?
698                     // special case - if signature enforces constant vararg we
699                     // have to ensure all args are indeed constant
700 
701 
702                     if (match != MATCH_EXACT_MATCH) {
703                         if (candidateSigArgTypeSum > sigArgTypeSum || bestMatch < match) {
704                             candidate = factory;
705                             candidateDescriptor = descriptor;
706                             candidateSigArgCount = sigArgCount;
707                             candidateSigVarArgConst = sigVarArgConst;
708                             candidateSigArgTypeSum = sigArgTypeSum;
709                         }
710                         bestMatch = match;
711                     } else {
712                         candidate = factory;
713                         candidateDescriptor = descriptor;
714                         candidateSigArgCount = sigArgCount;
715                         candidateSigVarArgConst = sigVarArgConst;
716                         break;
717                     }
718                 }
719             }
720         }
721 
722         if (candidate == null) {
723             // no signature match
724             throw invalidArgument(node, args, candidateDescriptor);
725         }
726 
727         if (candidateSigVarArgConst) {
728             for (int k = candidateSigArgCount; k < argCount; k++) {
729                 Function func = args.getQuick(k);
730                 if (!(func.isConstant() || func.isRuntimeConstant())) {
731                     Misc.freeObjList(args);
732                     throw SqlException.$(argPositions.getQuick(k), "constant expected");
733                 }
734             }
735         }
736 
737         // it is possible that we have more undefined variables than
738         // args in the descriptor, in case of vararg for example
739         for (int i = 0, n = undefinedVariables.size(); i < n; i++) {
740             final int pos = undefinedVariables.getQuick(i);
741             if (pos < candidateSigArgCount) {
742                 final int sigArgType = FunctionFactoryDescriptor.toType(candidateDescriptor.getArgTypeMask(pos));
743                 args.getQuick(pos).assignType(sigArgType, sqlExecutionContext.getBindVariableService());
744             } else {
745                 args.getQuick(pos).assignType(ColumnType.VAR_ARG, sqlExecutionContext.getBindVariableService());
746             }
747         }
748 
749         for (int k = 0; k < candidateSigArgCount; k++) {
750             final Function arg = args.getQuick(k);
751             final short sigArgTypeTag = FunctionFactoryDescriptor.toType(candidateDescriptor.getArgTypeMask(k));
752             final short argTypeTag = ColumnType.tagOf(arg.getType());
753 
754             if (argTypeTag == ColumnType.DOUBLE && arg.isConstant() && Double.isNaN(arg.getDouble(null))) {
755                 // substitute NaNs with appropriate types
756                 if (sigArgTypeTag == ColumnType.LONG) {
757                     args.setQuick(k, LongConstant.NULL);
758                 } else if (sigArgTypeTag == ColumnType.INT) {
759                     args.setQuick(k, IntConstant.NULL);
760                 }
761             } else if ((argTypeTag == ColumnType.STRING || argTypeTag == ColumnType.SYMBOL) && sigArgTypeTag == ColumnType.TIMESTAMP) {
762                 // convert arguments if necessary
763                 int position = argPositions.getQuick(k);
764                 if (arg.isConstant()) {
765                     long timestamp = convertToTimestamp(arg.getStr(null), position);
766                     args.set(k, TimestampConstant.newInstance(timestamp));
767                 } else {
768                     AbstractUnaryTimestampFunction castFn;
769                     if (argTypeTag == ColumnType.STRING) {
770                         castFn = new CastStrToTimestampFunctionFactory.Func(arg);
771                     } else {
772                         castFn = new CastSymbolToTimestampFunctionFactory.Func(arg);
773                     }
774                     args.setQuick(k, castFn);
775                 }
776             } else if (argTypeTag == ColumnType.UUID && sigArgTypeTag == ColumnType.STRING) {
777                 args.setQuick(k, new CastUuidToStrFunctionFactory.Func(arg));
778             }
779         }
780         return checkAndCreateFunction(candidate, args, argPositions, node, configuration);
781     }

785         int fromType = function.getType();

800                 if (toType == ColumnType.STRING) {
801                     return new CastUuidToStrFunctionFactory.Func(function);

805                 if (ColumnType.isGeoHash(fromType)) {

809                         return CastGeoHashToGeoHashFunctionFactory.newInstance(position, function, toType, fromType);
