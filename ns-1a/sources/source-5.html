92             if (!ff.exists(srcPath.slash$())) {

106             final CharSequence snapshotInstanceId = memFile.getStr(0);
107             if (Chars.empty(currentInstanceId) || Chars.empty(snapshotInstanceId) || Chars.equals(currentInstanceId, snapshotInstanceId)) {
108                 return;

113                     .$("`, previousId=`").$(snapshotInstanceId)

283     public void completeSnapshot() throws SqlException {
284         if (!lock.tryLock()) {
285             throw SqlException.position(0).put("Another snapshot command in progress");
286         }
287         try {
288             activePrepareFlag.set(false); // reset snapshot prepare flag
289             if (snapshotReaders.size() == 0) {
290                 LOG.info().$("Snapshot has no tables, SNAPSHOT COMPLETE is ignored.").$();
291             }
292 
293             // Delete snapshot/db directory.
294             path.of(configuration.getSnapshotRoot()).concat(configuration.getDbDirectory()).$();
295             ff.rmdir(path); // it's fine to ignore errors here
296 
297             // Release locked readers if any.
298             unsafeReleaseReaders();
299             // Resume the WalPurgeJob
300             if (walPurgeJobRunLock != null) {
301                 try {
302                     walPurgeJobRunLock.unlock();
303                 } catch (IllegalStateException ignore) {
304                     // not an error here
305                     // completeSnapshot can be called several time in a row.
306                 }
307             }
308         } finally {
309             lock.unlock();
310         }
311     }

313     public void prepareSnapshot(SqlExecutionContext executionContext) throws SqlException {
314         // Windows doesn't support sync() system call.
315         if (Os.isWindows()) {
316             throw SqlException.position(0).put("Snapshots are not supported on Windows");
317         }
318 
319         if (!lock.tryLock()) {
320             throw SqlException.position(0).put("Another snapshot command in progress");
321         }
322         try {
323             // activePrepareFlag is used to detect active snapshot prepare called on an empty DB
324             if (activePrepareFlag.get()) {
325                 throw SqlException.position(0).put("Waiting for SNAPSHOT COMPLETE to be called");
326             }
327 
328             path.of(configuration.getSnapshotRoot()).concat(configuration.getDbDirectory());
329             int snapshotLen = path.length();
330             // Delete all contents of the snapshot/db dir.
331             if (ff.exists(path.slash$())) {
332                 path.trimTo(snapshotLen).$();
333                 if (ff.rmdir(path) != 0) {
334                     throw CairoException.critical(ff.errno()).put("Could not remove snapshot dir [dir=").put(path).put(']');
335                 }
336             }
337             // Recreate the snapshot/db dir.
338             path.trimTo(snapshotLen).slash$();
339             if (ff.mkdirs(path, configuration.getMkDirMode()) != 0) {
340                 throw CairoException.critical(ff.errno()).put("Could not create [dir=").put(path).put(']');
341             }
342 
343             try (TableListRecordCursorFactory factory = new TableListRecordCursorFactory()) {
344                 final int tableNameIndex = factory.getMetadata().getColumnIndex(TableListRecordCursorFactory.TABLE_NAME_COLUMN);
345                 try (RecordCursor cursor = factory.getCursor(executionContext)) {
346                     final Record record = cursor.getRecord();
347 
348                     // Suspend the WalPurgeJob
349                     if (walPurgeJobRunLock != null) {
350                         final long timeout = configuration.getCircuitBreakerConfiguration().getTimeout();
351                         while (!walPurgeJobRunLock.tryLock(timeout, TimeUnit.MICROSECONDS)) {
352                             executionContext.getCircuitBreaker().statefulThrowExceptionIfTrippedNoThrottle();
353                         }
354                     }
355 
356                     try (MemoryCMARW mem = Vm.getCMARWInstance()) {
357                         // Copy metadata files for all tables.
358                         while (cursor.hasNext()) {
359                             CharSequence tableName = record.getStr(tableNameIndex);
360                             path.of(configuration.getRoot());
361                             TableToken tableToken = engine.getTableToken(tableName);
362                             if (
363                                     TableUtils.isValidTableName(tableName, tableName.length())
364                                             && ff.exists(path.concat(tableToken).concat(TableUtils.META_FILE_NAME).$())
365                             ) {
366                                 boolean isWalTable = engine.isWalTable(tableToken);
367                                 path.of(configuration.getSnapshotRoot()).concat(configuration.getDbDirectory());
368                                 LOG.info().$("preparing for snapshot [table=").$(tableName).I$();
369 
370                                 TableReader reader = engine.getReaderWithRepair(executionContext.getCairoSecurityContext(), tableToken);
371                                 snapshotReaders.add(reader);
372 
373                                 path.trimTo(snapshotLen).concat(tableToken);
374                                 int rootLen = path.length();
375 
376                                 if (isWalTable) {
377                                     path.concat(WalUtils.SEQ_DIR);
378                                 }
379                                 if (ff.mkdirs(path.slash$(), configuration.getMkDirMode()) != 0) {
380                                     throw CairoException.critical(ff.errno()).put("Could not create [dir=").put(path).put(']');
381                                 }
382 
383                                 // Copy _meta file.
384                                 path.trimTo(rootLen).concat(TableUtils.META_FILE_NAME).$();
385                                 mem.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
386                                 reader.getMetadata().dumpTo(mem);
387                                 mem.close(false);
388                                 // Copy _txn file.
389                                 path.trimTo(rootLen).concat(TableUtils.TXN_FILE_NAME).$();
390                                 mem.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
391                                 reader.getTxFile().dumpTo(mem);
392                                 mem.close(false);
393                                 // Copy _cv file.
394                                 path.trimTo(rootLen).concat(TableUtils.COLUMN_VERSION_FILE_NAME).$();
395                                 mem.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
396                                 reader.getColumnVersionReader().dumpTo(mem);
397                                 mem.close(false);
398 
399                                 if (isWalTable) {
400                                     metadata.clear();
401                                     long lastTxn = engine.getTableSequencerAPI().getTableMetadata(tableToken, metadata);
402                                     path.trimTo(rootLen).concat(WalUtils.SEQ_DIR);
403                                     metadata.switchTo(path, path.length()); // dump sequencer metadata to snapshot/db/tableName/txn_seq/_meta
404                                     metadata.close(Vm.TRUNCATE_TO_POINTER);
405 
406                                     mem.smallFile(ff, path.concat(TableUtils.TXN_FILE_NAME).$(), MemoryTag.MMAP_DEFAULT);
407                                     mem.putLong(lastTxn); // write lastTxn to snapshot/db/tableName/txn_seq/_txn
408                                     mem.close(true, Vm.TRUNCATE_TO_POINTER);
409                                 }
410                             } else {
411                                 LOG.error().$("skipping, invalid table name or missing metadata [table=").$(tableName).I$();
412                             }
413                         }
414 
415                         path.of(configuration.getSnapshotRoot()).concat(configuration.getDbDirectory()).concat(TableUtils.SNAPSHOT_META_FILE_NAME).$();
416                         mem.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
417                         mem.putStr(configuration.getSnapshotInstanceId());
418                         mem.close();
419 
420                         // Flush dirty pages and filesystem metadata to disk
421                         if (ff.sync() != 0) {
422                             throw CairoException.critical(ff.errno()).put("Could not sync");
423                         }
424 
425                         activePrepareFlag.set(true);
426                         LOG.info().$("snapshot copying finished").$();
427                     } catch (Throwable e) {
428                         // Resume the WalPurgeJob
429                         if (walPurgeJobRunLock != null) {
430                             walPurgeJobRunLock.unlock();
431                         }
432                         unsafeReleaseReaders();
433                         LOG.error()
434                                 .$("snapshot error [e=").$(e)
435                                 .I$();
436                         throw e;
437                     }
438                 }
439             }
440         } finally {
441             lock.unlock();
442         }
443     }
