69         boolean anyReadersBeforeCommittedTxn = tableWriter.checkScoreboardHasReadersBeforeLastCommittedTxn();
70         TableRecordMetadata tableMetadata = tableWriter.getMetadata();
71 
72         path.trimTo(rootLen);
73         long updateTxn = tableWriter.getTxn();
74 
75         try {
76             // Process updated column by column, one at the time
77             for (int updatedCol = 0, nn = updateColumnIndexes.size(); updatedCol < nn; updatedCol++) {
78                 int processColumnIndex = updateColumnIndexes.getQuick(updatedCol);
79                 CharSequence columnName = tableMetadata.getColumnName(processColumnIndex);
80                 int columnType = tableMetadata.getColumnType(processColumnIndex);
81                 cleanupColumnVersionsAsync.clear();
82 
83                 for (int i = 0, n = cleanupColumnVersions.size(); i < n; i += 4) {
84                     int columnIndex = (int) cleanupColumnVersions.getQuick(i);
85                     long columnVersion = cleanupColumnVersions.getQuick(i + 1);
86                     long partitionTimestamp = cleanupColumnVersions.getQuick(i + 2);
87                     long partitionNameTxn = cleanupColumnVersions.getQuick(i + 3);
88 
89                     // Process updated column by column, one at a time
90                     if (columnIndex == processColumnIndex) {
91                         boolean columnPurged = !anyReadersBeforeCommittedTxn;
92                         if (!anyReadersBeforeCommittedTxn) {
93                             path.trimTo(rootLen);
94                             TableUtils.setPathForPartition(path, tableWriter.getPartitionBy(), partitionTimestamp, false);
95                             TableUtils.txnPartitionConditionally(path, partitionNameTxn);
96                             int pathPartitionLen = path.length();
97                             TableUtils.dFile(path, columnName, columnVersion);
98                             if (!ff.remove(path.$())) {
99                                 columnPurged = false;
100                             }
101                             if (columnPurged && ColumnType.isVariableLength(columnType)) {
102                                 path.trimTo(pathPartitionLen);
103                                 TableUtils.iFile(path, columnName, columnVersion);
104                                 TableUtils.iFile(path.$(), columnName, columnVersion);
105                                 if (!ff.remove(path.$()) && ff.exists(path)) {
106                                     columnPurged = false;
107                                 }
108                             }
109                             if (columnPurged && tableMetadata.isColumnIndexed(columnIndex)) {
110                                 path.trimTo(pathPartitionLen);
111                                 BitmapIndexUtils.valueFileName(path, columnName, columnVersion);
112                                 if (!ff.remove(path.$()) && ff.exists(path)) {
113                                     columnPurged = false;
114                                 }
115 
116                                 path.trimTo(pathPartitionLen);
117                                 BitmapIndexUtils.keyFileName(path, columnName, columnVersion);
118                                 if (!ff.remove(path.$()) && ff.exists(path)) {
119                                     columnPurged = false;
120                                 }
121                             }
122                         }
123 
124                         if (!columnPurged) {
125                             cleanupColumnVersionsAsync.add(columnVersion, partitionTimestamp, partitionNameTxn, 0L);
126                         }
127                     }
128                 }
129 
130                 // if anything not purged, schedule async purge
131                 if (cleanupColumnVersionsAsync.size() > 0) {
132                     purgeColumnVersionAsync(
133                             tableWriter.getTableToken(),
134                             columnName,
135                             tableMetadata.getTableId(),
136                             (int) tableWriter.getTruncateVersion(),
137                             columnType,

139                             updateTxn,
140                             cleanupColumnVersionsAsync
141                     );
142                     log.info().$("column purge scheduled [table=").utf8(tableWriter.getTableToken().getTableName())
143                             .$(", column=").utf8(columnName)
144                             .$(", updateTxn=").$(updateTxn)
145                             .I$();
146                 } else {
147                     log.info().$("columns purged locally [table=").utf8(tableWriter.getTableToken().getTableName())
148                             .$(", column=").utf8(columnName)
149                             .$(", newColumnVersion=").$(updateTxn - 1)
150                             .I$();
151                 }
152             }
153         } finally {
154             path.trimTo(rootLen);
155         }
156     }

158     private void purgeColumnVersionAsync(
159             TableToken tableName,
160             CharSequence columnName,
161             int tableId,
162             int tableTruncateVersion,
163             int columnType,
164             int partitionBy,
165             long updateTxn,
166             LongList columnVersions
167     ) {
168         Sequence pubSeq = messageBus.getColumnPurgePubSeq();
169         while (true) {
170             long cursor = pubSeq.next();
171             if (cursor > -1L) {
172                 ColumnPurgeTask task = messageBus.getColumnPurgeQueue().get(cursor);
173                 task.of(tableName, columnName, tableId, tableTruncateVersion, columnType, partitionBy, updateTxn, columnVersions);
174                 pubSeq.done(cursor);
175                 return;
176             } else if (cursor == -1L) {
177                 // Queue overflow
178                 log.error().$("cannot schedule column purge, purge queue is full. Please run 'VACUUM TABLE \"").utf8(tableName.getTableName())
179                         .$("\"' [columnName=").utf8(columnName)
180                         .$(", updateTxn=").$(updateTxn)
181                         .I$();
182                 return;
183             }
184             Os.pause();
185         }
