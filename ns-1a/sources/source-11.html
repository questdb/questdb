258     private void checkType(int newNodeType) {
259         if (lastNodeType == NODE_NONE) {//start of new node
260             indent();
261             sink.put('"');
262             lastNodeType = newNodeType;
263             return;
264         }
265         if (newNodeType == lastNodeType &&
266                 newNodeType != NODE_CHILD) {
267             return;//continuation
268         }
269 
270         switch (lastNodeType) {
271             case NODE_TYPE:
272                 sink.putNoEsc("\",\n");
273                 indent();
274                 break;
275             case NODE_META:
276             case NODE_ATTR:
277                 sink.put("\": ");
278                 break;
279             case NODE_VALUE:
280                 if (quoteValue) {
281                     sink.put('"');
282                     sink.setCharAt(valueStartPos, '"');
283                 }
284                 sink.putNoEsc(",\n");
285                 indent();
286                 break;
287             case NODE_CHILD:
288                 indent();
289                 if (newNodeType == NODE_CHILD) {
290                     sink.putNoEsc("},\n");
291                 } else {
292                     sink.putNoEsc("} ]\n");
293                 }
294         }
295 
296         if (newNodeType == NODE_CHILD) {
297             if (lastNodeType != NODE_CHILD) {
298                 sink.putNoEsc("\"Plans\": [\n");
299                 indent();
300                 sink.putNoEsc("{\n");
301             } else {
302                 indent();
303                 sink.putNoEsc("{\n");
304             }
305         } else {
306             char c = '"';
307             if (newNodeType == NODE_VALUE) {
308                 valueStartPos = sink.length();
309                 if (!quoteValue) {
310                     c = ' ';
311                 }
312             }
313             sink.put(c);
314         }
315 
316         lastNodeType = newNodeType;
317         lastNodeDepth = depth;
318     }
