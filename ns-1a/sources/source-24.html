505     private boolean analyzeInInterval(
506             IntrinsicModel model,
507             ExpressionNode col,
508             ExpressionNode in,
509             boolean isNegated,
510             FunctionParser functionParser,
511             RecordMetadata metadata,
512             SqlExecutionContext executionContext
513     ) throws SqlException {
514 
515         if (!isTimestamp(col)) {
516             return false;
517         }
518 
519         if (in.paramCount == 2) {
520             // Single value ts in '2010-01-01' - treat string literal as an interval, not single Timestamp point
521             ExpressionNode inArg = in.rhs;
522             if (inArg.type == ExpressionNode.CONSTANT) {
523                 if (isNullKeyword(inArg.token)) {
524                     if (!isNegated) {
525                         model.intersectIntervals(Numbers.LONG_NaN, Numbers.LONG_NaN);
526                     } else {
527                         model.subtractIntervals(Numbers.LONG_NaN, Numbers.LONG_NaN);
528                     }
529                 } else {
530                     if (!isNegated) {
531                         model.intersectIntervals(inArg.token, 1, inArg.token.length() - 1, inArg.position);
532                     } else {
533                         model.subtractIntervals(inArg.token, 1, inArg.token.length() - 1, inArg.position);
534                     }
535                 }
536                 in.intrinsicValue = IntrinsicModel.TRUE;
537                 return true;
538             } else if (isFunc(inArg)) {
539                 // Single value ts in $1 - treat string literal as an interval, not single Timestamp point
540                 Function f1 = functionParser.parseFunction(inArg, metadata, executionContext);
541                 if (checkFunctionCanBeTimestampInterval(executionContext, f1)) {
542                     if (f1.isConstant()) {
543                         CharSequence funcVal = f1.getStr(null);
544                         if (!isNegated) {
545                             model.intersectIntervals(funcVal, 0, funcVal.length(), inArg.position);
546                         } else {
547                             model.subtractIntervals(funcVal, 0, funcVal.length(), inArg.position);
548                         }
549                     } else if (f1.isRuntimeConstant()) {
550                         if (!isNegated) {
551                             model.intersectRuntimeIntervals(f1);
552                         } else {
553                             model.subtractRuntimeIntervals(f1);
554                         }
555                     } else {
556                         return false;
557                     }
558 
559                     in.intrinsicValue = IntrinsicModel.TRUE;
560                     return true;
561                 } else {
562                     checkFunctionCanBeTimestamp(metadata, executionContext, f1, inArg.position);
563                     // This is IN (TIMESTAMP) one value which is timestamp and not a STRING
564                     // This is same as equals
565                     return analyzeTimestampEqualsFunction(model, in, f1, inArg.position);
566                 }
567             }
568         } else {
569             // Multiple values treat as multiple Timestamp points
570             // Only possible to translate if it's the only timestamp restriction atm
571             // NOT IN can be translated in any case as series of subtractions
572             if (!model.hasIntervalFilters() || isNegated) {
573                 int n = in.args.size() - 1;
574                 for (int i = 0; i < n; i++) {
575                     ExpressionNode inListItem = in.args.getQuick(i);
576                     if (inListItem.type != ExpressionNode.CONSTANT) {
577                         return false;
578                     }
579                 }
580 
581                 for (int i = 0; i < n; i++) {
582                     ExpressionNode inListItem = in.args.getQuick(i);
583                     long ts = parseTokenAsTimestamp(inListItem);
584                     if (!isNegated) {
585                         if (i == 0) {
586                             model.intersectIntervals(ts, ts);
587                         } else {
588                             model.unionIntervals(ts, ts);
589                         }
590                     } else {
591                         model.subtractIntervals(ts, ts);
592                     }
593                 }
594                 in.intrinsicValue = IntrinsicModel.TRUE;
595                 return true;
596             }
597         }
598         return false;
599     }
