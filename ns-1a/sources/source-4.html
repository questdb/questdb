79     @Override
80     public OperationFuture execute(SqlExecutionContext sqlExecutionContext, SCSequence eventSubSeq, boolean closeOnDone) throws SqlException {
81         switch (type) {
82             case INSERT:
83                 return insertOp.execute(sqlExecutionContext);
84             case UPDATE:
85                 updateOp.withSqlStatement(sqlStatement);
86                 return updateOperationDispatcher.execute(updateOp, sqlExecutionContext, eventSubSeq, closeOnDone);
87             case ALTER:
88                 alterOp.withSqlStatement(sqlStatement);
89                 return alterOperationDispatcher.execute(alterOp, sqlExecutionContext, eventSubSeq, closeOnDone);
90             default:
91                 return doneFuture.of(0);
92         }
93     }
