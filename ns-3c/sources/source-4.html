70         if (recordB == null) {
71             recordB = new UnionCastRecord(castRecord.getCastFunctionsA(), castRecord.getCastFunctionsB());
72             recordB.setAb(true);
73             // we do not need cursorB here, it is likely to be closed anyway
74             recordB.of(cursorA.getRecordB(), null);

79     @Override

81         if (!isCursorBHashed) {
82             hashCursorB();
83             castRecord.setAb(true);
84             toTop();

87         while (cursorA.hasNext()) {

89             key.put(castRecord, recordSink);

94         }
95         return false;
96     }

113     private void hashCursorB() {
114         while (cursorB.hasNext()) {
115             MapKey key = map.withKey();
116             key.put(castRecord, recordSink);
117             key.createValue();
118             circuitBreaker.statefulThrowExceptionIfTripped();
119         }
120         // this is an optimisation to release TableReader in case "this"
121         // cursor lingers around. If there is exception or circuit breaker fault
122         // we will rely on close() method to release reader.
123         cursorB = Misc.free(cursorB);
124     }
