172     public static long compute(
173             DateLocale locale,
174             int era,
175             int year,
176             int month,
177             int day,
178             int hour,
179             int minute,
180             int second,
181             int millis,
182             int timezone,
183             long offset,
184             int hourType
185     ) throws NumericException {
186 
187         if (era == 0) {
188             year = -(year - 1);
189         }
190 
191         boolean leap = Dates.isLeapYear(year);
192 
193         // wrong month
194         if (month < 1 || month > 12) {
195             throw NumericException.INSTANCE;
196         }
197 
198         switch (hourType) {
199             case HOUR_PM:
200                 hour += 12;
201             case HOUR_24:
202                 // wrong hour
203                 if (hour < 0 || hour > 23) {
204                     throw NumericException.INSTANCE;
205                 }
206                 break;
207             default:
208                 // wrong 12-hour clock hour
209                 if (hour < 0 || hour > 11) {
210                     throw NumericException.INSTANCE;
211                 }
212         }
213 
214         // wrong day of month
215         if (day < 1 || day > Dates.getDaysPerMonth(month, leap)) {
216             throw NumericException.INSTANCE;
217         }
218 
219         if (minute < 0 || minute > 59) {
220             throw NumericException.INSTANCE;
221         }
222 
223         if (second < 0 || second > 59) {
224             throw NumericException.INSTANCE;
225         }
226 
227         long datetime = Dates.yearMillis(year, leap)
228                 + Dates.monthOfYearMillis(month, leap)
229                 + (day - 1) * Dates.DAY_MILLIS
230                 + hour * Dates.HOUR_MILLIS
231                 + minute * Dates.MINUTE_MILLIS
232                 + second * Dates.SECOND_MILLIS
233                 + millis;
234 
235         if (timezone > -1) {
236             datetime -= locale.getZoneRules(timezone, RESOLUTION_MILLIS).getOffset(datetime, year, leap);
237         } else if (offset > Long.MIN_VALUE) {
238             datetime -= offset;
239         }
240 
241         return datetime;
242     }
