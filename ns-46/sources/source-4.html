350         if (mid < 0) {

352         }

354

358 
359         if (diff < 0) {
360             year--;
361         } else if (diff >= YEAR_MILLIS) {
362             yearStart += leap ? LEAP_YEAR_MILLIS : YEAR_MILLIS;
363             if (yearStart <= millis) {
364                 year++;
365             }
366         }
367

446                         }

553      * Calculated start of year in millis. For example of year 2008 this is
554      * equivalent to parsing "2008-01-01T00:00:00.000Z", except this method is faster.
555      *
556      * @param year the year
557      * @param leap true if give year is leap year
558      * @return millis for start of year.
559      */
560     public static long yearMillis(int year, boolean leap) {
561         int leapYears = year / 100;
562         if (year < 0) {
563             leapYears = ((year + 3) >> 2) - leapYears + ((leapYears + 3) >> 2) - 1;
564         } else {
565             leapYears = (year >> 2) - leapYears + (leapYears >> 2);
566             if (leap) {
567                 leapYears--;
568             }
569         }
570 
571         return (year * 365L + (leapYears - DAYS_0000_TO_1970)) * DAY_MILLIS;
572     }
