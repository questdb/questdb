50     public static void dumpThreadStacks() {
51         final ThreadMXBean threadMXBean = ManagementFactory.getThreadMXBean();
52         final ThreadInfo[] threadInfos = threadMXBean.getThreadInfo(threadMXBean.getAllThreadIds(), 20);
53         // Each thread stack on its own LOG message to avoid overrunning log buffer
54         // Generally overrun will truncate the log message. We are likely to overrun considering how
55         // many threads we could be running
56         for (ThreadInfo threadInfo : threadInfos) {
57             // it turns out it is possible to have null "infos"
58             if (threadInfo != null) {
59                 final LogRecord record = LOG.advisory();
60                 final Thread.State state = threadInfo.getThreadState();
61                 record.$('\n');
62                 record.$('\'').$(threadInfo.getThreadName()).$("': ").$(state);
63                 final StackTraceElement[] stackTraceElements = threadInfo.getStackTrace();
64                 for (final StackTraceElement stackTraceElement : stackTraceElements) {
65                     record.$("\n\t\tat ").$(stackTraceElement);
66                 }
67                 record.$("\n\n");
68                 record.$();
69             }
70         }
71     }
72
