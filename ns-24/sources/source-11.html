171         private boolean next0() {
172             do {
173                 if (readNextFileFromDisk) {
174                     foundMetadataFile = false;
175                     final long pUtf8NameZ = ff.findName(findFileStruct);
176                     if (hasNextFile) {
177                         if (ff.isDirOrSoftLinkDirNoDots(path, plimit, pUtf8NameZ, ff.findType(findFileStruct))) {
178                             if (ff.exists(path.concat(TableUtils.META_FILE_NAME).$())) {
179                                 foundMetadataFile = true;
180                                 metaMem.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
181                                 columnCount = metaMem.getInt(TableUtils.META_OFFSET_COUNT);
182                                 tableId = metaMem.getInt(TableUtils.META_OFFSET_TABLE_ID);
183                             }
184                         }
185                         hasNextFile = ff.findNext(findFileStruct) > 0;
186                     }
187                 }
188 
189                 if (foundMetadataFile) {
190                     long offset = TableUtils.getColumnNameOffset(columnCount);
191                     for (int i = 0; i < columnCount; i++) {
192                         CharSequence name = metaMem.getStr(offset);
193                         if (columnIndex == i) {
194                             int type = PGOids.getTypeOid(TableUtils.getColumnType(metaMem, i));
195                             diskReadingRecord.intValues[3] = type;
196                             diskReadingRecord.name = name;
197                             diskReadingRecord.shortValues[2] = (short) (i + 1);
198                             diskReadingRecord.shortValues[6] = (short) PG_TYPE_TO_SIZE_MAP.get(type);
199                             diskReadingRecord.intValues[0] = tableId;
200                             columnIndex++;
201                             if (columnIndex == columnCount) {
202                                 readNextFileFromDisk = true;
203                                 columnIndex = 0;
204                             } else {
205                                 readNextFileFromDisk = false;
206                             }
207                             return true;
208                         }
209                         offset += Vm.getStorageLength(name);
210                     }
211                 }
212             } while (hasNextFile);
213 
214             findFileStruct = ff.findClose(findFileStruct);
215             hasNextFile = true;
216             foundMetadataFile = false;
217             return false;
218         }
