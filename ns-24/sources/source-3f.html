219                     tableName = tableToken.getTableName();
220                     int rootLen = rootPath.length();
221                     rootPath.concat(tableToken).concat(SEQ_DIR);
222                     int metaFd = -1;

224                     try {

227                         suspendedFlag = ff.readNonNegativeByte(metaFd, SEQ_META_SUSPENDED) > 0;
228                         sequencerTxn = ff.readNonNegativeLong(txnFd, MAX_TXN_OFFSET);
229                     } finally {
