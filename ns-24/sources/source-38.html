47     public StringToStringArrayFunction(int position, CharSequence type) throws SqlException {
48         if (type == null) {
49             throw SqlException.$(position, "NULL is not allowed");
50         }
51 
52         int charIndex = findArrayOpeningBracketIndex(position, type);
53         int branch = BRANCH_BEFORE_ITEM;
54         int stringStartIndex = -1;
55         int stringEndIndex = -1;
56         int lastBackslashIndex = -1;
57         StringSink sink = Misc.getThreadLocalBuilder();
58         int len = type.length();
59 
60         out:
61         for (charIndex++; charIndex < len; charIndex++) {
62             if (lastBackslashIndex == charIndex - 1) {
63                 if (branch == BRANCH_ITEM || branch == BRANCH_DOUBLE_QUOTE) {
64                     sink.put(type, stringStartIndex, lastBackslashIndex);
65                 } else {
66                     branch = BRANCH_ITEM;
67                 }
68                 stringStartIndex = stringEndIndex = charIndex;
69                 continue;
70             }
71 
72             char ch = type.charAt(charIndex);
73             switch (ch) {
74                 case '\\':
75                     if (branch == BRANCH_AFTER_ITEM) {
76                         throw SqlException.$(position, "unexpected character after '\"'");
77                     }
78                     lastBackslashIndex = charIndex;
79                     break;
80                 case '"':
81                     if (branch == BRANCH_AFTER_ITEM) {
82                         throw SqlException.$(position, "unexpected character after '\"'");
83                     }
84                     if (branch == BRANCH_ITEM) {
85                         throw SqlException.$(position, "unexpected '\"' character");
86                     }
87                     if (branch == BRANCH_BEFORE_ITEM) {
88                         stringStartIndex = charIndex + 1;
89                         branch = BRANCH_DOUBLE_QUOTE;
90                     } else {
91                         stringEndIndex = charIndex - 1;
92                         branch = BRANCH_AFTER_ITEM;
93                     }
94                     break;
95                 case '{':
96                     if (branch != BRANCH_DOUBLE_QUOTE) {
97                         throw SqlException.$(position, "unexpected '{' character");
98                     }
99                     break;
100                 case '}':
101                     if (branch == BRANCH_DOUBLE_QUOTE) {
102                         break;
103                     }
104                     if (branch == BRANCH_BEFORE_ITEM && items.size() > 0) {
105                         throw SqlException.$(position, "unexpected '}' character");
106                     }
107                     if (branch == BRANCH_ITEM || branch == BRANCH_AFTER_ITEM) {
108                         commit(type, stringStartIndex, stringEndIndex, sink);
109                     }
110                     branch = BRANCH_AFTER_LAST_ITEM;
111                     break out;
112                 case ',':
113                     if (branch == BRANCH_DOUBLE_QUOTE) {
114                         break;
115                     }
116                     if (branch == BRANCH_BEFORE_ITEM) {
117                         throw SqlException.$(position, "unexpected ',' character");
118                     }
119                     commit(type, stringStartIndex, stringEndIndex, sink);
120                     branch = BRANCH_BEFORE_ITEM;
121                     break;
122                 default:
123                     if (!GenericLexer.WHITESPACE_CH.contains(ch)) {
124                         if (branch == BRANCH_AFTER_ITEM) {
125                             throw SqlException.$(position, "unexpected character after '\"'");
126                         }
127                         if (branch == BRANCH_BEFORE_ITEM) {
128                             stringStartIndex = charIndex;
129                             branch = BRANCH_ITEM;
130                         }
131                         stringEndIndex = charIndex;
132                     }
133             }
134         }
135 
136         if (branch != BRANCH_AFTER_LAST_ITEM) {
137             throw SqlException.$(position, "array must end with '}'");
138         }
139 
140         for (charIndex++; charIndex < len; charIndex++) {
141             char ch = type.charAt(charIndex);
142             if (!GenericLexer.WHITESPACE_CH.contains(ch)) {
143                 throw SqlException.$(position, "unexpected character after '}'");
144             }
145         }
146     }
