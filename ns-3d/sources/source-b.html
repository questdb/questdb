178         int left = 0;

181             int mid = (left + right) / 2;
182             long lo = getEncodedPeriodLo(intervals, mid * 2);

187                 left = mid + 1;

190             }

867      * Intervals to be chronologically ordered and result list will be ordered as well.

876         final int sizeA = dividerIndex / 2;
877         final int sizeB = sizeA + (concatenatedIntervals.size() - dividerIndex) / 2;

879

881         int writePoint = 0;
882

885 
886         while ((aLower < sizeA || aUpper < aUpperSize) && intervalB < sizeB) {
887

889             long aLo = concatenatedIntervals.getQuick(intervalA * 2);
890             long aHi = concatenatedIntervals.getQuick(intervalA * 2 + 1);
891

897                 // a loses

899                     aUpper++;
900                 } else {
901                     aLower++;
902                 }

915                     }
916                 } else {

918                     // a loses
