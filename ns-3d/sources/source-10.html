169     // Recursively clones the current value of whereClause for the model and its sub-models into the backupWhereClause field.
170     public static void backupWhereClause(final ObjectPool<ExpressionNode> pool, final QueryModel model) {
171         QueryModel current = model;
172         while (current != null) {
173             if (current.unionModel != null) {
174                 backupWhereClause(pool, current.unionModel);
175             }
176             if (current.updateTableModel != null) {
177                 backupWhereClause(pool, current.updateTableModel);
178             }
179             for (int i = 1, n = current.joinModels.size(); i < n; i++) {
180                 final QueryModel m = current.joinModels.get(i);
181                 if (m != null && current != m) {
182                     backupWhereClause(pool, m);
183                 }
184             }
185             current.backupWhereClause = ExpressionNode.deepClone(pool, current.whereClause);
186             current = current.nestedModel;
187         }
188     }

191     public static void restoreWhereClause(final ObjectPool<ExpressionNode> pool, final QueryModel model) {
192         QueryModel current = model;
193         while (current != null) {
194             if (current.unionModel != null) {
195                 restoreWhereClause(pool, current.unionModel);
196             }
197             if (current.updateTableModel != null) {
198                 restoreWhereClause(pool, current.updateTableModel);
199             }
200             for (int i = 1, n = current.joinModels.size(); i < n; i++) {
201                 final QueryModel m = current.joinModels.get(i);
202                 if (m != null && current != m) {
203                     restoreWhereClause(pool, m);
204                 }
205             }
206             current.whereClause = ExpressionNode.deepClone(pool, current.backupWhereClause);
207             current = current.nestedModel;

968      * Splits "where" clauses into "and" chunks
969      */
970     public ObjList<ExpressionNode> parseWhereClause() {
971         ExpressionNode n = getWhereClause();
972         // pre-order traversal
973         sqlNodeStack.clear();
974         while (!sqlNodeStack.isEmpty() || n != null) {
975             if (n != null && n.token != null) {
976                 if (isAndKeyword(n.token)) {
977                     if (n.rhs != null) {
978                         sqlNodeStack.push(n.rhs);
979                     }
980                     n = n.lhs;
981                 } else {
982                     addParsedWhereNode(n, false);
983                     n = null;
984 
985                 }
986             } else {
987                 n = sqlNodeStack.poll();
988             }
989         }
990         return getParsedWhere();
991     }

1185     private void sinkColumns(CharSink sink, ObjList<QueryColumn> columns) {
1186         for (int i = 0, n = columns.size(); i < n; i++) {
1187             if (i > 0) {
1188                 sink.put(", ");
1189             }
1190             QueryColumn column = columns.getQuick(i);
1191             CharSequence name = column.getName();
1192             CharSequence alias = column.getAlias();
1193             ExpressionNode ast = column.getAst();
1194             ast.toSink(sink);
1195             if (column instanceof AnalyticColumn || name == null) {
1196 
1197                 if (alias != null) {
1198                     aliasToSink(alias, sink);
1199                 }
1200 
1201                 // this can only be analytic column
1202                 if (name != null) {
1203                     AnalyticColumn ac = (AnalyticColumn) column;
1204                     sink.put(" over (");
1205                     final ObjList<ExpressionNode> partitionBy = ac.getPartitionBy();
1206                     if (partitionBy.size() > 0) {
1207                         sink.put("partition by ");
1208                         for (int k = 0, z = partitionBy.size(); k < z; k++) {
1209                             if (k > 0) {
1210                                 sink.put(", ");
1211                             }
1212                             partitionBy.getQuick(k).toSink(sink);
1213                         }
1214                     }
1215 
1216                     final ObjList<ExpressionNode> orderBy = ac.getOrderBy();
1217                     if (orderBy.size() > 0) {
1218                         if (partitionBy.size() > 0) {
1219                             sink.put(' ');
1220                         }
1221                         sink.put("order by ");
1222                         for (int k = 0, z = orderBy.size(); k < z; k++) {
1223                             if (k > 0) {
1224                                 sink.put(", ");
1225                             }
1226                             orderBy.getQuick(k).toSink(sink);
1227                             if (ac.getOrderByDirection().getQuick(k) == 1) {
1228                                 sink.put(" desc");
1229                             }
1230                         }
1231                     }
1232                     sink.put(')');
1233                 }
1234             } else {
1235                 // do not repeat alias when it is the same as AST token, provided AST is a literal
1236                 if (alias != null && (ast.type != ExpressionNode.LITERAL || !ast.token.equals(alias))) {
1237                     aliasToSink(alias, sink);
1238                 }
1239             }
1240         }
1241     }

1245         final boolean hasColumns = this.topDownColumns.size() > 0 || this.bottomUpColumns.size() > 0;
1246         if (hasColumns) {

1255                 sink.put(' ');
1256                 sinkColumns(sink, this.bottomUpColumns);
1257             }
1258             sink.put(" from ");
1259         }
1260         if (tableNameExpr != null) {

1263             sink.put('(');

1269         }

1272             sink.put(" timestamp (");
