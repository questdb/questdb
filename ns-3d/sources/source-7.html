76             ExpressionNode groupByExpr,
77             ExpressionNode columnExpr,
78             QueryModel translatingModel
79     ) {
80         if (groupByExpr == null && columnExpr == null) {
81             return true;
82         }
83 
84         if (groupByExpr == null || columnExpr == null || groupByExpr.type != columnExpr.type) {
85             return false;
86         }
87 
88         if (!Chars.equals(groupByExpr.token, columnExpr.token)) {
89 
90             int index = translatingModel.getAliasToColumnMap().keyIndex(columnExpr.token);
91             if (index > -1) {
92                 return false;
93             }
94 
95             final QueryColumn qc = translatingModel.getAliasToColumnMap().valueAt(index);
96             final CharSequence tok = groupByExpr.token;
97             final CharSequence qcTok = qc.getAst().token;
98             if (Chars.equals(qcTok, tok)) {
99                 return true;
100             }
101 
102             int dot = Chars.indexOf(tok, '.');
103 
104             if (dot > -1 &&
105                     translatingModel.getModelAliasIndex(tok, 0, dot) > -1
106                     && Chars.equals(qcTok, tok, dot + 1, tok.length())
107             ) {
108                 return compareArgs(groupByExpr, columnExpr, translatingModel);

112         }

260             ExpressionNode groupByExpr,
261             ExpressionNode columnExpr,
262             QueryModel translatingModel
263     ) {
264         final int groupByArgsSize = groupByExpr.args.size();
265         final int selectNodeArgsSize = columnExpr.args.size();
266 
267         if (groupByArgsSize != selectNodeArgsSize) {
268             return false;
269         }
270 
271         if (groupByArgsSize < 3) {
272             return compareNodesGroupBy(groupByExpr.lhs, columnExpr.lhs, translatingModel)
273                     && compareNodesGroupBy(groupByExpr.rhs, columnExpr.rhs, translatingModel);
274         }
275 
276         for (int i = 0; i < groupByArgsSize; i++) {
277             if (!compareNodesGroupBy(groupByExpr.args.get(i), columnExpr.args.get(i), translatingModel)) {
278                 return false;
279             }
280         }
281         return true;
282     }
