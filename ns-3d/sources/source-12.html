95     private void addEvaluateDynamicIntervals(LongList outIntervals, SqlExecutionContext sqlContext) throws SqlException {
96         int size = intervals.size();
97         int dynamicStart = size - dynamicRangeList.size() * STATIC_LONGS_PER_DYNAMIC_INTERVAL;
98         int dynamicIndex = 0;
99         boolean firstFuncApplied = false;
100 
101         for (int i = dynamicStart; i < size; i += STATIC_LONGS_PER_DYNAMIC_INTERVAL) {
102             Function dynamicFunction = dynamicRangeList.getQuick(dynamicIndex++);
103             short operation = IntervalUtils.getEncodedOperation(intervals, i);
104             boolean negated = operation > IntervalOperation.NEGATED_BORDERLINE;
105             int divider = outIntervals.size();
106 
107             if (dynamicFunction == null) {
108                 // copy 4 longs to output and apply the operation
109                 outIntervals.add(intervals, i, i + STATIC_LONGS_PER_DYNAMIC_INTERVAL);
110                 IntervalUtils.applyLastEncodedIntervalEx(outIntervals);
111             } else {
112                 long lo = IntervalUtils.getEncodedPeriodLo(intervals, i);
113                 long hi = IntervalUtils.getEncodedPeriodHi(intervals, i);
114                 short adjustment = IntervalUtils.getEncodedAdjustment(intervals, i);
115                 short dynamicHiLo = IntervalUtils.getEncodedDynamicIndicator(intervals, i);
116 
117                 dynamicFunction.init(null, sqlContext);
118 
119                 if (operation != IntervalOperation.INTERSECT_INTERVALS && operation != IntervalOperation.SUBTRACT_INTERVALS) {
120                     long dynamicValue = getTimestamp(dynamicFunction);
121                     long dynamicValue2 = 0;
122                     if (dynamicHiLo == IntervalDynamicIndicator.IS_LO_SEPARATE_DYNAMIC) {
123                         // Both ends of BETWEEN are dynamic and different values. Take next dynamic point.
124                         i += STATIC_LONGS_PER_DYNAMIC_INTERVAL;
125                         dynamicFunction = dynamicRangeList.getQuick(dynamicIndex++);
126                         dynamicFunction.init(null, sqlContext);
127                         dynamicValue2 = hi = getTimestamp(dynamicFunction);
128                         lo = dynamicValue;
129                     } else {
130                         if ((dynamicHiLo & IntervalDynamicIndicator.IS_HI_DYNAMIC) != 0) {
131                             hi = dynamicValue + adjustment;
132                         }
133                         if ((dynamicHiLo & IntervalDynamicIndicator.IS_LO_DYNAMIC) != 0) {
134                             lo = dynamicValue + adjustment;
135                         }
136                     }
137 
138                     if (dynamicValue == Numbers.LONG_NaN || dynamicValue2 == Numbers.LONG_NaN) {
139                         // functions evaluated to null.
140                         if (!negated) {
141                             // return empty set if it's not negated
142                             outIntervals.clear();
143                             return;
144                         } else {
145                             // or full set
146                             negatedNothing(outIntervals, divider);
147                             continue;
148                         }
149                     }
150 
151                     if (operation == IntervalOperation.INTERSECT_BETWEEN || operation == IntervalOperation.SUBTRACT_BETWEEN) {
152                         long tempHi = Math.max(hi, lo);
153                         lo = Math.min(hi, lo);
154                         hi = tempHi;
155                     }
156 
157                     outIntervals.extendAndSet(divider + 1, hi);
158                     outIntervals.setQuick(divider, lo);
159                     if (divider == 0 && negated) {
160                         // Divider == 0 means it's the first interval applied
161                         // Invert the interval, since it will not be applied negated to anything
162                         IntervalUtils.invert(outIntervals, divider);
163                     }
164                 } else {
165                     // This is subtract or intersect with a string interval (not a single timestamp)
166                     CharSequence strValue = dynamicFunction.getStr(null);
167                     if (operation == IntervalOperation.INTERSECT_INTERVALS) {
168                         // This is intersect
169                         if (parseIntervalFails(outIntervals, strValue)) {
170                             // return empty set
171                             outIntervals.clear();
172                             return;
173                         }
174                     } else {
175                         // This is subtract
176                         if (parseIntervalFails(outIntervals, strValue)) {
177                             // full set
178                             negatedNothing(outIntervals, divider);
179                             continue;
180                         }
181                         IntervalUtils.invert(outIntervals, divider);
182                     }
183                 }
184             }
185 
186             // Do not apply operation (intersect, subtract)
187             // if this is first element and no pre-calculated static intervals exist
188             if (firstFuncApplied || divider > 0) {
189                 switch (operation) {
190                     case IntervalOperation.INTERSECT:
191                     case IntervalOperation.INTERSECT_BETWEEN:
192                     case IntervalOperation.INTERSECT_INTERVALS:
193                     case IntervalOperation.SUBTRACT_INTERVALS:
194                         IntervalUtils.intersectInplace(outIntervals, divider);
195                         break;
196                     case IntervalOperation.SUBTRACT:
197                     case IntervalOperation.SUBTRACT_BETWEEN:
198                         IntervalUtils.subtract(outIntervals, divider);
199                         break;
200                     // UNION cannot be the first thing at the moment.
201                     default:
202                         throw new UnsupportedOperationException("Interval operation " + operation + " is not supported");
203                 }
204             }
205             firstFuncApplied = true;
206         }
207     }

209     private boolean allIntervalsHitOnePartition(PartitionBy.PartitionFloorMethod floorMethod) {
210         if (!isStatic()) {
211             return false;
212         }
213         if (intervals.size() == 0) {
214             return true;
215         }
216 
217         long floor = floorMethod.floor(intervals.getQuick(0));
218         for (int i = 1, n = intervals.size(); i < n; i++) {
219             if (floor != floorMethod.floor(intervals.getQuick(i))) {
220                 return false;
221             }
222         }
223         return true;
224     }

227         if (ColumnType.isString(dynamicFunction.getType())) {
