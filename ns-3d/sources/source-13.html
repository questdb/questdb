34 /**
35  * Collects interval during query parsing
36  * and records them as 2 types:
37  * - static list of intervals as 2 long points [lo, hi] in staticIntervals list
38  * - dynamic list of functions.
39  * <p>
40  * When first interval involving function is added all data starts to be encoded in 4 longs in staticPeriods
41  * 0: lo (long)
42  * 1: hi (long)
43  * 2: operation (short), period type (short), adjustment (short), dynamicIndicator (short)
44  * 3: period (int), count (int)
45  * <p>
46  * and the index when it happens stored in pureStaticCount
47  */
48 public class RuntimeIntervalModelBuilder implements Mutable {

169     public void setBetweenBoundary(long timestamp) {
170         if (!betweenBoundarySet) {
171             betweenBoundary = timestamp;
172             betweenBoundarySet = true;
173         } else {
174             if (betweenBoundaryFunc == null) {
175                 // Constant interval
176                 long lo = Math.min(timestamp, betweenBoundary);
177                 long hi = Math.max(timestamp, betweenBoundary);
178                 if (hi == Numbers.LONG_NaN || lo == Numbers.LONG_NaN) {
179                     if (!betweenNegated) {
180                         intersectEmpty();
181                     }
182                     // else {
183                     // NOT BETWEEN with NULL
184                     // to be consistent with non-designated filtering
185                     // do no filtering
186                     //  }
187                 } else {
188                     if (!betweenNegated) {
189                         intersect(lo, hi);
190                     } else {
191                         subtractInterval(lo, hi);
192                     }
193                 }
194             } else {
195                 intersectBetweenSemiDynamic(betweenBoundaryFunc, timestamp);
196             }
197             betweenBoundarySet = false;
198         }
199     }
