140     @Override
141     public long getTableMetadata(@NotNull TableRecordMetadataSink sink) {
142         int columnCount = metadata.getColumnCount();
143         int timestampIndex = metadata.getTimestampIndex();
144         int compressedTimestampIndex = -1;
145         sink.clear();
146 
147         int compressedColumnCount = 0;
148         for (int i = 0; i < columnCount; i++) {
149             int columnType = metadata.getColumnType(i);
150             sink.addColumn(
151                     metadata.getColumnName(i),
152                     columnType,
153                     metadata.isColumnIndexed(i),
154                     metadata.getIndexValueBlockCapacity(i),
155                     metadata.isSymbolTableStatic(i),
156                     i
157             );
158             if (columnType > -1) {
159                 if (i == timestampIndex) {
160                     compressedTimestampIndex = compressedColumnCount;
161                 }
162                 compressedColumnCount++;
163             }
164         }
165 
166         sink.of(
167                 tableToken,
168                 metadata.getTableId(),
169                 timestampIndex,
170                 compressedTimestampIndex,
171                 metadata.isSuspended(),
172                 metadata.getStructureVersion(),
173                 compressedColumnCount
174         );
175 
176         return tableTransactionLog.lastTxn();
177     }

211     @Override
212     public long nextStructureTxn(long expectedStructureVersion, TableMetadataChange change) {
213         // Writing to TableSequencer can happen from multiple threads, so we need to protect against concurrent writes.
214         assert !closed;
215         checkDropped();
216         long txn;
217         try {
218             if (metadata.getStructureVersion() == expectedStructureVersion) {
219                 tableTransactionLog.beginMetadataChangeEntry(expectedStructureVersion + 1, alterCommandWalFormatter, change, microClock.getTicks());
220 
221                 // Re-read serialised change to ensure it can be read.
222                 AlterOperation deserializedAlter = tableTransactionLog.readTableMetadataChangeLog(expectedStructureVersion, alterCommandWalFormatter);
223 
224                 applyToMetadata(deserializedAlter);
225                 if (metadata.getStructureVersion() != expectedStructureVersion + 1) {
226                     throw CairoException.critical(0)
227                             .put("applying structure change to WAL table failed [table=").put(tableToken.getDirName())
228                             .put(", oldVersion: ").put(expectedStructureVersion)
229                             .put(", newVersion: ").put(metadata.getStructureVersion())
230                             .put(']');
231                 }
232                 metadata.syncToDisk();
233                 txn = tableTransactionLog.endMetadataChangeEntry();
234             } else {
235                 return NO_TXN;
236             }
237         } catch (Throwable th) {
238             distressed = true;
239             LOG.critical().$("could not apply structure change to WAL table sequencer [table=").utf8(tableToken.getDirName())
240                     .$(", error=").$(th.getMessage())
241                     .I$();
242             throw th;
243         }
244 
245         if (!metadata.isSuspended()) {
246             engine.notifyWalTxnCommitted(tableToken, txn);
247         }
248         return txn;
249     }
