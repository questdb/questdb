221     private void loadSequencerMetadata(MemoryMR metaMem) {
222         columnMetadata.clear();
223         columnNameIndexMap.clear();
224 
225         try {
226             final long memSize = checkMemSize(metaMem, SEQ_META_OFFSET_COLUMNS);
227             validateMetaVersion(metaMem, SEQ_META_OFFSET_WAL_VERSION, WAL_FORMAT_VERSION);
228             final int columnCount = TableUtils.getColumnCount(metaMem, SEQ_META_OFFSET_COLUMN_COUNT);
229             final int timestampIndex = TableUtils.getTimestampIndex(metaMem, SEQ_META_OFFSET_TIMESTAMP_INDEX, columnCount);
230 
231             // load column types and names
232             long offset = SEQ_META_OFFSET_COLUMNS;
233             for (int i = 0; i < columnCount; i++) {
234                 final int type = TableUtils.getColumnType(metaMem, memSize, offset, i);
235                 offset += Integer.BYTES;
236 
237                 final String name = TableUtils.getColumnName(metaMem, memSize, offset, i).toString();
238                 offset += Vm.getStorageLength(name);
239 
240                 if (type > 0) {
241                     columnNameIndexMap.put(name, i);
242                 }
243 
244                 if (ColumnType.isSymbol(Math.abs(type))) {
245                     columnMetadata.add(new TableColumnMetadata(name, type, true, 1024, true, null));
246                 } else {
247                     columnMetadata.add(new TableColumnMetadata(name, type));
248                 }
249             }
250 
251             // validate designated timestamp column
252             if (timestampIndex != -1) {
253                 final int timestampType = columnMetadata.getQuick(timestampIndex).getType();
254                 if (!ColumnType.isTimestamp(timestampType)) {
255                     throw validationException(metaMem).put("Timestamp column must be TIMESTAMP, but found ").put(ColumnType.nameOf(timestampType));
256                 }
257             }
258         } catch (Throwable e) {
259             columnNameIndexMap.clear();
260             columnMetadata.clear();
261             throw e;
262         }
263     }
