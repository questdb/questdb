98     public void forAllWalTables(ObjList<TableToken> tableTokenBucket, boolean includeDropped, RegisteredTable callback) {
99         final CharSequence root = configuration.getRoot();
100         final FilesFacade ff = configuration.getFilesFacade();
101         Path path = Path.PATH.get();
102 
103         engine.getTableTokens(tableTokenBucket, includeDropped);
104         for (int i = 0, n = tableTokenBucket.size(); i < n; i++) {
105             TableToken tableToken = tableTokenBucket.getQuick(i);
106 
107             // Exclude locked entries.
108             // Use includeDropped argument to decide whether to include dropped tables.
109             String publicTableName = tableToken.getTableName();
110             boolean isDropped = includeDropped && engine.isTableDropped(tableToken);
111             if (engine.isWalTable(tableToken) && !isDropped) {
112                 long lastTxn;
113                 int tableId;
114 
115                 try {
116                     if (!seqRegistry.containsKey(tableToken.getDirName())) {
117                         // Fast path.
118                         // The following calls are racy, i.e. there might be a sequencer modifying both
119                         // metadata and log concurrently as we read the values. It's ok since we iterate
120                         // through the WAL tables periodically, so eventually we should see the updates.
121                         path.of(root).concat(tableToken.getDirName()).concat(SEQ_DIR);
122                         tableId = tableToken.getTableId();
123                         int fdTxn = TableUtils.openRO(ff, path, TXNLOG_FILE_NAME, LOG);
124                         lastTxn = ff.readNonNegativeLong(fdTxn, MAX_TXN_OFFSET); // does not throw
125                         ff.close(fdTxn);
126                     } else {
127                         // Slow path.
128                         try (TableSequencer tableSequencer = openSequencerLocked(tableToken, SequencerLockType.NONE)) {
129                             lastTxn = tableSequencer.lastTxn();
130                             tableId = tableSequencer.getTableId();
131                         }
132                     }
133                 } catch (CairoException ex) {
134                     LOG.critical().$("could not read WAL table transaction file [table=").utf8(publicTableName).$(", errno=").$(ex.getErrno())
135                             .$(", error=").$((Throwable) ex).I$();
136                     continue;
137                 }
138 
139                 if (tableId < 0 || lastTxn < 0) {
140                     LOG.critical().$("could not read WAL table metadata [table=").utf8(publicTableName).$(", tableId=").$(tableId)
141                             .$(", lastTxn=").$(lastTxn).I$();
142                     continue;
143                 }
144 
145                 try {
146                     callback.onTable(tableId, tableToken, lastTxn);
147                 } catch (CairoException ex) {
148                     LOG.critical().$("could not process table sequencer [table=").utf8(publicTableName).$(", errno=").$(ex.getErrno())
149                             .$(", error=").$((Throwable) ex).I$();
150                 }
151             } else if (isDropped) {
152                 try {
153                     callback.onTable(tableToken.getTableId(), tableToken, -1);
154                 } catch (CairoException ex) {
155                     LOG.critical().$("could not process table sequencer [table=").utf8(publicTableName).$(", errno=").$(ex.getErrno())
156                             .$(", error=").$((Throwable) ex).I$();
157                 }
158             }
159         }
160     }

327     public void resumeTable(TableToken tableToken, long resumeFromTxn, CairoSecurityContext cairoSecurityContext) {
328         try (TableSequencerImpl sequencer = openSequencerLocked(tableToken, SequencerLockType.WRITE)) {
329             try {
330                 if (!sequencer.isSuspended()) {
331                     return;
332                 }
333                 final long nextTxn = sequencer.lastTxn() + 1;
334                 if (resumeFromTxn > nextTxn) {
335                     throw CairoException.nonCritical().put("resume txn is higher than next available transaction [resumeFromTxn=").put(resumeFromTxn).put(", nextTxn=").put(nextTxn).put(']');
336                 }
337                 // resume from the latest on negative value
338                 if (resumeFromTxn > 0) {
339                     try (TableWriter tableWriter = engine.getWriter(cairoSecurityContext, tableToken, WAL_2_TABLE_RESUME_REASON)) {
340                         long seqTxn = tableWriter.getSeqTxn();
341                         if (resumeFromTxn - 1 > seqTxn) {
342                             // including resumeFromTxn 
343                             tableWriter.commitSeqTxn(resumeFromTxn - 1);
344                         }
345                     }
346                 }
347                 sequencer.resumeTable();
348             } finally {
349                 sequencer.unlockWrite();
350             }
351         }
352     }

375     @NotNull
376     private TableSequencerEntry getTableSequencerEntry(
377             TableToken tableToken,
378             SequencerLockType lock,
379             BiFunction<CharSequence, Object, TableSequencerEntry> getSequencerLambda
380     ) {
381         TableSequencerEntry entry;
382         int attempt = 0;
383         while (attempt < recreateDistressedSequencerAttempts) {
384             throwIfClosed();
385             entry = seqRegistry.computeIfAbsent(tableToken.getDirName(), tableToken, getSequencerLambda);
386             if (lock == SequencerLockType.READ) {
387                 entry.readLock();
388             } else if (lock == SequencerLockType.WRITE) {
389                 entry.writeLock();
390             }
391 
392             boolean isDistressed = entry.isDistressed();
393             if (!isDistressed && !entry.isClosed()) {
394                 return entry;
395             } else {
396                 if (lock == SequencerLockType.READ) {
397                     entry.unlockRead();
398                 } else if (lock == SequencerLockType.WRITE) {
399                     entry.unlockWrite();
400                 }
401             }
402             if (isDistressed) {
403                 attempt++;
404             }
405         }
406 
407         throw CairoException.critical(0).put("sequencer is distressed [table=").put(tableToken.getDirName()).put(']');
408     }

421     private boolean releaseEntries(long deadline) {
422         if (seqRegistry.size() == 0) {
423             // nothing to release
424             return true;
425         }
426         boolean removed = false;
427         for (CharSequence tableDir : seqRegistry.keySet()) {
428             final TableSequencerEntry sequencer = seqRegistry.get(tableDir);
429             if (sequencer != null && deadline >= sequencer.releaseTime && !sequencer.isClosed()) {
430                 // Remove from registry only if this thread closed the instance
431                 if (sequencer.checkClose()) {
432                     LOG.info().$("releasing idle table sequencer [tableDir=").utf8(tableDir).I$();
433                     seqRegistry.remove(tableDir, sequencer);
434                     removed = true;
435                 }
436             }
437         }
438         return removed;
439     }

463     private static class TableSequencerEntry extends TableSequencerImpl {
464         private final TableSequencerAPI pool;
465         private volatile long releaseTime = Long.MAX_VALUE;
466 
467         TableSequencerEntry(TableSequencerAPI pool, CairoEngine engine, TableToken tableToken) {
468             super(engine, tableToken);
469             this.pool = pool;
470         }
471 
472         @Override
473         public void close() {
474             if (!pool.closed) {
475                 if (!isDistressed() && !isDropped()) {
476                     releaseTime = pool.configuration.getMicrosecondClock().getTicks();
477                 } else {
478                     // Sequencer is distressed or dropped, close before removing from the pool.
479                     // Remove from registry only if this thread closed the instance.
480                     if (checkClose()) {
481                         LOG.info().$("closed distressed table sequencer [table=").$(getTableToken()).I$();
482                         pool.seqRegistry.remove(getTableToken().getDirName(), this);
483                     }
484                 }
485             } else {
486                 super.close();
487             }
488         }
489     }
