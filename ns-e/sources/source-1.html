43 /**
44  * ILP client to feed data to a remote QuestDB instance.
45  * <p>
46  * Use {@link #builder()} method to create a new instance.
47  * <br>
48  * How to use the Sender:
49  * <ol>
50  *     <li>Obtain an instance via {@link #builder()}</li>
51  *     <li>Use {@link #table(CharSequence)} to select a table</li>
52  *     <li>Use {@link #symbol(CharSequence, CharSequence)} to add all symbols. You must add symbols before adding other columns.</li>
53  *     <li>Use {@link #stringColumn(CharSequence, CharSequence)}, {@link #longColumn(CharSequence, long)},
54  *     {@link #doubleColumn(CharSequence, double)}, {@link #boolColumn(CharSequence, boolean)},
55  *     {@link #timestampColumn(CharSequence, long)} to add remaining columns columns</li>
56  *     <li>Use {@link #at(long)} (long)} to finish a row with an explicit timestamp.Alternatively, you can use use
57  *     {@link #atNow()} which will add a timestamp on a server.</li>
58  *     <li>Optionally: You can use {@link #flush()} to send locally buffered data into a server</li>
59  * </ol>
60  * <br>
61  * <p>
62  * Sender implements the <code>java.io.Closeable</code> interface. Thus, you must call the {@link #close()} method
63  * when you no longer need it.
64  * <br>
65  * Thread-safety: Sender is not thread-safe. Each thread-safe needs its own instance or you have to implement
66  * a mechanism for passing Sender instances among thread. An object pool could have this role.
67  * <br>
68  * Error-handling: Most errors throw an instance of {@link LineSenderException}.
69  */

248         /**
249          * Set address of a QuestDB server. It can be either a domain name or a textual representation of an IP address.
250          * Only IPv4 addresses are supported.
251          * <br>
252          * Optionally, you can also include a port. In this can you separate a port from the address by using a colon.
253          * Example: my.example.org:54321.
254          * <p>
255          * If you can include a port then you must not call {@link LineSenderBuilder#port(int)}.
256          *
257          * @param address address of a QuestDB server
258          * @return this instance for method chaining.
259          */
260         public LineSenderBuilder address(CharSequence address) {
261             if (this.host != null) {
262                 throw new LineSenderException("server address is already configured ")
263                         .put("[configured-address=").put(this.host).put("]");
264             }
265             if (address == null || address.length() == 0) {
266                 throw new LineSenderException("address cannot be empty nor null");
267             }
268             int portIndex = Chars.indexOf(address, ':');
269             if (portIndex + 1 == address.length()) {
270                 throw new LineSenderException("address cannot ends with : ")
271                         .put("[address=").put(address).put("]");
272             }
273             if (portIndex != -1) {
274                 if (port != 0) {
275                     throw new LineSenderException("address contains a port, but a port was already configured ")
276                             .put("[address=").put(address)
277                             .put(", configured-port=").put(port)
278                             .put("]");
279                 }
280                 this.host = address.subSequence(0, portIndex).toString();
281                 try {
282                     port = Numbers.parseInt(address, portIndex + 1, address.length());
283                 } catch (NumericException e) {
284                     throw new LineSenderException("cannot parse port from address ", e).
285                             put("[address=").put(address).put("]");
286                 }
287             } else {
288                 this.host = address.toString();
289             }
290             return this;
291         }

326         /**
327          * Build a Sender instance. This method construct a Sender instance.
328          *
329          * @return returns a configured instance of Sender.
330          */
331         public Sender build() {
332             configureDefaults();
333             validateParameters();
334 
335             NetworkFacade nf = NetworkFacadeImpl.INSTANCE;
336             LineChannel channel = new PlainTcpLineChannel(nf, host, port, bufferCapacity * 2);
337             LineTcpSender sender;
338             if (tlsEnabled) {
339                 assert (trustStorePath == null) == (trustStorePassword == null); //either both null or both non-null
340                 DelegatingTlsChannel tlsChannel;
341                 try {
342                     tlsChannel = new DelegatingTlsChannel(channel, trustStorePath, trustStorePassword, tlsValidationMode, host);
343                 } catch (Throwable t) {
344                     channel.close();
345                     throw rethrow(t);
346                 }
347                 channel = tlsChannel;
348             }
349             try {
350                 sender = new LineTcpSender(channel, bufferCapacity);
351             } catch (Throwable t) {
352                 channel.close();
353                 throw rethrow(t);
354             }
355             if (privateKey != null) {
356                 try {
357                     sender.authenticate(keyId, privateKey);
358                 } catch (Throwable t) {
359                     sender.close();
360                     throw rethrow(t);
361                 } finally {
362                     if (shouldDestroyPrivKey) {
363                         try {
364                             privateKey.destroy();
365                         } catch (DestroyFailedException e) {
366                             // not much we can do
367                         }
368                     }
369                 }
370             }
371             return sender;
372         }

433         private void validateParameters() {
434             if (host == null) {
435                 throw new LineSenderException("questdb server address not set");
436             }
437             if (!tlsEnabled && trustStorePath != null) {
438                 throw new LineSenderException("custom trust store configured, but TLS was not enabled ")
439                         .put("[configured-path=").put(LineSenderBuilder.this.trustStorePath).put("]");
440             }
441             if (!tlsEnabled && tlsValidationMode != TlsValidationMode.DEFAULT) {
442                 throw new LineSenderException("TSL validation disabled, but TLS was not enabled");
443             }
444             if (keyId != null && bufferCapacity < MIN_BUFFER_SIZE_FOR_AUTH) {
445                 throw new LineSenderException("Requested buffer too small ")
446                         .put("[minimal-capacity=").put(MIN_BUFFER_SIZE_FOR_AUTH)
447                         .put(", requested-capacity=").put(bufferCapacity)
