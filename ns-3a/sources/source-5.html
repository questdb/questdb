50     /**
51      * Generates byte code for record comparator. To avoid frequent calls to
52      * record field getters comparator caches values of left argument.
53      *
54      * @param columnTypes      types of columns in the cursor. All but BINARY types are supported
55      * @param keyColumnIndices indexes of columns in types object. Column indexes are 1-based.
56      *                         Index sign indicates direction of sort: negative - descending,
57      *                         positive - ascending.
58      * @return RecordComparator instance.
59      */
60     public RecordComparator compile(ColumnTypes columnTypes, @Transient IntList keyColumnIndices) {
61 
62         assert keyColumnIndices.size() < SqlParser.MAX_ORDER_BY_COLUMNS;
63 
64         asm.init(RecordComparator.class);
65         asm.setupPool();
66 
67         int stackMapTableIndex = asm.poolUtf8("StackMapTable");
68         int thisClassIndex = asm.poolClass(asm.poolUtf8("io/questdb/griffin/engine/comparator"));
69         int interfaceClassIndex = asm.poolClass(RecordComparator.class);
70         int recordClassIndex = asm.poolClass(Record.class);
71         // this is name re-use, it used on all static interfaces that compare values
72         int compareNameIndex = asm.poolUtf8("compare");
73         // our compare method signature
74         int compareDescIndex = asm.poolUtf8("(Lio/questdb/cairo/sql/Record;)I");
75         poolFieldArtifacts(compareNameIndex, thisClassIndex, recordClassIndex, columnTypes, keyColumnIndices);
76         // elements for setLeft() method
77         int setLeftNameIndex = asm.poolUtf8("setLeft");
78         int setLeftDescIndex = asm.poolUtf8("(Lio/questdb/cairo/sql/Record;)V");
79 
80         asm.finishPool();
81         asm.defineClass(thisClassIndex);
82         asm.interfaceCount(1);
83         asm.putShort(interfaceClassIndex);
84         asm.fieldCount(fieldNameIndices.size());
85         for (int i = 0, n = fieldNameIndices.size(); i < n; i++) {
86             asm.defineField(fieldNameIndices.getQuick(i), fieldTypeIndices.getQuick(i));
87         }
88         asm.methodCount(3);
89         asm.defineDefaultConstructor();
90         instrumentSetLeftMethod(setLeftNameIndex, setLeftDescIndex, keyColumnIndices, columnTypes);
91         instrumentCompareMethod(stackMapTableIndex, compareNameIndex, compareDescIndex, keyColumnIndices, columnTypes);
92 
93         // class attribute count
94         asm.putShort(0);
95         return asm.newInstance();
96     }

98     private void instrumentCompareMethod(int stackMapTableIndex, int nameIndex, int descIndex, IntList keyColumns, ColumnTypes columnTypes) {
99         branches.clear();
100         int sz = keyColumns.size();
101         int maxStack = sz + (fieldIndices.size() > sz ? 4 : 0) + 3;
102         asm.startMethod(nameIndex, descIndex, maxStack, 3);
103 
104         int fieldIndex = 0;
105         for (int i = 0; i < sz; i++) {
106             if (i > 0) {
107                 asm.iload(2);
108                 // last one does not jump
109                 branches.add(asm.ifne());
110             }
111             asm.aload(0);
112             asm.getfield(fieldIndices.getQuick(fieldIndex++));
113             asm.aload(1);
114             int index = keyColumns.getQuick(i);
115             int columnIndex = (index > 0 ? index : -index) - 1;
116             asm.iconst(columnIndex);
117             asm.invokeInterface(fieldRecordAccessorIndicesA.getQuick(i), 1);
118 
119             if (columnTypes.getColumnType(columnIndex) == ColumnType.LONG128 || columnTypes.getColumnType(columnIndex) == ColumnType.UUID) {
120                 asm.aload(0);
121                 asm.getfield(fieldIndices.getQuick(fieldIndex++));
122                 asm.aload(1);
123                 asm.iconst(columnIndex);
124                 asm.invokeInterface(fieldRecordAccessorIndicesB.getQuick(i), 1);
125             }
126 
127             asm.invokeStatic(comparatorAccessorIndices.getQuick(i));
128             if (index < 0) {
129                 asm.ineg();
130             }
131             asm.istore(2);
132         }
133         int p = asm.position();
134         asm.iload(2);
135         asm.ireturn();
136 
137         // update ifne jumps to jump to "p" position
138         for (int i = 0, n = branches.size(); i < n; i++) {
139             asm.setJmp(branches.getQuick(i), p);
140         }
141 
142         asm.endMethodCode();
143         // exceptions
144         asm.putShort(0);
145 
146         // we have to add stack map table as branch target
147         // jvm requires it
148 
149         // attributes: 1 - StackMapTable
150         asm.putShort(1);
151         // verification to ensure that return type is int and there is correct
152         // value present on stack
153         asm.startStackMapTables(stackMapTableIndex, 1);
154         // frame type APPEND
155         asm.append_frame(1, p - asm.getCodeStart());
156         asm.putITEM_Integer();
157         asm.endStackMapTables();
158         asm.endMethod();
159     }

161     /*
162      * setLeft(Record)
163      *
164      * This code generates method setLeft(Record), which assigns selected fields from
165      * the record to class fields. Generally this method looks like:
166      * f1 = record.getInt(3);
167      * f2 = record.getFlyweightStr(5);
168      *
169      * as you can see record fields are accessed by constants, to speed up the process.
170      *
171      * Class like that would translate to bytecode:
172      *
173      * aload_0
174      * aload_1
175      * invokeinterface index
176      * putfield index
177      * ...
178      *
179      * and so on for each field
180      * bytecode finishes with
181      *
182      * return
183      *
184      * all this complicated dancing around is to have class names, method names, field names
185      * method signatures in constant pool in bytecode.
186      */
187     private void instrumentSetLeftMethod(int nameIndex, int descIndex, IntList keyColumns, ColumnTypes columnTypes) {
188         asm.startMethod(nameIndex, descIndex, 3, 2);
189         int fieldIndex = 0;
190         for (int i = 0, n = keyColumns.size(); i < n; i++) {
191             asm.aload(0);
192             asm.aload(1);
193             int index = keyColumns.getQuick(i);
194             // make sure column index is valid in case of "descending sort" flag
195             int columnIndex = (index > 0 ? index : -index) - 1;
196             asm.iconst(columnIndex);
197             asm.invokeInterface(fieldRecordAccessorIndicesB.getQuick(i), 1);
198             asm.putfield(fieldIndices.getQuick(fieldIndex++));
199 
200             if (columnTypes.getColumnType(columnIndex) == ColumnType.LONG128 || columnTypes.getColumnType(columnIndex) == ColumnType.UUID) {
201                 asm.aload(0);
202                 asm.aload(1);
203                 asm.iconst(columnIndex);
204                 asm.invokeInterface(fieldRecordAccessorIndicesA.getQuick(i), 1);
205                 asm.putfield(fieldIndices.getQuick(fieldIndex++));
206             }
207         }
208         asm.return_();
209         asm.endMethodCode();
210         // exceptions
211         asm.putShort(0);
212         // attributes
213         asm.putShort(0);
214         asm.endMethod();
215     }

267                 case ColumnType.FLOAT:

272                 case ColumnType.INT:

277                 case ColumnType.LONG:

287                 case ColumnType.TIMESTAMP:

292                 case ColumnType.SHORT:

294                     getterNameA = "getShort";

296                     break;
297                 case ColumnType.CHAR:

299                     getterNameA = "getChar";

302                 case ColumnType.STRING:

312                     comparatorClass = Long256Util.class;

314                 case ColumnType.UUID:
315                 case ColumnType.LONG128:
316                     getterNameA = "getLong128Hi";

320                     comparatorClass = Long128.class;

322                 default:
323                     // SYMBOL
324                     getterNameA = "getSym";
325                     getterNameB = "getSymB";
326                     fieldType = "Ljava/lang/CharSequence;";
327                     comparatorClass = Chars.class;
328                     comparatorDesc = "(Ljava/lang/CharSequence;Ljava/lang/CharSequence;)I";

332             int keyIndex;
333             int nameIndex;
334             int typeIndex;
335 
336             keyIndex = typeMap.keyIndex(fieldType);
337             if (keyIndex > -1) {

339             } else {

341             }
342

344             fieldNameIndices.add(nameIndex = asm.poolUtf8().put('f').put(i).$());
345             fieldIndices.add(asm.poolField(thisClassIndex, asm.poolNameAndType(nameIndex, typeIndex)));

353                 fieldNameIndices.add(nameIndex2);
354                 int nameAndTypeIndex = asm.poolNameAndType(nameIndex2, typeIndex);
355                 fieldIndices.add(asm.poolField(thisClassIndex, nameAndTypeIndex));

361             methodMap.putIfAbsent(getterNameA, methodIndex = asm.poolInterfaceMethod(recordClassIndex, getterIndex));
362             fieldRecordAccessorIndicesA.add(methodIndex);
363 
364             if (getterNameB != null) {
365                 getterNameIndex = asm.poolUtf8(getterNameB);
366                 getterIndex = asm.poolNameAndType(getterNameIndex, getterSigIndex);
367                 methodMap.putIfAbsent(getterNameB, methodIndex = asm.poolInterfaceMethod(recordClassIndex, getterIndex));
368             }
369 
370             fieldRecordAccessorIndicesB.add(methodIndex);
371             comparatorAccessorIndices.add(
372                     asm.poolMethod(asm.poolClass(comparatorClass),
373                             asm.poolNameAndType(
374                                     compareMethodIndex, comparatorDesc == null ?
375                                             asm.poolUtf8().put('(').put(fieldType).put(fieldType).put(")I").$()
376                                             :
377                                             asm.poolUtf8(comparatorDesc))
378                     ));
379         }
380     }
