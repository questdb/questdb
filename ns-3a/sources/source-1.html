40 /**
41  * LongTreeChain with a size limit - used to keep only the necessary records
42  * instead of whole result set for queries with "limit L | limit L, H"  clause.
43  * <p>
44  * 1. "limit L" means we only need to keep :
45  * <p>
46  * L &gt;=0 - first L records
47  * L &lt; 0 - last L records
48  * 2. "limit L, H" means we need to keep :
49  * L &lt; 0          - last  L records (but skip last H records, if H &gt;=0 then don't skip anything    )
50  * L &gt;= 0, H &gt;= 0 - first H records (but skip first L later, if H &lt;= L then return empty set)
51  * L &gt;= 0, H &lt; 0  - we can't optimize this case (because it spans from record L-th from the beginning up to
52  * H-th from the end, and we don't  ) and need to revert to default behavior -
53  * produce the whole set and skip.
54  * </p>
55  * <p>
56  * TreeChain stores repeating values (rowids) on valueChain as a linked list :
57  * [latest rowid, offset to next] -&gt; [old rowid, offset to next] -&gt; [oldest rowid, -1L ]
58  * -1L marks end of current node's value chain .
59  * -2  marks an unused element on the value chain list for the current tree node
60  * but should only happen once . It's meant to limit value chain allocations on delete/insert.
61  * </p>
62  */

120                      Record placeholder,
121                      RecordComparator comparator) {
122         comparator.setLeft(searchedRecord);

132             cmp = comparator.compare(placeholder);
133             if (cmp < 0) {
134                 p = leftOf(p);

167     public void put(
168             Record leftRecord,
169             RecordCursor sourceCursor,
170             Record rightRecord,
171             RecordComparator comparator
172     ) {
173         if (maxValues == 0) {
174             return;
175         }
176 
177         comparator.setLeft(leftRecord);
178 
179         //if maxValues < 0 then there's no limit (unless there's more than 2^64 records, which is unlikely)
180         if (maxValues == currentValues) {
181             sourceCursor.recordAt(rightRecord, minMaxRowId);
182             int cmp = comparator.compare(rightRecord);
183 
184             if (isFirstN && cmp >= 0) {//bigger than max for firstN/bottomN
185                 return;
186             } else if (!isFirstN && cmp <= 0) { //smaller than min for lastN/topN
187                 return;
188             } else { //record has to be inserted so we've to remove current minMax
189                 removeAndCache(minMaxNode);
190             }
191         }
192 
193         if (root == EMPTY) {
194             putParent(leftRecord.getRowId());
195             minMaxNode = root;
196             minMaxRowId = leftRecord.getRowId();
197             currentValues++;
198             return;
199         }
200 
201         long p = root;
202         long parent;
203         int cmp;
204         do {
205             parent = p;
206             final long r = refOf(p);
207             sourceCursor.recordAt(rightRecord, valueChain.getLong(r));
208             cmp = comparator.compare(rightRecord);
209             if (cmp < 0) {
210                 p = leftOf(p);
211             } else if (cmp > 0) {
212                 p = rightOf(p);
213             } else {
214                 setRef(p, appendValue(leftRecord.getRowId(), r)); //appends value to chain, minmax shouldn't change
215                 if (minMaxRowId == -1) {
216                     refreshMinMaxNode();
217                 }
218                 currentValues++;
219                 return;
220             }
221         } while (p > -1);
222 
223         p = allocateBlock(parent, leftRecord.getRowId());
224 
225         if (cmp < 0) {
226             setLeft(parent, p);
227         } else {
228             setRight(parent, p);
229         }
230 
231         fixInsert(p);
232         refreshMinMaxNode();
233         currentValues++;
234     }

301     //if not empty - reuses most recently deleted node from freelist; otherwise allocates a new node
302     protected long allocateBlock(long parent, long recordRowId) {
303         if (freeList.size() > 0) {
304             long freeNode = freeList.get(freeList.size() - 1);
305             freeList.removeIndex(freeList.size() - 1);
306 
307             setParent(freeNode, parent);
308             valueChain.putLong(refOf(freeNode), recordRowId);
309 
310             return freeNode;
311         } else {
312             long newNode = super.allocateBlock();
313             setParent(newNode, parent);
314 
315             long chainOffset;
316 
317             if (chainFreeList.size() > 0) {
318                 chainOffset = chainFreeList.get(chainFreeList.size() - 1);
319                 chainFreeList.removeIndex(chainFreeList.size() - 1);
320                 valueChain.putLong(chainOffset, recordRowId);
321                 valueChain.putLong(chainOffset + 8, CHAIN_END);
322             } else {
323                 chainOffset = appendValue(recordRowId, CHAIN_END);
324             }
325 
326             setRef(newNode, chainOffset);
327 
328             return newNode;
329         }

373     //remove node and put on freelist (if holds only one value in chain)
374     protected void removeAndCache(long node) {
375         if (hasMoreThanOneValue(node)) {
376             removeMostRecentChainValue(node);//don't change minmax
377         } else {
378             long nodeToRemove = super.remove(node);
379             clearBlock(nodeToRemove);
380             freeList.add(nodeToRemove);//keep node on freelist to minimize allocations
381 
382             minMaxRowId = -1;//re-compute after inserting, there's no point doing it now
383             minMaxNode = -1;
384         }
385 
386         currentValues--;
387     }
