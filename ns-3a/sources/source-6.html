95     public void put(Record record) {
96         if (root == -1) {
97             putParent(record);
98             return;
99         }
100 
101         comparator.setLeft(record);
102 
103         long p = root;
104         long parent;
105         int cmp;
106         do {
107             parent = p;
108             long r = refOf(p);
109             recordChain.recordAt(recordChainRecord, r);
110             cmp = comparator.compare(recordChainRecord);
111             if (cmp < 0) {
112                 p = leftOf(p);
113             } else if (cmp > 0) {
114                 p = rightOf(p);
115             } else {
116                 setRef(p, recordChain.put(record, r));
117                 return;
118             }
119         } while (p > -1);
120 
121         p = allocateBlock();
122         setParent(p, parent);
123         long r = recordChain.put(record, -1L);
124         setTop(p, r);
125         setRef(p, r);
126 
127         if (cmp < 0) {
128             setLeft(parent, p);
129         } else {
130             setRight(parent, p);
131         }
132         fix(p);
133     }

193         long p = rightOf(current);

274         if (p != -1) {
275             long r = rightOf(p);
276             setRight(p, leftOf(r));
277             if (leftOf(r) != -1) {
278                 setParent(leftOf(r), p);
279             }
280             setParent(r, parentOf(p));
281             if (parentOf(p) == -1) {
282                 root = r;
283             } else if (leftOf(parentOf(p)) == p) {
284                 setLeft(parentOf(p), r);
285             } else {
286                 setRight(parentOf(p), r);
287             }
288             setLeft(r, p);
289             setParent(p, r);
290         }
