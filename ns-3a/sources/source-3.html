75     @Override
76     public RecordCursor getCursor(SqlExecutionContext executionContext) throws SqlException {
77         boolean preTouchEnabled = executionContext.isColumnPreTouchEnabled();
78         // Forcefully disable column pre-touch for LIMIT K,N queries for all downstream
79         // async filtered factories to avoid redundant disk reads.
80         executionContext.setColumnPreTouchEnabled(preTouchEnabled && hiFunction == null);
81         RecordCursor baseCursor = null;
82         try {
83             baseCursor = base.getCursor(executionContext);
84             initialize(executionContext, baseCursor);
85             cursor.of(baseCursor, executionContext);
86             return cursor;
87         } catch (Throwable ex) {
88             Misc.free(baseCursor);
89             Misc.free(cursor);
90             throw ex;
91         } finally {
92             executionContext.setColumnPreTouchEnabled(preTouchEnabled);
93         }
94     }

101     /*
102      * 1. "limit L" means we only need to keep :
103      * L >=0 - first L records
104      * L < 0 - last L records
105      * 2. "limit L, H" means we need to keep :
106      * L < 0          - last  L records (but skip last H records, if H >=0 then don't skip anything)
107      * L >= 0, H >= 0 - first H records (but skip first L later, if H <= L then return empty set)
108      * L >= 0, H < 0  - we can't optimize this case (because it spans from record L-th from the beginning up to
109      * H-th from the end, and we don't) and need to revert to default behavior - produce the whole set and skip.
110      * <p>
111      * Similar to LimitRecordCursorFactory.LimitRecordCursor, but doesn't check the underlying count.
112      */
113     public void initializeLimitedSizeCursor(SqlExecutionContext executionContext, RecordCursor base) throws SqlException {
114         loFunction.init(base, executionContext);
115         if (hiFunction != null) {
116             hiFunction.init(base, executionContext);
117         }
118 
119         long skipFirst = 0, skipLast = 0, limit;
120         boolean isFirstN = false;
121 
122         long lo = loFunction.getLong(null);
123         if (lo < 0 && hiFunction == null) {
124             // last N rows
125             // lo is negative, -5 for example
126             // if we have 12 records we need to skip 12-5 = 7
127             // if we have 4 records = return all of them
128             // set limit to return remaining rows
129             limit = -lo;
130         } else if (lo > -1 && hiFunction == null) {
131             // first N rows
132             isFirstN = true;
133             limit = lo;
134         } else {
135             // at this stage we also have 'hi'
136             long hi = hiFunction.getLong(null);
137             if (lo < 0) {
138                 // right, here we are looking for something like -10,-5 five rows away from tail
139                 if (lo < hi) {
140                     limit = -lo;
141                     skipLast = Math.max(-hi, 0);
142                     //}
143                 } else {
144                     // this is invalid bottom range, for example -3, -10
145                     limit = 0;//produce empty result
146                 }
147             } else { //lo >= 0
148                 if (hi < 0) {
149                     //if lo>=0 but hi<0 then we fall back to standard algorithm because we can't estimate result size
150                     // (it's from lo up to end-hi so probably whole result anyway )
151                     limit = -1;
152                     skipFirst = lo;
153                     skipLast = -hi;
154                 } else { //both lo and hi are positive
155                     if (hi <= lo) {
156                         limit = 0;//produce empty result
157                     } else {
158                         isFirstN = true;
159                         limit = hi;
160                         //but we've to skip to lo
161                         skipFirst = lo;
162                     }
163                 }
164             }
165         }
166 
167         LimitedSizeLongTreeChain chain = new LimitedSizeLongTreeChain(
168                 configuration.getSqlSortKeyPageSize(),
169                 configuration.getSqlSortKeyMaxPages(),
170                 configuration.getSqlSortLightValuePageSize(),
171                 configuration.getSqlSortLightValueMaxPages(),
172                 isFirstN,
173                 limit
174         );
175 
176         this.cursor = new LimitedSizeSortedLightRecordCursor(chain, comparator, limit, skipFirst, skipLast);
177     }

195     // Check if lo, hi is set and lo >=0 while hi < 0 (meaning - return whole result set except some rows at start and some at the end)
196     // because such case can't really be optimized by topN/bottomN
197     private boolean canBeOptimized(RecordCursor baseCursor, SqlExecutionContext executionContext) throws SqlException {
198         loFunction.init(baseCursor, executionContext);
199         if (hiFunction != null) {
200             hiFunction.init(baseCursor, executionContext);
201         }
202 
203         return !(loFunction.getLong(null) >= 0 &&
204                 hiFunction != null &&
205                 hiFunction.getLong(null) < 0);
206     }

208     private void initialize(SqlExecutionContext executionContext, RecordCursor baseCursor) throws SqlException {
209         if (isInitialized()) {
210             return;
211         }
212 
213         if (canBeOptimized(baseCursor, executionContext)) {
214             initializeLimitedSizeCursor(executionContext, baseCursor);
215         } else {
216             initializeUnlimitedSizeCursor();
217         }
218     }
