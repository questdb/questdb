33 class SortedLightRecordCursor implements DelegatingRecordCursor {
34     private final LongTreeChain chain;
35     private final LongTreeChain.TreeCursor chainCursor;
36     private final RecordComparator comparator;
37     private RecordCursor base;
38     private Record baseRecord;
39     private SqlExecutionCircuitBreaker circuitBreaker;
40     private boolean isChainBuilt;
41     private boolean isOpen;
42 
43     public SortedLightRecordCursor(LongTreeChain chain, RecordComparator comparator) {
44         this.chain = chain;
45         this.comparator = comparator;
46         // assign it once, it's the same instance anyway
47         this.chainCursor = chain.getCursor();
48         this.isOpen = true;
49     }
50 
51     @Override
52     public void close() {
53         if (isOpen) {
54             isOpen = false;
55             Misc.free(chain);
56             base = Misc.free(base);
57             baseRecord = null;
58         }
59     }
60 
61     @Override
62     public Record getRecord() {
63         return baseRecord;
64     }
65 
66     @Override
67     public Record getRecordB() {
68         return base.getRecordB();
69     }
70 
71     @Override
72     public SymbolTable getSymbolTable(int columnIndex) {
73         return base.getSymbolTable(columnIndex);
74     }
75 
76     @Override
77     public boolean hasNext() {
78         if (!isChainBuilt) {
79             buildChain();
80             isChainBuilt = true;
81         }
82         if (chainCursor.hasNext()) {
83             base.recordAt(baseRecord, chainCursor.next());
84             return true;
85         }
86         return false;
87     }
88 
89     @Override
90     public SymbolTable newSymbolTable(int columnIndex) {
91         return base.newSymbolTable(columnIndex);
92     }
93 
94     @Override
95     public void of(RecordCursor base, SqlExecutionContext executionContext) {
96         if (!isOpen) {
97             chain.reopen();
98             isOpen = true;
99         }
100 
101         this.base = base;
102         baseRecord = base.getRecord();
103         circuitBreaker = executionContext.getCircuitBreaker();
104         isChainBuilt = false;
105     }
106 
107     @Override
108     public void recordAt(Record record, long atRowId) {
109         base.recordAt(record, atRowId);
110     }
111 
112     @Override
113     public long size() {
114         return base.size();
115     }
116 
117     @Override
118     public void toTop() {
119         chainCursor.toTop();
120     }
121 
122     private void buildChain() {
123         final Record placeHolderRecord = base.getRecordB();
124         while (base.hasNext()) {
125             circuitBreaker.statefulThrowExceptionIfTripped();
126             // Tree chain is liable to re-position record to
127             // other rows to do record comparison. We must use our
128             // own record instance in case base cursor keeps
129             // state in the record it returns.
130             chain.put(
131                     baseRecord,
132                     base,
133                     placeHolderRecord,
134                     comparator
135             );
136         }
137         toTop();
138     }
