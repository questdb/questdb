130     private static class ConcatFunction extends StrFunction implements MultiArgFunction {
131         private final ObjList<TypeAdapter> adaptors;
132         private final int functionCount;
133         private final ObjList<Function> functions;
134         private final StringSink sinkA = new StringSink();

137         public ConcatFunction(ObjList<Function> functions) {
138             this.functions = functions;
139             this.functionCount = functions.size();
140             this.adaptors = new ObjList<>(functionCount);
141             for (int i = 0; i < functionCount; i++) {
142                 adaptors.add(adapterReferences.getQuick(functions.getQuick(i).getType()));
143             }
144         }
145 
146         @Override
147         public ObjList<Function> getArgs() {
148             return functions;
149         }
150 
151         @Override
152         public void getStr(Record rec, CharSink sink) {
153             for (int i = 0; i < functionCount; i++) {
154                 adaptors.getQuick(i).sink(sink, functions.getQuick(i), rec);
155             }
156         }
157
