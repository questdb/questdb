114         @Nullable
115         private StringSink rPad(CharSequence str, int len, StringSink sink) {
116             if (str != null && len >= 0) {
117                 if (len > maxLength) {
118                     throw CairoException.nonCritical()
119                             .put("breached memory limit set for ").put(SIGNATURE)
120                             .put(" [maxLength=").put(maxLength)
121                             .put(", requiredLength=").put(len).put(']');
122                 }
123                 sink.clear();
124                 if (len > str.length()) {
125                     sink.put(str);
126                     for (int i = 0; i < (len - str.length()); i++) {
127                         sink.put(' ');
128                     }
129                 } else {
130                     sink.put(str, str.length() - len, str.length());
131                 }
132                 return sink;
133             }
134             return null;
135         }
