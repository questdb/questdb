118         @Nullable
119         private StringSink lPadStr(CharSequence str, int len, CharSequence fillText, StringSink sink) {
120             if (str != null && len >= 0 && fillText != null && fillText.length() > 0) {
121                 if (len > maxLength) {
122                     throw CairoException.nonCritical()
123                             .put("breached memory limit set for ").put(SIGNATURE)

126                 }
127                 sink.clear();
128                 if (len > str.length()) {
129                     final int fillTextLen = fillText.length();
130                     for (int i = 0, n = (len - str.length()) / fillTextLen; i < n; i++) {
131                         sink.put(fillText);
132                     }
133                     for (int i = 0, n = (len - str.length()) % fillTextLen; i < n; i++) {
134                         sink.put(fillText.charAt(i));
135                     }
136                     sink.put(str);
137                 } else {
138                     sink.put(str, 0, len);
139                 }
140                 return sink;
141             }
