52         for (int i = 1; i < n; i++) {
53             Function func = args.getQuick(i);
54             switch (ColumnType.tagOf(func.getType())) {
55                 case ColumnType.NULL:
56                 case ColumnType.STRING:
57                 case ColumnType.SYMBOL:
58                     CharSequence value = func.getStr(null);
59                     if (value == null) {
60                         throw SqlException.$(argPositions.getQuick(i), "NULL is not allowed in IN list");
61                     }
62                     try {
63                         Uuid.checkDashesAndLength(value);
64                         set.add(Uuid.parseLo(value), Uuid.parseHi(value));
65                     } catch (NumericException e) {
66                         // the given string is not a valid UUID -> no UUID value can match it -> we can ignore it
67                     }
68                     break;
69                 case ColumnType.UUID:
70                     long lo = func.getLong128Lo(null);
71                     long hi = func.getLong128Hi(null);
72                     if (hi == Numbers.LONG_NaN && lo == Numbers.LONG_NaN) {
73                         throw SqlException.$(argPositions.getQuick(i), "NULL is not allowed in IN list");
74                     }
75                     set.add(lo, hi);
76                     break;
77                 default:
78                     throw SqlException.$(argPositions.getQuick(i), "STRING or UUID constant expected in IN list");
79             }
80         }
81         Function var = args.getQuick(0);
82         if (var.isConstant()) {
83             long lo = var.getLong128Lo(null);
