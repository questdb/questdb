57     public Function newInstance(int position, ObjList<Function> args, IntList argPositions, CairoConfiguration configuration, SqlExecutionContext sqlExecutionContext) throws SqlException {
58         boolean allConst = true;
59         for (int i = 1, n = args.size(); i < n; i++) {
60             Function func = args.getQuick(i);
61             switch (ColumnType.tagOf(func.getType())) {
62                 case ColumnType.NULL:
63                 case ColumnType.TIMESTAMP:
64                 case ColumnType.LONG:
65                 case ColumnType.INT:
66                 case ColumnType.STRING:
67                 case ColumnType.SYMBOL:
68                     break;
69                 default:
70                     throw SqlException.position(0).put("cannot compare TIMESTAMP with type ").put(ColumnType.nameOf(func.getType()));
71             }
72             if (!func.isConstant()) {
73                 allConst = false;
74                 break;
75             }
76         }
77 
78         if (allConst) {
79             return new InTimestampConstFunction(args.getQuick(0), parseToTs(args, argPositions));
80         }
81 
82         if (args.size() == 2 && ColumnType.isString(args.get(1).getType())) {
83             // special case - one argument and it a string
84             return new InTimestampStrFunctionFactory.EqTimestampStrFunction(args.get(0), args.get(1));
85         }
86 
87         // have to copy, args is mutable
88         return new InTimestampVarFunction(new ObjList<>(args));
89     }

91     private LongList parseToTs(ObjList<Function> args, IntList argPositions) throws SqlException {
92         LongList res = new LongList(args.size() - 1);
93         res.extendAndSet(args.size() - 2, 0);
94 
95         for (int i = 1, n = args.size(); i < n; i++) {
96             Function func = args.getQuick(i);
97             long val = Numbers.LONG_NaN;
98             switch (ColumnType.tagOf(func.getType())) {
99                 case ColumnType.TIMESTAMP:
100                 case ColumnType.LONG:
101                 case ColumnType.INT:
102                     val = func.getTimestamp(null);
103                     break;
104                 case ColumnType.STRING:
105                 case ColumnType.SYMBOL:
106                     CharSequence tsValue = func.getStr(null);
107                     val = (tsValue != null) ? tryParseTimestamp(tsValue, argPositions.getQuick(i)) : Numbers.LONG_NaN;
108                     break;
109             }
110             res.setQuick(i - 1, val);
111         }
112 
113         res.sort();
114         return res;
115     }
