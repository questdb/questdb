55             SqlExecutionContext sqlExecutionContext
56     ) throws SqlException {
57         CharSequenceHashSet set = new CharSequenceHashSet();
58         int n = args.size();
59 
60         if (n == 1) {
61             return BooleanConstant.FALSE;
62         }
63 
64         ObjList<Function> deferredValues = null;
65         for (int i = 1; i < n; i++) {
66             Function func = args.getQuick(i);
67             switch (ColumnType.tagOf(func.getType())) {
68                 case ColumnType.STRING:
69                     if (func.isRuntimeConstant()) {
70                         // string bind variable case
71                         if (deferredValues == null) {
72                             deferredValues = new ObjList<>();
73                         }
74                         deferredValues.add(func);
75                         continue;
76                     }
77                     // fall through
78                 case ColumnType.SYMBOL:
79                 case ColumnType.NULL:
80                     CharSequence value = func.getStr(null);
81                     if (value == null) {
82                         set.add(null);
83                     } else {
84                         set.add(Chars.toString(value));
85                     }
86                     break;
87                 case ColumnType.CHAR:
88                     set.add(String.valueOf(func.getChar(null)));
89                     break;
90                 default:
91                     throw SqlException.$(argPositions.getQuick(i), "STRING constant expected");
92             }
93         }
94         SymbolFunction var = (SymbolFunction) args.getQuick(0);
95         if (var.isConstant() && deferredValues == null) {
96             // Fast path for all constants case.
97             return BooleanConstant.of(set.contains(var.getSymbol(null)));
98         }
99         return new Func(var, set, deferredValues);
100     }
