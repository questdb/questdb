154     private void checkAndExtend(long address) {
155         assert appendAddress <= lim;
156         assert address >= pageAddress;
157         if (address <= lim) {
158             return;
159         }
160         extend0(address - pageAddress);
161     }
162 
163     private void extend0(long size) {
164         long nPages = size > 0 ? ((size - 1) >>> sizeMsb) + 1 : 1;
165         size = nPages << sizeMsb;
166         final long oldSize = size();
167 
168         // sometimes the resize request ends up being the same
169         // as existing memory size
170         if (size == oldSize) {
171             return;
172         }
173 
174         if (nPages > maxPages) {
175             throw LimitOverflowException.instance().put("Maximum number of pages (").put(maxPages).put(") breached in VirtualMemory");
176         }
177         final long newBaseAddress = reallocateMemory(pageAddress, size(), size);
178         if (oldSize > 0) {
179             LOG.debug().$("extended [oldBase=").$(pageAddress).$(", newBase=").$(newBaseAddress).$(", oldSize=").$(oldSize).$(", newSize=").$(size).$(']').$();
180         }
181         handleMemoryReallocation(newBaseAddress, size);
182     }
183 
184     protected final void handleMemoryReallocation(long newBaseAddress, long newSize) {
185         assert newBaseAddress != 0;
186         long appendOffset = appendAddress - pageAddress;
187         pageAddress = newBaseAddress;
188         lim = pageAddress + newSize;
189         appendAddress = pageAddress + appendOffset;
190         if (appendAddress > lim) {
191             appendAddress = lim;
192         }
193         this.size = newSize;
194     }
195 
196     protected final void handleMemoryReleased() {
197         pageAddress = 0;
198         lim = 0;
199         appendAddress = 0;
200         size = 0;
201     }

203     protected long reallocateMemory(long currentBaseAddress, long currentSize, long newSize) {
204         if (currentBaseAddress != 0) {
205             return Unsafe.realloc(currentBaseAddress, currentSize, newSize, memoryTag);
206         }
207         return Unsafe.malloc(newSize, memoryTag);
208     }
