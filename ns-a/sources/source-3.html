206         if (pageAddress != 0) {

208             final long fileSize = ff.length(fd);
209             long sz = Math.min(fileSize, minMappedMemorySize);
210             try {
211                 // we are remapping file to make it smaller, should not need
212                 // to allocate space; we already have it
213                 this.pageAddress = TableUtils.mremap(
214                         ff,

221             } catch (Throwable e) {

224                 if (truncatedToSize != 0) {
225                     if (truncatedToSize > 0) {
226                         Vect.memset(pageAddress, truncatedToSize, 0);
227                         this.size = sz;
228                     } else {

230                     }

237             this.lim = pageAddress + sz;

268     private void extend0(long newSize) {
269         long nPages = (newSize >>> extendSegmentMsb) + 1;
270         newSize = nPages << extendSegmentMsb;
271         long offset = appendAddress - pageAddress;
272         long previousSize = size;
273         assert size > 0;
274         TableUtils.allocateDiskSpace(ff, fd, newSize);
275         try {
276             this.pageAddress = TableUtils.mremap(
277                     ff,
278                     fd,
279                     this.pageAddress,
280                     previousSize,
281                     newSize,
282                     Files.MAP_RW,
283                     memoryTag
284             );
285             ff.madvise(pageAddress, newSize, madviseOpts);
286         } catch (Throwable e) {
287             appendAddress = pageAddress + previousSize;
288             close(false);
289             throw e;
290         }
291         size = newSize;
292         lim = pageAddress + newSize;
293         appendAddress = pageAddress + offset;
294     }

313     protected void map(FilesFacade ff, @Nullable CharSequence name, long size, int memoryTag) {
314         this.memoryTag = memoryTag;
315         // file either did not exist when length() was called or empty
316         if (size < 1) {
