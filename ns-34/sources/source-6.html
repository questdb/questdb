134         private long touchMemory(long pageSize, long baseAddress, long memorySize) {
135             final long pageCount = (memorySize + pageSize - 1) / pageSize;
136 
137             for (long i = 0; i < pageCount; i++) {

139                 // Use the same blackhole as in async offload's column pre-touch.

146         private void touchTable() {
147             clearCounters();
148             final long pageSize = Files.PAGE_SIZE;
149             // factory belongs to the function, do not close
150             final RecordCursorFactory recordCursorFactory = arg.getRecordCursorFactory();
151             try (PageFrameCursor pageFrameCursor = recordCursorFactory.getPageFrameCursor(sqlExecutionContext, ORDER_ASC)) {
152                 PageFrame frame;
153                 RecordMetadata metadata = recordCursorFactory.getMetadata();
154                 while ((frame = pageFrameCursor.next()) != null) {
155                     for (int columnIndex = 0, sz = metadata.getColumnCount(); columnIndex < sz; columnIndex++) {
156 
157                         final long columnMemorySize = frame.getPageSize(columnIndex);
158                         final long columnBaseAddress = frame.getPageAddress(columnIndex);
159                         dataPages += touchMemory(pageSize, columnBaseAddress, columnMemorySize);
160 
161                         if (metadata.isColumnIndexed(columnIndex)) {
162                             final BitmapIndexReader indexReader = frame.getBitmapIndexReader(columnIndex, BitmapIndexReader.DIR_BACKWARD);
163 
164                             final long keyBaseAddress = indexReader.getKeyBaseAddress();
165                             final long keyMemorySize = indexReader.getKeyMemorySize();
166                             indexKeyPages += touchMemory(pageSize, keyBaseAddress, keyMemorySize);
167 
168                             final long valueBaseAddress = indexReader.getValueBaseAddress();
169                             final long valueMemorySize = indexReader.getValueMemorySize();
170                             indexValuePages += touchMemory(pageSize, valueBaseAddress, valueMemorySize);
171                         }
172                     }
173                 }
