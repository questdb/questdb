205     private long repairMultiByteChar(long lo, long hi, byte b) throws LineProtoException {
206         int n = -1;
207         do {
208             // UTF8 error
209             if (utf8ErrorTop == -1) {
210                 utf8ErrorTop = utf8ErrorPos = dstPos + 1;
211             }
212             // store partial byte
213             dstPos = utf8ErrorPos;
214             utf8ErrorPos += 1;
215             sink.put((char) b);
216 
217             // try to decode partial bytes
218             long errorLen = utf8ErrorPos - utf8ErrorTop;
219             if (errorLen > 1) {
220                 dstPos = utf8ErrorTop - 1;
221                 n = Chars.utf8DecodeMultiByte(utf8ErrorTop, utf8ErrorPos, Unsafe.getUnsafe().getByte(utf8ErrorTop), sink);
222             }
223 
224             if (n == -1 && errorLen > 3) {
225                 errorCode = LineUdpParser.ERROR_ENCODING;
226                 throw LineProtoException.INSTANCE;
227             }
228 
229             if (n == -1 && ++lo < hi) {
230                 b = Unsafe.getUnsafe().getByte(lo);
231             } else {
232                 break;
233             }
234         } while (true);
235 
236         // we can only be in error when we ran out of bytes to read
237         // in which case we return array pointer to original position and exit method
238         dstPos = utf8ErrorTop - 1;
239 
240         if (n > 0) {
241             // if we are successful, reset error pointers
242             utf8ErrorTop = utf8ErrorPos = -1;
243             // bump pos by one more byte in addition to what we may have incremented in the loop
244             return lo + 1;
245         }
246         throw Utf8RepairContinue.INSTANCE;
247     }

249     private long utf8Decode(long lo, long hi, byte b) throws LineProtoException {
250         if (utf8ErrorPos > -1) {
251             return repairMultiByteChar(lo, hi, b);
252         }
253 
254         int n = Chars.utf8DecodeMultiByte(lo, hi, b, sink);
255         if (n == -1) {
256             return repairMultiByteChar(lo, hi, b);
257         } else {

267         if (!escapeQuote) {
268             switch (state) {
269                 case LineUdpParser.EVT_MEASUREMENT:
270                     chop();
271                     break;
272                 case LineUdpParser.EVT_TAG_VALUE:
273                 case LineUdpParser.EVT_FIELD_VALUE:
274                 case LineUdpParser.EVT_TIMESTAMP:
275                     fireEvent();
276                     parser.onLineEnd(charSequenceCache);
277                     clear();
278                     break;
279                 default:
280                     errorCode = LineUdpParser.ERROR_EXPECTED;
281                     throw LineProtoException.INSTANCE;
282             }
283         }

286     protected void parsePartial(final long bytesPtr, final long hi) {
287         long p = bytesPtr;
288 
289         byte lastByte = (byte) 0;
290         while (p < hi && !partialComplete()) {
291 
292             final byte b = Unsafe.getUnsafe().getByte(p);
293 
294             if (skipLine) {
295                 doSkipLine(b);
296                 p++;
297                 lastByte = (byte) 0;
298                 continue;
299             }
300 
301             if (escape) {
302                 dstPos -= 2;
303             }
304 
305             try {
306                 if (b > -1) {
307                     sink.put((char) b);
308                     p++;
309                 } else {
310                     try {
311                         p = utf8Decode(p, hi, b);
312                     } catch (Utf8RepairContinue e) {
313                         break;
314                     }
315                 }
316 
317                 dstPos += 2;
318 
319                 if (escape) {
320                     escape = false;
321                     lastByte = b;
322                     continue;
323                 }
324 
325                 switch (b) {
326                     case '"':
327                         onQuote(lastByte);
328                         break;
329                     case '\\':
330                         onEsc();
331                         break;
332                     case '\n':
333                     case '\r':
334                         onEol();
335                         break;
336                     case ' ':
337                         onSpace();
338                         break;
339                     case ',':
340                         onComma();
341                         break;
342                     case '=':
343                         onEquals();
344                         break;
345                     default:
346                         escapeQuote = false;
347                         break;
348                 }
349                 lastByte = b;
350             } catch (LineProtoException ex) {
351                 skipLine = true;
352                 parser.onError((int) (dstPos - 2 - buffer) / 2, state, errorCode);
353             }
354         }
