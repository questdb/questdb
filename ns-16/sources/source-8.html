33 public final class UdpLineChannel implements LineChannel {
34     private static final Log LOG = LogFactory.getLog(UdpLineChannel.class);
35     private final int fd;
36     private final NetworkFacade nf;
37     private final long sockaddr;
38 
39     public UdpLineChannel(NetworkFacade nf, int interfaceIPv4Address, int sendToAddress, int port, int ttl) {
40         this.nf = nf;
41         this.fd = nf.socketUdp();
42         if (fd == -1) {
43             throw new LineSenderException("could not create UDP socket").errno(nf.errno());
44         }
45         this.sockaddr = nf.sockaddr(sendToAddress, port);
46         if (nf.setMulticastInterface(fd, interfaceIPv4Address) != 0) {
47             final int errno = nf.errno();
48             close();
49             throw new LineSenderException("could not bind ")
50                     .put("[ip=").appendIPv4(interfaceIPv4Address).put("]")
51                     .errno(errno);
52         }
53 
54         if (nf.setMulticastTtl(fd, ttl) != 0) {
55             final int errno = nf.errno();
56             close();
57             throw new LineSenderException("could not set ttl ")
58                     .put("[fd=").put(fd)
59                     .put(", ttl=").put(ttl).put("]")
60                     .errno(errno);
61         }
62     }
63 
64     @Override
65     public void close() {
66         if (nf.close(fd) != 0) {
67             LOG.error().$("could not close network socket [fd=").$(fd).$(", errno=").$(nf.errno()).$(']').$();
68         }
69         nf.freeSockAddr(sockaddr);
70     }
71 
72     @Override
73     public int errno() {
74         return nf.errno();
75     }
76 
77     @Override
78     public int receive(long ptr, int len) {
79         throw new UnsupportedOperationException("Udp channel does not support receive()");
80     }
81 
82     @Override
83     public void send(long ptr, int len) {
84         if (nf.sendTo(fd, ptr, len, sockaddr) != len) {
85             throw new LineSenderException("send error").errno(nf.errno());
86         }
87     }
