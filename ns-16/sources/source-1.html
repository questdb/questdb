56     public AbstractLineProtoUdpReceiver(
57             LineUdpReceiverConfiguration configuration,
58             CairoEngine engine,
59             WorkerPool workerPool
60     ) {
61         this.configuration = configuration;
62         this.commitMode = configuration.getCommitMode();
63         nf = configuration.getNetworkFacade();
64         fd = nf.socketUdp();
65         if (fd < 0) {
66             int errno = nf.errno();
67             LOG.error().$("cannot open UDP socket [errno=").$(errno).$(']').$();
68             throw NetworkError.instance(errno, "Cannot open UDP socket");
69         }
70 
71         try {
72             // when listening for multicast packets bind address must be 0
73             bind(configuration);
74             this.commitRate = configuration.getCommitRate();
75 
76             if (configuration.getReceiveBufferSize() != -1 && nf.setRcvBuf(fd, configuration.getReceiveBufferSize()) != 0) {
77                 LOG.error()
78                         .$("could not set receive buffer size [fd=").$(fd)
79                         .$(", size=").$(configuration.getReceiveBufferSize())
80                         .$(", errno=").$(configuration.getNetworkFacade().errno())
81                         .I$();
82             }
83 
84             lexer = new LineUdpLexer(configuration.getMsgBufferSize());
85             parser = new LineUdpParserImpl(engine, configuration);
86             lexer.withParser(parser);
87 
88             if (!configuration.ownThread()) {
89                 workerPool.assign(this);
90                 logStarted(configuration);
91             }
92         } catch (Throwable e) {
93             close();
94             throw e;
95         }
96     }

98     @Override
99     public void close() {
100         if (fd > -1) {
101             if (running.compareAndSet(true, false)) {
102                 started.await();
103                 halted.await();
104             }
105             if (nf.close(fd) != 0) {
106                 LOG.error().$("could not close [fd=").$(fd).$(", errno=").$(nf.errno()).$(']').$();
107             } else {
108                 LOG.info().$("closed [fd=").$(fd).$(']').$();
109             }
110             if (parser != null) {
111                 parser.commitAll(commitMode);
112                 parser.close();
113             }
114             Misc.free(lexer);
115             LOG.info().$("closed [fd=").$(fd).$(']').$();
116             fd = -1;
117         }
118     }

120     public void start() {
121         if (configuration.ownThread() && running.compareAndSet(false, true)) {
122             new Thread(() -> {
123                 started.countDown();
124                 if (configuration.ownThreadAffinity() != -1) {
125                     Os.setCurrentThreadAffinity(configuration.ownThreadAffinity());
126                 }
127                 logStarted(configuration);
128                 while (running.get()) {
129                     runSerially();
130                 }
131                 LOG.info().$("shutdown").$();
132                 Path.clearThreadLocals();
133                 halted.countDown();
134             }).start();
135         }
136     }

138     private void bind(LineUdpReceiverConfiguration configuration) {
139         if (nf.bindUdp(fd, configuration.isUnicast() ? configuration.getBindIPv4Address() : 0, configuration.getPort())) {
140             if (!configuration.isUnicast() && !nf.join(fd, configuration.getBindIPv4Address(), configuration.getGroupIPv4Address())) {
141                 throw NetworkError.instance(nf.errno())
142                         .put("cannot join group ")
143                         .put("[fd=").put(fd)
144                         .put(", bind=").ip(configuration.getBindIPv4Address())
145                         .put(", group=").ip(configuration.getGroupIPv4Address())
146                         .put(']');
147             }
148         } else {
149             throw NetworkError.instance(nf.errno()).couldNotBindSocket("udp-line-server", configuration.getBindIPv4Address(), configuration.getPort());
150         }
151     }

172                     .$(", commitRate=").$(commitRate)
173                     .I$();
174         }
