41     public static int getValueType(CharSequence value, short defaultFloatColumnType, short defaultIntegerColumnType) {
42         // method called for inbound ilp messages on each value.
43         // returning UNDEFINED makes the whole line be skipped.
44         // 0 len values, return null type.
45         // the goal of this method is to guess the potential type
46         // and then it will be parsed accordingly by 'putValue'.
47         int valueLen = value.length();
48         if (valueLen > 0) {
49             char first = value.charAt(0);
50             char last = value.charAt(valueLen - 1); // see AbstractLineSender.field methods
51             switch (last) {
52                 case 'i':
53                     if (valueLen > 3 && value.charAt(0) == '0' && value.charAt(1) == 'x') {
54                         return ColumnType.LONG256;
55                     }
56                     return valueLen == 1 ? ColumnType.SYMBOL : defaultIntegerColumnType;
57                 case 't':
58                     if (valueLen > 1 && ((first >= '0' && first <= '9') || first == '-')) {
59                         return ColumnType.TIMESTAMP;
60                     }
61                     // fall through
62                 case 'T':
63                     // t
64                     // T
65                 case 'e':
66                 case 'E':
67                     // tru(e)
68                     // fals(e)
69                 case 'f':
70                 case 'F':
71                     // f
72                     // F
73                     if (valueLen == 1) {
74                         return last != 'e' ? ColumnType.BOOLEAN : ColumnType.SYMBOL;
75                     }
76                     return SqlKeywords.isTrueKeyword(value) || SqlKeywords.isFalseKeyword(value) ?
77                             ColumnType.BOOLEAN : ColumnType.SYMBOL;
78                 case '"':
79                     if (valueLen < 2 || value.charAt(0) != '\"') {
80                         LOG.error().$("incorrectly quoted string: ").$(value).$();
81                         return ColumnType.UNDEFINED;
82                     }
83                     return ColumnType.STRING;
84                 default:
85                     if (last >= '0' && last <= '9' && ((first >= '0' && first <= '9') || first == '-' || first == '.')) {
86                         return defaultFloatColumnType;
87                     }
88                     if (SqlKeywords.isNanKeyword(value)) {
89                         return defaultFloatColumnType;
90                     }
91                     if (value.charAt(0) == '"') {
92                         return ColumnType.UNDEFINED;
93                     }
94                     return ColumnType.SYMBOL;
95             }
96         }
97         return ColumnType.NULL;
98     }

101      * Writes column value to table row. CharSequence value is interpreted depending on
102      * column type and written to column, identified by columnIndex. If value cannot be
103      * cast to column type, #BadCastException is thrown.
104      *
105      * @param row            table row
106      * @param columnType     column type value will be cast to
107      * @param columnTypeMeta if columnType's tag is GeoHash it contains bits precision (low short)
108      *                       and tag size (high short negative), otherwise -1
109      * @param columnIndex    index of column to write value to
110      * @param value          value characters
111      */
112     public static void putValue(
113             TableWriter.Row row,
114             int columnType,
115             int columnTypeMeta,
116             int columnIndex,
117             CharSequence value
118     ) {
119         if (value.length() > 0) {
120             try {
121                 switch (ColumnType.tagOf(columnType)) {
122                     case ColumnType.LONG:
123                         row.putLong(columnIndex, Numbers.parseLong(value, 0, value.length() - 1));
124                         break;
125                     case ColumnType.BOOLEAN:
126                         row.putBool(columnIndex, isTrue(value));
127                         break;
128                     case ColumnType.STRING:
129                         row.putStr(columnIndex, value, 1, value.length() - 2);
130                         break;
131                     case ColumnType.SYMBOL:
132                         row.putSym(columnIndex, value);
133                         break;
134                     case ColumnType.DOUBLE:
135                         row.putDouble(columnIndex, Numbers.parseDouble(value));
136                         break;
137                     case ColumnType.FLOAT:
138                         row.putFloat(columnIndex, Numbers.parseFloat(value));
139                         break;
140                     case ColumnType.INT:
141                         row.putInt(columnIndex, Numbers.parseInt(value, 0, value.length() - 1));
142                         break;
143                     case ColumnType.SHORT:
144                         row.putShort(columnIndex, Numbers.parseShort(value, 0, value.length() - 1));
145                         break;
146                     case ColumnType.BYTE:
147                         long v = Numbers.parseLong(value, 0, value.length() - 1);
148                         if (v < Byte.MIN_VALUE || v > Byte.MAX_VALUE) {
149                             throw CairoException.nonCritical()
150                                     .put("line protocol integer is out of byte bounds [columnIndex=")
151                                     .put(columnIndex)
152                                     .put(", v=")
153                                     .put(v)
154                                     .put(']');
155                         }
156                         row.putByte(columnIndex, (byte) v);
157                         break;
158                     case ColumnType.DATE:
159                         row.putDate(columnIndex, Numbers.parseLong(value, 0, value.length() - 1));
160                         break;
161                     case ColumnType.LONG256:
162                         int limit = value.length() - 1;
163                         if (value.charAt(limit) != 'i') {
164                             limit++;
165                         }
166                         row.putLong256(columnIndex, value, 2, limit);
167                         break;
168                     case ColumnType.TIMESTAMP:
169                         row.putTimestamp(columnIndex, Numbers.parseLong(value, 0, value.length() - 1));
170                         break;
171                     case ColumnType.CHAR:
172                         row.putChar(columnIndex, value.length() == 2 ? (char) 0 : value.charAt(1)); // skip quotes
173                         break;
174                     case ColumnType.GEOBYTE:
175                         row.putByte(
176                                 columnIndex,  // skip quotes
177                                 (byte) GeoHashes.fromStringTruncatingNl(
178                                         value,
179                                         1,
180                                         value.length() - 1,
181                                         columnTypeMeta
182                                 )
183                         );
184                         break;
185                     case ColumnType.GEOSHORT:
186                         row.putShort(
187                                 columnIndex,
188                                 (short) GeoHashes.fromStringTruncatingNl(
189                                         value,
190                                         1,
191                                         value.length() - 1,
192                                         columnTypeMeta
193                                 )
194                         );
195                         break;
196                     case ColumnType.GEOINT:
197                         row.putInt(
198                                 columnIndex,
199                                 (int) GeoHashes.fromStringTruncatingNl(
200                                         value,
201                                         1,
202                                         value.length() - 1,
203                                         columnTypeMeta
204                                 )
205                         );
206                         break;
207                     case ColumnType.GEOLONG:
208                         row.putLong(
209                                 columnIndex,
210                                 GeoHashes.fromStringTruncatingNl(
211                                         value,
212                                         1,
213                                         value.length() - 1,
214                                         columnTypeMeta
215                                 )
216                         );
217                         break;
218                     case ColumnType.NULL:
219                     default:
220                         // unsupported types and null are ignored
221                         break;
222                 }
223             } catch (NumericException | ImplicitCastException e) {
224                 LOG.info()
225                         .$("cast error [value=")
226                         .$(value).$(", toType=")
227                         .$(ColumnType.nameOf(columnType))
228                         .$(']')
229                         .$();
230             }
231         } else {
232             putNullValue(row, columnIndex, columnType);
233         }
234     }
