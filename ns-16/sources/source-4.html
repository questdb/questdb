137     public void onEvent(CachedCharSequence token, int eventType, CharSequenceCache cache) {
138 
139         switch (eventType) {
140             case EVT_MEASUREMENT:
141                 int wrtIndex = writerCache.keyIndex(token);
142                 // this condition relies on the fact that this.cacheEntryIndex is always negative
143                 // which indicates that entry is in cache
144                 if (wrtIndex == this.cacheEntryIndex) {
145                     // same table as from last line?
146                     // make sure we append it in case it was in "create" mode
147                     if (writer != null) {
148                         switchModeToAppend();
149                     } else {
150                         initCacheEntry(token, writerCache.valueAtQuick(wrtIndex));
151                     }
152                 } else {
153                     switchTable(token, wrtIndex);
154                 }
155                 break;
156             case EVT_FIELD_NAME:
157             case EVT_TAG_NAME:
158                 onFieldName.parse(token);
159                 break;
160             case EVT_TAG_VALUE:
161                 onTagValue.parse(token, cache);
162                 break;
163             case EVT_FIELD_VALUE:
164                 onFieldValue.parse(token, cache);
165                 break;
166             case EVT_TIMESTAMP:
167                 columnValues.add(token.getCacheAddress());
168                 geoHashBitsSizeByColIdx.add(0);
169                 break;
170             default:
171                 break;
172         }

282         TableToken tableToken = engine.getTableTokenIfExists(token);
283         switch (entry.state) {
284             case 0:
285                 int exists = engine.getStatus(cairoSecurityContext, path, tableToken);
286                 switch (exists) {
287                     case TABLE_EXISTS:
288                         entry.state = 1;
289                         cacheWriter(entry, token, tableToken);

291                     case TABLE_DOES_NOT_EXIST:
292                         if (!autoCreateNewTables) {
293                             throw CairoException.nonCritical()

296                         }
297                         if (!autoCreateNewColumns) {
298                             throw CairoException.nonCritical()
299                                     .put("table does not exist, cannot create table, creating new columns is disabled [table=").put(token)
300                                     .put(']');
301                         }
302                         tableName = token.getCacheAddress();
303                         if (onLineEnd != MY_NEW_LINE_END) {
304                             onLineEnd = MY_NEW_LINE_END;

306                             onFieldValue = MY_NEW_FIELD_VALUE;

308                         }
309                         break;
310                     default:
311                         entry.state = 3;
312                         switchModeToSkipLine();

314                 }
315                 break;
316             case 1:

373     private void parseValue(CachedCharSequence value, int valueType, CharSequenceCache cache, boolean isForField) {
374         assert valueType > ColumnType.UNDEFINED;
375         if (columnType > ColumnType.UNDEFINED) {
376             boolean valid;
377             int geoHashBits = 0;
378             if (valueType != ColumnType.NULL) {
379                 final int valueTypeTag = ColumnType.tagOf(valueType);
380                 final int columnTypeTag = ColumnType.tagOf(columnType);
381                 switch (valueTypeTag) {
382                     case ColumnType.LONG:
383                         valid = columnTypeTag == ColumnType.LONG
384                                 || columnTypeTag == ColumnType.INT
385                                 || columnTypeTag == ColumnType.SHORT
386                                 || columnTypeTag == ColumnType.BYTE
387                                 || columnTypeTag == ColumnType.TIMESTAMP
388                                 || columnTypeTag == ColumnType.DATE;
389                         break;
390                     case ColumnType.INT:
391                         valid = columnTypeTag == ColumnType.INT
392                                 || columnTypeTag == ColumnType.SHORT
393                                 || columnTypeTag == ColumnType.BYTE;
394                         break;
395                     case ColumnType.SHORT:
396                         valid = columnTypeTag == ColumnType.SHORT
397                                 || columnTypeTag == ColumnType.BYTE;
398                         break;
399                     case ColumnType.BYTE:
400                         valid = columnTypeTag == ColumnType.BYTE;
401                         break;
402                     case ColumnType.BOOLEAN:
403                         valid = columnTypeTag == ColumnType.BOOLEAN;
404                         break;
405                     case ColumnType.STRING:
406                         valid = columnTypeTag == ColumnType.STRING ||
407                                 columnTypeTag == ColumnType.CHAR ||
408                                 isForField &&
409                                         (geoHashBits = ColumnType.getGeoHashBits(columnType)) != 0;
410                         break;
411                     case ColumnType.DOUBLE:
412                         valid = columnTypeTag == ColumnType.DOUBLE || columnTypeTag == ColumnType.FLOAT;
413                         break;
414                     case ColumnType.FLOAT:
415                         valid = columnTypeTag == ColumnType.FLOAT;
416                         break;
417                     case ColumnType.SYMBOL:
418                         valid = columnTypeTag == ColumnType.SYMBOL;
419                         break;
420                     case ColumnType.LONG256:
421                         valid = columnTypeTag == ColumnType.LONG256;
422                         break;
423                     case ColumnType.TIMESTAMP:
424                         valid = columnTypeTag == ColumnType.TIMESTAMP;
425                         break;
426                     default:
427                         valid = false;
428                 }
429             } else {
430                 valid = true; // null is valid, the storage value is assigned later
431             }
432             if (valid) {
433                 columnIndexAndType.add(Numbers.encodeLowHighInts(columnIndex, columnType));
434                 columnValues.add(value.getCacheAddress());
435                 geoHashBitsSizeByColIdx.add(geoHashBits);
436             } else {
437                 LOG.error().$("mismatched column and value types [table=").utf8(writer.getTableToken().getTableName())
438                         .$(", column=").$(metadata.getColumnName(columnIndex))
439                         .$(", columnType=").$(ColumnType.nameOf(columnType))
440                         .$(", valueType=").$(ColumnType.nameOf(valueType))
441                         .$(']').$();
442                 switchModeToSkipLine();
443             }
444         } else {
445             CharSequence colNameAsChars = cache.get(columnName);
446             if (autoCreateNewColumns && TableUtils.isValidColumnName(colNameAsChars, udpConfiguration.getMaxFileNameLength())) {
447                 writer.addColumn(colNameAsChars, valueType);
448                 // Writer index can be different from column count, it keeps deleted columns in metadata
449                 int columnIndex = writer.getColumnIndex(colNameAsChars);
450                 columnIndexAndType.add(Numbers.encodeLowHighInts(columnIndex, valueType));
451                 columnValues.add(value.getCacheAddress());
452                 geoHashBitsSizeByColIdx.add(0);
453             } else if (!autoCreateNewColumns) {
454                 throw CairoException.nonCritical()
455                         .put("column does not exist, creating new columns is disabled [table=").put(writer.getTableToken().getTableName())
456                         .put(", columnName=").put(colNameAsChars)
457                         .put(']');
458             } else {
459                 LOG.error().$("invalid column name [table=").utf8(writer.getTableToken().getTableName())
460                         .$(", columnName=").$(colNameAsChars)
461                         .$(']').$();
462                 switchModeToSkipLine();
463             }
464         }
465     }

497     private void switchTable(CachedCharSequence tableName, int entryIndex) {
498         if (this.cacheEntryIndex != 0) {
499             // add previous writer to commit list
500             CacheEntry e = writerCache.valueAtQuick(cacheEntryIndex);
501             if (e.writer != null) {
502                 if (Chars.equals(tableName, e.writer.getTableToken().getTableName())) {
503                     commitList.put(e.writer.getTableToken().getTableName(), e.writer);
504                 } else {
505                     // Cannot happen except with WAL table rename and out of date TableWriter.tableToken.
506                     commitList.put(Chars.toString(tableName), e.writer);
507                 }
508             }
509         }
510 
511         CacheEntry entry;
512         if (entryIndex < 0) {
513             entry = writerCache.valueAtQuick(entryIndex);
514         } else {
515             entry = new CacheEntry();
516             writerCache.putAt(entryIndex, Chars.toString(tableName), entry);
517             // adjust writer map index to negative, which indicates that entry exists
518             entryIndex = -entryIndex - 1;
519         }
520 
521         this.cacheEntryIndex = entryIndex;
522 
523         if (entry.writer == null) {
524             initCacheEntry(tableName, entry);
525         } else {
526             createState(entry);
527         }
528     }
