54         if (hasContentLengthMarker) {
55             // take the body length and format it into the ###### contentLength marker
56             int bodyLen = (int) (_wptr - bodyStart);
57             long p = contentLengthEnd; // note, we replace # from the lowest significant digit (right to left)
58             if (bodyLen == 0) {
59                 Unsafe.getUnsafe().putByte(p--, (byte) '0');
60             } else {
61                 // note, we reserved CL_MARKER_LEN (9) positions for the value of
62                 // http header attribute Content-Length. With 9 positions, the max
63                 // length is assumed to be way larger than needed, to avoid
64                 // additional checks for overflow here.
65                 int rem = bodyLen % 10;
66                 while (bodyLen > 0) {
67                     Unsafe.getUnsafe().putByte(p--, (byte) ('0' + rem));
68                     bodyLen /= 10;
69                     rem = bodyLen % 10;
70                 }
71             }
72             int lpadLen = (int) (CL_MARKER_LEN - contentLengthEnd + p); // remaining # become white space
73             for (int lpad = 0; lpad < lpadLen; lpad++) {
74                 Unsafe.getUnsafe().putByte(p--, (byte) ' ');
75             }
76         }
77         return length(); // length of the http log record
78     }
