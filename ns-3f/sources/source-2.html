155     public boolean send(int len, Runnable onReconnect) {
156         if (len < 1) {
157             return false;
158         }
159 
160         final int maxSendAttempts = 2 * alertHostsCount;
161         int sendAttempts = maxSendAttempts; // empirical, say twice per host at most
162         while (sendAttempts > 0) {
163             if (socketFd > 0) {
164                 int remaining = len;
165                 long p = outBufferPtr;
166                 boolean sendFail = false;
167                 while (remaining > 0) {
168                     int n = nf.send(socketFd, p, remaining);
169                     if (n > 0) {
170                         remaining -= n;
171                         p += n;
172                     } else {
173                         $currentAlertHost(log.info().$("Could not send"))
174                                 .$(" [errno=").$(nf.errno())
175                                 .$(", size=").$(n)
176                                 .$(", log=").$utf8(outBufferPtr, outBufferPtr + len).I$();
177                         sendFail = true;
178                         // do fail over, could not send
179                         break;
180                     }
181                 }
182                 if (!sendFail) {
183                     // receive ack
184                     p = inBufferPtr;
185                     final int n = nf.recv(socketFd, p, inBufferSize);
186                     if (n > 0) {
187                         logResponse(n);
188                         break;
189                     }
190                     // do fail over, ack was not received
191                 }
192             }
193 
194             // fail to the next host and attempt to send again
195             freeSocketAndAddress();
196             int alertHostIdx = this.alertHostIdx;
197             this.alertHostIdx = (this.alertHostIdx + 1) % alertHostsCount;
198             LogRecord logFailOver = $alertHost(
199                     this.alertHostIdx,
200                     $alertHost(
201                             alertHostIdx,
202                             log.info().$("Failing over from")
203                     ).$(" to"));
204             if (alertHostIdx == this.alertHostIdx) {
205                 logFailOver.$(" with a delay of ")
206                         .$(reconnectDelay / 1000000)
207                         .$(" millis (as it is the same alert manager)")
208                         .$();
209                 onReconnect.run();
210             } else {
211                 logFailOver.$();
212             }
213             connect();
214             sendAttempts--;
215         }
216         boolean success = sendAttempts > 0;
217         if (!success) {
218             log.info()
219                     .$("None of the configured alert managers are accepting alerts.\n")
220                     .$("Giving up sending after ")
221                     .$(maxSendAttempts)
222                     .$(" attempts: [")
223                     .$utf8(outBufferPtr, outBufferPtr + len)
224                     .I$();
225         }
226         return success;
227     }

274     private void parseAlertTargets() {
275         if (alertTargets == null || alertTargets.isEmpty()) {
276             setDefaultHostPort();
277             return;
278         }
279         int startIdx = 0;
280         int endIdx = alertTargets.length();
281 
282         if (Chars.isQuoted(alertTargets)) {
283             startIdx++;
284             endIdx--;
285         }
286         while (alertTargets.charAt(startIdx) == ' ' && startIdx < endIdx - 1) {
287             startIdx++;
288         }
289         while (alertTargets.charAt(endIdx - 1) == ' ' && endIdx > startIdx) {
290             endIdx--;
291         }
292         final int len = endIdx - startIdx;
293         if (len == 0) {
294             setDefaultHostPort();
295             return;
296         }
297 
298         // expected format: host[:port](,host[:port])*
299         int hostIdx = startIdx;
300         int portIdx = -1;
301         for (int i = startIdx; i < endIdx; ++i) {
302             char c = alertTargets.charAt(i);
303             switch (c) {
304                 case ':':
305                     if (portIdx != -1) {
306                         throw new LogError(String.format(
307                                 "Unexpected ':' found at position %d: %s",
308                                 i,
309                                 alertTargets));
310                     }
311                     portIdx = i;
312                     break;
313 
314                 case ',':
315                     setHostPort(hostIdx, portIdx, i);
316                     hostIdx = i + 1;
317                     portIdx = -1;
318                     break;
319             }
320         }
321         setHostPort(hostIdx, portIdx, len);
322         alertHostIdx = rand.nextInt(alertHostsCount);
323     }

334     private void setHostPort(int hostIdx, int portLimit, int hostLimit) {
335         // host0:port0, host1 : port1 , ..., host9:port9
336         //              ^     ^       ^
337         //              |     |       hostEnd
338         //              |     portLimit
339         //              hostIdx
340 
341         boolean hostResolved = false;
342         int hostEnd = hostLimit;
343         if (portLimit == -1) { // no ':' was found
344             if (hostIdx + 1 > hostLimit) {
345                 alertHosts[alertHostsCount] = defaultHost;
346                 hostResolved = true;
347             }
348             alertPorts[alertHostsCount] = defaultPort;
349         } else {
350             if (hostIdx + 1 > portLimit) {
351                 alertHosts[alertHostsCount] = defaultHost;
352                 hostResolved = true;
353             } else {
354                 hostEnd = portLimit;
355             }
356             if (portLimit + 2 > hostLimit) {
357                 alertPorts[alertHostsCount] = defaultPort;
358             } else {
359                 int port = 0;
360                 int scale = 1;
361                 for (int i = hostLimit - 1; i > portLimit; i--) {
362                     int c = alertTargets.charAt(i) - '0';
363                     if (c > -1 && c < 10) {
364                         port += c * scale;
365                         scale *= 10;
366                     } else {
367                         throw new LogError(String.format(
368                                 "Invalid port value [%s] at position %d for alertTargets: %s",
369                                 alertTargets.substring(portLimit + 1, hostLimit),
370                                 portLimit + 1,
371                                 alertTargets
372                         ));
373                     }
374                 }
375                 alertPorts[alertHostsCount] = port;
376             }
377         }
378         LogRecord logRecord = log.info()
379                 .$("Added alert manager [")
380                 .$(alertHostsCount)
381                 .$("]: ");
382         try {
383             if (!hostResolved) {
384                 String host = alertTargets.substring(hostIdx, hostEnd).trim();
385                 try {
386                     alertHosts[alertHostsCount] = InetAddress.getByName(host).getHostAddress();
387                     logRecord.$(host).$(" (").$(alertHosts[alertHostsCount]).$(')');
388                 } catch (UnknownHostException e) {
389                     throw new LogError(String.format(
390                             "Invalid host value [%s] at position %d for alertTargets: %s",
391                             host,
392                             hostIdx,
393                             alertTargets
394                     ));
395                 }
396             } else {
397                 logRecord.$(alertHosts[alertHostsCount]);
398             }
399             logRecord.$(':').$(alertPorts[alertHostsCount]);
400             alertHostsCount++;
401         } finally {
402             logRecord.$();
403         }
404     }

441     @TestOnly
442     void logResponse(int len) {
443         responseSink.clear();
444         Chars.utf8Decode(inBufferPtr, inBufferPtr + len, responseSink);
445         final int responseLen = responseSink.length();
446         int contentLength = 0;
447         int lineStart = 0;
448         int colonIdx = -1;
449         boolean headerEndFound = false;
450         for (int i = 0; i < responseLen; i++) {
451             switch (responseSink.charAt(i)) {
452                 case ':':
453                     if (colonIdx == -1) { // values may contain ':', e.g. Date: Thu, 09 Dec 2021 09:37:22 GMT
454                         colonIdx = i;
455                     }
456                     break;
457 
458                 case '\n':
459                     if (colonIdx != -1) {
460                         if (isContentLength(responseSink, lineStart, colonIdx)) {
461                             int startSize = colonIdx + 1;
462                             int limSize = i - 1;
463                             while (startSize < responseLen && responseSink.charAt(startSize) == ' ') {
464                                 startSize++;
465                             }
466                             while (limSize > startSize) {
467                                 char c = responseSink.charAt(limSize);
468                                 if (c == '\r' || c == ' ') {
469                                     limSize--;
470                                 } else {
471                                     break;
472                                 }
473                             }
474                             try {
475                                 contentLength = Numbers.parseInt(responseSink, startSize, limSize + 1);
476                             } catch (NumericException e) {
477                                 $currentAlertHost(log.info().$("Received")).$(": ").$(responseSink).$();
478                                 return;
479                             }
480                         }
481                         colonIdx = -1;
482                     } else if (i - lineStart == 1 && responseSink.charAt(i - 1) == '\r') {
483                         lineStart = i + 1;
484                         headerEndFound = true;
485                         break; // for loop
486                     }
487                     lineStart = i + 1;
488                     break;
489             }
490         }
491         int start = headerEndFound && contentLength == responseLen - lineStart ? lineStart : 0;
492         $currentAlertHost(log.info().$("Received"))
493                 .$(": ")
494                 .$(responseSink, start, responseLen)
495                 .$();
496     }
