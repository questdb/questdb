402     private static CharSequence compressScope(CharSequence key, StringSink builder) {
403         builder.clear();
404         char c = 0;
405         boolean pick = true;
406         int z = 0;
407         for (int i = 0, n = key.length(); i < n; i++) {
408             char a = key.charAt(i);
409             if (a == '.') {
410                 if (!pick) {
411                     builder.put(c).put('.');
412                     pick = true;
413                 }
414             } else if (pick) {
415                 c = a;
416                 z = i;
417                 pick = false;
418             }
419         }
420 
421         for (; z < key.length(); z++) {
422             builder.put(key.charAt(z));
423         }
424 
425         builder.put(' ');
426 
427         return builder.toString();
428     }

432         final String writer = "w." + writerName + '.';
433         final String clazz = getProperty(properties, writer + "class");
434         final String levelStr = getProperty(properties, writer + "level");
435         final String scope = getProperty(properties, writer + "scope");
436 
437         if (clazz == null) {
438             return null;
439         }
440 
441         final Class<?> cl;
442         final Constructor constructor;
443         try {
444             cl = Class.forName(clazz);
445             constructor = cl.getDeclaredConstructor(RingQueue.class, SCSequence.class, int.class);
446         } catch (ClassNotFoundException e) {
447             throw new LogError("Class not found " + clazz, e);
448         } catch (NoSuchMethodException e) {
449             throw new LogError("Constructor(RingQueue, Sequence, int) expected: " + clazz, e);
450         }
451 
452         int level = 0;
453         if (levelStr != null) {
454             for (String s : levelStr.split(",")) {
455                 switch (s.toUpperCase()) {
456                     case "DEBUG":
457                         level |= LogLevel.DEBUG;
458                         break;
459                     case "INFO":
460                         level |= LogLevel.INFO;
461                         break;
462                     case "ERROR":
463                         level |= LogLevel.ERROR;
464                         break;
465                     case "CRITICAL":
466                         level |= LogLevel.CRITICAL;
467                         break;
468                     case "ADVISORY":
469                         level |= LogLevel.ADVISORY;
470                         break;
471                     default:
472                         throw new LogError("Unknown level: " + s);
473                 }
474             }
475         }
476 
477         if (isForcedDebug()) {
478             level = level | LogLevel.DEBUG;
479         }
480 
481         // enable all LOG levels above the minimum set one
482         // ((-1 >>> (msb-1)) << msb) | level
483         final int msb = Numbers.msb(level);
484         level = (((-1 >>> (msb - 1)) << msb) | level) & LogLevel.MASK;
485 
486         return new LogWriterConfig(scope == null ? EMPTY_STR : scope, level, (ring, seq, level1) -> {
487             try {
488                 LogWriter w1 = (LogWriter) constructor.newInstance(ring, seq, level1);
489 
490                 for (String n : properties.stringPropertyNames()) {
491                     if (n.startsWith(writer)) {
492                         String p = n.substring(writer.length());
493 
494                         if (reserved.contains(p)) {
495                             continue;
496                         }
497 
498                         try {
499                             Field f = cl.getDeclaredField(p);
500                             if (f.getType() == String.class) {
501 
502                                 String value = getProperty(properties, n);
503                                 if (logDir != null && value.contains(LOG_DIR_VAR)) {
504                                     value = value.replace(LOG_DIR_VAR, logDir);
505                                 }
506 
507                                 Unsafe.getUnsafe().putObject(w1, Unsafe.getUnsafe().objectFieldOffset(f), value);
508                             }
509                         } catch (Exception e) {
510                             throw new LogError("Unknown property: " + n, e);
511                         }
512                     }
513                 }
514                 return w1;
515             } catch (Exception e) {
516                 throw new LogError("Error creating log writer", e);
517             }
518         });
519     }

935     private static class ScopeConfiguration implements Closeable {
936         private final int[] channels;
937         private final ObjList<Holder> holderList = new ObjList<>();
938         private final IntObjHashMap<Holder> holderMap = new IntObjHashMap<>();
939         private final ObjList<LogWriterConfig> writerConfigs = new ObjList<>();
940         private int ci = 0;
941 
942         public ScopeConfiguration(int levels) {
943             this.channels = new int[levels];
944         }
945 
946         public void bind(ObjHashSet<LogWriter> jobs, int queueDepth, int recordLength) {
947             // create queues for processed channels
948             for (int index : channels) {
949                 if (index > 0) {
950                     int keyIndex = holderMap.keyIndex(index);
951                     if (keyIndex > -1) {
952                         Holder h = new Holder(queueDepth, recordLength);
953                         holderMap.putAt(keyIndex, index, h);
954                         holderList.add(h);
955                     }
956                 }
957             }
958 
959             for (int i = 0, n = writerConfigs.size(); i < n; i++) {
960                 LogWriterConfig c = writerConfigs.getQuick(i);
961                 // the channels array has a guarantee that
962                 // all bits in level mask will point to the same queue,
963                 // so we just get most significant bit number
964                 // and dereference queue on its index
965                 Holder h = holderMap.get(channels[Numbers.msb(c.getLevel())]);
966                 // check if this queue was used by another writer
967                 if (h.wSeq != null) {
968                     // yes, it was
969                     if (h.fanOut == null) {
970                         h.fanOut = FanOut.to(h.wSeq).and(h.wSeq = new SCSequence());
971                     } else {
972                         h.fanOut.and(h.wSeq = new SCSequence());
973                     }
974                 } else {
975                     // we are here first!
976                     h.wSeq = new SCSequence();
977                 }
978                 // now h.wSeq contains out writer's sequence
979                 jobs.add(c.getFactory().createLogWriter(h.ring, h.wSeq, c.getLevel()));
980             }
981 
982             // and the last step is to link dependent sequences
983             for (int i = 0, n = holderList.size(); i < n; i++) {
984                 Holder h = holderList.getQuick(i);
985                 if (h.fanOut != null) {
986                     h.lSeq.then(h.fanOut).then(h.lSeq);
987                 } else {
988                     h.lSeq.then(h.wSeq).then(h.lSeq);
989                 }
990             }
991         }

993         @Override
994         public void close() {
995             for (int i = 0, n = holderList.size(); i < n; i++) {
996                 Misc.free(holderList.getQuick(i));
997             }
998         }

1000         /**
1001          * Aggregates channels into set of queues. Consumer interest is represented by
1002          * level, where consumer sets bits corresponding to channel indexes is it interested in.
1003          * <p>
1004          * Consumer 1 requires channels D & E. So its interest looks like {1,0,1}
1005          * Consumer 2 requires channel I, so its interest is {0,1,0}
1006          * <p>
1007          * This method combines these interests as follows:
1008          * <p>
1009          * channels = {1,2,1}
1010          * <p>
1011          * which means that there will be need to 2 queues (1 and 2) and that Consumer 1
1012          * will be using queue 1 and consumer 2 will be using queue 2.
1013          * <p>
1014          * More complex scenario where consumer interests overlap, for example:
1015          * <p>
1016          * consumer 1 {1,1,0}
1017          * consumer 2 {0,1,1}
1018          * <p>
1019          * these interests will be combined as follows:
1020          * <p>
1021          * channels = {1,1,1}
1022          * <p>
1023          * which means that both consumers will be sharing same queue and they will have to
1024          * filter applicable messages as they get them.
1025          * <p>
1026          * Algorithm iterates over set of bits in "level" twice. First pass is to establish
1027          * minimum number of channel[] element out of those entries where bit in level is set.
1028          * Additionally, this pass will set channel[] elements to current consumer index where
1029          * channel[] element is zero.
1030          * <p>
1031          * Second pass sets channel[] element to min value found on first pass.
1032          *
1033          * @param conf LogWriterConfig
1034          */
1035         private void add(LogWriterConfig conf) {
1036             int mask = conf.getLevel();
1037             int min = Integer.MAX_VALUE;
1038             int q = ++ci;
1039 
1040             for (int i = 0, n = channels.length; i < n; i++) {
1041                 if (((mask >> i) & 1) == 1) {
1042                     int that = channels[i];
1043                     if (that == 0) {
1044                         channels[i] = q;
1045                     }
1046 
1047                     if (that > 0 && that < min) {
1048                         min = that;
1049                     }
1050                 }
1051             }
1052 
1053             if (mask > 1 && min < Integer.MAX_VALUE) {
1054                 for (int i = 0, n = channels.length; i < n; i++) {
1055                     if (((mask >> i) & 1) == 1) {
1056                         channels[i] = min;
1057                     }
1058                 }
1059             }
1060 
1061             writerConfigs.add(conf);
1062         }
1063 
1064         private Holder getHolder(int index) {
1065             return holderMap.get(channels[index]);
1066         }
