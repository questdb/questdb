179     private void loadLogAlertTemplate() {
180         final long now = clock.getTicks();
181         if (location == null || location.isEmpty()) {
182             location = DEFAULT_ALERT_TPT_FILE;
183         }
184         location = alertTemplate.parseEnv(location, now).toString(); // location may contain dollar expressions
185 
186         // read template, resolve env vars within (except $ALERT_MESSAGE)
187         boolean needsReading = true;
188         try (InputStream is = LogAlertSocketWriter.class.getResourceAsStream(location)) {
189             if (is != null) {
190                 byte[] buff = new byte[LogAlertSocket.IN_BUFFER_SIZE];
191                 int len = is.read(buff, 0, buff.length);
192                 String template = new String(buff, 0, len, Files.UTF_8);
193                 alertTemplate.parse(template, now, properties);
194                 needsReading = false;
195             }
196         } catch (IOException e) {
197             // it was not a resource ("/resource_name")
198         }
199         if (needsReading) {
200             sink.clear();
201             readFile(
202                     location,
203                     socket.getInBufferPtr(),
204                     socket.getInBufferSize(),
205                     ff,
206                     sink
207             );
208             alertTemplate.parse(sink, now, properties);
209         }
210         if (alertTemplate.getKeyOffset(MESSAGE_ENV) < 0) {
211             throw new LogError(String.format(
212                     "Bad template, no %s declaration found %s",
213                     MESSAGE_ENV_VALUE,
214                     location));
215         }
216         alertTemplateNodes = alertTemplate.getTemplateNodes();
217         alertTemplateNodesLen = alertTemplateNodes.size();
218     }

220     @TestOnly
221     static void readFile(String location, long address, long addressSize, FilesFacade ff, CharSink sink) {
222         int templateFd = -1;
223         try (Path path = new Path()) {
224             path.of(location);
225             templateFd = ff.openRO(path.$());
226             if (templateFd == -1) {
227                 throw new LogError(String.format(
228                         "Cannot read %s [errno=%d]",
229                         location,
230                         ff.errno()
231                 ));
232             }
233             long size = ff.length(templateFd);
234             if (size > addressSize) {
235                 throw new LogError("Template file is too big");
236             }
237             if (size < 0 || size != ff.read(templateFd, address, size, 0)) {
238                 throw new LogError(String.format(
239                         "Cannot read %s [errno=%d, size=%d]",
240                         location,
241                         ff.errno(),
242                         size
243                 ));
244             }
245             Chars.utf8Decode(address, address + size, sink);
246         } finally {
247             ff.close(templateFd);
248         }
249     }

291     @TestOnly
292     void onLogRecord(LogRecordSink logRecord) {
293         final int len = logRecord.length();
294         if ((logRecord.getLevel() & level) != 0 && len > 0) {
295             alertTemplate.setDateValue(clock.getTicks());
296             alertSink.rewindToMark();
297             for (int i = 0; i < alertTemplateNodesLen; i++) {
298                 TemplateNode comp = alertTemplateNodes.getQuick(i);
299                 if (comp.isEnv(MESSAGE_ENV)) {
300                     alertSink.put(logRecord);
301                 } else {
302                     alertSink.put(comp);
303                 }
304             }
305             sink.clear();
306             sink.put(logRecord);
307             sink.clear(sink.length() - Misc.EOL.length());
308             log.info().$("Sending: ").$(sink).$();
309             socket.send(alertSink.$());
310         }
311     }
