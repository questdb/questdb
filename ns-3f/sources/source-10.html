99     private void addDateTemplateNode(int start, int end) {
100         if (end - start < 1) {
101             throw new LogError("Missing expression at position " + start);
102         }
103         int actualStart = start;
104         int actualEnd = end;
105         while (originalTxt.charAt(actualStart) == ' ' && actualStart < actualEnd) {
106             actualStart++;
107         }
108         while (originalTxt.charAt(actualEnd - 1) == ' ' && actualEnd > actualStart) {
109             actualEnd--;
110         }
111         if (actualEnd - actualStart < 1) {
112             throw new LogError("Missing expression at position " + actualStart);
113         }
114         final DateFormat dateFormat = dateCompiler.compile(originalTxt, actualStart, actualEnd, false);
115         templateNodes.add(new TemplateNode(TemplateNode.TYPE_DATE, DATE_FORMAT_KEY) {

151     private TemplateParser parse(
152             CharSequence txt,
153             long dateValue,
154             CharSequenceObjHashMap<CharSequence> props,
155             boolean needsUtf8Encoding
156     ) {
157         originalTxt = txt;
158         this.dateValue.set(dateValue);
159         this.props = props;
160         templateNodes.clear();
161         envStartIdxs.clear();
162         int dollarStart = NIL; // points at $
163         int keyStart = NIL;   // points at the first char after {
164         int lastExprEnd = 0;   // points to the char right after the expression
165         int curlyBraces = 0;
166         final int locationLen = originalTxt.length();
167         for (int i = 0; i < locationLen; i++) {
168             char c = originalTxt.charAt(i);
169             switch (c) {
170                 case '$':
171                     if (dollarStart != NIL) { // already found a $
172                         if (i - dollarStart > 1) {
173                             addEnvTemplateNode(dollarStart, dollarStart + 1, i);
174                             lastExprEnd = i + 1;
175                         } else {
176                             throw new LogError("Unexpected '$' at position " + i);
177                         }
178                     } else {
179                         if (i - lastExprEnd > 0) {
180                             addStaticTemplateNode(lastExprEnd, i, needsUtf8Encoding);
181                             lastExprEnd = i + 1;
182                         }
183                     }
184                     dollarStart = i;
185                     break;
186                 case '{':
187                     curlyBraces++;
188                     if (dollarStart == NIL) {
189                         continue;
190                     }
191                     keyStart = i + 1;
192                     break;
193                 case '}':
194                     curlyBraces--;
195                     if (dollarStart == NIL) {
196                         continue;
197                     }
198                     if (keyStart == NIL) {
199                         addEnvTemplateNode(dollarStart, dollarStart + 1, i);
200                         lastExprEnd = i;
201                     } else {
202                         int exprLen = i - keyStart;
203                         if (exprLen == 0) {
204                             throw new LogError("Missing expression at position " + keyStart);
205                         }
206                         int formatStart = keyStart + DATE_FORMAT_KEY.length();
207                         if (Chars.startsWith(originalTxt, keyStart, formatStart, DATE_FORMAT_KEY)) {
208                             addDateTemplateNode(formatStart, i);
209                         } else {
210                             addEnvTemplateNode(dollarStart, dollarStart + 2, i);
211                         }
212                         lastExprEnd = i + 1;
213                     }
214                     keyStart = NIL;
215                     dollarStart = NIL;
216             }
217         }
218         if (dollarStart == NIL) {
219             if (curlyBraces != 0) {
220                 throw new LogError("Mismatched '{}' at position " + lastExprEnd);
221             }
222             if (locationLen - lastExprEnd > 0) {
223                 addStaticTemplateNode(lastExprEnd, locationLen, needsUtf8Encoding);
224             }
225         } else {
226             if (keyStart != NIL) {
227                 throw new LogError("Missing '}' at position " + locationLen);
228             }
229             if (locationLen - dollarStart > 1) {
230                 addEnvTemplateNode(dollarStart, dollarStart + 1, locationLen);
231             } else {
232                 throw new LogError("Unexpected '$' at position " + dollarStart);
233             }
234         }
235         return this;
236     }
