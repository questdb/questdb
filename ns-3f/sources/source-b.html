278     private void flush() {
279         long ticks = Long.MIN_VALUE;
280         if (currentSize > nRollSize || (ticks = clock.getTicks()) > rollDeadline) {
281             ff.close(fd);
282             removeOldLogs();
283             if (ticks > rollDeadline) {
284                 rollDeadline = rollDeadlineFunction.getDeadline();
285                 locationParser.setDateValue(ticks);
286             }
287             openFile();
288             rolledCounter.incrementAndGet();
289         }
290 
291         int len = (int) (_wptr - buf);
292         if (ff.append(fd, buf, len) != len) {
293             throw new LogError("Could not append log [fd=" + fd + "]");
294         }
295         currentSize += len;
296         _wptr = buf;
297     }

332     private void pushFileStackUp() {
333         // find max file index that has space above it
334         // for files like:
335         // myapp.6.log
336         // myapp.5.log
337         // myapp.3.log
338         // myapp.2.log
339         // myapp.1.log
340         // myapp.log
341         //
342         // max file will be .3 because it can be renamed to .4
343         // without needing to shift anything
344 
345         int index = 1;
346         while (true) {
347             buildFilePath(path);
348 
349             path.put('.').put(index);
350             if (ff.exists(path.$())) {
351                 index++;
352             } else {
353                 break;
354             }
355         }
356 
357         // rename files
358         while (index > 1) {
359             buildFilePath(path);
360             buildFilePath(renameToPath);
361 
362             path.put('.').put(index - 1);
363             renameToPath.put('.').put(index);
364             if (ff.rename(path.$(), renameToPath.$()) != Files.FILES_RENAME_OK) {
365                 throw new LogError("Could not rename " + path + " to " + renameToPath);
366             }
367             index--;
368         }
369 
370         // finally move original file to .1
371         buildFilePath(path);
372         buildFilePath(renameToPath);
373         renameToPath.put(".1");
374         if (ff.rename(path.$(), renameToPath.$()) != Files.FILES_RENAME_OK) {
375             throw new LogError("Could not rename " + path + " to " + renameToPath);
376         }
377     }

379     private void removeExcessiveLogs() {
380         path.of(logDir);
381         // The list is sorted by last modification ts ASC, so we iterate in reverse order
382         // starting with the newest files.
383         long totalSize = 0;
384         for (long i = logFileList.size() - 1; i > -1; i -= 2) {
385             final long packedOffsets = logFileList.get(i);
386             final int startOffset = Numbers.decodeLowInt(packedOffsets);
387             final int endOffset = Numbers.decodeHighInt(packedOffsets);
388             CharSequence fileName = logFileNameSink.subSequence(startOffset, endOffset);
389             path.trimTo(logDir.length()).concat(fileName).$();
390             if ((totalSize += Files.length(path)) > nSizeLimit) {
391                 if (!ff.remove(path)) {
392                     throw new LogError("cannot remove: " + path);
393                 }
394             }
395         }
396     }

398     private void removeExcessiveLogs(long filePointer, int type) {
399         if (type == Files.DT_FILE && Files.notDots(filePointer)) {
400             logFileName.of(filePointer);
401             if (Chars.contains(logFileName, logFileTemplate)) {
402                 path.trimTo(logDir.length()).concat(filePointer).$();
403                 int startOffset = logFileNameSink.length();
404                 logFileNameSink.put(logFileName);
405                 int endOffset = logFileNameSink.length();
406                 logFileList.add(ff.getLastModified(path));
407                 logFileList.add(Numbers.encodeLowHighInts(startOffset, endOffset));
408             }
409         }

425     private void removeOldLogs() {
426         if (lifeDuration != null) {
427             ff.iterateDir(path.of(logDir).$(), removeExpiredLogsRef);
428         }
429         if (sizeLimit != null) {
430             if (logFileList == null) {
431                 logFileList = new DirectLongList(2 * INITIAL_LOG_FILE_LIST_SIZE, MemoryTag.NATIVE_LONG_LIST);
432             }
433             if (logFileNameSink == null) {
434                 logFileNameSink = new DirectCharSink(INITIAL_LOG_FILE_NAME_SINK_SIZE);
435             }
436 
437             ff.iterateDir(path.of(logDir).$(), removeExcessiveLogsRef);
438             // Sort log files by last modification timestamp.
439             Vect.sort128BitAscInPlace(logFileList.getAddress(), logFileList.size() / 2);
440             removeExcessiveLogs();
441 
442             logFileNameSink.resetCapacity();
443             logFileList.clear();
444             logFileList.resetCapacity();
445         }
