32 /**
33  * A copy implementation of CharSequenceIntHashMap. It is there to work with concrete classes
34  * and avoid incurring performance penalty of megamorphic virtual calls. These calls originate
35  * from calling charAt() on CharSequence interface. C2 compiler cannot inline these calls due to
36  * multiple implementations of charAt(). It resorts to a virtual method call via itable. ILP is the
37  * main victim of itable, suffering from non-deterministic performance loss. With this specific
38  * implementation of the map C2 compiler seems to be able to inline chatAt() calls and itables are
39  * no longer present in the async profiler.
40  * <p>
41  * This map is optimized for ASCII and UTF8 DirectByteCharSequence lookups.
42  */
43 public class ByteCharSequenceIntHashMap implements Mutable {

163     public void removeAt(int index) {
164         if (index < 0) {
165             int from = -index - 1;
166             erase(from);
167             free++;
168 
169             // after we have freed up a slot
170             // consider non-empty keys directly below
171             // they may have been a direct hit but because
172             // directly hit slot wasn't empty these keys would
173             // have moved.
174             //
175             // After slot if freed these keys require re-hash
176             from = (from + 1) & mask;
177             for (
178                     ByteCharSequence key = keys[from];
179                     key != null;
180                     from = (from + 1) & mask, key = keys[from]
181             ) {
182                 int hashCode = Hash.hashMem32(key);
183                 int idealHit = hashCode & mask;
184                 if (idealHit != from) {
185                     int to;
186                     if (keys[idealHit] != null) {
187                         to = probe(key, hashCode, idealHit);
188                     } else {
189                         to = idealHit;
190                     }
191 
192                     if (to > -1) {
193                         move(from, to);
194                     }
195                 }
196             }
197         }
198     }

265         keys[index] = key;

273     private void rehash() {
274         int[] oldValues = values;
275         ByteCharSequence[] oldKeys = keys;
276         int[] oldHashCodes = hashCodes;
277         int size = capacity - free;
278         capacity = capacity * 2;
279         free = capacity - size;
280         mask = Numbers.ceilPow2((int) (capacity / loadFactor)) - 1;
281         keys = new ByteCharSequence[mask + 1];
282         hashCodes = new int[mask + 1];
283         values = new int[mask + 1];
284         for (int i = oldKeys.length - 1; i > -1; i--) {
285             ByteCharSequence key = oldKeys[i];
286             if (key != null) {
287                 final int index = keyIndex(key);
288                 keys[index] = key;
289                 hashCodes[index] = oldHashCodes[i];
290                 values[index] = oldValues[i];
291             }
292         }
293     }
