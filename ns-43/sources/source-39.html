48      * Sorts the specified range of the array.
49      *
50      * @param vec   vector of long values
51      * @param left  the index of the first element, inclusive, to be sorted
52      * @param right the index of the last element, inclusive, to be sorted
53      */
54     public static void sort(LongVec vec, int left, int right) {
55         // Use Quicksort on small arrays
56         if (right - left < QUICKSORT_THRESHOLD) {
57             sort(vec, left, right, true);
58             return;
59         }
60 
61         /*
62          * Index run[i] is the start of i-th run
63          * (ascending or descending sequence).
64          */
65         int[] run = new int[MAX_RUN_COUNT + 1];
66         int count = 0;
67         run[0] = left;
68 
69         // Check if the array is nearly sorted
70         for (int k = left; k < right; run[count] = k) {
71             if (vec.getQuick(k) < vec.getQuick(k + 1)) { // ascending
72                 //noinspection StatementWithEmptyBody
73                 while (++k <= right && vec.getQuick(k - 1) <= vec.getQuick(k)) ;
74             } else if (vec.getQuick(k) > vec.getQuick(k + 1)) { // descending
75                 //noinspection StatementWithEmptyBody
76                 while (++k <= right && vec.getQuick(k - 1) >= vec.getQuick(k)) ;
77                 for (int lo = run[count] - 1, hi = k; ++lo < --hi; ) {
78                     swap(vec, lo, hi);
79                 }
80             } else { // equal
81                 for (int m = MAX_RUN_LENGTH; ++k <= right && vec.getQuick(k - 1) == vec.getQuick(k); ) {
82                     if (--m == 0) {
83                         sort(vec, left, right, true);
84                         return;
85                     }
86                 }
87             }
88 
89             /*
90              * The array is not highly structured,
91              * use Quicksort instead of merge sort.
92              */
93             if (++count == MAX_RUN_COUNT) {
94                 sort(vec, left, right, true);
95                 return;
96             }
97         }
98 
99         // Check special cases
100         if (run[count] == right++) { // The last run contains one element
101             run[++count] = right;
102         } else if (count == 1) { // The array is already sorted
103             return;
104         }
105 
106         /*
107          * Create temporary array, which is used for merging.
108          * Implementation note: variable "right" is increased by 1.
109          */
110 
111         LongVec a;
112         LongVec b;
113 
114         byte odd = 0;
115         //noinspection StatementWithEmptyBody
116         for (int n = 1; (n <<= 1) < count; odd ^= 1) ;
117 
118         if (odd == 0) {
119             b = vec;
120             a = vec.newInstance();
121 
122             //noinspection StatementWithEmptyBody
123             for (int i = left - 1; ++i < right; a.setQuick(i, b.getQuick(i))) ;
124         } else {
125             a = vec;
126             b = vec.newInstance();
127         }
128 
129         // Merging
130         for (int last; count > 1; count = last) {
131             for (int k = (last = 0) + 2; k <= count; k += 2) {
132                 int hi = run[k], mi = run[k - 1];
133                 for (int i = run[k - 2], p = i, q = mi; i < hi; ++i) {
134                     if (q >= hi || p < mi && a.getQuick(p) <= a.getQuick(q)) {
135                         b.setQuick(i, a.getQuick(p++));
136                     } else {
137                         b.setQuick(i, a.getQuick(q++));
138                     }
139                 }
140                 run[++last] = hi;
141             }
142             if ((count & 1) != 0) {
143                 //noinspection StatementWithEmptyBody
144                 for (int i = right, lo = run[count - 1]; --i >= lo; b.setQuick(i, a.getQuick(i))) ;
145                 run[++last] = right;
146             }
147             LongVec t = a;
148             a = b;
149             b = t;
150         }
151     }

157     /**
158      * Sorts the specified range of the array by Dual-Pivot Quicksort.
159      *
160      * @param left     the index of the first element, inclusive, to be sorted
161      * @param right    the index of the last element, inclusive, to be sorted
162      * @param leftmost indicates if this part is the leftmost in the range
163      */
164     private static void sort(LongVec vec, int left, int right, boolean leftmost) {
165         int length = right - left + 1;
166 
167         // Use insertion sort on tiny arrays
168         if (length < INSERTION_SORT_THRESHOLD) {
169             if (leftmost) {
170                 /*
171                  * Traditional (without sentinel) insertion sort,
172                  * optimized for server VM, is used in case of
173                  * the leftmost part.
174                  */
175                 for (int i = left, j = i; i < right; j = ++i) {
176                     long ai = vec.getQuick(i + 1);
177                     while (ai < vec.getQuick(j)) {
178                         let(vec, j + 1, j);
179                         if (j-- == left) {
180                             break;
181                         }
182                     }
183                     vec.setQuick(j + 1, ai);
184                 }
185             } else {
186                 /*
187                  * Skip the longest ascending sequence.
188                  */
189                 do {
190                     if (left >= right) {
191                         return;
192                     }
193                 } while (vec.getQuick(++left) >= vec.getQuick(left - 1));
194 
195                 /*
196                  * Every element from adjoining part plays the role
197                  * of sentinel, therefore this allows us to avoid the
198                  * left range check on each iteration. Moreover, we use
199                  * the more optimized algorithm, so called pair insertion
200                  * sort, which is faster (in the context of Quicksort)
201                  * than traditional implementation of insertion sort.
202                  */
203                 for (int k = left; ++left <= right; k = ++left) {
204                     long a1 = vec.getQuick(k), a2 = vec.getQuick(left);
205 
206                     if (a1 < a2) {
207                         a2 = a1;
208                         a1 = vec.getQuick(left);
209                     }
210                     while (a1 < vec.getQuick(--k)) {
211                         let(vec, k + 2, k);
212                     }
213                     vec.setQuick(++k + 1, a1);
214 
215                     while (a2 < vec.getQuick(--k)) {
216                         let(vec, k + 1, k);
217                     }
218                     vec.setQuick(k + 1, a2);
219                 }
220                 long last = vec.getQuick(right);
221 
222                 while (last < vec.getQuick(--right)) {
223                     let(vec, right + 1, right);
224                 }
225                 vec.setQuick(right + 1, last);
226             }
227             return;
228         }
229 
230         // Inexpensive approximation of length / 7
231         int seventh = (length >> 3) + (length >> 6) + 1;
232 
233         /*
234          * Sort five evenly spaced elements around (and including) the
235          * center element in the range. These elements will be used for
236          * pivot selection as described below. The choice for spacing
237          * these elements was empirically determined to work well on
238          * a wide variety of inputs.
239          */
240         int e3 = (left + right) >>> 1; // The midpoint
241         int e2 = e3 - seventh;
242         int e1 = e2 - seventh;
243         int e4 = e3 + seventh;
244         int e5 = e4 + seventh;
245 
246         // Sort these elements using insertion sort
247         if (vec.getQuick(e2) < vec.getQuick(e1)) {
248             swap(vec, e2, e1);
249         }
250 
251         if (vec.getQuick(e3) < vec.getQuick(e2)) {
252             long t = vec.getQuick(e3);
253             let(vec, e3, e2);
254             vec.setQuick(e2, t);
255             if (t < vec.getQuick(e1)) {
256                 let(vec, e2, e1);
257                 vec.setQuick(e1, t);
258             }
259         }
260         if (vec.getQuick(e4) < vec.getQuick(e3)) {
261             long t = vec.getQuick(e4);
262             let(vec, e4, e3);
263             vec.setQuick(e3, t);
264             if (t < vec.getQuick(e2)) {
265                 let(vec, e3, e2);
266                 vec.setQuick(e2, t);
267                 if (t < vec.getQuick(e1)) {
268                     let(vec, e2, e1);
269                     vec.setQuick(e1, t);
270                 }
271             }
272         }
273         if (vec.getQuick(e5) < vec.getQuick(e4)) {
274             long t = vec.getQuick(e5);
275             let(vec, e5, e4);
276             vec.setQuick(e4, t);
277             if (t < vec.getQuick(e3)) {
278                 let(vec, e4, e3);
279                 vec.setQuick(e3, t);
280                 if (t < vec.getQuick(e2)) {
281                     let(vec, e3, e2);
282                     vec.setQuick(e2, t);
283                     if (t < vec.getQuick(e1)) {
284                         let(vec, e2, e1);
285                         vec.setQuick(e1, t);
286                     }
287                 }
288             }
289         }
290 
291         // Pointers
292         int less = left;  // The index of the first element of center part
293         int great = right; // The index before the first element of right part
294 
295         if (vec.getQuick(e1) != vec.getQuick(e2) && vec.getQuick(e2) != vec.getQuick(e3) && vec.getQuick(e3) != vec.getQuick(e4) && vec.getQuick(e4) != vec.getQuick(e5)) {
296             /*
297              * Use the second and fourth of the five sorted elements as pivots.
298              * These values are inexpensive approximations of the first and
299              * second terciles of the array. Note that pivot1 <= pivot2.
300              */
301             long pivot1 = vec.getQuick(e2);
302             long pivot2 = vec.getQuick(e4);
303 
304             /*
305              * The first and the last elements to be sorted are moved to the
306              * locations formerly occupied by the pivots. When partitioning
307              * is complete, the pivots are swapped back into their final
308              * positions, and excluded from subsequent sorting.
309              */
310             let(vec, e2, left);
311             let(vec, e4, right);
312 
313             /*
314              * Skip elements, which are less or greater than pivot values.
315              */
316             //noinspection StatementWithEmptyBody
317             while (vec.getQuick(++less) < pivot1) ;
318             //noinspection StatementWithEmptyBody
319             while (vec.getQuick(--great) > pivot2) ;
320 
321             /*
322              * Partitioning:
323              *
324              *   left part           center part                   right part
325              * +--------------------------------------------------------------+
326              * |  < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2  |
327              * +--------------------------------------------------------------+
328              *               ^                          ^       ^
329              *               |                          |       |
330              *              less                        k     great
331              *
332              * Invariants:
333              *
334              *              all in (left, less)   < pivot1
335              *    pivot1 <= all in [less, k)     <= pivot2
336              *              all in (great, right) > pivot2
337              *
338              * Pointer k is the first index of ?-part.
339              */
340             outer:
341             for (int k = less - 1; ++k <= great; ) {
342                 long ak = vec.getQuick(k);
343                 if (ak < pivot1) { // Move a[k] to left part
344                     let(vec, k, less);
345                     /*
346                      * Here and below we use "a[i] = b; i++;" instead
347                      * of "a[i++] = b;" due to performance issue.
348                      */
349                     vec.setQuick(less, ak);
350                     ++less;
351                 } else if (ak > pivot2) { // Move a[k] to right part
352                     while (vec.getQuick(great) > pivot2) {
353                         if (great-- == k) {
354                             break outer;
355                         }
356                     }
357                     if (vec.getQuick(great) < pivot1) { // a[great] <= pivot2
358                         let(vec, k, less);
359                         let(vec, less, great);
360                         ++less;
361                     } else { // pivot1 <= a[great] <= pivot2
362                         let(vec, k, great);
363                     }
364                     /*
365                      * Here and below we use "a[i] = b; i--;" instead
366                      * of "a[i--] = b;" due to performance issue.
367                      */
368                     vec.setQuick(great, ak);
369                     --great;
370                 }
371             }
372 
373             // Swap pivots into their final positions
374             let(vec, left, less - 1);
375             vec.setQuick(less - 1, pivot1);
376             let(vec, right, great + 1);
377             vec.setQuick(great + 1, pivot2);
378 
379             // Sort left and right parts recursively, excluding known pivots
380             sort(vec, left, less - 2, leftmost);
381             sort(vec, great + 2, right, false);
382 
383             /*
384              * If center part is too large (comprises > 4/7 of the array),
385              * swap internal pivot values to ends.
386              */
387             if (less < e1 && e5 < great) {
388                 /*
389                  * Skip elements, which are equal to pivot values.
390                  */
391                 while (vec.getQuick(less) == pivot1) {
392                     ++less;
393                 }
394 
395                 while (vec.getQuick(great) == pivot2) {
396                     --great;
397                 }
398 
399                 /*
400                  * Partitioning:
401                  *
402                  *   left part         center part                  right part
403                  * +----------------------------------------------------------+
404                  * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
405                  * +----------------------------------------------------------+
406                  *              ^                        ^       ^
407                  *              |                        |       |
408                  *             less                      k     great
409                  *
410                  * Invariants:
411                  *
412                  *              all in (*,  less) == pivot1
413                  *     pivot1 < all in [less,  k)  < pivot2
414                  *              all in (great, *) == pivot2
415                  *
416                  * Pointer k is the first index of ?-part.
417                  */
418                 outer:
419                 for (int k = less - 1; ++k <= great; ) {
420                     long ak = vec.getQuick(k);
421                     if (ak == pivot1) { // Move a[k] to left part
422                         let(vec, k, less);
423                         vec.setQuick(less, ak);
424                         ++less;
425                     } else if (ak == pivot2) { // Move a[k] to right part
426                         while (vec.getQuick(great) == pivot2) {
427                             if (great-- == k) {
428                                 break outer;
429                             }
430                         }
431                         if (vec.getQuick(great) == pivot1) { // a[great] < pivot2
432                             let(vec, k, less);
433                             /*
434                              * Even though a[great] equals to pivot1, the
435                              * assignment a[less] = pivot1 may be incorrect,
436                              * if a[great] and pivot1 are floating-point zeros
437                              * of different signs. Therefore in float and
438                              * double sorting methods we have to use more
439                              * accurate assignment a[less] = a[great].
440                              */
441                             vec.setQuick(less, pivot1);
442                             ++less;
443                         } else { // pivot1 < a[great] < pivot2
444                             let(vec, k, great);
445                         }
446                         vec.setQuick(great, ak);
447                         --great;
448                     }
449                 }
450             }
451 
452             // Sort center part recursively
453             sort(vec, less, great, false);
454 
455         } else { // Partitioning with one pivot
456             /*
457              * Use the third of the five sorted elements as pivot.
458              * This value is inexpensive approximation of the median.
459              */
460             long pivot = vec.getQuick(e3);
461 
462             /*
463              * Partitioning degenerates to the traditional 3-way
464              * (or "Dutch National Flag") schema:
465              *
466              *   left part    center part              right part
467              * +-------------------------------------------------+
468              * |  < pivot  |   == pivot   |     ?    |  > pivot  |
469              * +-------------------------------------------------+
470              *              ^              ^        ^
471              *              |              |        |
472              *             less            k      great
473              *
474              * Invariants:
475              *
476              *   all in (left, less)   < pivot
477              *   all in [less, k)     == pivot
478              *   all in (great, right) > pivot
479              *
480              * Pointer k is the first index of ?-part.
481              */
482             for (int k = less; k <= great; ++k) {
483                 if (vec.getQuick(k) == pivot) {
484                     continue;
485                 }
486                 long ak = vec.getQuick(k);
487                 if (ak < pivot) { // Move a[k] to left part
488                     let(vec, k, less);
489                     vec.setQuick(less, ak);
490                     ++less;
491                 } else { // a[k] > pivot - Move a[k] to right part
492                     while (vec.getQuick(great) > pivot) {
493                         --great;
494                     }
495                     if (vec.getQuick(great) < pivot) { // a[great] <= pivot
496                         let(vec, k, less);
497                         let(vec, less, great);
498                         ++less;
499                     } else { // a[great] == pivot
500                         /*
501                          * Even though a[great] equals to pivot, the
502                          * assignment a[k] = pivot may be incorrect,
503                          * if a[great] and pivot are floating-point
504                          * zeros of different signs. Therefore in float
505                          * and double sorting methods we have to use
506                          * more accurate assignment a[k] = a[great].
507                          */
508                         vec.setQuick(k, pivot);
509                     }
510                     vec.setQuick(great, ak);
511                     --great;
512                 }
513             }
514 
515             /*
516              * Sort left and right parts recursively.
517              * All elements from center part are equal
518              * and, therefore, already sorted.
519              */
520             sort(vec, left, less - 1, leftmost);
521             sort(vec, great + 1, right, false);
522         }
523     }
