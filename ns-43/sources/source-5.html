101     public void removeAt(int index) {
102         if (index < 0) {
103             int from = -index - 1;
104             erase(from);
105             free++;
106 
107             // after we have freed up a slot
108             // consider non-empty keys directly below
109             // they may have been a direct hit but because
110             // directly hit slot wasn't empty these keys would
111             // have moved.
112             //
113             // After slot if freed these keys require re-hash
114             from = (from + 1) & mask;
115             for (
116                     CharSequence key = keys[from];
117                     key != noEntryKey;
118                     from = (from + 1) & mask, key = keys[from]
119             ) {
120                 int idealHit = Chars.lowerCaseAsciiHashCode(key) & mask;
121                 if (idealHit != from) {
122                     int to;
123                     if (keys[idealHit] != noEntryKey) {
124                         to = probe(key, idealHit);
125                     } else {
126                         to = idealHit;
127                     }
128 
129                     if (to > -1) {
130                         move(from, to);
131                     }
132                 }
133             }
134         }
135     }
