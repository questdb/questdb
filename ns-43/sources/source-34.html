126     public int binarySearch(long value, int scanDir) {
127 
128         // this is the same algorithm as implemented in C (util.h)
129         // template<class T, class V>
130         // inline int64_t binary_search(T *data, V value, int64_t low, int64_t high, int32_t scan_dir)
131         // please ensure these implementations are in sync
132 
133         int low = 0;
134         int high = pos - 1;
135         while (high - low > 65) {
136             final int mid = (low + high) / 2;
137             final long midVal = data[mid];
138 
139             if (midVal < value) {
140                 low = mid;
141             } else if (midVal > value) {
142                 high = mid - 1;
143             } else {
144                 // In case of multiple equal values, find the first
145                 return scanDir == BinarySearch.SCAN_UP ?
146                         scrollUp(mid, midVal) :
147                         scrollDown(mid, high, midVal);
148             }
149         }
150         return scanDir == BinarySearch.SCAN_UP ?
151                 scanUp(value, low, high + 1) :
152                 scanDown(value, low, high + 1);

156         // Binary searches using 2^shl blocks
157         // e.g. when shl == 2
158         // this method treats 4 longs as 1 entry

173     public int binarySearchBlock(int offset, int shl, long value, int scanDir) {
174         int low = offset >> shl;
175         int high = (pos - 1) >> shl;
176         while (high - low > 65) {
177             final int mid = (low + high) / 2;
178             final long midVal = data[mid << shl];
179 
180             if (midVal < value) {
181                 low = mid;
182             } else if (midVal > value) {
183                 high = mid - 1;
184             } else {
185                 // In case of multiple equal values, find the first
186                 return scanDir == BinarySearch.SCAN_UP ?
187                         scrollUpBlock(shl, mid, midVal) :
188                         scrollDownBlock(shl, mid, high, midVal);
189             }
190         }
191         return scanDir == BinarySearch.SCAN_UP ?
192                 scanUpBlock(shl, value, low, high + 1) :
193                 scanDownBlock(shl, value, low, high + 1);

432             final int ii = (i << shl) + k1;
