87     public void removeAt(int index) {
88         if (index < 0) {
89             int from = -index - 1;
90             erase(from);
91             free++;
92 
93             // after we have freed up a slot
94             // consider non-empty keys directly below
95             // they may have been a direct hit but because
96             // directly hit slot wasn't empty these keys would
97             // have moved.
98             //
99             // After slot if freed these keys require re-hash
100             from = (from + 1) & mask;
101             for (
102                     long key = keys[from];
103                     key != noEntryKeyValue;
104                     from = (from + 1) & mask, key = keys[from]
105             ) {
106                 int idealHit = (int) (key & mask);
107                 if (idealHit != from) {
108                     int to;
109                     if (keys[idealHit] != noEntryKeyValue) {
110                         to = probe(key, idealHit);
111                     } else {
112                         to = idealHit;
113                     }
114 
115                     if (to > -1) {
116                         move(from, to);
117                     }
118                 }
119             }
120         }
121     }
