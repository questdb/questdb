49     @SuppressWarnings("unchecked")
50     public AssociativeCache(int blocks, int rows, LongGauge cachedGauge) {
51         this.blocks = Math.max(MIN_BLOCKS, Numbers.ceilPow2(blocks));
52         rows = Math.max(MIN_ROWS, Numbers.ceilPow2(rows));
53 
54         int size = rows * this.blocks;
55         if (size < 0) {
56             throw new OutOfMemoryError();
57         }
58         this.keys = new CharSequence[size];
59         this.values = (V[]) new Object[size];
60         this.rmask = rows - 1;
61         this.bmask = this.blocks - 1;
62         this.bshift = Numbers.msb(this.blocks);
63         this.cachedGauge = cachedGauge;
64     }

66     @Override

68         long freed = 0;
69         for (int i = 0, n = keys.length; i < n; i++) {
70             if (keys[i] != null) {

72                 if (values[i] != null) {

94     public V poll(CharSequence key) {
95         int index = getIndex(key);
96         if (index == NOT_FOUND) {
97             return null;
98         }
99         V value = values[index];
100         values[index] = null;
101         if (value != null) {
102             // The value is present, so we're removing.
103             cachedGauge.dec();
104         }
105         // We do not null the key reference to avoid creating another immutable key.
106         return value;
107     }

109     public CharSequence put(CharSequence key, V value) {
110         final int lo = lo(key);
111 
112         if (Chars.equalsNc(key, keys[lo])) {
113             // Present entry case.
114             if (values[lo] != value) {
115                 if (values[lo] == null) {
116                     // The value was previously cleared by poll(), so we're inserting.
117                     cachedGauge.inc();
118                 } else {
119                     // We're replacing the value with another one, no need to change the gauge.
120                     Misc.freeIfCloseable(values[lo]);
121                 }
122                 values[lo] = value;
123             }
124             return null;
125         }
126 
127         // New entry case.
128 
129         final CharSequence outgoingKey = keys[lo + bmask];
130         if (outgoingKey != null) {
131             int idx = lo + bmask;
132             if (values[idx] == null) {
133                 // The value for the outgoing key was previously cleared by poll(), so we're inserting.
134                 cachedGauge.inc();
135             } else {
136                 // We're replacing the value with another one, no need to change the gauge.
137                 values[idx] = Misc.freeIfCloseable(values[idx]);
138             }
139         } else {
140             // The block has empty entries, so we're inserting.
141             cachedGauge.inc();
142         }
143 
144         System.arraycopy(keys, lo, keys, lo + 1, bmask);
145         System.arraycopy(values, lo, values, lo + 1, bmask);
146         if (value == null) {
147             keys[lo] = null;
148         } else {
149             keys[lo] = Chars.toString(key);
150         }
151         values[lo] = value;
152 
153         return outgoingKey;
154     }

156     private int getIndex(CharSequence key) {
157         int lo = lo(key);
158         for (int i = lo, hi = lo + blocks; i < hi; i++) {
159             CharSequence k = keys[i];
160             if (k == null) {
161                 return NOT_FOUND;
162             }
163 
164             if (Chars.equals(k, key)) {
165                 return i;
166             }
167         }
168         return NOT_FOUND;
169     }
