121     @Override
122     public boolean nextCqe() {
123         if (++cachedIndex < cachedSize) {
124             return true;
125         }
126         // Consume all available cqes and store them in the cache.
127         final int tail = Unsafe.getUnsafe().getInt(cqKtailAddr);
128         Unsafe.getUnsafe().loadFence();
129         final int head = Unsafe.getUnsafe().getInt(cqKheadAddr);
130         if (tail == head) {
131             return false;
132         }
133         for (int i = head; i < tail; i++) {
134             final long cqeAddr = cqesAddr + (long) (i & cqKringMask) * SIZEOF_CQE;
135             cachedCqes[2 * (i - head)] = Unsafe.getUnsafe().getLong(cqeAddr + CQE_USER_DATA_OFFSET);
136             cachedCqes[2 * (i - head) + 1] = Unsafe.getUnsafe().getInt(cqeAddr + CQE_RES_OFFSET);
137         }
138         cachedSize = tail - head;
139         cachedIndex = 0;
140         Unsafe.getUnsafe().putInt(cqKheadAddr, tail);
141         Unsafe.getUnsafe().storeFence();
142         return true;
143     }

155     private long enqueueSqe(byte op, int fd, long offset, long bufAddr, int len) {
156         final long sqeAddr = nextSqe();
157         if (sqeAddr == 0) {
158             return -1;
159         }
160         Unsafe.getUnsafe().putByte(sqeAddr + SQE_OPCODE_OFFSET, op);
161         Unsafe.getUnsafe().putInt(sqeAddr + SQE_FD_OFFSET, fd);
162         Unsafe.getUnsafe().putLong(sqeAddr + SQE_OFF_OFFSET, offset);
163         Unsafe.getUnsafe().putLong(sqeAddr + SQE_ADDR_OFFSET, bufAddr);
164         Unsafe.getUnsafe().putInt(sqeAddr + SQE_LEN_OFFSET, len);
165         final long id = idSeq++;
166         Unsafe.getUnsafe().putLong(sqeAddr + SQE_USER_DATA_OFFSET, id);
167         return id;
168     }

173     private long nextSqe() {
