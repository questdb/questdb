27 import io.questdb.std.str.CharSink;

29 import java.util.Arrays;

31 /**
32  * Thread-safe grow-only integer list. Assumes multiple reader threads
33  * and a single writer thread.
34  */
35 public class AtomicIntList {
36 
37     private static final int DEFAULT_ARRAY_SIZE = 16;
38     private static final int NO_ENTRY_VALUE = -1;
39 
40     private volatile int[] data;
41     private volatile int pos;
42 
43     public AtomicIntList() {
44         this(DEFAULT_ARRAY_SIZE);
45     }
46 
47     public AtomicIntList(int capacity) {
48         int[] data = new int[capacity];
49         Arrays.fill(data, 0, data.length, NO_ENTRY_VALUE);
50         this.data = data;
51     }
52 
53     public void add(int value) {
54         int[] data = this.data;
55         int pos = this.pos;
56         if (pos >= data.length) {
57             int[] dataCopy = new int[2 * data.length];
58             System.arraycopy(data, 0, dataCopy, 0, data.length);
59             dataCopy[pos] = value;
60             this.data = dataCopy;
61         } else {
62             Unsafe.arrayPutOrdered(data, pos, value);
63         }
64         this.pos = pos + 1;
65     }
66 
67     public int capacity() {
68         int[] data = this.data;
69         return data.length;
70     }
71 
72     public void extendAndSet(int index, int value) {
73         int[] data = this.data;
74         int pos = this.pos;
75         if (index >= data.length) {
76             int[] dataCopy = new int[2 * (index + 1)];
77             System.arraycopy(data, 0, dataCopy, 0, data.length);
78             Arrays.fill(dataCopy, data.length, dataCopy.length, NO_ENTRY_VALUE);
79             dataCopy[index] = value;
80             this.data = dataCopy;
81         } else {
82             Unsafe.arrayPutOrdered(data, index, value);
83         }
84         if (pos <= index) {
85             this.pos = index + 1;
86         }
87     }
88 
89     public int get(int index) {
90         // First we read position, then data as they're written in the reverse order.
91         // That's because it's a grow-only list, so the data array can only grow
92         // once we've read the position.
93         int pos = this.pos;
94         int[] data = this.data;
95         if (index >= pos) {
96             throw new IndexOutOfBoundsException("Index out of range: " + index);
97         }
98         return Unsafe.arrayGetVolatile(data, index);
99     }
100 
101     public void set(int index, int value) {
102         int[] data = this.data;
103         int pos = this.pos;
104         if (index < pos) {
105             Unsafe.arrayPutOrdered(data, index, value);
106             return;
107         }
108         throw new ArrayIndexOutOfBoundsException(index);
109     }
110 
111     public int size() {
112         return this.pos;
113     }
114 
115     @Override
116     public String toString() {
117         CharSink b = Misc.getThreadLocalBuilder();
118         int[] data = this.data;
119         int pos = this.pos;
120         b.put('[');
121         for (int i = 0; i < pos; i++) {
122             if (i > 0) {
123                 b.put(',');
124             }
125             b.put(data[i]);
126         }
127         b.put(']');
128         return b.toString();
129     }
