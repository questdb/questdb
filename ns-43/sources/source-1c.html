371     @Override
372     public int unlinkOrRemove(Path path, int checkedType, Log LOG) {
373         if (checkedType == Files.DT_LNK) {
374             // in Windows ^ ^ will return DT_DIR, but that is ok as the behaviour
375             // is to delete the link, not the contents of the target. in *nix
376             // systems we can simply unlink, which deletes the link and leaves
377             // the contents of the target intact
378             if (unlink(path) == 0) {
379                 LOG.info().$("removed by unlink [path=").utf8(path).I$();
380                 return 0;
381             } else {
382                 LOG.error().$("failed to unlink, will remove [path=").utf8(path).I$();
383             }
384         }
385 
386         int errno;
387         if ((errno = rmdir(path)) == 0) {
388             LOG.info().$("removed [path=").utf8(path).I$();
389         } else {
390             LOG.error().$("cannot remove [path=").utf8(path).$(", errno=").$(errno).I$();
391         }
392         return errno;

395     public void walk(Path path, FindVisitor func) {
396         int len = path.length();
397         long p = findFirst(path);
398         if (p > 0) {
399             try {
400                 do {
401                     long name = findName(p);
402                     if (Files.notDots(name)) {
403                         int type = findType(p);
404                         path.trimTo(len);
405                         if (type == Files.DT_FILE) {
406                             func.onFind(name, type);
407                         } else {
408                             walk(path.concat(name).$(), func);
409                         }
410                     }
411                 } while (findNext(p) > 0);
412             } finally {
413                 findClose(p);
414             }
415         }

423     private long computeMapPageSize() {
424         long pageSize = getPageSize();
425         long mapPageSize = pageSize * pageSize;
426         if (mapPageSize < pageSize || mapPageSize > _16M) {
427             if (_16M % pageSize == 0) {
428                 return _16M;

430             return pageSize;
431         } else {
432             return mapPageSize;

436     private int runRecursive(Path src, Path dst, int dirMode, FsOperation operation) {
437         int dstLen = dst.length();
438         int srcLen = src.length();
439         int len = src.length();
440         long p = findFirst(src.$());
441 
442         if (!exists(dst.$()) && -1 == mkdir(dst, dirMode)) {
443             return -1;
444         }
445 
446         if (p > 0) {
447             try {
448                 int res;
449                 do {
450                     long name = findName(p);
451                     if (Files.notDots(name)) {
452                         int type = findType(p);
453                         src.trimTo(len);
454                         src.concat(name);
455                         dst.concat(name);
456                         if (type == Files.DT_FILE) {
457                             if ((res = operation.invoke(src.$(), dst.$())) < 0) {
458                                 return res;
459                             }
460                         } else {
461 
462                             // Ignore if subfolder already exists
463                             mkdir(dst.$(), dirMode);
464 
465                             if ((res = runRecursive(src, dst, dirMode, operation)) < 0) {
466                                 return res;
467                             }
468 
469                         }
470                         src.trimTo(srcLen);
471                         dst.trimTo(dstLen);
472                     }
473                 } while (findNext(p) > 0);
474             } finally {
475                 findClose(p);
476                 src.trimTo(srcLen);
477                 dst.trimTo(dstLen);
478             }
479         }
480 
481         return 0;
482     }
