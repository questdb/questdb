256     public static int mkdirs(Path path, int mode) {
257         for (int i = 0, n = path.length(); i < n; i++) {
258             char c = path.charAt(i);
259             if (c == Files.SEPARATOR) {
260                 // do not attempt to create '/' on linux or 'C:\' on Windows
261                 if ((i == 0 && Os.isPosix()) || (i == 2 && Os.isWindows() && path.charAt(1) == ':')) {
262                     continue;
263                 }
264 
265                 // replace separator we just found with \0
266                 // temporarily truncate path to the directory we need to create
267                 path.$at(i);
268                 if (path.length() > 0 && !Files.exists(path)) {
269                     int r = Files.mkdir(path, mode);
270                     if (r != 0) {
271                         path.put(i, Files.SEPARATOR);
272                         return r;
273                     }
274                 }
275                 path.put(i, Files.SEPARATOR);
276             }
277         }
278         return 0;
279     }

370     public static int rmdir(Path path) {
371         long p = findFirst(path.address());
372         int len = path.length();
373         int errno = -1;
374         if (p > 0) {
375             try {
376                 do {
377                     long lpszName = findName(p);
378                     path.trimTo(len).concat(lpszName).$();
379                     if (findType(p) == DT_DIR) {
380                         if (strcmp(lpszName, "..") || strcmp(lpszName, ".")) {
381                             continue;
382                         }
383 
384                         if ((errno = rmdir(path)) == 0) {
385                             continue;
386                         }
387 
388                     } else {
389                         if ((remove(path.address()))) {
390                             continue;
391                         }
392                         errno = Os.errno();
393                     }
394                     return errno;
395                 } while (findNext(p) > 0);
396             } finally {
397                 findClose(p);
398             }
399             if (rmdir(path.trimTo(len).$().address())) {
400                 return 0;
401             }
402             return Os.errno();
403         }
404 
405         return errno;
406     }

518     static boolean strcmp(long lpsz, CharSequence s) {
519         int len = s.length();
520         for (int i = 0; i < len; i++) {
521             byte b = Unsafe.getUnsafe().getByte(lpsz + i);
522             if (b == 0 || b != (byte) s.charAt(i)) {
523                 return false;
524             }
525         }
526         return Unsafe.getUnsafe().getByte(lpsz + len) == 0;
