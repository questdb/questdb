32 /**
33  * A copy implementation of CharSequenceObjHashMap. It is there to work with concrete classes
34  * and avoid incurring performance penalty of megamorphic virtual calls. These calls originate
35  * from calling charAt() on CharSequence interface. C2 compiler cannot inline these calls due to
36  * multiple implementations of charAt(). It resorts to a virtual method call via itable. ILP is the
37  * main victim of itable, suffering from non-deterministic performance loss. With this specific
38  * implementation of the map C2 compiler seems to be able to inline chatAt() calls and itables are
39  * no longer present in the async profiler.
40  * <p>
41  * This map is optimized for ASCII and UTF8 DirectByteCharSequence lookups.
42  */
43 public class ByteCharSequenceObjHashMap<V> implements Mutable {

169     public void removeAt0(int index) {
170         if (index < 0) {
171             int from = -index - 1;
172             erase(from);
173             free++;
174 
175             // after we have freed up a slot
176             // consider non-empty keys directly below
177             // they may have been a direct hit but because
178             // directly hit slot wasn't empty these keys would
179             // have moved.
180             //
181             // After slot if freed these keys require re-hash
182             from = (from + 1) & mask;
183             for (
184                     ByteCharSequence key = keys[from];
185                     key != null;
186                     from = (from + 1) & mask, key = keys[from]
187             ) {
188                 int hashCode = Hash.hashMem32(key);
189                 int idealHit = Hash.spread(hashCode) & mask;
190                 if (idealHit != from) {
191                     int to;
192                     if (keys[idealHit] != null) {
193                         to = probe(key, hashCode, idealHit);
194                     } else {
195                         to = idealHit;
196                     }
197 
198                     if (to > -1) {
199                         move(from, to);
200                     }
201                 }
202             }
203         }
204     }

259     private boolean putAt0(int index, ByteCharSequence key, V value) {
260         if (index < 0) {
261             values[-index - 1] = value;
262             return false;
263         } else {
264             keys[index] = key;
265             hashCodes[index] = Hash.hashMem32(key);
266             values[index] = value;
267             if (--free == 0) {
268                 rehash();
269             }
270             return true;
271         }
272     }

274     @SuppressWarnings({"unchecked"})
275     private void rehash() {
276         int size = size();
277         int newCapacity = capacity * 2;
278         free = capacity = newCapacity;
279         int len = Numbers.ceilPow2((int) (newCapacity / loadFactor));
280 
281         V[] oldValues = values;
282         ByteCharSequence[] oldKeys = keys;
283         int[] oldHashCodes = hashCodes;
284         keys = new ByteCharSequence[len];
285         hashCodes = new int[len];
286         values = (V[]) new Object[len];
287         Arrays.fill(keys, null);
288         mask = len - 1;
289 
290         free -= size;
291         for (int i = oldKeys.length; i-- > 0; ) {
292             ByteCharSequence key = oldKeys[i];
293             if (key != null) {
294                 final int index = keyIndex(key);
295                 keys[index] = key;
296                 hashCodes[index] = oldHashCodes[i];
297                 values[index] = oldValues[i];
298             }
299         }
300     }
