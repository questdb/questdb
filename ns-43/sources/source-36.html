31 /**
32  * Open addressing linear probing hash set specialized in storing long tuples.
33  *
34  * <p>
35  * Currently, it implements only limited amount of methods. Feel free to add more as you need them.
36  * <p>
37  * <b>Note:</b>
38  * Semantic of <code>capacity</code> differs from collections in the JDK. It says how many elements you
39  * can store in the set without triggering rehashing. This is consistent with other collections
40  * in the QuestDB project.
41  * <p>
42  * <b>Implementation notes:</b>
43  * The whole set is backed by a single long array. It uses a concept of slots: Each slot owns a tuple of two longs
44  * stored along each other for spatial locality.  Hence, when you want to get an index into a backing array you need to
45  * multiply slot by two to get the index of the first long and add one to get the index of the second long.
46  * <p>
47  * This class is not thread safe.
48  */

50     public static final SinkStrategy LONG_LONG_STRATEGY = (key1, key2, sink) -> {
51         sink.put('[');
52         Numbers.append(sink, key1, false);
53         sink.put(',');
54         Numbers.append(sink, key2, false);
55         sink.put(']');
56     };

58         sink.put('\'');
59         Numbers.appendUuid(key1, key2, sink);
60         sink.put('\'');

71     /**
72      * Creates a new set with a given capacity, load factor and no entry sentinel value.
73      * <p>
74      * No entry sentinel value is used to indicate that a slot is empty. It means that you cannot store a tuple
75      * where both longs are equal to no entry sentinel value.
76      *
77      * @param initialCapacity initial capacity of the set.
78      * @param loadFactor      load factor of the set.
79      * @param noEntryValue    no entry sentinel value.
80      * @param sinkStrategy    sink strategy
81      */
82     public LongLongHashSet(int initialCapacity, double loadFactor, long noEntryValue, SinkStrategy sinkStrategy) {
83         if (loadFactor <= 0d || loadFactor >= 1d) {
84             throw new IllegalArgumentException("0 < load factor < 1");
85         }
86         this.noEntryKeyValue = noEntryValue;
87         this.loadFactor = loadFactor;
88         this.capacity = Math.max(initialCapacity, MIN_INITIAL_CAPACITY);
89         int slots = Numbers.ceilPow2((int) (this.capacity / loadFactor));
90         this.values = new long[2 * slots];
91         this.mask = slots - 1;
92         this.sinkStrategy = sinkStrategy;
93         Arrays.fill(values, noEntryKeyValue);
94     }

97      * Adds a tuple to the set.
98      *
99      * @param key1 first long of the tuple
100      * @param key2 second long of the tuple
101      * @return true if the tuple was added, false if it was already in the set
102      */
103     public boolean add(long key1, long key2) {
104         if (key1 == noEntryKeyValue && key2 == noEntryKeyValue) {
105             throw new IllegalArgumentException("keys cannot be NO_ENTRY_KEY (" + noEntryKeyValue + ")");
106         }
107         int slot = keySlot(key1, key2);
108         if (slot < 0) {
109             return false;
110         }
111 
112         addAt(slot, key1, key2);
113         if (size == capacity) {
114             rehash();
115         }
116         return true;
117     }

151      * <p>
152      * If the tuple is already in the set, the slot is negative and the value is the index of the tuple.
153      * If the tuple is not in the set, the slot is positive and the value is the index of the first empty slot.

161         int slot = (hash & mask);

176         sink.put('[');
177         boolean pastFirst = false;
178         for (int i = 0, n = values.length; i < n; i += 2) {
179             long key1 = values[i];
180             long key2 = values[i + 1];
181             if (key1 != noEntryKeyValue || key2 != noEntryKeyValue) {
182                 if (pastFirst) {
183                     sink.put(',');
184                 }
185                 sinkStrategy.put(key1, key2, sink);

188         }

200     private int probe(long key1, long key2, int slot) {
201         do {
202             if (firstValue(values, slot) == noEntryKeyValue && secondValue(values, slot) == noEntryKeyValue) {
203                 return slot;
204             }
205             if (firstValue(values, slot) == key1 && secondValue(values, slot) == key2) {
206                 return -slot - 1;
207             }
208             slot = (slot + 1) & mask;
209         } while (true);
210     }

212     private void rehash() {
213         int newCapacity = capacity * 2;
214         int slots = Numbers.ceilPow2((int) (newCapacity / loadFactor));
215         if (slots < 0 || slots * 2 < 0) {
216             throw new IllegalStateException("cannot rehash, required capacity is too large. [current-capacity=" + capacity + ", load-factor=" + loadFactor + "]");
217         }
218         long[] newValues = new long[2 * slots];
219         Arrays.fill(newValues, noEntryKeyValue);
220         mask = slots - 1;
221         long[] oldKeys = this.values;
222         this.values = newValues;
223         int oldSlots = oldKeys.length / 2;
224         for (int i = 0; i < oldSlots; i++) {
225             long key1 = firstValue(oldKeys, i);
226             long key2 = secondValue(oldKeys, i);
227             if (key1 != noEntryKeyValue || key2 != noEntryKeyValue) {
228                 int slot = keySlot(key1, key2);
229                 set(slot, key1, key2);
230             }
231         }
232         capacity = newCapacity;
233     }

235     private void set(int slot, long key1, long key2) {
236         values[slot * 2] = key1;
237         values[slot * 2 + 1] = key2;
238     }
