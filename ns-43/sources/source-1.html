108     public void removeAt(int index) {
109         if (index < 0) {
110             int from = -index - 1;
111             erase(from);
112             free++;
113 
114             // after we have freed up a slot
115             // consider non-empty keys directly below
116             // they may have been a direct hit but because
117             // directly hit slot wasn't empty these keys would
118             // have moved.
119             //
120             // After slot if freed these keys require re-hash
121             from = (from + 1) & mask;
122             for (
123                     CharSequence key = keys[from];
124                     key != noEntryKey;
125                     from = (from + 1) & mask, key = keys[from]
126             ) {
127                 int idealHit = Hash.spread(Chars.hashCode(key)) & mask;
128                 if (idealHit != from) {
129                     int to;
130                     if (keys[idealHit] != noEntryKey) {
131                         to = probe(key, idealHit);
132                     } else {
133                         to = idealHit;
134                     }
135 
136                     if (to > -1) {

139                 }
140             }
141         }
