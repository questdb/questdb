398         int m = term.length();
399         if (m == 0) {
400             return -1;
401         }
402 
403         if (occurrence == 0) {
404             return -1;
405         }
406 
407         int foundIndex = -1;
408         int count = 0;
409         if (occurrence > 0) {
410             for (int i = sequenceLo; i < sequenceHi; i++) {
411                 if (foundIndex == -1) {
412                     if (sequenceHi - i < m) {
413                         return -1;
414                     }
415                     if (sequence.charAt(i) == term.charAt(0)) {
416                         foundIndex = i;
417                     }
418                 } else {    // first character matched, try to match the rest of the term
419                     if (sequence.charAt(i) != term.charAt(i - foundIndex)) {
420                         // start again from after where the first character was found
421                         i = foundIndex;
422                         foundIndex = -1;
423                     }
424                 }
425 
426                 if (foundIndex != -1 && i - foundIndex == m - 1) {
427                     count++;
428                     if (count == occurrence) {
429                         return foundIndex;
430                     } else {
431                         foundIndex = -1;
432                     }
433                 }
434             }
435         } else {    // if occurrence is negative, search in reverse
436             for (int i = sequenceHi - 1; i >= sequenceLo; i--) {
437                 if (foundIndex == -1) {
438                     if (i - sequenceLo + 1 < m) {
439                         return -1;
440                     }
441                     if (sequence.charAt(i) == term.charAt(m - 1)) {
442                         foundIndex = i;
443                     }
444                 } else {    // last character matched, try to match the rest of the term
445                     if (sequence.charAt(i) != term.charAt(m - 1 + i - foundIndex)) {
446                         // start again from after where the first character was found
447                         i = foundIndex;
448                         foundIndex = -1;

450                 }
