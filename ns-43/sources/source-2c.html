78         int h = head;

80         if (result == noEntryValue) {

103     private void allocateElements(int capacity) {
104         capacity = capacity < MIN_INITIAL_CAPACITY ? MIN_INITIAL_CAPACITY : Numbers.ceilPow2(capacity);
105         elements = new int[capacity];

116     private void doubleCapacity() {
117         assert head == tail;
118         int h = head;
119         int n = elements.length;
120         int r = n - h;
121         int newCapacity = n << 1;
122         if (newCapacity < 0) {
123             throw new IllegalStateException("Stack is too big");
124         }
125         int[] next = new int[newCapacity];
126         System.arraycopy(elements, h, next, 0, r);

128         Arrays.fill(next, r + h, newCapacity, noEntryValue);
129         elements = next;

132         mask = newCapacity - 1;
