80      * <p>

82      * Use {@link #checkDashesAndLength(CharSequence)} to validate that before calling this method.
83      * <p>

90     public static long parseHi(CharSequence uuid) throws NumericException {
91         long hi1;
92         long hi2;
93         long hi3;
94         hi1 = Numbers.parseHexLong(uuid, 0, FIRST_DASH_POS);
95         hi2 = Numbers.parseHexLong(uuid, FIRST_DASH_POS + 1, SECOND_DASH_POS);
96         hi3 = Numbers.parseHexLong(uuid, SECOND_DASH_POS + 1, THIRD_DASH_POS);
97         return (hi1 << 32) | (hi2 << 16) | hi3;

112     public static long parseLo(CharSequence uuid) throws NumericException {
