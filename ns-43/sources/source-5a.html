25 package io.questdb.std;
26 
27 import java.util.ArrayDeque;
28 
29 /**
30  * Weak object pool can be used in multithreaded environment. While
31  * weak object pool is a factory of new instances, when object is popped
32  * out of pool there is no reference to tie this object back to pool.
33  * <p>
34  * In multithreaded environment weak pools must be local to thread and
35  * objects they create can travel between pools, e.g. one thread gets
36  * object from pool and last thread to finish with this object then
37  * returns it to its own local pool.
38  * <p>
39  * In an extreme situation this object migration could keep feeding some
40  * pools until we run out of memory. Hence, we limit the maximum number of
41  * objects a pool can hold.
42  */
43 abstract class WeakObjectPoolBase<T> {
44     // package private for testing
45     final ArrayDeque<T> cache = new ArrayDeque<>();
46 
47     private final int initSize;
48     private final int maxSize;
49 
50     public WeakObjectPoolBase(int initSize) {
51         this.initSize = initSize;
52         this.maxSize = 2 * initSize;
53     }
54 
55     public T pop() {
56         final T obj = cache.poll();
57         return obj == null ? newInstance() : obj;
58     }
59 
60     void clear(T obj) {
61     }
62 
63     void close(T obj) {
64     }
65 
66     final void fill() {
67         for (int i = 0; i < initSize; i++) {
68             cache.add(newInstance());
69         }
70     }
71 
72     abstract T newInstance();
73 
74     boolean push(T obj) {
75         assert obj != null;
76         if (cache.size() < maxSize) {
77             clear(obj);
78             cache.push(obj);
79             return true;
80         } else {
81             close(obj);
82             return false;
83         }
84     }
85 }
