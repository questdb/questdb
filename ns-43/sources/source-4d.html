59     public static long forkExec(CharSequence args) {
60         ObjList<Path> paths = Chars.splitLpsz(args);
61         int n = paths.size();
62         try {
63             long argv = Unsafe.malloc((n + 1) * 8L, MemoryTag.NATIVE_DEFAULT);
64             try {
65                 long p = argv;
66                 for (int i = 0; i < n; i++) {
67                     Unsafe.getUnsafe().putLong(p, paths.getQuick(i).address());
68                     p += 8;
69                 }
70                 Unsafe.getUnsafe().putLong(p, 0);
71                 return forkExec(argv);
72             } finally {
73                 Unsafe.free(argv, n + 1, MemoryTag.NATIVE_DEFAULT);
74             }
75         } finally {

181     private static void loadLib(String lib) {
182         InputStream is = Os.class.getResourceAsStream(lib);
183         if (is == null) {
184             throw new FatalError("Internal error: cannot find " + lib + ", broken package?");
185         }
186 
187         try {
188             File tempLib = null;
189             try {
190                 int dot = lib.indexOf('.');
191                 tempLib = File.createTempFile(lib.substring(0, dot), lib.substring(dot));
192                 // copy to tempLib
193                 try (FileOutputStream out = new FileOutputStream(tempLib)) {
194                     byte[] buf = new byte[4096];
195                     while (true) {
196                         int read = is.read(buf);
197                         if (read == -1) {
198                             break;
199                         }
200                         out.write(buf, 0, read);
201                     }
202                 } finally {
203                     tempLib.deleteOnExit();
204                 }
205                 System.load(tempLib.getAbsolutePath());
206             } catch (IOException e) {
207                 throw new FatalError("Internal error: cannot unpack " + tempLib, e);
208             }
209         } finally {
210             Misc.free(is);
211         }
