263     private static boolean memeq0(long a, long b, long len) {
264         int i = 0;
265         for (; i + 7 < len; i += 8) {
266             if (Unsafe.getUnsafe().getLong(a + i) != Unsafe.getUnsafe().getLong(b + i)) {
267                 return false;
268             }
269         }
270         if (i + 3 < len) {
271             if (Unsafe.getUnsafe().getInt(a + i) != Unsafe.getUnsafe().getInt(b + i)) {
272                 return false;
273             }
274             i += 4;
275         }
276         for (; i < len; i++) {
277             if (Unsafe.getUnsafe().getByte(a + i) != Unsafe.getUnsafe().getByte(b + i)) {
278                 return false;
279             }
280         }
281         return true;
282     }
