36 /**
37  * This lock is not reentrant.
38  * If a thread holds a write lock and it tries to grab a lock again it will deadlock.
39  * If a thread holding a read lock tries to upgrade its lock to a write lock it must first release its read lock or it will deadlock.
40  * Threads waiting on a write lock have priority over threads waiting on a read lock.
41  * Threads waiting on a write lock are not resumed fairly.
42  *
43  * @author Patrick Mackinlay
44  */
45 public class SimpleReadWriteLock implements ReadWriteLock {
46     private final int MAX_READERS = Integer.MAX_VALUE / 2 - 1;
47     private final AtomicBoolean lock = new AtomicBoolean();
48     private final AtomicInteger nReaders = new AtomicInteger();

51     private final WriteLock writeLock = new WriteLock();
