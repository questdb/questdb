196     private void removeAt(int index) {
197         if (index < 0) {
198             int from = -index - 1;
199             erase(from);
200             free++;
201 
202             // after we have freed up a slot
203             // consider non-empty keys directly below
204             // they may have been a direct hit but because
205             // directly hit slot wasn't empty these keys would
206             // have moved.
207             //
208             // After slot if freed these keys require re-hash
209             from = (from + 1) & mask;
210             for (

212                     key != noEntryKey;
