557     public int poolUtf8(CharSequence cs) {
558         int index = utf8Cache.keyIndex(cs);
559         if (index > -1) {
560             putByte(0x01);
561             int n = cs.length();
562             int pos = buf.position();
563             putShort(0);
564             for (int i = 0; i < n; ) {
565                 final char c = cs.charAt(i++);
566                 if (c < 128) {
567                     putByte(c);
568                 } else {
569                     if (c < 2048) {
570                         putByte((char) (192 | c >> 6));
571                         putByte((char) (128 | c & 63));
572                     } else if (Character.isSurrogate(c)) {
573                         i = encodeSurrogate(c, cs, i, n);
574                     } else {
575                         putByte((char) (224 | c >> 12));
576                         putByte((char) (128 | c >> 6 & 63));
577                         putByte((char) (128 | c & 63));
578                     }
579                 }
580             }
581             buf.putShort(pos, (short) (buf.position() - pos - 2));
582             utf8Cache.putAt(index, cs, poolCount);
583             return this.poolCount++;
584         }
585 
586         return utf8Cache.valueAt(index);
587     }

718     private int encodeSurrogate(char c, CharSequence in, int pos, int hi) {
719         int dword;
720         if (Character.isHighSurrogate(c)) {
721             if (hi - pos < 1) {
722                 putByte('?');
723                 return pos;
724             } else {
725                 char c2 = in.charAt(pos++);
726                 if (Character.isLowSurrogate(c2)) {
727                     dword = Character.toCodePoint(c, c2);
728                 } else {
729                     putByte('?');
