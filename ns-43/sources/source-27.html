41     public IntLongAssociativeCache(int blocks, int rows) {
42         this.blocks = Math.max(MIN_BLOCKS, Numbers.ceilPow2(blocks));
43         rows = Math.max(MIN_ROWS, Numbers.ceilPow2(rows));
44 
45         int size = rows * this.blocks;
46         if (size < 0) {

49         this.keys = new int[size];
50         this.values = new long[size];

54     }

74     public int put(int key, long value) {
75         int lo = lo(key);
76         int ok = keys[lo + bmask];
77         System.arraycopy(keys, lo, keys, lo + 1, bmask);

86         for (int i = lo, hi = lo + blocks; i < hi; i++) {
