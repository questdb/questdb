51      * Hash function based on Knuth's multiplicative method.
52      * It provides good distribution even for a sequence of keys.
53      *
54      * @param k the long for which the hash will be calculated
55      * @return the hash
56      */
57     public static long fastLongMix(long k) {
58         // phi = 2^64 / goldenRatio
59         final long phi = 0x9E3779B97F4A7C15L;
60         long h = k * phi;
61         h ^= h >>> 32;
62         return h ^ (h >>> 16);
63     }

80      */
81     public static int hashMem32(ByteCharSequence seq) {
82         final int len = seq.length();
83         long h = 0;
84         int i = 0;
85         for (; i + 7 < len; i += 8) {
86             h = h * M2 + seq.longAt(i);
87         }
88         if (i + 3 < len) {
89             h = h * M2 + seq.intAt(i);
90             i += 4;
91         }
92         for (; i < len; i++) {
93             h = h * M2 + seq.byteAt(i);
94         }
95         h *= M2;
96         return (int) h ^ (int) (h >>> 32);
97     }

102      * <p>

149     /**
150      * (copied from ConcurrentHashMap)
151      * Spreads (XORs) higher bits of hash to lower and also forces top
152      * bit to 0. Because the table uses power-of-two masking, sets of
153      * hashes that vary only in bits above the current mask will
154      * always collide. (Among known examples are sets of Float keys
155      * holding consecutive whole numbers in small tables.)  So we
156      * apply a transform that spreads the impact of higher bits
157      * downward. There is a trade-off between speed, utility, and
158      * quality of bit-spreading. Because many common sets of hashes
159      * are already reasonably distributed (so don't benefit from
160      * spreading), and because we use trees to handle large sets of
161      * collisions in bins, we just XOR some shifted bits in the
162      * cheapest possible way to reduce systematic lossage, as well as
163      * to incorporate impact of the highest bits that would otherwise
164      * never be used in index calculations because of table bounds.
165      *
166      * @param h hash code
167      * @return adjusted hash code
168      */
169     public static int spread(int h) {
170         return (h ^ (h >>> 16)) & SPREAD_HASH_BITS;
171     }
