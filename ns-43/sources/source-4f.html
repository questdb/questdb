34     public static long alloc(ColumnTypes types, long capacity) {
35         // min capacity that works on all platforms is 16  
36         assert capacity >= 16;
37 
38         final int columnCount = types.getColumnCount();
39         final long mem = Unsafe.malloc(4L * columnCount, MemoryTag.NATIVE_ROSTI);
40         try {
41             long p = mem;
42             for (int i = 0; i < columnCount; i++) {
43                 Unsafe.getUnsafe().putInt(p, types.getColumnType(i));
44                 p += Integer.BYTES;
45             }
46             // this is not an exact size of memory allocated for Rosti, but this is useful to
47             // track that we free these maps
48             long pRosti = alloc(mem, columnCount, Numbers.ceilPow2(capacity) - 1);
49             if (pRosti != 0) {
50                 Unsafe.recordMemAlloc(getAllocMemory(pRosti), MemoryTag.NATIVE_ROSTI);
51             }
52             return pRosti;
53         } finally {
54             Unsafe.free(mem, 4L * columnCount, MemoryTag.NATIVE_ROSTI);
55         }
56     }

261     public static void printRosti(long pRosti) {
262         final long slots = getSlots(pRosti);
263         final long shift = getSlotShift(pRosti);
264         long ctrl = getCtrl(pRosti);
265         final long start = ctrl;
266         long count = getSize(pRosti);
267         System.out.println("size=" + count);
268         System.out.println("capacity=" + getCapacity(pRosti));
269         System.out.println("slot size=" + getSlotSize(pRosti));
270         System.out.println("slot shift=" + getSlotShift(pRosti));
271         System.out.print("initial slot=");
272         long initialSlot = getInitialValuesSlot(pRosti);
273         for (long i = 0, n = getSlotSize(pRosti); i < n; i++) {
274 
275             byte b = Unsafe.getUnsafe().getByte(initialSlot + i);
276             final int v;
277             if (b < 0) {
278                 v = 256 + b;
279             } else {
280                 v = b;
281             }
282 
283             if (v < 0x10) {
284                 System.out.print('0');
285                 System.out.print(hexDigits[b]);
286             } else {
287                 System.out.print(hexDigits[v / 0x10]);
288                 System.out.print(hexDigits[v % 0x10]);
289             }
290 
291             System.out.print(' ');
292         }
293         System.out.println();
294         while (count > 0) {
295             byte b = Unsafe.getUnsafe().getByte(ctrl);
296             if ((b & 0x80) == 0) {
297                 long p = slots + ((ctrl - start) << shift);
298                 System.out.println(Unsafe.getUnsafe().getInt(p) + " -> " + Unsafe.getUnsafe().getDouble(p + 12));
299                 count--;
300             }
301             ctrl++;
302         }
303     }
