79     private static Map createMap(
80             boolean smallMap,
81             CairoConfiguration configuration,
82             @Transient @NotNull ColumnTypes keyTypes
83     ) {
84         final int keyCapacity = smallMap ? configuration.getSqlSmallMapKeyCapacity() : configuration.getSqlMapKeyCapacity();
85         final int pageSize = smallMap ? configuration.getSqlSmallMapPageSize() : configuration.getSqlMapPageSize();
86         CharSequence mapType = configuration.getDefaultMapType();
87         if (Chars.equalsLowerCaseAscii(mapType, "fast")) {
88             return new FastMap(
89                     pageSize,
90                     keyTypes,
91                     keyCapacity,
92                     configuration.getSqlFastMapLoadFactor(),
93                     configuration.getSqlMapMaxResizes()
94             );
95         }
96 
97         if (Chars.equalsLowerCaseAscii(mapType, "compact")) {
98             return new CompactMap(
99                     pageSize,
100                     keyTypes,
101                     EmptyRecordMetadata.INSTANCE,
102                     keyCapacity,
103                     configuration.getSqlCompactMapLoadFactor(),
104                     configuration.getSqlMapMaxResizes(),
105                     configuration.getSqlMapMaxPages()
106             );
107         }
108         throw CairoException.critical(0).put("unknown map type: ").put(mapType);
109     }
