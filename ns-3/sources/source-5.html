36 /**
37  * FastMap is a general purpose off-heap hash table used to store intermediate data of join,
38  * group by, sample by queries, but not only. It provides {@link MapKey} and {@link MapValue},
39  * as well as {@link RecordCursor} interfaces for data access and modification.
40  * The preferred way to create a FastMap is {@link MapFactory}.
41  * <p>
42  * <strong>Important!</strong>
43  * Key and value structures must match the ones provided via lists of columns ({@link ColumnTypes})
44  * to the map constructor. Later put* calls made on {@link MapKey} and {@link MapValue} must match
45  * the declared column types to guarantee memory access safety.
46  * <p>
47  * Keys may be var-size, i.e. a key may contain string or binary columns, while values are expected
48  * to be fixed-size. Only insertions and updates operations are supported meaning that a key can't
49  * be removed from the map once it was inserted.
50  * <p>
51  * Map iteration provided by {@link RecordCursor} preserves the key insertion order.
52  * <p>
53  * The hash table is organized into two main parts:
54  * <ul>
55  * <li>1. Off-heap list for heap offsets</li>
56  * <li>2. Off-heap memory for key-value pairs a.k.a. "key memory"</li>
57  * </ul>
58  * The offset list contains [compressed_offset, hash_code] pairs. An offset value contains an offset to
59  * the address of a key-value pair in the key memory compressed to an int. Key-value pair addresses are
60  * 8 byte aligned, so a FastMap is capable of holding up to 32GB of data.
61  * <p>
62  * The offset list is used as a hash table with linear probing. So, a table resize allocates a new
63  * offset list and copies offsets there while the key memory stays as is.
64  * <p>
65  * Key-value pairs stored in the key memory may have the following layout:
66  * <pre>
67  * |       length         | Value columns 0..V | Key columns 0..K |
68  * +----------------------+--------------------+------------------+
69  * |      4 bytes         |         -          |        -         |
70  * +----------------------+--------------------+------------------+
71  * </pre>
72  * Length field is present for var-size keys only. It stores full key-value pair length in bytes.
73  */

317     private FastMapValue asNew(BaseKey keyWriter, int index, int hashCode, FastMapValue value) {
318         kPos = keyWriter.appendAddress;
319         // Align current pointer to 8 bytes, so that we can store compressed offsets.
320         if ((kPos & 0x7) != 0) {
321             kPos |= 0x7;
322             kPos++;
323         }
324         setPackedOffset(offsets, index, keyWriter.startAddress - kStart, hashCode);
325         if (--free == 0) {
326             rehash();
327         }
328         size++;
329         return valueOf(keyWriter.startAddress, true, value);
330     }

332     private FastMapValue probe0(BaseKey keyWriter, int index, int hashCode, FastMapValue value) {
333         long packedOffset;
334         long offset;
335         while ((offset = unpackOffset(packedOffset = getPackedOffset(offsets, index = (++index & mask)))) > -1) {
336             if (hashCode == unpackHashCode(packedOffset) && keyWriter.eq(offset)) {
337                 return valueOf(kStart + offset, false, value);
338             }
339         }
340         return asNew(keyWriter, index, hashCode, value);
341     }

354     private void rehash() {
355         int capacity = keyCapacity << 1;
356         mask = capacity - 1;
357         DirectLongList newOffsets = new DirectLongList(capacity, listMemoryTag);
358         newOffsets.setPos(capacity);
359         newOffsets.zero(0);
360 
361         for (int i = 0, k = (int) offsets.size(); i < k; i++) {
362             long packedOffset = getPackedOffset(offsets, i);
363             long offset = unpackOffset(packedOffset);
364             if (offset < 0) {
365                 continue;
366             }
367             int hashCode = unpackHashCode(packedOffset);
368             int index = hashCode & mask;
369             while (unpackOffset(getPackedOffset(newOffsets, index)) > -1) {
370                 index = (index + 1) & mask;
371             }
372             setPackedOffset(newOffsets, index, packedOffset);
373         }
374         offsets.close();
375         offsets = newOffsets;
376         free += (capacity - keyCapacity) * loadFactor;
377         keyCapacity = capacity;
378     }

380     private void resize(int size) {
381         if (nResizes < maxResizes) {
382             nResizes++;
383             long kCapacity = (kLimit - kStart) << 1;
384             long target = key.appendAddress + size - kStart;
385             if (kCapacity < target) {
386                 kCapacity = Numbers.ceilPow2(target);
387             }
388             if (kCapacity > MAX_HEAP_SIZE) {
389                 throw LimitOverflowException.instance().put("limit of ").put(MAX_HEAP_SIZE).put(" memory exceeded in FastMap");
390             }
391             long kAddress = Unsafe.realloc(this.kStart, this.capacity, kCapacity, mapMemoryTag);
392 
393             this.capacity = kCapacity;
394             long d = kAddress - this.kStart;
395             kPos += d;
396             key.startAddress += d;
397             key.appendAddress += d;
398 
399             assert kPos > 0;
400             assert key.startAddress > 0;
401             assert key.appendAddress > 0;
402 
403             this.kStart = kAddress;
404             this.kLimit = kAddress + kCapacity;
405         } else {
406             throw LimitOverflowException.instance().put("limit of ").put(maxResizes).put(" resizes exceeded in FastMap");
407         }
408     }

480         private MapValue createValue(FastMapValue value) {
481             commit();
482             // calculate hash remembering "key" structure
483             // [ len | value block | key offset block | key data block ]
484             int hashCode = hash();
485             int index = hashCode & mask;
486             long packedOffset = getPackedOffset(offsets, index);
487             long offset = unpackOffset(packedOffset);
488 
489             if (offset < 0) {
490                 return asNew(this, index, hashCode, value);
491             } else if (hashCode == unpackHashCode(packedOffset) && eq(offset)) {
492                 return valueOf(kStart + offset, false, value);
493             } else {
494                 return probe0(this, index, hashCode, value);
495             }
496         }

500             int hashCode = hash();

502             long packedOffset = getPackedOffset(offsets, index);
503             long offset = unpackOffset(packedOffset);
504
