75         this.split = valueOffsets != null ? valueOffsets.length : 0;
76 
77         int nColumns;
78         int keyIndexOffset;
79         if (valueTypes != null) {
80             keyIndexOffset = valueTypes.getColumnCount();
81             nColumns = keyTypes.getColumnCount() + valueTypes.getColumnCount();
82         } else {
83             keyIndexOffset = 0;
84             nColumns = keyTypes.getColumnCount();
85         }
86 
87         DirectCharSequence[] csA = null;
88         DirectCharSequence[] csB = null;
89         DirectBinarySequence[] bs = null;
90         Long256Impl[] long256A = null;
91         Long256Impl[] long256B = null;
92 
93         final ArrayColumnTypes keyTypesCopy = new ArrayColumnTypes();
94         for (int i = 0, n = keyTypes.getColumnCount(); i < n; i++) {
95             final int columnType = keyTypes.getColumnType(i);
96             keyTypesCopy.add(columnType);
97             switch (ColumnType.tagOf(columnType)) {
98                 case ColumnType.STRING:
99                     if (csA == null) {
100                         csA = new DirectCharSequence[nColumns];
101                         csB = new DirectCharSequence[nColumns];
102                     }
103                     csA[i + keyIndexOffset] = new DirectCharSequence();
104                     csB[i + keyIndexOffset] = new DirectCharSequence();
105                     break;
106                 case ColumnType.BINARY:
107                     if (bs == null) {
108                         bs = new DirectBinarySequence[nColumns];
109                     }
110                     bs[i + keyIndexOffset] = new DirectBinarySequence();
111                     break;
112                 case ColumnType.LONG256:
113                     if (long256A == null) {
114                         long256A = new Long256Impl[nColumns];
115                         long256B = new Long256Impl[nColumns];
116                     }
117                     long256A[i + keyIndexOffset] = new Long256Impl();
118                     long256B[i + keyIndexOffset] = new Long256Impl();
119                     break;
120                 default:
121                     break;
122             }
123         }
124         this.keyTypes = keyTypesCopy;
125 
126         if (valueTypes != null) {
127             for (int i = 0, n = valueTypes.getColumnCount(); i < n; i++) {
128                 if (ColumnType.tagOf(valueTypes.getColumnType(i)) == ColumnType.LONG256) {
129                     if (long256A == null) {
130                         long256A = new Long256Impl[nColumns];
131                         long256B = new Long256Impl[nColumns];
132                     }

134                     long256B[i] = new Long256Impl();
135                 }
136             }

141         this.bs = bs;
142         this.keyLong256A = long256A;
143         this.keyLong256B = long256B;

328     private long addressOfColumn(int index) {
329         if (index < split) {
330             return valueAddress + valueOffsets[index];
331         }
332 
333         if (index == split) {
334             return keyAddress;
335         }
336 
337         return addressOfKeyColumn(index - split);
338     }
339 
340     private long addressOfKeyColumn(int index) {
341         long addr = keyAddress;
342         int i = 0;
343         if (lastKeyIndex > -1 && index >= lastKeyIndex) {
344             addr += lastKeyOffset;
345             i = lastKeyIndex;
346         }
347         while (i < index) {
348             final int columnType = keyTypes.getColumnType(i);
349             final int size = ColumnType.sizeOf(columnType);
350             if (size > 0) {
351                 // Fixed-size type.
352                 addr += size;
353             } else {
354                 // Var-size type: string or binary.
355                 final int len = Unsafe.getUnsafe().getInt(addr);
356                 addr += Integer.BYTES;
357                 if (len != TableUtils.NULL_LEN) {
358                     if (ColumnType.isString(columnType)) {
359                         addr += (long) len << 1;
360                     } else {
361                         addr += len;
362                     }
363                 }
364             }
365             i++;
366         }
367         lastKeyOffset = (int) (addr - keyAddress);
368         lastKeyIndex = i;
369         return addr;
370     }
