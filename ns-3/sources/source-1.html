37 /**
38  * Storage structure to support queries such as "select distinct ...",
39  * group by queries and analytic functions. It can be thought of as a
40  * hash map with composite keys and values. Composite key is allowed
41  * to contain any number of fields of any type. In practice key will
42  * be a record of columns, including both of variable-length (string and binary)
43  * and of fixed-length types. Composite values can be any combination of
44  * fixed-length types only.
45  * <p>
46  * QMap constructor requires ColumnTypes instances for both key and values
47  * to determine optimal storage structure. Both keys and values are stored
48  * in individually sized memory cells. Types written to these cells are not
49  * yet validated. User must make sure correct types are written to correct cells.
50  * Failing to do so will lead to memory corruption. But lets not dwell on that.
51  * Map memory structure looks like this:
52  * <pre>
53  *     union cell {
54  *         byte b;
55  *         short s;
56  *         int i;
57  *         long l;
58  *         float f;
59  *         double d;
60  *         BOOL bool;
61  *     } cell;
62  * *
63  *     struct string {
64  *         int size;
65  *         char[] chars;
66  *     } string;
67  *
68  *     struct bin {
69  *         long size;
70  *         byte[] bytes;
71  *     } bin;
72  *
73  *     union entry_var {
74  *         string s;
75  *         bin b;
76  *     }
77  *
78  *     struct entry {
79  *         byte flag;
80  *         long size;
81  *         cell[] cells;
82  *         entry_var[] var;
83  *     }
84  * </pre>
85  * <p>
86  * QMap uses open addressing to keep track of entry offsets. Key hash
87  * code determines bucket. Entries in the same bucket are stored
88  * as mono-directional linked list. In this list the reference part
89  * is a one-byte distance from parent to the next list entry. The
90  * value of this byte is an index in fixed jump distance table. QMap
91  * also provides and maintains guarantee that each hash code root
92  * entry will be stored in bucket, which can be computed directly from
93  * this hash code.
94  */

319         @Override
320         public CompactMapValue createValue() {
321             long slot = calculateEntrySlot(currentEntryOffset, currentEntrySize);
322             long offset = getOffsetAt(slot);
323 
324             if (offset == -1) {
325                 // great, slot is empty, create new entry as direct hit
326                 return putNewEntryAt(slot, BITS_DIRECT_HIT);
327             }
328 
329             // check if this was a direct hit
330             final byte flag = entries.getByte(offset);
331             if ((flag & BITS_DIRECT_HIT) == 0) {
332                 // this is not a direct hit slot, reshuffle entries to free this slot up
333                 // then create new entry here with direct hit flag
334                 // we don't have to compare keys, because this isn't our hash code
335 
336                 // steps to take
337                 // 1. find parent of this rogue entry: compute hash code on key, find direct hit entry and
338                 //    descend until we find the sucker just above this one. We need this in order to change
339                 //    distance byte to keep structure consistent
340                 // 2. Find empty slot from parent
341                 // 3. Move current entry there
342                 // 4. For next entry - current will be parent
343                 // 5. Find empty slot from new parent
344                 // 6. Move entry there
345                 // 7. etc
346                 // as we shuffle these things we have to be careful not to use
347                 // entry we originally set out to free
348 
349                 if (moveForeignEntries(slot, offset)) {
350                     return putNewEntryAt(slot, BITS_DIRECT_HIT);
351                 }
352 
353                 grow();
354                 return createValue();
355             }
356 
357             // this is direct hit, scroll down all keys with same hashcode
358             // and exit this loop as soon as equality operator scores
359             // in simple terms check key equality on this key
360             if (cmp(offset)) {
361                 return found(offset);
362             }
363 
364             return appendEntry(offset, slot, flag);
365         }

367         @Override
368         public CompactMapValue findValue() {
369             long slot = calculateEntrySlot(currentEntryOffset, currentEntrySize);
370             long offset = getOffsetAt(slot);
371 
372             if (offset == -1) {
373                 return null;
374             } else {
375                 // check if this was a direct hit
376                 byte flag = entries.getByte(offset);
377                 if ((flag & BITS_DIRECT_HIT) == 0) {
378                     // not a direct hit? not our value
379                     return null;
380                 } else {
381                     // this is direct hit, scroll down all keys with same hashcode
382                     // and exit this loop as soon as equality operator scores
383 
384                     // in simple terms check key equality on this key
385                     if (cmp(offset)) {
386                         return found(offset);
387                     } else {
388 
389                         // then go down the list until either list ends or we find value
390                         int distance = flag & BITS_DISTANCE;
391                         while (distance > 0) {
392                             slot = nextSlot(slot, distance);
393                             offset = getOffsetAt(slot);
394 
395                             // this offset cannot be 0 when data structure is consistent
396                             assert offset != 0;
397 
398                             if (cmp(offset)) {
399                                 return found(offset);
400                             }
401                             distance = entries.getByte(offset) & BITS_DISTANCE;
402                         }
403                         // reached the end of the list, nothing found
404                         return null;
405                     }
406                 }
407             }

521         private CompactMapValue appendEntry(long offset, long slot, byte flag) {
522             int distance = flag & BITS_DISTANCE;
523             long original = offset;
524 
525             while (distance > 0) {
526                 slot = nextSlot(slot, distance);
527                 offset = getOffsetAt(slot);
528 
529                 // this offset cannot be 0 when data structure is consistent
530                 assert offset != 0;
531 
532                 distance = entries.getByte(offset) & BITS_DISTANCE;
533 
534                 if (cmp(offset)) {
535                     return found(offset);
536                 }
537             }
538 
539             // create entry at "nextOffset"
540             distance = findFreeSlot(slot);
541 
542             // we must have space here because to get to this place
543             // we must have checked available capacity
544             // anyway there is a test that ensures that going
545             // down the chain will not hit problems.
546             assert distance != 0;
547 
548             slot = nextSlot(slot, distance);
549 
550             // update distance on last entry in linked list
551             if (original == offset) {
552                 distance = distance | BITS_DIRECT_HIT;
553             }
554             entries.putByte(offset, (byte) distance);
555 
556             // add new entry
557             return putNewEntryAt(slot, (byte) 0);
558         }

568         private boolean cmp(long offset1, long offset2, long size) {
569             final long lim = size - size % 8L;
570 
571             for (long i = 0; i < lim; i += 8L) {
572                 if (entries.getLong(offset1 + i) != entries.getLong(offset2 + i)) {
573                     return false;
574                 }
575             }
576 
577             for (long i = lim; i < size; i++) {
578                 if (entries.getByte(offset1 + i) != entries.getByte(offset2 + i)) {
579                     return false;
580                 }
581             }
582 
583             return true;

603          * @param targetSlot slot of entry
604          * @return parent slot - offset in entrySlots

606         private long findParentSlot(long offset, long targetSlot) {
607             long parentSlot = calculateEntrySlot(offset, getEntrySize(offset));
608 
609             do {
610                 final int distance = entries.getByte(getOffsetAt(parentSlot)) & BITS_DISTANCE;
611                 assert distance != 0;
612                 final long nextSlot = nextSlot(parentSlot, distance);
613                 if (nextSlot == targetSlot) {
614                     return parentSlot;
615                 }
616                 parentSlot = nextSlot;
617             } while (true);
618         }

637         private void grow() {
638             if (nResizes < maxResizes) {
639                 nResizes++;
640                 // resize offsets virtual memory
641                 long appendPosition = entries.getAppendOffset();
642                 try {
643                     keyCapacity = keyCapacity * 2;
644                     configureCapacity();
645                     long target = size;
646                     long offset = 0L;
647                     while (target > 0) {
648                         final long entrySize = getEntrySize(offset);
649                         rehashEntry(offset, entrySize);
650                         offset += entrySize;
651                         target--;
652                     }
653                 } finally {
654                     entries.jumpTo(appendPosition);
655                 }
656             } else {
657                 throw LimitOverflowException.instance().put("limit of ").put(maxResizes).put(" resizes exceeded in CompactMap");
658             }
659         }

661         private boolean moveForeignEntries(final long slot, final long offset) {
662             // find parent slot for our direct hit
663             long parentSlot = findParentSlot(offset, slot);
664             // find entry for the parent slot, we will be updating distance here
665             long parentOffset = getOffsetAt(parentSlot);

675                     return false;

679                 if ((entries.getByte(parentOffset) & BITS_DIRECT_HIT) == 0) {

682                     entries.putByte(parentOffset, (byte) (dist | BITS_DIRECT_HIT));
683                 }

686                 final long nextSlot = nextSlot(parentSlot, dist);
687                 setOffsetAt(nextSlot, currentOffset);

690                 dist = entries.getByte(currentOffset) & BITS_DISTANCE;
691

693                     setOffsetAt(currentSlot, -1);
694                 }
695 
696                 if (dist == 0) {
697                     // done
698                     break;
699                 }
700 
701                 // parent of next entry will be current entry
702                 parentSlot = nextSlot;
703                 parentOffset = currentOffset;
704                 currentSlot = nextSlot(currentSlot, dist);
705                 currentOffset = getOffsetAt(currentSlot);
706             }
707

740         private void rehashEntry(long entryOffset, long currentEntrySize) {
741             long slot = calculateEntrySlot(entryOffset, currentEntrySize);
742             long offset = getOffsetAt(slot);
743 
744             if (offset == -1) {
745                 // great, slot is empty, create new entry as direct hit
746                 putEntryAt(entryOffset, slot, BITS_DIRECT_HIT);
747             } else {
748                 // check if this was a direct hit
749                 final byte flag = entries.getByte(offset);
750                 if ((flag & BITS_DIRECT_HIT) == 0) {
751                     moveForeignEntries(slot, offset);
752                     putEntryAt(entryOffset, slot, BITS_DIRECT_HIT);
753                 } else {
754                     // Our entries are now guaranteed to be unique. In case of direct hit we simply append
755                     // entry to end of list.
756                     int distance = flag & BITS_DISTANCE;
757                     long original = offset;
758 
759                     while (distance > 0) {
760                         slot = nextSlot(slot, distance);
761                         distance = entries.getByte(offset = getOffsetAt(slot)) & BITS_DISTANCE;
762                     }
763 
764                     // create entry at "nextOffset"
765                     distance = findFreeSlot(slot);
766                     assert distance != 0;
767                     slot = nextSlot(slot, distance);
768 
769                     // update distance on last entry in linked list
770                     if (original == offset) {
771                         entries.putByte(offset, (byte) (distance | BITS_DIRECT_HIT));
772                     } else {
773                         entries.putByte(offset, (byte) distance);
774                     }
775                     // add new entry
776                     putEntryAt(entryOffset, slot, (byte) 0);
777                 }
778             }
779         }
