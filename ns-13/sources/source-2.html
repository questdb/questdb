89     public void parse(long lo, long hi, JsonParser listener) throws JsonException {
90 
91         if (lo >= hi) {
92             return;
93         }
94 
95         long p = lo;
96         long valueStart = useCache ? lo : 0;
97         int posAtStart = position;
98         int state = this.state;
99         boolean quoted = this.quoted;
100         boolean ignoreNext = this.ignoreNext;
101         boolean useCache = this.useCache;
102         int objDepth = this.objDepth;
103         int arrayDepth = this.arrayDepth;
104 
105         while (p < hi) {
106             char c = (char) Unsafe.getUnsafe().getByte(p++);
107 
108             if (ignoreNext) {
109                 ignoreNext = false;
110                 continue;
111             }
112 
113             if (valueStart > 0) {
114                 if (quoted) {
115                     if (c == '\\') {
116                         ignoreNext = true;
117                         continue;
118                     }
119 
120                     if (c != '"') {
121                         continue;
122                     }
123                 } else if (isNotATerminator(c)) {
124                     continue;
125                 }
126 
127                 int vp = (int) (posAtStart + valueStart - lo + 1 - cacheSize);
128                 if (state == S_EXPECT_NAME || state == S_EXPECT_FIRST_NAME) {
129                     listener.onEvent(EVT_NAME, getCharSequence(valueStart, p, vp), vp);
130                     state = S_EXPECT_COLON;
131                 } else {
132                     listener.onEvent(arrayDepth > 0 ? EVT_ARRAY_VALUE : EVT_VALUE, getCharSequence(valueStart, p, vp), vp);
133                     state = S_EXPECT_COMMA;
134                 }
135 
136                 valueStart = 0;
137                 cacheSize = 0;
138                 useCache = false;
139 
140                 if (quoted) {
141                     // skip the quote mark
142                     continue;
143                 }
144             }
145 
146             switch (c) {
147                 case '{':
148                     if (state != S_START && state != S_EXPECT_VALUE) {
149                         throw JsonException.$((int) (posAtStart + p - lo), "{ is not expected here");
150                     }
151                     arrayDepthStack.push(arrayDepth);
152                     arrayDepth = 0;
153 
154                     listener.onEvent(EVT_OBJ_START, null, (int) (posAtStart + p - lo));
155                     objDepth++;
156                     state = S_EXPECT_FIRST_NAME;
157                     break;
158                 case '}':
159                     if (arrayDepth > 0) {
160                         throw JsonException.$((int) (posAtStart + p - lo), "} is not expected here. You have non-terminated array");
161                     }
162 
163                     if (objDepth > 0) {
164                         switch (state) {
165                             case S_EXPECT_VALUE:
166                                 throw JsonException.$((int) (posAtStart + p - lo - 1), "Attribute value expected");
167                             case S_EXPECT_NAME:
168                                 throw JsonException.$((int) (posAtStart + p - lo - 1), "Attribute name expected");
169                             default:
170                                 break;
171                         }
172                         listener.onEvent(EVT_OBJ_END, null, (int) (posAtStart + p - lo));
173                         objDepth--;
174                         arrayDepth = arrayDepthStack.pop();
175                         state = S_EXPECT_COMMA;
176                     } else {
177                         throw JsonException.$((int) (posAtStart + p - lo), "Dangling }");
178                     }
179                     break;
180                 case '[':
181                     if (state != S_START && state != S_EXPECT_VALUE) {
182                         throw JsonException.$((int) (posAtStart + p - lo), "[ is not expected here");
183                     }
184 
185                     listener.onEvent(EVT_ARRAY_START, null, (int) (posAtStart + p - lo));
186                     objDepthStack.push(objDepth);
187                     objDepth = 0;
188                     arrayDepth++;
189                     state = S_EXPECT_VALUE;
190                     break;
191                 case ']':
192                     if (objDepth > 0) {
193                         throw JsonException.$((int) (posAtStart + p - lo), "] is not expected here. You have non-terminated object");
194                     }
195 
196                     if (arrayDepth == 0) {
197                         throw JsonException.$((int) (posAtStart + p - lo), "Dangling ]");
198                     }
199 
200                     listener.onEvent(EVT_ARRAY_END, null, (int) (posAtStart + p - lo));
201                     arrayDepth--;
202                     objDepth = objDepthStack.pop();
203                     state = S_EXPECT_COMMA;
204                     break;
205                 case ' ':
206                 case '\t':
207                 case '\n':
208                 case '\r':
209                     break;
210                 case ',':
211                     if (state != S_EXPECT_COMMA) {
212                         throw JsonException.$((int) (posAtStart + p - lo), "Unexpected comma");
213                     }
214                     if (arrayDepth > 0) {
215                         state = S_EXPECT_VALUE;
216                     } else {
217                         state = S_EXPECT_NAME;
218                     }
219                     break;
220                 case ':':
221                     if (state != S_EXPECT_COLON) {
222                         throw JsonException.$((int) (posAtStart + p - lo), "Misplaced ':'");
223                     }
224                     state = S_EXPECT_VALUE;
225                     break;
226                 case '"':
227                     if (state != S_EXPECT_NAME && state != S_EXPECT_FIRST_NAME && state != S_EXPECT_VALUE) {
228                         throw JsonException.$((int) (posAtStart + p - lo), "Unexpected quote '\"'");
229                     }
230                     valueStart = p;
231                     quoted = true;
232                     break;
233                 default:
234                     if (state != S_EXPECT_VALUE) {
235                         throw JsonException.$((int) (posAtStart + p - lo), "Unexpected symbol");
236                     }
237                     // this isn't a quote, include this character
238                     valueStart = p - 1;
239                     quoted = false;
240                     break;
241             }
242         }
243 
244         this.position = (int) (posAtStart + p - lo);
245 
246         this.state = state;
247         this.quoted = quoted;
248         this.ignoreNext = ignoreNext;
249         this.objDepth = objDepth;
250         this.arrayDepth = arrayDepth;
251 
252         if (valueStart > 0) {
253             // stash
254             addToStash(valueStart, hi);
255             useCache = true;
256         }
257         this.useCache = useCache;
258     }

282     private void addToStash(long lo, long hi) throws JsonException {
283         final int len = (int) (hi - lo);
284         int n = len + cacheSize;
285         if (n > cacheCapacity) {
286             extendCache(Numbers.ceilPow2(n));
287         }
288 
289         if (len > 0) {
290             Vect.memcpy(cache + cacheSize, lo, len);
291             cacheSize += len;
292         }
293     }

295     private void extendCache(int n) throws JsonException {
296         if (n > cacheSizeLimit) {
297             throw JsonException.$(position, "String is too long");
298         }
299         long ptr = Unsafe.malloc(n, MemoryTag.NATIVE_TEXT_PARSER_RSS);
300         if (cacheCapacity > 0) {
301             Vect.memcpy(ptr, cache, cacheSize);
302             Unsafe.free(cache, cacheCapacity, MemoryTag.NATIVE_TEXT_PARSER_RSS);
303         }
304         cacheCapacity = n;
305         cache = ptr;
306     }

308     private CharSequence getCharSequence(long lo, long hi, int position) throws JsonException {
309         sink.clear();
310         if (cacheSize == 0) {
311             if (!Chars.utf8Decode(lo, hi - 1, sink)) {
312                 throw unsupportedEncoding(position);
313             }
314         } else {
315             utf8DecodeCacheAndBuffer(lo, hi - 1, position);
316         }
317         return sink;
318     }

320     private void utf8DecodeCacheAndBuffer(long lo, long hi, int position) throws JsonException {
321         long p = cache;
322         long lim = cache + cacheSize;
323         int loOffset = 0;
324         while (p < lim) {
325             byte b = Unsafe.getUnsafe().getByte(p);
326             if (b < 0) {
327                 int len = Chars.utf8DecodeMultiByte(p, lim, b, sink);
328                 if (len != -1) {
329                     p += len;
330                 } else {
331                     // UTF8 error, check if we can switch to main buffer
332                     final int cacheRemaining = (int) (lim - p);
333                     if (cacheRemaining > 4) {
334                         throw unsupportedEncoding(position);
335                     } else {
336                         // add up to four bytes to stash and try again
337                         int n = (int) Math.max(4, hi - lo);
338 
339                         // keep offset of 'p' in case stash re-sizes and updates pointers
340                         long offset = p - cache;
341                         addToStash(lo, lo + n);
342                         assert offset < cacheSize;
343                         assert cacheSize <= cacheCapacity;
344                         len = Chars.utf8DecodeMultiByte(cache + offset, cache + cacheSize, b, sink);
345                         if (len == -1) {
346                             // definitely UTF8 error
347                             throw unsupportedEncoding(position);
348                         }
349                         // right, decoding was a success, we must continue with decoding main buffer from
350                         // non-zero offset, because we used some of the bytes to decode cache
351                         loOffset = len - cacheRemaining;
352                         p += cacheRemaining;
353                     }
354                 }
355             } else {
356                 sink.put((char) b);
357                 ++p;
358             }
359         }
360 
361         p = lo + loOffset;
362         while (p < hi) {
363             byte b = Unsafe.getUnsafe().getByte(p);
364             if (b < 0) {
365                 int len = Chars.utf8DecodeMultiByte(p, hi, b, sink);
366                 if (len == -1) {
367                     throw unsupportedEncoding(position);
368                 }
369                 p += len;
370             } else {
371                 sink.put((char) b);
372                 ++p;
373             }
374         }
375     }
