44     @Override
45     public Function newInstance(
46             int position,
47             ObjList<Function> args,
48             IntList argPositions,
49             CairoConfiguration configuration,
50             SqlExecutionContext sqlExecutionContext
51     ) throws SqlException {
52         if (args.size() < 2) {
53             throw SqlException.$(position, "coalesce can be used with 2 or more arguments");
54         }
55         if (args.size() > 2) {
56             // copy, args collection will be reused by sql parser
57             args = new ObjList<>(args);
58         }
59 
60         // Similar to CASE function.
61         // compute return type in this loop
62         final int argsSize = args.size();
63         int returnType = -1;
64         for (int i = 0; i < argsSize; i++) {
65             returnType = CaseCommon.getCommonType(returnType, args.getQuick(i).getType(), argPositions.getQuick(i));
66         }
67         switch (ColumnType.tagOf(returnType)) {
68             case ColumnType.DOUBLE:
69                 return argsSize == 2 ? new TwoDoubleCoalesceFunction(args) : new DoubleCoalesceFunction(args, argsSize);
70             case ColumnType.DATE:
71                 return argsSize == 2 ? new TwoDateCoalesceFunction(args) : new DateCoalesceFunction(args, argsSize);
72             case ColumnType.TIMESTAMP:
73                 return argsSize == 2 ? new TwoTimestampCoalesceFunction(args) : new TimestampCoalesceFunction(args);
74             case ColumnType.LONG:
75                 return argsSize == 2 ? new TwoLongCoalesceFunction(args) : new LongCoalesceFunction(args, argsSize);
76             case ColumnType.LONG256:
77                 return argsSize == 2 ? new TwoLong256CoalesceFunction(args) : new Long256CoalesceFunction(args);
78             case ColumnType.INT:
79                 return argsSize == 2 ? new TwoIntCoalesceFunction(args) : new IntCoalesceFunction(args, argsSize);
80             case ColumnType.FLOAT:
81                 return argsSize == 2 ? new TwoFloatCoalesceFunction(args) : new FloatCoalesceFunction(args, argsSize);
82             case ColumnType.STRING:
83             case ColumnType.SYMBOL:
84                 if (argsSize == 2) {
85                     int type0 = ColumnType.tagOf(args.getQuick(0).getType());
86                     if (type0 != ColumnType.tagOf(args.getQuick(1).getType())) {
87                         return new TwoSymStrCoalesceFunction(args);
88                     } else if (type0 == ColumnType.SYMBOL) {
89                         return new TwoSymCoalesceFunction(args);
90                     } else {
91                         return new TwoStrCoalesceFunction(args);
92                     }
93                 }
94                 return new SymStrCoalesceFunction(args, argsSize);
95             case ColumnType.UUID:
96                 return argsSize == 2 ? new TwoUuidCoalesceFunction(args) : new UuidCoalesceFunction(args, argsSize);
97             case ColumnType.BOOLEAN:
98             case ColumnType.SHORT:
99             case ColumnType.BYTE:
100             case ColumnType.CHAR:
101                 // Null on these data types not supported
102                 return args.getQuick(0);
103             default:
104                 throw SqlException.$(position, "coalesce cannot be used with ")
105                         .put(ColumnType.nameOf(returnType))
106                         .put(" data type");
107         }
108     }
