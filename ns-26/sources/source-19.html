167     private Function getCharSequenceKeyedFunction(
168             ObjList<Function> args,
169             IntList argPositions,
170             int position,
171             int n,
172             Function keyFunction,
173             int valueType,
174             Function elseBranch,
175             CharSequenceMethod method
176     ) throws SqlException {
177         final CharSequenceObjHashMap<Function> map = new CharSequenceObjHashMap<>();
178         final ObjList<Function> argsToPoke = new ObjList<>();
179         Function nullFunc = null;
180         for (int i = 1; i < n; i += 2) {
181             final Function fun = args.getQuick(i);
182             final CharSequence key = method.getKey(fun, null);
183             if (key == null) {
184                 nullFunc = args.getQuick(i + 1);
185             } else {
186                 final int index = map.keyIndex(key);
187                 if (index < 0) {
188                     throw SqlException.$(argPositions.getQuick(i), "duplicate branch");
189                 }
190                 map.putAt(index, key, args.getQuick(i + 1));
191                 argsToPoke.add(args.getQuick(i + 1));
192             }
193         }
194 
195         final Function elseB = getElseFunction(valueType, elseBranch);
196         final CaseFunctionPicker picker;
197         if (nullFunc == null) {
198             picker = record -> {
199                 final CharSequence value = method.getKey(keyFunction, record);
200                 if (value != null) {
201                     final int index = map.keyIndex(value);
202                     if (index < 0) {
203                         return map.valueAtQuick(index);
204                     }
205                 }
206                 return elseB;
207             };
208         } else {
209             final Function nullFuncRef = nullFunc;
210             picker = record -> {
211                 final CharSequence value = method.getKey(keyFunction, record);
212                 if (value == null) {
213                     return nullFuncRef;
214                 }
215                 final int index = map.keyIndex(value);
216                 if (index < 0) {
217                     return map.valueAtQuick(index);
218                 }
219                 return elseB;
220             };
221             argsToPoke.add(nullFunc);
222         }
223         argsToPoke.add(elseB);
224         argsToPoke.add(keyFunction);
225         return CaseCommon.getCaseFunction(position, valueType, picker, argsToPoke);
226     }

232     private Function getIfElseFunction(
233             ObjList<Function> args,
234             IntList argPositions,
235             int position,
236             int n,
237             Function keyFunction,
238             int returnType,
239             Function elseBranch
240     ) throws SqlException {
241         final CaseFunctionPicker picker;
242         final ObjList<Function> argsToPoke;
243         if (n == 3) {
244             // only one conditional branch
245             boolean value = args.getQuick(1).getBool(null);
246             final Function branch = args.getQuick(2);
247 
248             final Function elseB = getElseFunction(returnType, elseBranch);
249 
250             if (value) {
251                 picker = record -> keyFunction.getBool(record) ? branch : elseB;
252             } else {
253                 picker = record -> keyFunction.getBool(record) ? elseB : branch;
254             }
255 
256             argsToPoke = new ObjList<>();
257             argsToPoke.add(keyFunction);
258             argsToPoke.add(elseB);
259             argsToPoke.add(branch);
260 
261         } else if (n == 5) {
262             final boolean a = args.getQuick(1).getBool(null);
263             final Function branchA = args.getQuick(2);
264             final boolean b = args.getQuick(3).getBool(null);
265             final Function branchB = args.getQuick(4);
266 
267             if (a && b || !a && !b) {
268                 throw SqlException.$(argPositions.getQuick(3), "duplicate branch");
269             }
270 
271             if (a) {
272                 picker = record -> keyFunction.getBool(record) ? branchA : branchB;
273             } else {
274                 picker = record -> keyFunction.getBool(record) ? branchB : branchA;
275             }
276 
277             argsToPoke = new ObjList<>();
278             argsToPoke.add(keyFunction);
279             argsToPoke.add(branchA);
280             argsToPoke.add(branchB);
281         } else {
282             throw SqlException.$(argPositions.getQuick(5), "too many branches");
283         }
284 
285         return CaseCommon.getCaseFunction(position, returnType, picker, argsToPoke);
286     }
