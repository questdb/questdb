55     static Function getCaseFunction(int position, int returnType, CaseFunctionPicker picker, ObjList<Function> args) throws SqlException {
56         if (ColumnType.isGeoHash(returnType)) {
57             switch (ColumnType.tagOf(returnType)) {
58                 case ColumnType.GEOBYTE:
59                     return new GeoByteCaseFunction(returnType, picker, args);
60                 case ColumnType.GEOSHORT:
61                     return new GeoShortCaseFunction(returnType, picker, args);
62                 case ColumnType.GEOINT:
63                     return new GeoIntCaseFunction(returnType, picker, args);
64                 default:
65                     return new GeoLongCaseFunction(returnType, picker, args);
66             }
67         }
68 
69         return getCaseFunctionConstructor(position, returnType).getInstance(position, picker, args);
70     }

72     static Function getCastFunction(
73             Function arg,
74             int argPosition,
75             int toType,
76             CairoConfiguration configuration,
77             SqlExecutionContext sqlExecutionContext
78     ) throws SqlException {
79         if (ColumnType.isNull(arg.getType())) {
80             return Constants.getNullConstant(toType);
81         }
82         final int keyIndex = castFactories.keyIndex(Numbers.encodeLowHighInts(arg.getType(), toType));
83         if (keyIndex < 0) {
84             FunctionFactory fact = castFactories.valueAt(keyIndex);
85             ObjList<Function> args = tlArgs.get();
86             args.clear();
87             args.add(arg);
88 
89             IntList argPositions = tlArgPositions.get();
90             argPositions.clear();
91             argPositions.add(argPosition);
92             return fact.newInstance(0, args, argPositions, configuration, sqlExecutionContext);
93         }
94 
95         return arg;
96     }

98     static int getCommonType(int commonType, int valueType, int valuePos) throws SqlException {
99         if (commonType == -1 || ColumnType.isNull(commonType)) {
100             return valueType;
101         }
102         if (ColumnType.isNull(valueType)) {
103             return commonType;
104         }
105         final int type = typeEscalationMap.get(Numbers.encodeLowHighInts(commonType, valueType));
106 
107         if (type == LongIntHashMap.NO_ENTRY_VALUE) {
108             throw SqlException.$(valuePos, "inconvertible types ").put(ColumnType.nameOf(valueType)).put(" to ").put(ColumnType.nameOf(commonType));
109         }
110         return type;
111     }
