777     /**
778      * Tries to compute {@code significand * 10^power} exactly using
779      * a fast algorithm; and if {@code isNegative} is true, negate the result.
780      * <p>
781      * This function will only work in some cases, when it does not work it
782      * returns NaN. This should work *most of the time* (like 99% of the time).
783      * We assume that power is in the
784      * [{@value #DOUBLE_MIN_EXPONENT_POWER_OF_TEN}, {@value #DOUBLE_MAX_EXPONENT_POWER_OF_TEN}]
785      * interval: the caller is responsible for this check.
786      *
787      * @param isNegative  whether the number is negative
788      * @param significand uint64 the significand
789      * @param power       the exponent number (the power)
790      * @return the computed double on success, {@link Double#NaN} on failure
791      */
792     static double tryDecFloatToDouble(boolean isNegative, long significand, int power) {
793         // we start with a fast path
794         // It was described in Clinger WD (1990).
795         if (-22 <= power && power <= 22 && Long.compareUnsigned(significand, (1L << DOUBLE_SIGNIFICAND_WIDTH) - 1) <= 0) {
796             // convert the integer into a double. This is lossless since
797             // 0 <= i <= 2^53 - 1.
798             double d = (double) significand;
799             //
800             // The general idea is as follows.
801             // If 0 <= s < 2^53 and if 10^0 <= p <= 10^22 then
802             // 1) Both s and p can be represented exactly as 64-bit floating-point values
803             // 2) Because s and p can be represented exactly as floating-point values,
804             // then s * p and s / p will produce correctly rounded values.
805             //
806             if (power < 0) {
807                 d = d / DOUBLE_POWERS_OF_TEN[-power];
808             } else {
809                 d = d * DOUBLE_POWERS_OF_TEN[power];
810             }
811             return (isNegative) ? -d : d;
812         }
813 
814 
815         // The fast path has now failed, so we are falling back on the slower path.
816 
817         // We are going to need to do some 64-bit arithmetic to get a more precise product.
818         // We use a table lookup approach.
819         // It is safe because
820         // power >= DOUBLE_MIN_EXPONENT_POWER_OF_TEN
821         // and power <= DOUBLE_MAX_EXPONENT_POWER_OF_TEN
822         // We recover the mantissa of the power, it has a leading 1. It is always
823         // rounded down.
824         long factorMantissa = MANTISSA_64[power - DOUBLE_MIN_EXPONENT_POWER_OF_TEN];
825 
826 
827         // The exponent is 1023 + 64 + power + floor(log(5**power)/log(2)).
828         //
829         // 1023 is the exponent bias.
830         // The 64 comes from the fact that we use a 64-bit word.
831         //
832         // Computing floor(log(5**power)/log(2)) could be
833         // slow. Instead, we use a fast function.
834         //
835         // For power in (-400,350), we have that
836         // (((152170 + 65536) * power ) >> 16);
837         // is equal to
838         //  floor(log(5**power)/log(2)) + power when power >= 0,
839         // and it is equal to
840         //  ceil(log(5**-power)/log(2)) + power when power < 0
841         //
842         //
843         // The 65536 is (1<<16) and corresponds to
844         // (65536 * power) >> 16 ---> power
845         //
846         // ((152170 * power ) >> 16) is equal to
847         // floor(log(5**power)/log(2))
848         //
849         // Note that this is not magic: 152170/(1<<16) is
850         // approximately equal to log(5)/log(2).
851         // The 1<<16 value is a power of two; we could use a
852         // larger power of 2 if we wanted to.
853         //
854         long exponent = (((152170L + 65536L) * power) >> 16) + DOUBLE_EXPONENT_BIAS + 64;
855         // We want the most significant bit of digits to be 1. Shift if needed.
856         int lz = Long.numberOfLeadingZeros(significand);
857         significand <<= lz;
858         // We want the most significant 64 bits of the product. We know
859         // this will be non-zero because the most significant bit of digits is
860         // 1.
861         long x01 = significand & 0xffffffffL, x11 = significand >>> 32;
862         long y01 = factorMantissa & 0xffffffffL, y11 = factorMantissa >>> 32;
863         long p111 = x11 * y11, p011 = x01 * y11;
864         long p101 = x11 * y01, p001 = x01 * y01;
865 
866         // 64-bit product + two 32-bit values
867         long middle1 = p101 + (p001 >>> 32) + (p011 & 0xffffffffL);
868         long lower = (middle1 << 32) | (p001 & 0xffffffffL);
869         long upper = p111 + (middle1 >>> 32) + (p011 >>> 32);
870         // We know that upper has at most one leading zero because
871         // both i and factor_mantissa have a leading one. This means
872         // that the result is at least as large as ((1<<63)*(1<<63))/(1<<64).
873 
874         // As long as the first 9 bits of "upper" are not "1", then we
875         // know that we have an exact computed value for the leading
876         // 55 bits because any imprecision would play out as a +1, in
877         // the worst case.
878         // Having 55 bits is necessary because
879         // we need 53 bits for the mantissa, but we have to have one rounding bit and
880         // we can waste a bit if the most significant bit of the product is zero.
881         // We expect this next branch to be rarely taken (say 1% of the time).
882         // When (upper & 0x1FF) == 0x1FF, it can be common for
883         // lower + i < lower to be true (proba. much higher than 1%).
884         if ((upper & 0x1ffL) == 0x1ffL && Long.compareUnsigned(lower + significand, lower) < 0) {
885             long factorMantissaLow =
886                     MANTISSA_128[power - DOUBLE_MIN_EXPONENT_POWER_OF_TEN];
887             // next, we compute the 64-bit x 128-bit multiplication, getting a 192-bit
888             // result (three 64-bit values)
889             //before Java 18
890             long x0 = significand & 0xffffffffL, x1 = significand >>> 32;
891             long y0 = factorMantissaLow & 0xffffffffL, y1 = factorMantissaLow >>> 32;
892             long p11 = x1 * y1, p01 = x0 * y1;
893             long p10 = x1 * y0, p00 = x0 * y0;
894 
895             // 64-bit product + two 32-bit values
896             long middle = p10 + (p00 >>> 32) + (p01 & 0xffffffffL);
897             long productLow = (middle << 32) | (p00 & 0xffffffffL);
898             long productMiddle2 = p11 + (middle >>> 32) + (p01 >>> 32);
899             long productHigh = upper;
900             long productMiddle = lower + productMiddle2;
901             if (Long.compareUnsigned(productMiddle, lower) < 0) {
902                 productHigh++; // overflow carry
903             }
904 
905             // we want to check whether mantissa *i + i would affect our result
