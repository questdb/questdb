85      * char[] chars = ...;

95         long fval = first - 0x0030_0030_0030_0030L;
96         long sval = second - 0x0030_0030_0030_0030L;

99         // or greater than '9' (0x0039).
100         // We have 0x007f - 0x0039 = 0x0046.
101         // The predicate is true if the hsb of a byte is set: (predicate & 0xff80) != 0.
102         long fpre = first + 0x0046_0046_0046_0046L | fval;

113      * Tries to parse eight digits from a long using the
114      * 'SIMD within a register technique' (SWAR).
115      *
116      * <pre>{@literal
117      * byte[] bytes = ...;
118      * long value  = ((bytes[7]&0xffL)<<56)
119      *             | ((bytes[6]&0xffL)<<48)
120      *             | ((bytes[5]&0xffL)<<40)
121      *             | ((bytes[4]&0xffL)<<32)
122      *             | ((bytes[3]&0xffL)<<24)
123      *             | ((bytes[2]&0xffL)<<16)
124      *             | ((bytes[1]&0xffL)<< 8)
125      *             |  (bytes[0]&0xffL);
126      * }</pre>
127      *
128      * @param chunk contains 8 ascii characters in little endian order
129      * @return the parsed number,
130      * returns a negative value if {@code value} does not contain 8 digits
131      */
132     static int tryToParseEightDigitsUtf8(long chunk) {
133         // Create a predicate for all bytes which are greater than '0' (0x30).
134         // The predicate is true if the hsb of a byte is set: (predicate & 0x80) != 0.
135         long val = chunk - 0x3030303030303030L;
136         long predicate = ((chunk + 0x4646464646464646L) | val) & 0x8080808080808080L;
137         if (predicate != 0L) {
138             return -1;
139         }
140 
141         // The last 2 multiplications are independent of each other.
142         val = val * (1 + (10 << 8)) >>> 8;
143         val = (val & 0xff_000000ffL) * (100 + (100_0000L << 32))
144                 + (val >>> 16 & 0xff_000000ffL) * (1 + (1_0000L << 32)) >>> 32;
145         return (int) val;
146     }

228     /**
229      * Tries to parse eight digits from a long using the
230      * 'SIMD within a register technique' (SWAR).
231      *
232      * @param chunk contains 8 ascii characters in big endian order
233      * @return the parsed number,
234      * returns a negative value if {@code value} does not contain 8 digits
235      */
236     static long tryToParseEightHexDigitsUtf8(long chunk) {
237         // The following code is based on the technique presented in the paper
238         // by Leslie Lamport.
239 
240 
241         // Subtract character '0' (0x30) from each of the eight characters
242         long vec = chunk - 0x30_30_30_30_30_30_30_30L;
243 
244         // Create a predicate for all bytes which are greater than '9'-'0' (0x09).
245         // The predicate is true if the hsb of a byte is set: (predicate & 0x80) != 0.
246         long gt_09 = vec + (0x09_09_09_09_09_09_09_09L ^ 0x7f_7f_7f_7f_7f_7f_7f_7fL);
247         gt_09 &= 0x80_80_80_80_80_80_80_80L;
248         // Create a predicate for all bytes which are greater or equal 'a'-'0' (0x30).
249         // The predicate is true if the hsb of a byte is set.
250         long ge_30 = vec + (0x30303030_30303030L ^ 0x7f_7f_7f_7f_7f_7f_7f_7fL);
251         ge_30 &= 0x80_80_80_80_80_80_80_80L;
252 
253         // Create a predicate for all bytes which are smaller equal than 'f'-'0' (0x37).
254         long le_37 = 0x37_37_37_37_37_37_37_37L + (vec ^ 0x7f_7f_7f_7f_7f_7f_7f_7fL);
255         // we don't need to 'and' with 0x80…L here, because we 'and' this with ge_30 anyway.
256         //le_37 &= 0x80_80_80_80_80_80_80_80L;
257 
258 
259         // If a character is greater than '9' then it must be greater equal 'a'
260         // and smaller  'f'.
261         if (gt_09 != (ge_30 & le_37)) {
262             return -1;
263         }
264 
265         // Expand the predicate to a byte mask
266         long gt_09mask = (gt_09 >>> 7) * 0xffL;
267 
268         // Subtract 'a'-'0'+10 (0x27) from all bytes that are greater than 0x09.
269         long v = vec & ~gt_09mask | vec - (0x27272727_27272727L & gt_09mask);
270 
271         // Compact all nibbles
272         long v2 = v | v >>> 4;
273         long v3 = v2 & 0x00ff00ff_00ff00ffL;
274         long v4 = v3 | v3 >>> 8;
275         return ((v4 >>> 16) & 0xffff_0000L) | v4 & 0xffffL;
276     }

288         // by Leslie Lamport.
289 
290 
291         // Subtract character '0' (0x0030) from each of the four characters
292         long vec = chunk - 0x0030_0030_0030_0030L;
293 
294         // Create a predicate for all bytes which are greater than '9'-'0' (0x0009).
295         // The predicate is true if the hsb of a byte is set: (predicate & 0xa000) != 0.
296         long gt_09 = vec + (0x0009_0009_0009_0009L ^ 0x7fff_7fff_7fff_7fffL);
297         gt_09 = gt_09 & 0x8000_8000_8000_8000L;
298         // Create a predicate for all bytes which are greater or equal 'a'-'0' (0x0030).
299         // The predicate is true if the hsb of a byte is set.
300         long ge_30 = vec + (0x0030_0030_0030_0030L ^ 0x7fff_7fff_7fff_7fffL);
301         ge_30 = ge_30 & 0x8000_8000_8000_8000L;
302 
303         // Create a predicate for all bytes which are smaller equal than 'f'-'0' (0x0037).
304         long le_37 = 0x0037_0037_0037_0037L + (vec ^ 0x7fff_7fff_7fff_7fffL);
305         // Not needed, because we are going to and this value with ge_30 anyway.
306         //le_37 = le_37 & 0x8000_8000_8000_8000L;
307 
308 
309         // If a character is greater than '9' then it must be greater equal 'a'
310         // and smaller equal 'f'.
311         if (gt_09 != (ge_30 & le_37)) {
312             return -1;
313         }
314 
315         // Expand the predicate to a char mask
316         long gt_09mask = (gt_09 >>> 15) * 0xffffL;
317 
318         // Subtract 'a'-'0'+10 (0x0027) from all bytes that are greater than 0x09.
319         long v = vec & ~gt_09mask | vec - (0x0027_0027_0027_0027L & gt_09mask);
320 
321         // Compact all nibbles
322         long v2 = v | v >>> 12;
323         return (v2 | v2 >>> 24) & 0xffffL;
324     }
