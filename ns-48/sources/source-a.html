44      * Parses a {@code DecimalFloatingPointLiteral} production with optional
45      * trailing white space until the end of the text.
46      * Given that we have already consumed the optional leading zero of
47      * the {@code DecSignificand}.
48      * <blockquote>
49      * <dl>
50      * <dt><i>DecimalFloatingPointLiteralWithWhiteSpace:</i></dt>
51      * <dd><i>DecimalFloatingPointLiteral [WhiteSpace] EOT</i></dd>
52      * </dl>
53      * </blockquote>
54      * See {@link io.questdb.std.fastdouble} for the grammar of
55      * {@code DecimalFloatingPointLiteral} and {@code DecSignificand}.
56      *
57      * @param str            a string
58      * @param index          start index inclusive of the {@code DecimalFloatingPointLiteralWithWhiteSpace}
59      * @param endIndex       end index (exclusive)
60      * @param isNegative     true if the float value is negative
61      * @param hasLeadingZero true if we have consumed the optional leading zero
62      * @param rejectOverflow reject parsed values that overflow double type
63      * @return the bit pattern of the parsed value, if the input is legal;
64      * otherwise, {@code -1L}.
65      */
66     private static float parseDecFloatLiteral(
67             byte[] str,
68             int index,
69             int startIndex,
70             int endIndex,
71             boolean isNegative,
72             boolean hasLeadingZero,
73             boolean rejectOverflow
74     ) throws NumericException {
75         // Parse significand
76         // -----------------
77         // Note: a multiplication by a constant is cheaper than an
78         //       arbitrary integer multiplication.
79         long significand = 0;// significand is treated as an unsigned long
80         final int significandStartIndex = index;
81         int virtualIndexOfPoint = -1;
82         boolean illegal = false;
83         byte ch = 0;
84         for (; index < endIndex; index++) {
85             ch = str[index];
86             if (isDigit(ch)) {
87                 // This might overflow, we deal with it later.
88                 significand = 10 * significand + ch - '0';
89             } else if (ch == '.') {
90                 illegal |= virtualIndexOfPoint >= 0;
91                 virtualIndexOfPoint = index;
92                 for (; index < endIndex - 8; index += 8) {
93                     int eightDigits = tryToParseEightDigits(str, index + 1);
94                     if (eightDigits < 0) {
95                         break;
96                     }
97                     // This might overflow, we deal with it later.
98                     significand = 100_000_000L * significand + eightDigits;
99                 }
100             } else {
101                 break;
102             }
103         }
104         final int digitCount;
105         final int significandEndIndex = index;
106         int exponent;
107         if (virtualIndexOfPoint < 0) {
108             digitCount = index - significandStartIndex;
109             virtualIndexOfPoint = index;
110             exponent = 0;
111         } else {
112             digitCount = index - significandStartIndex - 1;
113             exponent = virtualIndexOfPoint - index + 1;
114         }
115 
116         // Parse exponent number
117         // ---------------------
118         int expNumber = 0;
119         if (ch == 'e' || ch == 'E') {
120             ch = ++index < endIndex ? str[index] : 0;
121             boolean neg_exp = ch == '-';
122             if (neg_exp || ch == '+') {
123                 ch = ++index < endIndex ? str[index] : 0;
124             }
125             illegal |= !isDigit(ch);
126             do {
127                 // Guard against overflow
128                 if (expNumber < FastDoubleUtils.MAX_EXPONENT_NUMBER) {
129                     expNumber = 10 * expNumber + ch - '0';
130                 }
131                 ch = ++index < endIndex ? str[index] : 0;
132             } while (isDigit(ch));
133             if (neg_exp) {
134                 expNumber = -expNumber;
135             }
136             exponent += expNumber;
137         }
138 
139         // Skip optional FloatTypeSuffix
140         // ------------------------
141         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
142             index++;
143         }
144 
145         // Skip trailing whitespace and check if FloatingPointLiteral is complete
146         // ------------------------
147         index = skipWhitespace(str, index, endIndex);
148         if (illegal || index < endIndex
149                 || !hasLeadingZero && digitCount == 0) {
150             throw NumericException.INSTANCE;
151         }
152 
153         // Re-parse significand in case of a potential overflow
154         // -----------------------------------------------
155         final boolean isSignificandTruncated;
156         int skipCountInTruncatedDigits = 0;//counts +1 if we skipped over the decimal point
157         int exponentOfTruncatedSignificand;
158         if (digitCount > 19) {
159             significand = 0;
160             for (index = significandStartIndex; index < significandEndIndex; index++) {
161                 ch = str[index];
162                 if (ch == '.') {
163                     skipCountInTruncatedDigits++;
164                 } else {
165                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
166                         significand = 10 * significand + ch - '0';
167                     } else {
168                         break;
169                     }
170                 }
171             }
172             isSignificandTruncated = (index < significandEndIndex);
173             exponentOfTruncatedSignificand = virtualIndexOfPoint - index + skipCountInTruncatedDigits + expNumber;
174         } else {
175             isSignificandTruncated = false;
176             exponentOfTruncatedSignificand = 0;
177         }
178         return valueOfFloatLiteral(
179                 str,
180                 startIndex,
181                 endIndex,
182                 isNegative,
183                 significand,
184                 exponent,
185                 isSignificandTruncated,
186                 exponentOfTruncatedSignificand,
187                 rejectOverflow
188         );
189     }

220             boolean isNegative,
221             boolean rejectOverflow
222     ) throws NumericException {

225         // ------------
226         long significand = 0;// significand is treated as an unsigned long
227         int exponent = 0;
228         final int significandStartIndex = index;
229         int virtualIndexOfPoint = -1;
230         final int digitCount;
231         boolean illegal = false;
232         byte ch = 0;
233         for (; index < endIndex; index++) {
234             ch = str[index];
235             // Table look up is faster than a sequence of if-else-branches.
236             int hexValue = ch < 0 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
237             if (hexValue >= 0) {

239             } else if (hexValue == FastDoubleUtils.DECIMAL_POINT_CLASS) {

244             }
245         }

249             virtualIndexOfPoint = significandEndIndex;

252             exponent = Math.min(virtualIndexOfPoint - index + 1, FastDoubleUtils.MAX_EXPONENT_NUMBER) * 4;
253         }

264             }

269             } while (isDigit(ch));
270             if (neg_exp) {
271                 expNumber = -expNumber;
272             }
273             exponent += expNumber;
274         }
275

277         // ------------------------
278         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {

346         if (index + 7 < endIndex
347                 && str[index] == 'I'
348                 && str[index + 1] == 'n'
349                 && str[index + 2] == 'f'
350                 && str[index + 3] == 'i'
351                 && str[index + 4] == 'n'
352                 && str[index + 5] == 'i'
353                 && str[index + 6] == 't'
354                 && str[index + 7] == 'y'
355         ) {
356             index = skipWhitespace(str, index + 8, endIndex);
357             if (index == endIndex) {
358                 return negative ? negativeInfinity() : positiveInfinity();
359             }
360         }
361         throw NumericException.INSTANCE;
362     }

426      * Parses a {@code FloatingPointLiteral} production with optional leading and trailing
427      * white space.
428      * <blockquote>
429      * <dl>
430      * <dt><i>FloatingPointLiteralWithWhiteSpace:</i></dt>
431      * <dd><i>[WhiteSpace] FloatingPointLiteral [WhiteSpace]</i></dd>
432      * </dl>
433      * </blockquote>
434      * See {@link io.questdb.std.fastdouble} for the grammar of
435      * {@code FloatingPointLiteral}.
436      *
437      * @param str            a string containing a {@code FloatingPointLiteralWithWhiteSpace}
438      * @param offset         start offset of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
439      * @param length         length of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
440      * @param rejectOverflow reject parsed values that overflow double type
441      * @return the bit pattern of the parsed value, if the input is legal;
442      * otherwise, {@code -1L}.
443      */
444     static float parseFloatingPointLiteral(byte[] str, int offset, int length, boolean rejectOverflow) throws NumericException {
445         final int endIndex = offset + length;
446         if (offset < 0 || endIndex > str.length) {
447             throw NumericException.INSTANCE;
448         }
449 
450         // Skip leading whitespace
451         // -------------------
452         int index = skipWhitespace(str, offset, endIndex);
453         if (index == endIndex) {
454             throw NumericException.INSTANCE;
455         }
456         byte ch = str[index];
457 
458         // Parse optional sign
459         // -------------------
460         final boolean isNegative = ch == '-';
461         if (isNegative || ch == '+') {
462             ch = ++index < endIndex ? str[index] : 0;
463             if (ch == 0) {
464                 throw NumericException.INSTANCE;
465             }
466         }
467 
468         // Parse NaN or Infinity
469         // ---------------------
470         if (ch >= 'I') {
471             return ch == 'N'
472                     ? parseNaN(str, index, endIndex)
473                     : parseInfinity(str, index, endIndex, isNegative);
474         }
475 
476         // Parse optional leading zero
477         // ---------------------------
478         final boolean hasLeadingZero = ch == '0';
479         if (hasLeadingZero) {
480             ch = ++index < endIndex ? str[index] : 0;
481             if (ch == 'x' || ch == 'X') {
482                 return parseHexFloatingPointLiteral(str, index + 1, offset, endIndex, isNegative, rejectOverflow);
483             }
484         }
485 
486         return parseDecFloatLiteral(str, index, offset, endIndex, isNegative, hasLeadingZero, rejectOverflow);
487     }
