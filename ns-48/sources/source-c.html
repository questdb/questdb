131     /**
132      * Attempts to compute {@literal digits * 10^(power)} exactly;
133      * and if "negative" is true, negate the result.
134      * <p>
135      * This function will only work in some cases, when it does not work it
136      * returns null. This should work *most of the time* (like 99% of the time).
137      * We assume that power is in the
138      * [{@value FastDoubleMath#DOUBLE_MIN_EXPONENT_POWER_OF_TEN},
139      * {@value FastDoubleMath#DOUBLE_MAX_EXPONENT_POWER_OF_TEN}]
140      * interval: the caller is responsible for this check.
141      *
142      * @param isNegative whether the number is negative
143      * @param digits     uint64 the digits of the number
144      * @param power      int32 the exponent of the number
145      * @return the computed double on success, {@link Double#NaN} on failure
146      */
147     static float tryDecToFloatWithFastAlgorithm(boolean isNegative, long digits, int power) {
148 
149         // we start with a fast path
150         if (-10 <= power && power <= 10 && Long.compareUnsigned(digits, (1L << FLOAT_SIGNIFICAND_WIDTH) - 1L) <= 0) {
151             // convert the integer into a float. This is lossless since
152             // 0 <= i <= 2^24 - 1.
153             float d = (float) digits;
154             //
155             // The general idea is as follows.
156             // If 0 <= s < 2^24 and if 10^0 <= p <= 10^10 then
157             // 1) Both s and p can be represented exactly as 32-bit floating-point values
158             // 2) Because s and p can be represented exactly as floating-point values,
159             // then s * p and s / p will produce correctly rounded values.
160             //
161             if (power < 0) {
162                 d = d / FLOAT_POWER_OF_TEN[-power];
163             } else {
164                 d = d * FLOAT_POWER_OF_TEN[power];
165             }
166             return (isNegative) ? -d : d;
167         }
168 
169 
170         // The fast path has now failed, so we are falling back on the slower path.
171 
172         // We are going to need to do some 64-bit arithmetic to get a more precise product.
173         // We use a table lookup approach.
174         // It is safe because
175         // power >= DOUBLE_MIN_EXPONENT_POWER_OF_TEN
176         // and power <= DOUBLE_MAX_EXPONENT_POWER_OF_TEN
177         // We recover the mantissa of the power, it has a leading 1. It is always
178         // rounded down.
179         long factorMantissa = MANTISSA_64[power - DOUBLE_MIN_EXPONENT_POWER_OF_TEN];
180 
181 
182         // The exponent is 127 + 64 + power
183         //     + floor(log(5**power)/log(2)).
184         // The 127 is the exponent bias.
185         // The 64 comes from the fact that we use a 64-bit word.
186         //
187         // Computing floor(log(5**power)/log(2)) could be
188         // slow. Instead ,we use a fast function.
189         //
190         // For power in (-400,350), we have that
191         // (((152170 + 65536) * power ) >> 16);
192         // is equal to
193         //  floor(log(5**power)/log(2)) + power when power >= 0
194         // and it is equal to
195         //  ceil(log(5**-power)/log(2)) + power when power < 0
196         //
197         //
198         // The 65536 is (1<<16) and corresponds to
199         // (65536 * power) >> 16 ---> power
200         //
201         // ((152170 * power ) >> 16) is equal to
202         // floor(log(5**power)/log(2))
203         //
204         // Note that this is not magic: 152170/(1<<16) is
205         // approximately equal to log(5)/log(2).
206         // The 1<<16 value is a power of two; we could use a
207         // larger power of 2 if we wanted to.
208         //
209         long exponent = (((152170L + 65536L) * power) >> 16) + FLOAT_EXPONENT_BIAS + 64;
210         // We want the most significant bit of digits to be 1. Shift if needed.
211         int lz = Long.numberOfLeadingZeros(digits);
212         digits <<= lz;
213         // We want the most significant 64 bits of the product. We know
214         // this will be non-zero because the most significant bit of i is
215         // 1.
216         //before Java 18
217         long x01 = digits & 0xffffffffL, x11 = digits >>> 32;
218         long y01 = factorMantissa & 0xffffffffL, y11 = factorMantissa >>> 32;
219         long p111 = x11 * y11, p011 = x01 * y11;
220         long p101 = x11 * y01, p001 = x01 * y01;
221 
222         // 64-bit product + two 32-bit values
223         long middle1 = p101 + (p001 >>> 32) + (p011 & 0xffffffffL);
224         long lower = (middle1 << 32) | (p001 & 0xffffffffL);
225         long upper = p111 + (middle1 >>> 32) + (p011 >>> 32);
226         // We know that upper has at most one leading zero because
227         // both i and factor_mantissa have a leading one. This means
228         // that the result is at least as large as ((1<<63)*(1<<63))/(1<<64).
229 
230         // As long as the first 39 bits of "upper" are not "1", then we
231         // know that we have an exact computed value for the leading
232         // 25 bits because any imprecision would play out as a +1, in
233         // the worst case.
234         // Having 25 bits is necessary because
235         // we need 24 bits for the mantissa, but we have to have one rounding bit, and
236         // we can waste a bit if the most significant bit of the product is zero.
237         // We expect this next branch to be rarely taken (say 1% of the time).
238         // When (upper &0x3FFFFFFFFF) == 0x3FFFFFFFFF, it can be common for
239         // lower + i < lower to be true (proba. much higher than 1%).
240         if ((upper & 0x3_FFFFF_FFFFL) == 0x3_FFFFF_FFFFL && Long.compareUnsigned(lower + digits, lower) < 0) {
241             long factor_mantissa_low =
242                     MANTISSA_128[power - DOUBLE_MIN_EXPONENT_POWER_OF_TEN];
243             // next, we compute the 64-bit x 128-bit multiplication, getting a 192-bit
244             // result (three 64-bit values)
245             //before Java 18
246             long x0 = digits & 0xffffffffL, x1 = digits >>> 32;
247             long y0 = factor_mantissa_low & 0xffffffffL, y1 = factor_mantissa_low >>> 32;
248             long p11 = x1 * y1, p01 = x0 * y1;
249             long p10 = x1 * y0, p00 = x0 * y0;
250 
251             // 64-bit product + two 32-bit values
252             long middle = p10 + (p00 >>> 32) + (p01 & 0xffffffffL);
253             long product_low = (middle << 32) | (p00 & 0xffffffffL);
254             long product_middle2 = p11 + (middle >>> 32) + (p01 >>> 32);
255             long product_high = upper;
256             long product_middle = lower + product_middle2;
257             if (Long.compareUnsigned(product_middle, lower) < 0) {
258                 product_high++; // overflow carry
259             }
260 
261 
262             // we want to check whether mantissa *i + i would affect our result
263             // This does happen, e.g. with 7.3177701707893310e+15 ????
264             if (((product_middle + 1 == 0) && ((product_high & 0x7_FFFFF_FFFFL) == 0x7_FFFFF_FFFFL) &&
265                     (product_low + Long.compareUnsigned(digits, product_low) < 0))) { // let us be prudent and bail out.
266                 return Float.NaN;
267             }
268             upper = product_high;
269             //lower = product_middle;
270         }
271 
272         // The final mantissa should be 24 bits with a leading 1.
273         // We shift it so that it occupies 25 bits with a leading 1.
274         long upperbit = upper >>> 63;
275         long mantissa = upper >>> (upperbit + 38);
276         lz += (int) (1 ^ upperbit);
277         // Here we have mantissa < (1<<25).
278         //assert mantissa < (1<<25);
279 
280         // We have to round to even. The "to even" part
281         // is only a problem when we are right in between two floating-point values
282         // which we guard against.
283         // If we have lots of trailing zeros, we may fall right between two
284         // floating-point values.
285         if (((upper & 0x3_FFFFF_FFFFL) == 0x3_FFFFF_FFFFL)
286                 || ((upper & 0x3_FFFFF_FFFFL) == 0) && (mantissa & 3) == 1) {
287             // if mantissa & 1 == 1 we might need to round up.
288             //
289             // Scenarios:
290             // 1. We are not in the middle. Then we should round up.
291             //
292             // 2. We are right in the middle. Whether we round up depends
293             // on the last significant bit: if it is "one" then we round
294             // up (round to even) otherwise, we do not.
295             //
296             // So if the last significant bit is 1, we can safely round up.
297             // Hence, we only need to bail out if (mantissa & 3) == 1.
298             // Otherwise, we may need more accuracy or analysis to determine whether
299             // we are exactly between two floating-point numbers.
300             // It can be triggered with 1e23. ??
301             // Note: because the factor_mantissa and factor_mantissa_low are
302             // almost always rounded down (except for small positive powers),
303             // almost always should round up.
304             return Float.NaN;
305         }
306 
307         mantissa += 1;
308         mantissa >>>= 1;
309 
310         // Here we have mantissa < (1<<24), unless there was an overflow
311         if (mantissa >= (1L << FLOAT_SIGNIFICAND_WIDTH)) {
312             // This will happen when parsing values such as 7.2057594037927933e+16 ??
313             mantissa = (1L << (FLOAT_SIGNIFICAND_WIDTH - 1));
314             lz--; // undo previous addition
315         }
316 
317         mantissa &= ~(1L << (FLOAT_SIGNIFICAND_WIDTH - 1));
318 
319 
320         long real_exponent = exponent - lz;
321         // we have to check that real_exponent is in range, otherwise we bail out
322         if ((real_exponent < 1) || (real_exponent > FLOAT_MAX_EXPONENT_POWER_OF_TWO + FLOAT_EXPONENT_BIAS)) {
323             return Float.NaN;
324         }
325 
326         int bits = (int) (mantissa | real_exponent << (FLOAT_SIGNIFICAND_WIDTH - 1)
327                 | (isNegative ? 1L << 31 : 0));
328         return Float.intBitsToFloat(bits);
329     }

331     static float tryHexToFloatWithFastAlgorithm(boolean isNegative, long digits, int power) {
332         if (digits == 0 || power < Float.MIN_EXPONENT - 54) {
333             return isNegative ? -0.0f : 0.0f;
334         }
335         if (power > Float.MAX_EXPONENT) {
336             return isNegative ? Float.NEGATIVE_INFINITY : Float.POSITIVE_INFINITY;
337         }
338 
339         // we start with a fast path
340         // We try to mimic the fast described by Clinger WD for decimal
341         // float number literals. How to read floating point numbers accurately.
342         // ACM SIGPLAN Notices. 1990
343         if (Long.compareUnsigned(digits, 0x1fffffffffffffL) <= 0) {
344             // convert the integer into a double. This is lossless since
345             // 0 <= i <= 2^53 - 1.
346             float d = (float) digits;
347             //
348             // The general idea is as follows.
349             // If 0 <= s < 2^53  then
350             // 1) Both s and p can be represented exactly as 64-bit floating-point
351             // values (binary64).
352             // 2) Because s and p can be represented exactly as floating-point values,
353             // then s * p will produce correctly rounded values.
354             //
355             d = d * Math.scalb(1f, power);
356             if (isNegative) {
357                 d = -d;
358             }
359             return d;
360         }
361 
362         // The fast path has failed
363         return Float.NaN;
364     }
