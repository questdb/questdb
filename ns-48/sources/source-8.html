27 /**
28  * Abstract base class for parsers that parse a {@code FloatingPointLiteral} from a
29  * character sequence ({@code str}).
30  * <p>
31  * This is a C++ to Java port of Daniel Lemire's fast_double_parser.
32  * <p>
33  * References:
34  * <dl>
35  *     <dt>Daniel Lemire, fast_double_parser, 4x faster than strtod.
36  *     Apache License 2.0 or Boost Software License.</dt>
37  *     <dd><a href="https://github.com/lemire/fast_double_parser">github.com</a></dd>
38  *
39  *     <dt>Daniel Lemire, fast_float number parsing library: 4x faster than strtod.
40  *     Apache License 2.0.</dt>
41  *     <dd><a href="https://github.com/fastfloat/fast_float">github.com</a></dd>
42  *
43  *     <dt>Daniel Lemire, Number Parsing at a Gigabyte per Second,
44  *     Software: Practice and Experience 51 (8), 202    1.
45  *     arXiv.2101.11408v3 [cs.DS] 24 Feb 2021</dt>
46  *     <dd><a href="https://arxiv.org/pdf/2101.11408.pdf">arxiv.org</a></dd>
47  * </dl>
48  */
49 final class FastDoubleUtils {
50     /**
51      * Includes all non-negative values of a {@code byte}, so that we only have
52      * to check for byte values {@literal <} 0 before accessing this array.
53      */
54     static final byte[] CHAR_TO_HEX_MAP = new byte[128];
55     /**
56      * Special value in {@link #CHAR_TO_HEX_MAP} for
57      * the decimal point character.
58      */
59     static final byte DECIMAL_POINT_CLASS = -4;
60     /**
61      * The decimal exponent of a double has a range of -324 to +308.
62      * The hexadecimal exponent of a double has a range of -1022 to +1023.
63      */
64     final static int MAX_EXPONENT_NUMBER = 1024;
65     final static long MINIMAL_NINETEEN_DIGIT_INTEGER = 1000_00000_00000_00000L;
66     /**
67      * Special value in {@link #CHAR_TO_HEX_MAP} for
68      * characters that are neither a hex digit nor
69      * a decimal point character.
70      */
71     static final byte OTHER_CLASS = -1;
72 
73     static {
74         for (char ch = 0; ch < FastDoubleUtils.CHAR_TO_HEX_MAP.length; ch++) {
75             FastDoubleUtils.CHAR_TO_HEX_MAP[ch] = FastDoubleUtils.OTHER_CLASS;
76         }
77         for (char ch = '0'; ch <= '9'; ch++) {
78             FastDoubleUtils.CHAR_TO_HEX_MAP[ch] = (byte) (ch - '0');
79         }
80         for (char ch = 'A'; ch <= 'F'; ch++) {
81             FastDoubleUtils.CHAR_TO_HEX_MAP[ch] = (byte) (ch - 'A' + 10);
82         }
83         for (char ch = 'a'; ch <= 'f'; ch++) {
84             FastDoubleUtils.CHAR_TO_HEX_MAP[ch] = (byte) (ch - 'a' + 10);
85         }
86         for (char ch = '.'; ch <= '.'; ch++) {
87             FastDoubleUtils.CHAR_TO_HEX_MAP[ch] = FastDoubleUtils.DECIMAL_POINT_CLASS;
88         }
89     }
90 }
