66      * Parses a {@code DecimalFloatingPointLiteral} production with optional
67      * trailing white space until the end of the text.
68      * Given that we have already consumed the optional leading zero of
69      * the {@code DecSignificand}.
70      * <blockquote>
71      * <dl>
72      * <dt><i>DecimalFloatingPointLiteralWithWhiteSpace:</i></dt>
73      * <dd><i>DecimalFloatingPointLiteral [WhiteSpace] EOT</i></dd>
74      * </dl>
75      * </blockquote>
76      * See {@link io.questdb.std.fastdouble} for the grammar of
77      * {@code DecimalFloatingPointLiteral} and {@code DecSignificand}.
78      *
79      * @param str            a string
80      * @param index          start index inclusive of the {@code DecimalFloatingPointLiteralWithWhiteSpace}
81      * @param endIndex       end index (exclusive)
82      * @param isNegative     true if the float value is negative
83      * @param hasLeadingZero true if we have consumed the optional leading zero
84      * @return the bit pattern of the parsed value, if the input is legal;
85      * otherwise, {@code -1L}.
86      */
87     private static float parseDecFloatLiteral(
88             final CharSequence str,
89             int index,
90             int startIndex,
91             int endIndex,
92             boolean isNegative,
93             boolean hasLeadingZero,
94             boolean rejectOverflow
95     ) throws NumericException {
96         // Parse significand
97         // -----------------
98         // Note: a multiplication by a constant is cheaper than an
99         //       arbitrary integer multiplication.
100         long significand = 0;// significand is treated as an unsigned long
101         final int significandStartIndex = index;
102         int virtualIndexOfPoint = -1;
103         boolean illegal = false;
104         char ch = 0;
105         for (; index < endIndex; index++) {
106             ch = str.charAt(index);
107             if (isDigit(ch)) {
108                 // This might overflow, we deal with it later.
109                 significand = 10 * significand + ch - '0';
110             } else if (ch == '.') {
111                 illegal |= virtualIndexOfPoint >= 0;
112                 virtualIndexOfPoint = index;
113                 for (; index < endIndex - 8; index += 8) {
114                     int eightDigits = tryToParseEightDigits(str, index + 1);
115                     if (eightDigits < 0) {
116                         break;
117                     }
118                     // This might overflow, we deal with it later.
119                     significand = 100_000_000L * significand + eightDigits;
120                 }
121             } else {
122                 break;
123             }
124         }
125         final int digitCount;
126         final int significandEndIndex = index;
127         int exponent;
128         if (virtualIndexOfPoint < 0) {
129             digitCount = significandEndIndex - significandStartIndex;
130             virtualIndexOfPoint = significandEndIndex;
131             exponent = 0;
132         } else {
133             digitCount = significandEndIndex - significandStartIndex - 1;
134             exponent = virtualIndexOfPoint - significandEndIndex + 1;
135         }
136 
137         // Parse exponent number
138         // ---------------------
139         int expNumber = 0;
140         if (ch == 'e' || ch == 'E') {
141             ch = ++index < endIndex ? str.charAt(index) : 0;
142             boolean neg_exp = ch == '-';
143             if (neg_exp || ch == '+') {
144                 ch = ++index < endIndex ? str.charAt(index) : 0;
145             }
146             illegal |= !isDigit(ch);
147             do {
148                 // Guard against overflow
149                 if (expNumber < FastDoubleUtils.MAX_EXPONENT_NUMBER) {
150                     expNumber = 10 * expNumber + ch - '0';
151                 }
152                 ch = ++index < endIndex ? str.charAt(index) : 0;
153             } while (isDigit(ch));
154             if (neg_exp) {
155                 expNumber = -expNumber;
156             }
157             exponent += expNumber;
158         }
159 
160         // Skip optional FloatTypeSuffix
161         // ------------------------
162         if (index < endIndex && (ch == 'f' || ch == 'F')) {
163             index++;
164         }
165 
166         // Skip trailing whitespace and check if FloatingPointLiteral is complete
167         // ------------------------
168         index = skipWhitespace(str, index, endIndex);
169         if (illegal || index < endIndex
170                 || !hasLeadingZero && digitCount == 0) {
171             throw NumericException.INSTANCE;
172         }
173 
174         // Re-parse significand in case of a potential overflow
175         // -----------------------------------------------
176         final boolean isSignificandTruncated;
177         int skipCountInTruncatedDigits = 0;//counts +1 if we skipped over the decimal point
178         int exponentOfTruncatedSignificand;
179         if (digitCount > 19) {
180             significand = 0;
181             for (index = significandStartIndex; index < significandEndIndex; index++) {
182                 ch = str.charAt(index);
183                 if (ch == '.') {
184                     skipCountInTruncatedDigits++;
185                 } else {
186                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
187                         significand = 10 * significand + ch - '0';
188                     } else {
189                         break;
190                     }
191                 }
192             }
193             isSignificandTruncated = (index < significandEndIndex);
194             exponentOfTruncatedSignificand = virtualIndexOfPoint - index + skipCountInTruncatedDigits + expNumber;
195         } else {
196             isSignificandTruncated = false;
197             exponentOfTruncatedSignificand = 0;
198         }
199         return valueOfFloatLiteral(
200                 str,
201                 startIndex,
202                 endIndex,
203                 isNegative,
204                 significand,
205                 exponent,
206                 isSignificandTruncated,
207                 exponentOfTruncatedSignificand,
208                 rejectOverflow
209         );
210     }

214      * (more rules are defined in {@link FastDoubleUtils}):
215      * <dl>
216      * <dt><i>RestOfHexFloatingPointLiteral</i>:
217      * <dd><i>RestOfHexSignificand BinaryExponent</i>
218      * </dl>
219      *
220      * <dl>
221      * <dt><i>RestOfHexSignificand:</i>
222      * <dd><i>HexDigits</i>
223      * <dd><i>HexDigits</i> {@code .}
224      * <dd><i>[HexDigits]</i> {@code .} <i>HexDigits</i>
225      * </dl>
226      *
227      * @param str            the input string
228      * @param index          index to the first character of RestOfHexFloatingPointLiteral
229      * @param startIndex     the start index of the string
230      * @param endIndex       the end index of the string
231      * @param isNegative     if the resulting number is negative
232      * @param rejectOverflow reject parsed values that overflow double type
233      * @return the parsed value, if the input is legal;
234      * @throws NumericException when input is illegal
235      */
236     private static float parseHexFloatLiteral(
237             CharSequence str,
238             int index,
239             int startIndex,
240             int endIndex,
241             boolean isNegative,
242             boolean rejectOverflow
243     ) throws NumericException {
244 
245         // Parse HexSignificand
246         // ------------
247         long significand = 0;// significand is treated as an unsigned long
248         int exponent = 0;
249         final int significandStartIndex = index;
250         int virtualIndexOfPoint = -1;
251         final int digitCount;
252         boolean illegal = false;
253         char ch = 0;
254         for (; index < endIndex; index++) {
255             ch = str.charAt(index);
256             // Table look up is faster than a sequence of if-else-branches.
257             int hexValue = ch > 127 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
258             if (hexValue >= 0) {
259                 significand = (significand << 4) | hexValue;// This might overflow, we deal with it later.
260             } else if (hexValue == FastDoubleUtils.DECIMAL_POINT_CLASS) {
261                 illegal |= virtualIndexOfPoint >= 0;
262                 virtualIndexOfPoint = index;
263             } else {
264                 break;
265             }
266         }
267         final int significandEndIndex = index;
268         if (virtualIndexOfPoint < 0) {
269             digitCount = significandEndIndex - significandStartIndex;
270             virtualIndexOfPoint = significandEndIndex;
271         } else {
272             digitCount = significandEndIndex - significandStartIndex - 1;
273             exponent = Math.min(virtualIndexOfPoint - index + 1, FastDoubleUtils.MAX_EXPONENT_NUMBER) * 4;
274         }
275 
276         // Parse exponent
277         // --------------
278         int expNumber = 0;
279         final boolean hasExponent = (ch == 'p') || (ch == 'P');
280         if (hasExponent) {
281             ch = ++index < endIndex ? str.charAt(index) : 0;
282             boolean neg_exp = ch == '-';
283             if (neg_exp || ch == '+') {
284                 ch = ++index < endIndex ? str.charAt(index) : 0;
285             }
286             illegal |= !isDigit(ch);
287             do {
288                 expNumber = 10 * expNumber + ch - '0';
289                 ch = ++index < endIndex ? str.charAt(index) : 0;
290             } while (isDigit(ch));
291             if (neg_exp) {
292                 expNumber = -expNumber;
293             }
294             exponent += expNumber;
295         }
296 
297         // Skip optional FloatTypeSuffix
298         // ------------------------
299         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
300             index++;
301         }
302 
303         // Skip trailing whitespace and check if FloatingPointLiteral is complete
304         // ------------------------
305         index = skipWhitespace(str, index, endIndex);
306         if (illegal || index < endIndex
307                 || digitCount == 0
308                 || !hasExponent) {
309             throw NumericException.INSTANCE;
310         }
311 
312         // Re-parse significand in case of a potential overflow
313         // -----------------------------------------------
314         final boolean isSignificandTruncated;
315         int skipCountInTruncatedDigits = 0;//counts +1 if we skipped over the decimal point
316         if (digitCount > 16) {
317             significand = 0;
318             for (index = significandStartIndex; index < significandEndIndex; index++) {
319                 ch = str.charAt(index);
320                 // Table look up is faster than a sequence of if-else-branches.
321                 int hexValue = ch > 127 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
322                 if (hexValue >= 0) {
323                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
324                         significand = (significand << 4) | hexValue;
325                     } else {
326                         break;
327                     }
328                 } else {
329                     skipCountInTruncatedDigits++;
330                 }
331             }
332             isSignificandTruncated = (index < significandEndIndex);
333         } else {
334             isSignificandTruncated = false;
335         }
336 
337         return valueOfHexLiteral(
338                 str,
339                 startIndex,
340                 endIndex,
341                 isNegative,
342                 significand,
343                 exponent,
344                 isSignificandTruncated,
345                 virtualIndexOfPoint - index + skipCountInTruncatedDigits + expNumber,
346                 rejectOverflow
347         );

351      * Parses a {@code Infinity} production with optional trailing white space
352      * until the end of the text.
353      * <blockquote>
354      * <dl>
355      * <dt><i>InfinityWithWhiteSpace:</i></dt>
356      * <dd>{@code Infinity} <i>[WhiteSpace] EOT</i></dd>
357      * </dl>
358      * </blockquote>
359      *
360      * @param str      a string
361      * @param index    index of the "I" character
362      * @param endIndex end index (exclusive)
363      * @return a positive or negative infinity value
364      * @throws NumberFormatException on parsing failure
365      */
366     private static float parseInfinity(CharSequence str, int index, int endIndex, boolean negative) throws NumericException {
367         if (index + 7 < endIndex
368                 && str.charAt(index) == 'I'
369                 && str.charAt(index + 1) == 'n'
370                 && str.charAt(index + 2) == 'f'
371                 && str.charAt(index + 3) == 'i'
372                 && str.charAt(index + 4) == 'n'
373                 && str.charAt(index + 5) == 'i'
374                 && str.charAt(index + 6) == 't'
375                 && str.charAt(index + 7) == 'y'
376         ) {
377             index = skipWhitespace(str, index + 8, endIndex);
378             if (index == endIndex) {
379                 return negative ? negativeInfinity() : positiveInfinity();
380             }
381         }
382         throw NumericException.INSTANCE;
383     }

434     private static int tryToParseEightDigits(CharSequence str, int offset) {
435         // Performance: We extract the chars in two steps so that we
436         //              can benefit from out of order execution in the CPU.
437         long first = str.charAt(offset)
438                 | (long) str.charAt(offset + 1) << 16
439                 | (long) str.charAt(offset + 2) << 32
440                 | (long) str.charAt(offset + 3) << 48;
441 
442         long second = str.charAt(offset + 4)
443                 | (long) str.charAt(offset + 5) << 16
444                 | (long) str.charAt(offset + 6) << 32
445                 | (long) str.charAt(offset + 7) << 48;
446 
447         return FastDoubleSwar.tryToParseEightDigitsUtf16(first, second);
448     }

459      * Parses a {@code FloatingPointLiteral} production with optional leading and trailing
460      * white space.
461      * <blockquote>
462      * <dl>
463      * <dt><i>FloatingPointLiteralWithWhiteSpace:</i></dt>
464      * <dd><i>[WhiteSpace] FloatingPointLiteral [WhiteSpace]</i></dd>
465      * </dl>
466      * </blockquote>
467      * See {@link io.questdb.std.fastdouble} for the grammar of
468      * {@code FloatingPointLiteral}.
469      *
470      * @param str    a string containing a {@code FloatingPointLiteralWithWhiteSpace}
471      * @param offset start offset of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
472      * @param length length of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
473      * @return the bit pattern of the parsed value, if the input is legal;
474      * otherwise, {@code -1L}.
475      */
476     static float parseFloatingPointLiteral(CharSequence str, int offset, int length, boolean rejectOverflow) throws NumericException {
477         final int endIndex = offset + length;
478         if (offset < 0 || endIndex > str.length()) {
479             throw NumericException.INSTANCE;
480         }
481 
482         // Skip leading whitespace
483         // -------------------
484         int index = skipWhitespace(str, offset, endIndex);
485         if (index == endIndex) {
486             throw NumericException.INSTANCE;
487         }
488         char ch = str.charAt(index);
489 
490         // Parse optional sign
491         // -------------------
492         final boolean isNegative = ch == '-';
493         if (isNegative || ch == '+') {
494             ch = ++index < endIndex ? str.charAt(index) : 0;
495             if (ch == 0) {
496                 throw NumericException.INSTANCE;
497             }
498         }
499 
500         // Parse NaN or Infinity
501         // ---------------------
502         if (ch >= 'I') {
503             return ch == 'N'
504                     ? parseNaN(str, index, endIndex)
505                     : parseInfinity(str, index, endIndex, isNegative);
506         }
507 
508         // Parse optional leading zero
509         // ---------------------------
510         final boolean hasLeadingZero = ch == '0';
511         if (hasLeadingZero) {
512             ch = ++index < endIndex ? str.charAt(index) : 0;
513             if (ch == 'x' || ch == 'X') {
514                 return parseHexFloatLiteral(str, index + 1, offset, endIndex, isNegative, rejectOverflow);
515             }
516         }
517 
518         return parseDecFloatLiteral(str, index, offset, endIndex, isNegative, hasLeadingZero, rejectOverflow);
519     }
