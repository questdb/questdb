59      * Parses a {@code DecimalFloatingPointLiteral} production with optional
60      * trailing white space until the end of the text.
61      * Given that we have already consumed the optional leading zero of
62      * the {@code DecSignificand}.
63      * <blockquote>
64      * <dl>
65      * <dt><i>DecimalFloatingPointLiteralWithWhiteSpace:</i></dt>
66      * <dd><i>DecimalFloatingPointLiteral [WhiteSpace] EOT</i></dd>
67      * </dl>
68      * </blockquote>
69      * See {@link io.questdb.std.fastdouble} for the grammar of
70      * {@code DecimalFloatingPointLiteral} and {@code DecSignificand}.
71      *
72      * @param str            a string
73      * @param index          start index inclusive of the {@code DecimalFloatingPointLiteralWithWhiteSpace}
74      * @param endIndex       end index (exclusive)
75      * @param isNegative     true if the float value is negative
76      * @param hasLeadingZero true if we have consumed the optional leading zero
77      * @param rejectOverflow reject parsed values that overflow double type
78      * @return the parsed value, if the input is legal;
79      * @throws NumericException if the input is illegal.
80      */
81     private static double parseDecFloatLiteral(
82             byte[] str,
83             int index,
84             int startIndex,
85             int endIndex,
86             boolean isNegative,
87             boolean hasLeadingZero,
88             boolean rejectOverflow
89     ) throws NumericException {
90         // Parse significand
91         // -----------------
92         // Note: a multiplication by a constant is cheaper than an
93         //       arbitrary integer multiplication.
94         long significand = 0;// significand is treated as an unsigned long
95         final int significandStartIndex = index;
96         int virtualIndexOfPoint = -1;
97         boolean illegal = false;
98         byte ch = 0;
99         for (; index < endIndex; index++) {
100             ch = str[index];
101             if (isDigit(ch)) {
102                 // This might overflow, we deal with it later.
103                 significand = 10 * significand + ch - '0';
104             } else if (ch == '.') {
105                 illegal |= virtualIndexOfPoint >= 0;
106                 virtualIndexOfPoint = index;
107                 for (; index < endIndex - 8; index += 8) {
108                     int eightDigits = tryToParseEightDigits(str, index + 1);
109                     if (eightDigits < 0) {
110                         break;
111                     }
112                     // This might overflow, we deal with it later.
113                     significand = 100_000_000L * significand + eightDigits;
114                 }
115             } else {
116                 break;
117             }
118         }
119         final int digitCount;
120         final int significandEndIndex = index;
121         int exponent;
122         if (virtualIndexOfPoint < 0) {
123             digitCount = index - significandStartIndex;
124             virtualIndexOfPoint = index;
125             exponent = 0;
126         } else {
127             digitCount = index - significandStartIndex - 1;
128             exponent = virtualIndexOfPoint - index + 1;
129         }
130 
131         // Parse exponent number
132         // ---------------------
133         int expNumber = 0;
134         if (ch == 'e' || ch == 'E') {
135             ch = ++index < endIndex ? str[index] : 0;
136             boolean neg_exp = ch == '-';
137             if (neg_exp || ch == '+') {
138                 ch = ++index < endIndex ? str[index] : 0;
139             }
140             illegal |= !isDigit(ch);
141             do {
142                 // Guard against overflow
143                 if (expNumber < FastDoubleUtils.MAX_EXPONENT_NUMBER) {
144                     expNumber = 10 * expNumber + ch - '0';
145                 }
146                 ch = ++index < endIndex ? str[index] : 0;
147             } while (isDigit(ch));
148             if (neg_exp) {
149                 expNumber = -expNumber;
150             }
151             exponent += expNumber;
152         }
153 
154         // Skip optional FloatTypeSuffix
155         // ------------------------
156         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
157             index++;
158         }
159 
160         // Skip trailing whitespace and check if FloatingPointLiteral is complete
161         // ------------------------
162         index = skipWhitespace(str, index, endIndex);
163         if (illegal
164                 || index < endIndex
165                 || !hasLeadingZero
166                 && digitCount == 0
167         ) {
168             throw NumericException.INSTANCE;
169         }
170 
171         // Re-parse significand in case of a potential overflow
172         // -----------------------------------------------
173         final boolean isSignificandTruncated;
174         int skipCountInTruncatedDigits = 0;//counts +1 if we skipped over the decimal point
175         int exponentOfTruncatedSignificand;
176         if (digitCount > 19) {
177             significand = 0;
178             for (index = significandStartIndex; index < significandEndIndex; index++) {
179                 ch = str[index];
180                 if (ch == '.') {
181                     skipCountInTruncatedDigits++;
182                 } else {
183                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
184                         significand = 10 * significand + ch - '0';
185                     } else {
186                         break;
187                     }
188                 }
189             }
190             isSignificandTruncated = (index < significandEndIndex);
191             exponentOfTruncatedSignificand = virtualIndexOfPoint - index + skipCountInTruncatedDigits + expNumber;
192         } else {
193             isSignificandTruncated = false;
194             exponentOfTruncatedSignificand = 0;
195         }
196         return valueOfFloatLiteral(
197                 str,
198                 startIndex,
199                 endIndex,
200                 isNegative,
201                 significand,
202                 exponent,
203                 isSignificandTruncated,
204                 exponentOfTruncatedSignificand,
205                 rejectOverflow
206         );
207     }

235             int index,
236             int startIndex,

238             boolean isNegative,
239             boolean rejectOverflow
240     ) throws NumericException {
241

243         // ------------
244         long significand = 0;// significand is treated as an unsigned long
245         int exponent = 0;
246         final int significandStartIndex = index;
247         int virtualIndexOfPoint = -1;
248         final int digitCount;
249         boolean illegal = false;
250         byte ch = 0;
251         for (; index < endIndex; index++) {
252             ch = str[index];
253             // Table look up is faster than a sequence of if-else-branches.
254             int hexValue = ch < 0 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
255             if (hexValue >= 0) {

257             } else if (hexValue == FastDoubleUtils.DECIMAL_POINT_CLASS) {

259                 virtualIndexOfPoint = index;
260             } else {
261                 break;
262             }
263         }

265         if (virtualIndexOfPoint < 0) {
266             digitCount = significandEndIndex - significandStartIndex;
267             virtualIndexOfPoint = significandEndIndex;
268         } else {
269             digitCount = significandEndIndex - significandStartIndex - 1;
270             exponent = Math.min(virtualIndexOfPoint - index + 1, FastDoubleUtils.MAX_EXPONENT_NUMBER) * 4;
271         }
272

274         // --------------
275         int expNumber = 0;
276         final boolean hasExponent = (ch == 'p') || (ch == 'P');
277         if (hasExponent) {
278             ch = ++index < endIndex ? str[index] : 0;
279             boolean neg_exp = ch == '-';
280             if (neg_exp || ch == '+') {
281                 ch = ++index < endIndex ? str[index] : 0;
282             }
283             illegal |= !isDigit(ch);
284             do {
285                 expNumber = 10 * expNumber + ch - '0';
286                 ch = ++index < endIndex ? str[index] : 0;
287             } while (isDigit(ch));
288             if (neg_exp) {
289                 expNumber = -expNumber;
290             }
291             exponent += expNumber;
292         }
293 
294         // Skip optional FloatTypeSuffix
295         // ------------------------
296         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
297             index++;
298         }
299

301         // ------------------------
302         index = skipWhitespace(str, index, endIndex);
303         if (illegal || index < endIndex
304                 || digitCount == 0

310         // -----------------------------------------------

315             for (index = significandStartIndex; index < significandEndIndex; index++) {

318                 int hexValue = ch < 0 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
319                 if (hexValue >= 0) {
320                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
321                         significand = (significand << 4) | hexValue;
322                     } else {
323                         break;
324                     }
325                 } else {
326                     skipCountInTruncatedDigits++;
327                 }
328             }
329             isSignificandTruncated = (index < significandEndIndex);
330         } else {
331             isSignificandTruncated = false;

353      * <dd>{@code Infinity} <i>[WhiteSpace] EOT</i></dd>
354      * </dl>
355      * </blockquote>
356      *
357      * @param str      a string
358      * @param index    index of the "I" character
359      * @param endIndex end index (exclusive)

361      * @throws NumberFormatException on parsing failure
362      */
363     private static double parseInfinity(byte[] str, int index, int endIndex, boolean negative) throws NumericException {
364         if (index + 7 < endIndex
365                 && str[index] == 'I'
366                 && str[index + 1] == 'n'
367                 && str[index + 2] == 'f'
368                 && str[index + 3] == 'i'
369                 && str[index + 4] == 'n'
370                 && str[index + 5] == 'i'
371                 && str[index + 6] == 't'
372                 && str[index + 7] == 'y'
373         ) {
374             index = skipWhitespace(str, index + 8, endIndex);
375             if (index == endIndex) {
376                 return negative ? Double.NEGATIVE_INFINITY : Double.POSITIVE_INFINITY;
377             }
378         }
379         throw NumericException.INSTANCE;
380     }

436      * Parses a {@code FloatingPointLiteral} production with optional leading and trailing
437      * white space.
438      * <blockquote>
439      * <dl>
440      * <dt><i>FloatingPointLiteralWithWhiteSpace:</i></dt>
441      * <dd><i>[WhiteSpace] FloatingPointLiteral [WhiteSpace]</i></dd>
442      * </dl>
443      * </blockquote>
444      * See {@link io.questdb.std.fastdouble} for the grammar of
445      * {@code FloatingPointLiteral}.
446      *
447      * @param str            a string containing a {@code FloatingPointLiteralWithWhiteSpace}
448      * @param offset         start offset of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
449      * @param length         length of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
450      * @param rejectOverflow reject parsed values that overflow double type
451      * @return the parsed value, if the input is legal;
452      * @throws NumericException is the input is illegal.
453      */
454     static double parseFloatingPointLiteral(byte[] str, int offset, int length, boolean rejectOverflow) throws NumericException {
455         final int endIndex = offset + length;
456         if (offset < 0 || endIndex > str.length) {
457             throw NumericException.INSTANCE;
458         }
459 
460         // Skip leading whitespace
461         // -------------------
462         int index = skipWhitespace(str, offset, endIndex);
463         if (index == endIndex) {
464             throw NumericException.INSTANCE;
465         }
466         byte ch = str[index];
467 
468         // Parse optional sign
469         // -------------------
470         final boolean isNegative = ch == '-';
471         if (isNegative || ch == '+') {
472             ch = ++index < endIndex ? str[index] : 0;
473             if (ch == 0) {
474                 throw NumericException.INSTANCE;
475             }
476         }
477 
478         // Parse NaN or Infinity
479         // ---------------------
480         if (ch >= 'I') {
481             return ch == 'N'
482                     ? parseNaN(str, index, endIndex)
483                     : parseInfinity(str, index, endIndex, isNegative);
484         }
485 
486         // Parse optional leading zero
487         // ---------------------------
488         final boolean hasLeadingZero = ch == '0';
489         if (hasLeadingZero) {
490             ch = ++index < endIndex ? str[index] : 0;
491             if (ch == 'x' || ch == 'X') {
492                 return parseHexFloatingPointLiteral(str, index + 1, offset, endIndex, isNegative, rejectOverflow);
493             }
494         }
495 
496         return parseDecFloatLiteral(str, index, offset, endIndex, isNegative, hasLeadingZero, rejectOverflow);
497     }
