40     private static double dealWithNaN(double value, long str, int startIndex, int endIndex, boolean rejectOverflow) throws NumericException {
41         if (Double.isNaN(value)) {
42             int len = endIndex - startIndex;
43             byte[] b = new byte[len];
44             for (int i = 0; i < len; i++) {
45                 b[i] = Unsafe.getUnsafe().getByte(str + startIndex + i);
46             }
47             double v = Double.parseDouble(new String(b, StandardCharsets.ISO_8859_1));
48             if (rejectOverflow && (
49                     v == Double.POSITIVE_INFINITY
50                             || v == Double.NEGATIVE_INFINITY
51                             || v == 0.0
52             )
53             ) {
54                 throw NumericException.INSTANCE;
55             }
56             return v;
57         }
58         return value;
59     }

66      * Parses a {@code DecimalFloatingPointLiteral} production with optional
67      * trailing white space until the end of the text.
68      * Given that we have already consumed the optional leading zero of
69      * the {@code DecSignificand}.
70      * <blockquote>
71      * <dl>
72      * <dt><i>DecimalFloatingPointLiteralWithWhiteSpace:</i></dt>
73      * <dd><i>DecimalFloatingPointLiteral [WhiteSpace] EOT</i></dd>
74      * </dl>
75      * </blockquote>
76      * See {@link io.questdb.std.fastdouble} for the grammar of
77      * {@code DecimalFloatingPointLiteral} and {@code DecSignificand}.
78      *
79      * @param str            a string
80      * @param index          start index inclusive of the {@code DecimalFloatingPointLiteralWithWhiteSpace}
81      * @param endIndex       end index (exclusive)
82      * @param isNegative     true if the float value is negative
83      * @param hasLeadingZero true if we have consumed the optional leading zero
84      * @param rejectOverflow reject parsed values that overflow double type
85      * @return the parsed value, if the input is legal;
86      * @throws NumericException if the input is illegal.
87      */
88     private static double parseDecFloatLiteral(
89             long str,
90             int index,
91             int startIndex,
92             int endIndex,
93             boolean isNegative,
94             boolean hasLeadingZero,
95             boolean rejectOverflow
96     ) throws NumericException {
97         // Parse significand
98         // -----------------
99         // Note: a multiplication by a constant is cheaper than an
100         //       arbitrary integer multiplication.
101         long significand = 0;// significand is treated as an unsigned long
102         final int significandStartIndex = index;
103         int virtualIndexOfPoint = -1;
104         boolean illegal = false;
105         byte ch = 0;
106         for (; index < endIndex; index++) {
107             ch = Unsafe.getUnsafe().getByte(str + index);
108             if (isDigit(ch)) {
109                 // This might overflow, we deal with it later.
110                 significand = 10 * significand + ch - '0';
111             } else if (ch == '.') {
112                 illegal |= virtualIndexOfPoint >= 0;
113                 virtualIndexOfPoint = index;
114                 for (; index < endIndex - 8; index += 8) {
115                     int eightDigits = tryToParseEightDigits(str, index + 1);
116                     if (eightDigits < 0) {
117                         break;
118                     }
119                     // This might overflow, we deal with it later.
120                     significand = 100_000_000L * significand + eightDigits;
121                 }
122             } else {
123                 break;
124             }
125         }
126         final int digitCount;
127         final int significandEndIndex = index;
128         int exponent;
129         if (virtualIndexOfPoint < 0) {
130             digitCount = index - significandStartIndex;
131             virtualIndexOfPoint = index;
132             exponent = 0;
133         } else {
134             digitCount = index - significandStartIndex - 1;
135             exponent = virtualIndexOfPoint - index + 1;
136         }
137 
138         // Parse exponent number
139         // ---------------------
140         int expNumber = 0;
141         if (ch == 'e' || ch == 'E') {
142             ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
143             boolean neg_exp = ch == '-';
144             if (neg_exp || ch == '+') {
145                 ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
146             }
147             illegal |= !isDigit(ch);
148             do {
149                 // Guard against overflow
150                 if (expNumber < FastDoubleUtils.MAX_EXPONENT_NUMBER) {
151                     expNumber = 10 * expNumber + ch - '0';
152                 }
153                 ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
154             } while (isDigit(ch));
155             if (neg_exp) {
156                 expNumber = -expNumber;
157             }
158             exponent += expNumber;
159         }
160 
161         // Skip optional FloatTypeSuffix
162         // ------------------------
163         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
164             index++;
165         }
166 
167         // Skip trailing whitespace and check if FloatingPointLiteral is complete
168         // ------------------------
169         index = skipWhitespace(str, index, endIndex);
170         if (illegal
171                 || index < endIndex
172                 || !hasLeadingZero
173                 && digitCount == 0
174         ) {
175             throw NumericException.INSTANCE;
176         }
177 
178         // Re-parse significand in case of a potential overflow
179         // -----------------------------------------------
180         final boolean isSignificandTruncated;
181         int skipCountInTruncatedDigits = 0;//counts +1 if we skipped over the decimal point
182         int exponentOfTruncatedSignificand;
183         if (digitCount > 19) {
184             significand = 0;
185             for (index = significandStartIndex; index < significandEndIndex; index++) {
186                 ch = Unsafe.getUnsafe().getByte(str + index);
187                 if (ch == '.') {
188                     skipCountInTruncatedDigits++;
189                 } else {
190                     if (Long.compareUnsigned(significand, FastDoubleUtils.MINIMAL_NINETEEN_DIGIT_INTEGER) < 0) {
191                         significand = 10 * significand + ch - '0';
192                     } else {
193                         break;
194                     }
195                 }
196             }
197             isSignificandTruncated = (index < significandEndIndex);
198             exponentOfTruncatedSignificand = virtualIndexOfPoint - index + skipCountInTruncatedDigits + expNumber;
199         } else {
200             isSignificandTruncated = false;
201             exponentOfTruncatedSignificand = 0;
202         }
203         return valueOfFloatLiteral(
204                 str,
205                 startIndex,
206                 endIndex,
207                 isNegative,
208                 significand,
209                 exponent,
210                 isSignificandTruncated,
211                 exponentOfTruncatedSignificand,
212                 rejectOverflow
213         );
214     }

252         int exponent = 0;

255         final int digitCount;
256         boolean illegal = false;
257         byte ch = 0;

259             ch = Unsafe.getUnsafe().getByte(str + index);

261             int hexValue = ch < 0 ? FastDoubleUtils.OTHER_CLASS : FastDoubleUtils.CHAR_TO_HEX_MAP[ch];
262             if (hexValue >= 0) {

264             } else if (hexValue == FastDoubleUtils.DECIMAL_POINT_CLASS) {

267             } else {

275         } else {

278         }

288                 ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
289             }
290             illegal |= !isDigit(ch);
291             do {
292                 expNumber = 10 * expNumber + ch - '0';
293                 ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
294             } while (isDigit(ch));
295             if (neg_exp) {
296                 expNumber = -expNumber;
297             }
298             exponent += expNumber;
299         }
300 
301         // Skip optional FloatTypeSuffix
302         // ------------------------
303         if (index < endIndex && (ch == 'd' || ch == 'D' || ch == 'f' || ch == 'F')) {
304             index++;
305         }
306 
307         // Skip trailing whitespace and check if FloatingPointLiteral is complete
308         // ------------------------
309         index = skipWhitespace(str, index, endIndex);
310         if (illegal || index < endIndex
311                 || digitCount == 0

356      * until the end of the text.
357      * <blockquote>
358      * <dl>
359      * <dt><i>InfinityWithWhiteSpace:</i></dt>
360      * <dd>{@code Infinity} <i>[WhiteSpace] EOT</i></dd>
361      * </dl>
362      * </blockquote>
363      *
364      * @param str      a string
365      * @param index    index of the "I" character
366      * @param endIndex end index (exclusive)
367      * @return a positive or negative infinity value
368      * @throws NumberFormatException on parsing failure
369      */
370     private static double parseInfinity(long str, int index, int endIndex, boolean negative) throws NumericException {
371         if (index + 7 < endIndex
372                 && Unsafe.getUnsafe().getByte(str + index) == 'I'
373                 && Unsafe.getUnsafe().getByte(str + index + 1) == 'n'
374                 && Unsafe.getUnsafe().getByte(str + index + 2) == 'f'
375                 && Unsafe.getUnsafe().getByte(str + index + 3) == 'i'
376                 && Unsafe.getUnsafe().getByte(str + index + 4) == 'n'
377                 && Unsafe.getUnsafe().getByte(str + index + 5) == 'i'
378                 && Unsafe.getUnsafe().getByte(str + index + 6) == 't'
379                 && Unsafe.getUnsafe().getByte(str + index + 7) == 'y'
380         ) {
381             index = skipWhitespace(str, index + 8, endIndex);
382             if (index == endIndex) {
383                 return negative ? Double.NEGATIVE_INFINITY : Double.POSITIVE_INFINITY;
384             }
385         }
386         throw NumericException.INSTANCE;
387     }

443      * Parses a {@code FloatingPointLiteral} production with optional leading and trailing
444      * white space.
445      * <blockquote>
446      * <dl>
447      * <dt><i>FloatingPointLiteralWithWhiteSpace:</i></dt>
448      * <dd><i>[WhiteSpace] FloatingPointLiteral [WhiteSpace]</i></dd>
449      * </dl>
450      * </blockquote>
451      * See {@link io.questdb.std.fastdouble} for the grammar of
452      * {@code FloatingPointLiteral}.
453      *
454      * @param str            a string containing a {@code FloatingPointLiteralWithWhiteSpace}
455      * @param offset         start offset of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
456      * @param length         length of {@code FloatingPointLiteralWithWhiteSpace} in {@code str}
457      * @param rejectOverflow reject parsed values that overflow double type
458      * @return the parsed value, if the input is legal;
459      * @throws NumericException is the input is illegal.
460      */
461     static double parseFloatingPointLiteral(long str, int offset, int length, boolean rejectOverflow) throws NumericException {
462         final int endIndex = offset + length;
463         if (offset < 0) {
464             throw NumericException.INSTANCE;
465         }
466 
467         // Skip leading whitespace
468         // -------------------
469         int index = skipWhitespace(str, offset, endIndex);
470         if (index == endIndex) {
471             throw NumericException.INSTANCE;
472         }
473         byte ch = Unsafe.getUnsafe().getByte(str + index);
474 
475         // Parse optional sign
476         // -------------------
477         final boolean isNegative = ch == '-';
478         if (isNegative || ch == '+') {
479             ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
480             if (ch == 0) {
481                 throw NumericException.INSTANCE;
482             }
483         }
484 
485         // Parse NaN or Infinity
486         // ---------------------
487         if (ch >= 'I') {
488             return ch == 'N'
489                     ? parseNaN(str, index, endIndex)
490                     : parseInfinity(str, index, endIndex, isNegative);
491         }
492 
493         // Parse optional leading zero
494         // ---------------------------
495         final boolean hasLeadingZero = ch == '0';
496         if (hasLeadingZero) {
497             ch = ++index < endIndex ? Unsafe.getUnsafe().getByte(str + index) : 0;
498             if (ch == 'x' || ch == 'X') {
499                 return parseHexFloatingPointLiteral(str, index + 1, offset, endIndex, isNegative, rejectOverflow);
500             }
501         }
502 
503         return parseDecFloatLiteral(str, index, offset, endIndex, isNegative, hasLeadingZero, rejectOverflow);
504     }
