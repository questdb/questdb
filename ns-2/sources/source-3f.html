406     private static void copyFixedSizeCol(
407             FilesFacade ff,
408             long src,
409             long srcLo,
410             long srcHi,
411             long dstFixAddr,
412             long dstFixFileOffset,
413             int dstFd,
414             final int shl,
415             boolean directIoFlag
416     ) {
417         final long len = (srcHi - srcLo + 1) << shl;
418         final long fromAddress = src + (srcLo << shl);
419         if (directIoFlag) {
420             if (ff.write(Math.abs(dstFd), fromAddress, len, dstFixFileOffset) != len) {

423             }
424         } else {
425             Vect.memcpy(dstFixAddr, fromAddress, len);
426         }

529     private static void copyVarSizeCol(
530             FilesFacade ff,
531             long srcFixAddr,
532             long srcVarAddr,
533             long srcLo,
534             long srcHi,
535             long dstFixAddr,
536             int dstFixFd,
537             long dstFixFileOffset,
538             long dstVarAddr,
539             int dstVarFd,
540             long dstVarOffset,
541             long dstVarAdjust,
542             long dstVarSize,
543             boolean directIoFlag
544     ) {
545         final long lo = O3Utils.findVarOffset(srcFixAddr, srcLo);
546         assert lo >= 0;
547         final long hi = O3Utils.findVarOffset(srcFixAddr, srcHi + 1);
548         assert hi >= lo;
549         // copy this before it changes
550         final long len = hi - lo;
551         assert len <= Math.abs(dstVarSize) - dstVarOffset;
552         final long offset = dstVarOffset + dstVarAdjust;
553         if (directIoFlag) {
554             if (ff.write(Math.abs(dstVarFd), srcVarAddr + lo, len, offset) != len) {
555                 throw CairoException.critical(ff.errno()).put("cannot copy var data column prefix [fd=").put(dstVarFd).put(", offset=").put(offset).put(", len=").put(len).put(']');
556             }
557         } else {
558             Vect.memcpy(dstVarAddr + dstVarOffset, srcVarAddr + lo, len);
559         }
560         if (lo == offset) {
561             copyFixedSizeCol(
562                     ff,
563                     srcFixAddr,
564                     srcLo,
565                     srcHi + 1,
566                     dstFixAddr,
567                     dstFixFileOffset,
568                     dstFixFd,
569                     3,
570                     directIoFlag
571             );
572         } else {
573             O3Utils.shiftCopyFixedSizeColumnData(lo - offset, srcFixAddr, srcLo, srcHi + 1, dstFixAddr);
574         }
575     }

577     private static void mergeCopy(
578             int columnType,
579             long timestampMergeIndexAddr,
580             long srcDataFixAddr,
581             long srcDataVarAddr,
582             long srcDataLo,
583             long srcDataHi,
584             long srcOooFixAddr,
585             long srcOooVarAddr,
586             long srcOooLo,
587             long srcOooHi,
588             long dstFixAddr,
589             long dstVarAddr,
590             long dstVarOffset,
591             long dstVarOffsetEnd
592     ) {
593         final long rowCount = srcOooHi - srcOooLo + 1 + srcDataHi - srcDataLo + 1;
594         switch (ColumnType.tagOf(columnType)) {
595             case ColumnType.BOOLEAN:
596             case ColumnType.BYTE:
597             case ColumnType.GEOBYTE:
598                 Vect.mergeShuffle8Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
599                 break;
600             case ColumnType.SHORT:
601             case ColumnType.CHAR:
602             case ColumnType.GEOSHORT:
603                 Vect.mergeShuffle16Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
604                 break;
605             case ColumnType.STRING:
606                 Vect.oooMergeCopyStrColumn(
607                         timestampMergeIndexAddr,
608                         rowCount,
609                         srcDataFixAddr,
610                         srcDataVarAddr,
611                         srcOooFixAddr,
612                         srcOooVarAddr,
613                         dstFixAddr,
614                         dstVarAddr,
615                         dstVarOffset
616                 );
617                 // multiple threads could be writing to this location as var index segments overlap,
618                 // but they will be writing the same value
619                 Unsafe.getUnsafe().putLong(dstFixAddr + rowCount * 8, dstVarOffsetEnd);
620                 break;
621             case ColumnType.BINARY:
622                 Vect.oooMergeCopyBinColumn(
623                         timestampMergeIndexAddr,
624                         rowCount,
625                         srcDataFixAddr,
626                         srcDataVarAddr,
627                         srcOooFixAddr,
628                         srcOooVarAddr,
629                         dstFixAddr,
630                         dstVarAddr,
631                         dstVarOffset
632                 );
633                 Unsafe.getUnsafe().putLong(dstFixAddr + rowCount * 8, dstVarOffsetEnd);
634                 break;
635             case ColumnType.INT:
636             case ColumnType.FLOAT:
637             case ColumnType.SYMBOL:
638             case ColumnType.GEOINT:
639                 Vect.mergeShuffle32Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
640                 break;
641             case ColumnType.DOUBLE:
642             case ColumnType.LONG:
643             case ColumnType.DATE:
644             case ColumnType.GEOLONG:
645                 Vect.mergeShuffle64Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
646                 break;
647             case ColumnType.TIMESTAMP:
648                 final boolean designated = ColumnType.isDesignatedTimestamp(columnType);
649                 if (designated) {
650                     Vect.oooCopyIndex(timestampMergeIndexAddr, rowCount, dstFixAddr);
651                 } else {
652                     Vect.mergeShuffle64Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
653                 }
654                 break;
655             case ColumnType.UUID:
656             case ColumnType.LONG128:
657                 Vect.mergeShuffle128Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
658                 break;
659             case ColumnType.LONG256:
660                 Vect.mergeShuffle256Bit(srcDataFixAddr, srcOooFixAddr, dstFixAddr, timestampMergeIndexAddr, rowCount);
661                 break;
662             default:
663                 break;
664         }
665     }
