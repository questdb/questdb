45     private static boolean readWalEnabled(Path path, FilesFacade ff) {
46         int fd = -1;
47         try {
48             fd = ff.openRO(path);
49             if (fd < 1) {
50                 throw CairoException.critical(ff.errno()).put("Could not open file [path=").put(path).put(']');
51             }
52 
53             final byte walType = ff.readNonNegativeByte(fd, 0);
54             switch (walType) {
55                 case TABLE_TYPE_WAL:
56                     return true;
57                 case TABLE_TYPE_NON_WAL:
58                     return false;
59                 default:
60                     throw CairoException.critical(ff.errno()).put("Could not read walType from file [path=").put(path).put(']');
61             }
62         } finally {
63             ff.close(fd);
64         }
65     }

67     private static void removeWalPersistence(Path path, int rootLen, FilesFacade ff, StringSink sink, String dirName) {
68         path.trimTo(rootLen).concat(dirName).concat(WalUtils.SEQ_DIR).$();
69         if (ff.rmdir(path) != 0) {
70             LOG.error()
71                     .$("Could not remove sequencer dir [errno=").$(ff.errno())
72                     .$(", path=").$(path)
73                     .I$();
74         }
75 
76         path.trimTo(rootLen).concat(dirName).$();
77         final long pFind = ff.findFirst(path);
78         if (pFind > 0) {
79             try {
80                 do {
81                     sink.clear();
82                     Chars.utf8DecodeZ(ff.findName(pFind), sink);
83                     if (Chars.startsWith(sink, WalUtils.WAL_NAME_BASE)) {
84                         path.trimTo(rootLen).concat(dirName).concat(sink).$();
85                         if (Chars.endsWith(sink, ".lock")) {
86                             if (!ff.remove(path)) {

88                                         .$("Could not remove wal lock file [errno=").$(ff.errno())
89                                         .$(", path=").$(path)

92                         } else {
93                             if (ff.rmdir(path) != 0) {
94                                 LOG.error()

96                                         .$(", path=").$(path)
97                                         .I$();
98                             }
99                         }
100                     }
101                 } while (ff.findNext(pFind) > 0);
102             } finally {

108     static ObjList<TableToken> convertTables(CairoConfiguration configuration, TableSequencerAPI tableSequencerAPI) {
109         final ObjList<TableToken> convertedTables = new ObjList<>();
110         if (!configuration.isTableTypeConversionEnabled()) {
111             LOG.info().$("Table type conversion is disabled").$();
112             return null;
113         }
114         if (configuration.isReadOnlyInstance()) {
115             LOG.info().$("Read only instance is not allowed to perform table type conversion").$();
116             return null;
117         }
118 
119         final Path path = Path.getThreadLocal(configuration.getRoot());
120         final int rootLen = path.length();
121         final StringSink sink = Misc.getThreadLocalBuilder();
122         final FilesFacade ff = configuration.getFilesFacade();
123         final long findPtr = ff.findFirst(path.$());
124         try {
125             do {
126                 if (ff.isDirOrSoftLinkDirNoDots(path, rootLen, ff.findName(findPtr), ff.findType(findPtr), sink)) {
127                     if (!ff.exists(path.concat(WalUtils.CONVERT_FILE_NAME).$())) {
128                         continue;
129                     }
130                     try {
131                         final String dirName = sink.toString();
132                         final boolean walEnabled = readWalEnabled(path, ff);
133                         LOG.info().$("Converting table [dirName=").utf8(dirName).$(", walEnabled=").$(walEnabled).I$();
134 
135                         path.trimTo(rootLen).concat(dirName);
136                         try (final MemoryMARW metaMem = Vm.getMARWInstance()) {
137                             openSmallFile(ff, path, rootLen, metaMem, META_FILE_NAME, MemoryTag.MMAP_SEQUENCER_METADATA);
138                             if (metaMem.getBool(TableUtils.META_OFFSET_WAL_ENABLED) == walEnabled) {
139                                 LOG.info().$("Skipping conversion, table already has the expected type [dirName=").utf8(dirName).$(", walEnabled=").$(walEnabled).I$();
140                             } else {
141                                 final String tableName;
142                                 try (final MemoryCMR mem = Vm.getCMRInstance()) {
143                                     final String name = TableUtils.readTableName(path.of(configuration.getRoot()).concat(dirName), rootLen, mem, ff);
144                                     tableName = name != null ? name : dirName;
145                                 }
146 
147                                 final int tableId = metaMem.getInt(TableUtils.META_OFFSET_TABLE_ID);
148                                 final TableToken token = new TableToken(tableName, dirName, tableId, walEnabled);
149 
150                                 if (walEnabled) {
151                                     try (TableReaderMetadata metadata = new TableReaderMetadata(configuration, token)) {
152                                         metadata.load();
153                                         tableSequencerAPI.registerTable(tableId, new TableDescriptorImpl(metadata), token);
154                                     }
155                                 } else {
156                                     tableSequencerAPI.deregisterTable(token);
157                                     removeWalPersistence(path, rootLen, ff, sink, dirName);
158                                 }
159                                 metaMem.putBool(TableUtils.META_OFFSET_WAL_ENABLED, walEnabled);
160                                 convertedTables.add(token);
161                             }
162 
163                             path.trimTo(rootLen).concat(dirName).concat(CONVERT_FILE_NAME).$();
164                             if (!ff.remove(path)) {
165                                 LOG.error().$("Could not remove _convert file [path=").utf8(path).I$();
166                             }
167                         }
168                     } catch (Exception e) {
169                         LOG.error().$("Table conversion failed [path=").utf8(path).$(", e=").$(e).I$();
170                     }
171                 }
172             } while (ff.findNext(findPtr) > 0);
173         } finally {
174             ff.findClose(findPtr);
175         }
176         return convertedTables;
177     }
