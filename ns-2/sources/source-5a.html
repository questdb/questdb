31 public class TableNameRegistryRW extends AbstractTableNameRegistry {
32     private final ConcurrentHashMap<TableToken> nameTableTokenMap = new ConcurrentHashMap<>(false);
33     private final ConcurrentHashMap<ReverseTableMapItem> reverseTableNameTokenMap = new ConcurrentHashMap<>();
34 
35     public TableNameRegistryRW(CairoConfiguration configuration) {
36         super(configuration);
37         if (!this.nameStore.lock()) {
38             if (!configuration.getAllowTableRegistrySharedWrite()) {
39                 throw CairoException.critical(0).put("cannot lock table name registry file [path=").put(configuration.getRoot()).put(']');
40             }
41         }
42         setNameMaps(nameTableTokenMap, reverseTableNameTokenMap);
43     }
44 
45     @Override
46     public boolean dropTable(TableToken token) {
47         final ReverseTableMapItem reverseMapItem = reverseTableNameTokenMap.get(token.getDirName());
48         if (reverseMapItem != null && nameTableTokenMap.remove(token.getTableName(), token)) {
49             if (token.isWal()) {
50                 nameStore.logDropTable(token);
51                 reverseTableNameTokenMap.put(token.getDirName(), ReverseTableMapItem.ofDropped(token));
52             } else {
53                 reverseTableNameTokenMap.remove(token.getDirName(), reverseMapItem);
54             }
55             return true;
56         }
57         return false;
58     }
59 
60     @Override
61     public TableToken lockTableName(String tableName, String dirName, int tableId, boolean isWal) {
62         TableToken newNameRecord = new TableToken(tableName, dirName, tableId, isWal);
63         TableToken registeredRecord = nameTableTokenMap.putIfAbsent(tableName, LOCKED_TOKEN);
64         return registeredRecord == null ? newNameRecord : null;
65     }
66 
67     @Override
68     public void purgeToken(TableToken token) {
69         reverseTableNameTokenMap.remove(token.getDirName());
70     }
71 
72     @Override
73     public void registerName(TableToken tableToken) {
74         String tableName = tableToken.getTableName();
75         if (!nameTableTokenMap.replace(tableName, LOCKED_TOKEN, tableToken)) {
76             throw CairoException.critical(0).put("cannot register table, name is not locked [name=").put(tableName).put(']');
77         }
78         if (tableToken.isWal()) {
79             nameStore.appendEntry(tableToken);
80         }
81         reverseTableNameTokenMap.put(tableToken.getDirName(), ReverseTableMapItem.of(tableToken));
82     }
83 
84     @Override
85     public synchronized void reloadTableNameCache(ObjList<TableToken> convertedTables) {
86         nameTableTokenMap.clear();
87         reverseTableNameTokenMap.clear();
88         if (!nameStore.isLocked()) {
89             nameStore.lock();
90         }
91         nameStore.reload(nameTableTokenMap, reverseTableNameTokenMap, convertedTables);
92     }

94     @Override
95     public TableToken rename(CharSequence oldName, CharSequence newName, TableToken tableToken) {
96         String newTableNameStr = Chars.toString(newName);
97         TableToken newNameRecord = new TableToken(newTableNameStr, tableToken.getDirName(), tableToken.getTableId(), tableToken.isWal());
98 
99         if (nameTableTokenMap.putIfAbsent(newTableNameStr, newNameRecord) == null) {
100             if (nameTableTokenMap.remove(oldName, tableToken)) {
101                 // Persist to file
102                 nameStore.logDropTable(tableToken);
103                 nameStore.appendEntry(newNameRecord);
104                 reverseTableNameTokenMap.put(newNameRecord.getDirName(), ReverseTableMapItem.of(newNameRecord));
105                 return newNameRecord;
106             } else {
107                 // Already renamed by another thread. Revert new name reservation.
108                 nameTableTokenMap.remove(newTableNameStr, newNameRecord);
109                 throw CairoException.tableDoesNotExist(oldName);
110             }
111         } else {
112             throw CairoException.nonCritical().put("table '").put(newName).put("' already exists");
113         }
114     }
