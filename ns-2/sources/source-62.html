226     public void toBinary(TableWriterTask sink) {
227         sink.putInt(tableAction);
228         sink.putLong(dataVersion);
229         sink.putLong(maxTimestamp);
230 
231         int n = columnTops.size();
232         sink.putInt(n); // column top count
233         if (n > 0) {
234             for (int i = 0; i < n; i += SLOTS_PER_COLUMN_TOP) {
235                 sink.putLong(columnTops.getQuick(i)); // partition timestamp
236                 sink.putInt((int) columnTops.getQuick(i + 1)); // column index
237                 sink.putLong(columnTops.getQuick(i + 2)); // column top
238             }
239         }
240 
241         n = varColumnSizes.size();
242         sink.putInt(n);
243         if (n > 0) {
244             for (int i = 0; i < n; i += SLOTS_PER_VAR_COLUMN_SIZE) {
245                 sink.putLong(varColumnSizes.getQuick(i)); // partition timestamp
246                 sink.putInt((int) varColumnSizes.getQuick(i + 1)); // column index
247                 sink.putLong(varColumnSizes.getQuick(i + 2)); // column top
248             }
249         }
250 
251         n = partitions.size();
252         sink.putInt(n); // partition count
253         for (int i = 0; i < n; i += SLOTS_PER_PARTITION) {
254             sink.putInt((int) partitions.getQuick(i)); // action
255             sink.putLong(partitions.getQuick(i + 1)); // partition timestamp
256             sink.putLong(partitions.getQuick(i + 2)); // start row
257             sink.putLong(partitions.getQuick(i + 3)); // row count
258             sink.putLong(partitions.getQuick(i + 4)); // name txn (suffix for partition name)
259             sink.putLong(partitions.getQuick(i + 5)); // column version
260         }
261 
262         n = addedColumnMetadata.size();
263         sink.putInt(n); // column metadata count - this is metadata only for column that have been added
264         for (int i = 0; i < n; i++) {
265             final TableColumnMetadata metadata = addedColumnMetadata.getQuick(i);
266             sink.putStr(metadata.getName());
267             sink.putInt(metadata.getType()); // column type
268             sink.putLong(0); // placeholder
269             sink.putByte((byte) (metadata.isIndexed() ? 0 : 1)); // column indexed flag
270             sink.putInt(metadata.getIndexValueBlockCapacity());
271         }
272 
273         n = columnMetaIndex.size();
274         sink.putInt(n); // column metadata shuffle index - drives rearrangement of existing columns on the slave
275         for (int i = 0; i < n; i += SLOTS_PER_COLUMN_META_INDEX) {
276             sink.putInt((int) columnMetaIndex.getQuick(i)); // action
277             sink.putLong(columnMetaIndex.getQuick(i + 1)); // (int,int) pair on where (from,to) column needs to move
278         }
279     }
