321             partitionTableVersion = getLong(TableUtils.TX_OFFSET_PARTITION_TABLE_VERSION_64);
322             final long prevColumnVersion = this.columnVersion;

339     public boolean unsafeLoadBaseOffset() {
340         version = unsafeReadVersion();
341         Unsafe.getUnsafe().loadFence();
342 
343         boolean isA = (version & 1L) == 0L;
344         baseOffset = isA ? roTxMemBase.getInt(TX_BASE_OFFSET_A_32) : roTxMemBase.getInt(TX_BASE_OFFSET_B_32);
345         symbolsSize = isA ? roTxMemBase.getInt(TX_BASE_OFFSET_SYMBOLS_SIZE_A_32) : roTxMemBase.getInt(TX_BASE_OFFSET_SYMBOLS_SIZE_B_32);
346         partitionSegmentSize = isA ? roTxMemBase.getInt(TX_BASE_OFFSET_PARTITIONS_SIZE_A_32) : roTxMemBase.getInt(TX_BASE_OFFSET_PARTITIONS_SIZE_B_32);
347 
348         // Before extending file, check that values read are not dirty
349         Unsafe.getUnsafe().loadFence();
350         if (unsafeReadVersion() != version) {
351             return false;
352         }
353 
354         size = calculateTxRecordSize(symbolsSize, partitionSegmentSize);
355         if (size + baseOffset > roTxMemBase.size()) {
356             roTxMemBase.extend(size + baseOffset);
357         }
358         return true;
359     }

419     private void unsafeLoadPartitions(long prevPartitionTableVersion, long prevColumnVersion, int partitionTableSize) {
420         if (PartitionBy.isPartitioned(partitionBy)) {
421             int txAttachedPartitionsSize = partitionTableSize / Long.BYTES;
422             if (txAttachedPartitionsSize > 0) {
423                 if (prevPartitionTableVersion != partitionTableVersion || prevColumnVersion != columnVersion) {
424                     attachedPartitions.clear();
425                     unsafeLoadPartitions0(0, txAttachedPartitionsSize);
426                 } else {
427                     if (attachedPartitionsSize < txAttachedPartitionsSize) {
428                         unsafeLoadPartitions0(
429                                 Math.max(attachedPartitionsSize - LONGS_PER_TX_ATTACHED_PARTITION, 0),
430                                 txAttachedPartitionsSize
431                         );
432                     }
433                 }
434                 int offset = txAttachedPartitionsSize - LONGS_PER_TX_ATTACHED_PARTITION + PARTITION_MASKED_SIZE_OFFSET;
435                 long mask = attachedPartitions.getQuick(offset) & PARTITION_FLAGS_MASK;
436                 attachedPartitions.setQuick(offset, mask | (transientRowCount & PARTITION_SIZE_MASK)); // preserve mask
437                 attachedPartitions.setPos(txAttachedPartitionsSize);
438             } else {
439                 attachedPartitionsSize = 0;
440                 attachedPartitions.clear();
441             }
442         } else {
443             // Add transient row count as the only partition in attached partitions list
444             attachedPartitions.setPos(LONGS_PER_TX_ATTACHED_PARTITION);
445             initPartitionAt(0, DEFAULT_PARTITION_TIMESTAMP, transientRowCount, -1L, columnVersion);
446         }
447     }
