68     public void applyTransitionIndex() {
69         // swap meta and transitionMeta
70         MemoryMR temp = this.metaMem;
71         this.metaMem = this.transitionMeta;
72         transitionMeta = temp;
73         transitionMeta.close(); // Memory is safe to double close, do not assign null to transitionMeta
74         this.columnNameIndexMap.clear();
75         int existingColumnCount = this.columnCount;
76 
77         int columnCount = metaMem.getInt(TableUtils.META_OFFSET_COUNT);
78         assert columnCount >= existingColumnCount;
79         columnMetadata.setPos(columnCount);
80         int timestampIndex = metaMem.getInt(TableUtils.META_OFFSET_TIMESTAMP_INDEX);
81         this.tableId = metaMem.getInt(TableUtils.META_OFFSET_TABLE_ID);
82         this.structureVersion = metaMem.getLong(TableUtils.META_OFFSET_STRUCTURE_VERSION);
83         this.maxUncommittedRows = metaMem.getInt(TableUtils.META_OFFSET_MAX_UNCOMMITTED_ROWS);
84         this.o3MaxLag = metaMem.getLong(TableUtils.META_OFFSET_O3_MAX_LAG);
85         this.walEnabled = metaMem.getBool(TableUtils.META_OFFSET_WAL_ENABLED);
86         long offset = TableUtils.getColumnNameOffset(columnCount);
87 
88         int shiftLeft = 0, existingIndex = 0;
89         for (int metaIndex = 0; metaIndex < columnCount; metaIndex++) {
90             CharSequence name = metaMem.getStr(offset);
91             offset += Vm.getStorageLength(name);
92             assert name != null;
93             int columnType = TableUtils.getColumnType(metaMem, metaIndex);
94             boolean isIndexed = TableUtils.isColumnIndexed(metaMem, metaIndex);
95             int indexBlockCapacity = TableUtils.getIndexBlockCapacity(metaMem, metaIndex);
96             TableColumnMetadata existing = null;
97             String newName;
98 
99             if (existingIndex < existingColumnCount) {
100                 existing = columnMetadata.getQuick(existingIndex);
101 
102                 if (existing.getWriterIndex() > metaIndex) {
103                     // This column must be deleted so existing dense columns do not contain it
104                     assert columnType < 0;
105                     continue;
106                 }
107             }
108             assert existing == null || existing.getWriterIndex() == metaIndex; // Same column
109 
110             // exiting column
111             if (columnType < 0) {
112                 shiftLeft++;
113             } else {
114                 boolean rename = existing != null && !Chars.equals(existing.getName(), name);
115                 newName = rename || existing == null ? Chars.toString(name) : existing.getName();
116                 if (rename
117                         || existing == null
118                         || existing.isIndexed() != isIndexed
119                         || existing.getIndexValueBlockCapacity() != indexBlockCapacity
120                 ) {
121                     columnMetadata.setQuick(existingIndex - shiftLeft,
122                             new TableColumnMetadata(
123                                     newName,
124                                     columnType,
125                                     isIndexed,
126                                     indexBlockCapacity,
127                                     true,
128                                     null,
129                                     metaIndex
130                             )
131                     );
132                 } else if (shiftLeft > 0) {
133                     columnMetadata.setQuick(existingIndex - shiftLeft, existing);
134                 }
135                 this.columnNameIndexMap.put(newName, existingIndex - shiftLeft);
136                 if (timestampIndex == metaIndex) {
137                     this.timestampIndex = existingIndex - shiftLeft;
138                 }
139             }
140             existingIndex++;
141         }
142         columnMetadata.setPos(existingIndex - shiftLeft);
143         this.columnCount = columnMetadata.size();
144         if (timestampIndex < 0) {
145             this.timestampIndex = timestampIndex;
146         }
147     }

208     public void dumpTo(MemoryMA mem) {
209         // Since _meta files are immutable and get updated with a single atomic rename
210         // operation replacing the old file with the new one, it's ok to clone the metadata
211         // by copying metaMem's contents. Even if _meta file was already replaced, the file
212         // should be still kept on disk until inode's ref counter is above zero.
213         long len = metaMem.size();
214         for (long p = 0; p < len; p++) {
215             mem.putByte(metaMem.getByte(p));
216         }
217     }

257     public void load() {
258         final long timeout = configuration.getSpinLockTimeout();
259         final MillisecondClock millisecondClock = configuration.getMillisecondClock();
260         long deadline = configuration.getMillisecondClock().getTicks() + timeout;
261         this.path.trimTo(plen).concat(TableUtils.META_FILE_NAME).$();
262         boolean existenceChecked = false;
263         while (true) {
264             try {
265                 load(path);
266                 return;
267             } catch (CairoException ex) {
268                 if (!existenceChecked) {
269                     path.trimTo(plen).slash$();
270                     if (!ff.exists(path)) {
271                         throw CairoException.tableDoesNotExist(tableToken.getTableName());
272                     }
273                     path.trimTo(plen).concat(TableUtils.META_FILE_NAME).$();
274                 }
275                 existenceChecked = true;
276                 TableUtils.handleMetadataLoadException(tableToken.getTableName(), deadline, ex, millisecondClock, timeout);
277             }
278         }
279     }

281     public void load(Path path) {
282         try {
283             this.metaMem.smallFile(ff, path, MemoryTag.NATIVE_TABLE_READER);
284             this.columnNameIndexMap.clear();
285             TableUtils.validateMeta(metaMem, this.columnNameIndexMap, ColumnType.VERSION);
286             int columnCount = metaMem.getInt(TableUtils.META_OFFSET_COUNT);
287             int timestampIndex = metaMem.getInt(TableUtils.META_OFFSET_TIMESTAMP_INDEX);
288             this.partitionBy = metaMem.getInt(TableUtils.META_OFFSET_PARTITION_BY);
289             this.tableId = metaMem.getInt(TableUtils.META_OFFSET_TABLE_ID);
290             this.maxUncommittedRows = metaMem.getInt(TableUtils.META_OFFSET_MAX_UNCOMMITTED_ROWS);
291             this.o3MaxLag = metaMem.getLong(TableUtils.META_OFFSET_O3_MAX_LAG);
292             this.structureVersion = metaMem.getLong(TableUtils.META_OFFSET_STRUCTURE_VERSION);
293             this.walEnabled = metaMem.getBool(TableUtils.META_OFFSET_WAL_ENABLED);
294             this.columnMetadata.clear();
295             long offset = TableUtils.getColumnNameOffset(columnCount);
296             this.timestampIndex = -1;
297 
298             // don't create strings in this loop, we already have them in columnNameIndexMap
299             for (int i = 0; i < columnCount; i++) {
300                 CharSequence name = metaMem.getStr(offset);
301                 assert name != null;
302                 int columnType = TableUtils.getColumnType(metaMem, i);
303                 if (columnType > -1) {
304                     columnMetadata.add(
305                             new TableColumnMetadata(
306                                     Chars.toString(name),
307                                     columnType,
308                                     TableUtils.isColumnIndexed(metaMem, i),
309                                     TableUtils.getIndexBlockCapacity(metaMem, i),
310                                     true,
311                                     null,
312                                     i
313                             )
314                     );
315                     if (i == timestampIndex) {
316                         this.timestampIndex = columnMetadata.size() - 1;
317                     }
318                 }
319                 offset += Vm.getStorageLength(name);
320             }
321             this.columnCount = columnMetadata.size();
322         } catch (Throwable e) {
323             clear();
324             throw e;
325         }
326     }
