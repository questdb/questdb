90     public final void reload(MemoryMR metaMem) {
91         this.columnCount = metaMem.getInt(TableUtils.META_OFFSET_COUNT);
92         this.columnNameIndexMap.clear();
93         this.version = metaMem.getInt(TableUtils.META_OFFSET_VERSION);
94         this.tableId = metaMem.getInt(TableUtils.META_OFFSET_TABLE_ID);
95         this.maxUncommittedRows = metaMem.getInt(TableUtils.META_OFFSET_MAX_UNCOMMITTED_ROWS);
96         this.o3MaxLag = metaMem.getLong(TableUtils.META_OFFSET_O3_MAX_LAG);
97         TableUtils.validateMeta(metaMem, columnNameIndexMap, ColumnType.VERSION);
98         this.timestampIndex = metaMem.getInt(TableUtils.META_OFFSET_TIMESTAMP_INDEX);
99         this.columnMetadata.clear();
100         this.structureVersion = metaMem.getLong(TableUtils.META_OFFSET_STRUCTURE_VERSION);
101         this.walEnabled = metaMem.getBool(TableUtils.META_OFFSET_WAL_ENABLED);
102 
103         long offset = TableUtils.getColumnNameOffset(columnCount);
104         this.symbolMapCount = 0;
105         columnNameIndexMap.clear();
106         // don't create strings in this loop, we already have them in columnNameIndexMap
107         for (int i = 0; i < columnCount; i++) {
108             CharSequence name = metaMem.getStr(offset);
109             assert name != null;
110             int type = TableUtils.getColumnType(metaMem, i);
111             String nameStr = Chars.toString(name);
112             columnMetadata.add(
113                     new TableColumnMetadata(
114                             nameStr,
115                             type,
116                             TableUtils.isColumnIndexed(metaMem, i),
117                             TableUtils.getIndexBlockCapacity(metaMem, i),
118                             true,
119                             null,
120                             i
121                     )
122             );
123             columnNameIndexMap.put(nameStr, i);
124             if (ColumnType.isSymbol(type)) {
125                 symbolMapCount++;
126             }
127             offset += Vm.getStorageLength(name);
128         }
129     }
