31     public static final int SCAN_DOWN = 1;
32     // Up is decreasing direction

35     /**
36      * Performs binary search on column of Long values.
37      *
38      * @param column        the column
39      * @param value         the search value
40      * @param low           the low boundary of the search, inclusive of value
41      * @param high          the high boundary of the search inclusive of value
42      * @param scanDirection logical direction in which column is searched. UP means we are looking for
43      *                      the bottom boundary of the values that are lower or equal the search value. DOWN means
44      *                      we are looking for upper boundary of the values that are greater or equal the search
45      *                      value.
46      * @return index in column where value is less or equal to the search value. If column contains
47      * multiple exact matches the scanDirection determines whether top or bottom of these matches is returned.
48      * When scan direction is DOWN - the last index of exact matches is returns, when UP - the first index
49      */
50     public static long find(MemoryR column, long value, long low, long high, int scanDirection) {
51         while (low < high) {
52             long mid = (low + high) / 2;
53             long midVal = column.getLong(mid * Long.BYTES);
54 
55             if (midVal < value) {
56                 if (low < mid) {
57                     low = mid;
58                 } else {
59                     if (column.getLong(high * Long.BYTES) > value) {
60                         return low;
61                     }
62                     return high;
63                 }
64             } else if (midVal > value)
65                 high = mid;
66             else {
67                 // In case of multiple equal values, find the first
68                 mid += scanDirection;
69                 while (mid > 0 && mid <= high && midVal == column.getLong(mid * Long.BYTES)) {
70                     mid += scanDirection;
71                 }
72                 return mid - scanDirection;
73             }
74         }
75 
76         if (column.getLong(low * Long.BYTES) > value) {
77             return low - 1;
78         }
79         return low;
80     }
