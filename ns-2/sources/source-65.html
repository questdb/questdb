663      * Attaches a partition to the table. If size is given, partition file data is not validated.
664      *
665      * @param timestamp     partition timestamp
666      * @param partitionSize partition size in rows. Negative means unknown size.
667      * @return attached status code
668      */
669     public AttachDetachStatus attachPartition(long timestamp, long partitionSize) {
670         // Partitioned table must have a timestamp
671         // SQL compiler will check that table has it
672         assert metadata.getTimestampIndex() > -1;
673 
674         if (txWriter.attachedPartitionsContains(timestamp)) {
675             LOG.info().$("partition is already attached [path=").$(path).I$();
676             // TODO: potentially we can merge with existing data
677             return AttachDetachStatus.ATTACH_ERR_PARTITION_EXISTS;
678         }
679 
680         if (inTransaction()) {
681             LOG.info().$("committing open transaction before applying attach partition command [table=").utf8(tableToken.getTableName())
682                     .$(", partition=").$ts(timestamp).I$();
683             commit();
684 
685             // Check that partition we're about to attach hasn't appeared after commit
686             if (txWriter.attachedPartitionsContains(timestamp)) {
687                 LOG.info().$("partition is already attached [path=").$(path).I$();
688                 return AttachDetachStatus.ATTACH_ERR_PARTITION_EXISTS;
689             }
690         }
691 
692         // final name of partition folder after attach
693         setPathForPartition(path.trimTo(rootLen), partitionBy, timestamp, false);
694         TableUtils.txnPartitionConditionally(path, getTxn());
695         path.$();
696 
697         if (ff.exists(path)) {
698             // Very unlikely since txn is part of the folder name
699             return AttachDetachStatus.ATTACH_ERR_DIR_EXISTS;
700         }
701 
702         Path detachedPath = Path.PATH.get().of(configuration.getRoot()).concat(tableToken);
703         setPathForPartition(detachedPath, partitionBy, timestamp, false);
704         detachedPath.put(configuration.getAttachPartitionSuffix()).$();
705         int detachedRootLen = detachedPath.length();
706         boolean forceRenamePartitionDir = partitionSize < 0;
707 
708         boolean checkPassed = false;
709         boolean isSoftLink;
710         try {
711             if (ff.exists(detachedPath)) {
712 
713                 isSoftLink = ff.isSoftLink(detachedPath); // returns false regardless in Windows
714 
715                 // detached metadata files validation
716                 CharSequence timestampColName = metadata.getColumnMetadata(metadata.getTimestampIndex()).getName();
717                 if (partitionSize > -1L) {
718                     // read detachedMinTimestamp and detachedMaxTimestamp
719                     readPartitionMinMax(ff, timestamp, detachedPath.trimTo(detachedRootLen), timestampColName, partitionSize);
720                 } else {
721                     // read size, detachedMinTimestamp and detachedMaxTimestamp
722                     partitionSize = readPartitionSizeMinMax(ff, timestamp, detachedPath.trimTo(detachedRootLen), timestampColName);
723                 }
724 
725                 if (partitionSize < 1) {
726                     return AttachDetachStatus.ATTACH_ERR_EMPTY_PARTITION;
727                 }
728 
729                 if (forceRenamePartitionDir && !attachPrepare(timestamp, partitionSize, detachedPath, detachedRootLen)) {
730                     attachValidateMetadata(partitionSize, detachedPath.trimTo(detachedRootLen), timestamp);
731                 }
732 
733                 // main columnVersionWriter is now aligned with the detached partition values read from partition _cv file
734                 // in case of an error it has to be clean up
735 
736                 if (forceRenamePartitionDir && configuration.attachPartitionCopy() && !isSoftLink) { // soft links are read-only, no copy involved
737                     // Copy partition if configured to do so and it's not CSV import
738                     if (ff.copyRecursive(detachedPath.trimTo(detachedRootLen), path, configuration.getMkDirMode()) == 0) {
739                         LOG.info().$("copied partition dir [from=").$(detachedPath).$(", to=").$(path).I$();
740                     } else {
741                         LOG.error().$("could not copy [errno=").$(ff.errno()).$(", from=").$(detachedPath).$(", to=").$(path).I$();
742                         return AttachDetachStatus.ATTACH_ERR_COPY;
743                     }
744                 } else {
745                     if (ff.rename(detachedPath.trimTo(detachedRootLen).$(), path) == FILES_RENAME_OK) {
746                         LOG.info().$("renamed partition dir [from=").$(detachedPath).$(", to=").$(path).I$();
747                     } else {
748                         LOG.error().$("could not rename [errno=").$(ff.errno()).$(", from=").$(detachedPath).$(", to=").$(path).I$();
749                         return AttachDetachStatus.ATTACH_ERR_RENAME;
750                     }
751                 }
752 
753                 checkPassed = true;
754             } else {
755                 LOG.info().$("attach partition command failed, partition to attach does not exist [path=").$(detachedPath).I$();
756                 return AttachDetachStatus.ATTACH_ERR_MISSING_PARTITION;
757             }
758         } finally {
759             path.trimTo(rootLen);
760             if (!checkPassed) {
761                 columnVersionWriter.readUnsafe();
762             }
763         }
764 
765         try {
766             // find out lo, hi ranges of partition attached as well as size
767             assert timestamp <= attachMinTimestamp && attachMinTimestamp <= attachMaxTimestamp;
768             long nextMinTimestamp = Math.min(attachMinTimestamp, txWriter.getMinTimestamp());
769             long nextMaxTimestamp = Math.max(attachMaxTimestamp, txWriter.getMaxTimestamp());
770             boolean appendPartitionAttached = size() == 0 || getPartitionLo(nextMaxTimestamp) > getPartitionLo(txWriter.getMaxTimestamp());
771 
772             txWriter.beginPartitionSizeUpdate();
773             txWriter.updatePartitionSizeByTimestamp(timestamp, partitionSize, getTxn());
774             txWriter.finishPartitionSizeUpdate(nextMinTimestamp, nextMaxTimestamp);
775             if (isSoftLink) {
776                 txWriter.setPartitionReadOnlyByTimestamp(timestamp, true);
777             }
778             txWriter.bumpTruncateVersion();
779 
780             columnVersionWriter.commit();
781             txWriter.setColumnVersion(columnVersionWriter.getVersion());
782             txWriter.commit(defaultCommitMode, denseSymbolMapWriters);
783 
784             LOG.info().$("partition attached [table=").utf8(tableToken.getTableName())
785                     .$(", partition=").$ts(timestamp).I$();
786 
787             if (appendPartitionAttached) {
788                 LOG.info().$("switch partition after partition attach [tableName=").utf8(tableToken.getTableName())
789                         .$(", partition=").$ts(timestamp).I$();
790                 freeColumns(true);
791                 configureAppendPosition();
792             }
793             return AttachDetachStatus.OK;
794         } catch (Throwable e) {
795             // This is pretty serious, after partition copied there are no OS operations to fail
796             // Do full rollback to clean up the state
797             LOG.critical().$("failed on attaching partition to the table and rolling back [tableName=").utf8(tableToken.getTableName())
798                     .$(", error=").$(e).I$();
799             rollback();
800             throw e;
801         }
802     }

867     public AttachDetachStatus detachPartition(long timestamp) {
868         // Should be checked by SQL compiler
869         assert metadata.getTimestampIndex() > -1;
870         assert PartitionBy.isPartitioned(partitionBy);
871 
872         if (inTransaction()) {
873             LOG.info()
874                     .$("committing open transaction before applying detach partition command [table=")
875                     .utf8(tableToken.getTableName())
876                     .$(", partition=").$ts(timestamp)
877                     .I$();
878             commit();
879         }
880 
881         int partitionIndex = txWriter.getPartitionIndex(timestamp);
882         if (partitionIndex == -1) {
883             assert !txWriter.attachedPartitionsContains(timestamp);
884             return AttachDetachStatus.DETACH_ERR_MISSING_PARTITION;
885         }
886 
887         long maxTimestamp = txWriter.getMaxTimestamp();
888         if (timestamp == getPartitionLo(maxTimestamp)) {
889             return AttachDetachStatus.DETACH_ERR_ACTIVE;
890         }
891         long minTimestamp = txWriter.getMinTimestamp();
892 
893         long partitionNameTxn = txWriter.getPartitionNameTxn(partitionIndex);
894         Path detachedPath = Path.PATH.get();
895 
896         try {
897             // path: partition folder to be detached
898             setPathForPartition(path, rootLen, partitionBy, timestamp, partitionNameTxn);
899             if (!ff.exists(path.$())) {
900                 LOG.error().$("partition folder does not exist [path=").$(path).I$();
901                 return AttachDetachStatus.DETACH_ERR_MISSING_PARTITION_DIR;
902             }
903 
904             final int detachedPathLen;
905             AttachDetachStatus attachDetachStatus;
906             if (ff.isSoftLink(path)) {
907                 detachedPathLen = 0;
908                 attachDetachStatus = AttachDetachStatus.OK;
909                 LOG.info().$("detaching partition via unlink [path=").$(path).I$();
910             } else {
911 
912                 detachedPath.of(configuration.getRoot()).concat(tableToken.getDirName());
913                 int detachedRootLen = detachedPath.length();
914                 // detachedPath: detached partition folder
915                 if (!ff.exists(detachedPath.slash$())) {
916                     // the detached and standard folders can have different roots
917                     // (server.conf: cairo.sql.detached.root)
918                     if (0 != ff.mkdirs(detachedPath, mkDirMode)) {
919                         LOG.error().$("could no create detached partition folder [errno=").$(ff.errno())
920                                 .$(", path=").$(detachedPath).I$();
921                         return AttachDetachStatus.DETACH_ERR_MKDIR;
922                     }
923                 }
924                 setPathForPartition(detachedPath.trimTo(detachedRootLen), partitionBy, timestamp, false);
925                 detachedPath.put(DETACHED_DIR_MARKER).$();
926                 detachedPathLen = detachedPath.length();
927                 if (ff.exists(detachedPath)) {
928                     LOG.error().$("detached partition folder already exist [path=").$(detachedPath).I$();
929                     return AttachDetachStatus.DETACH_ERR_ALREADY_DETACHED;
930                 }
931 
932                 // Hard link partition folder recursive to partition.detached
933                 if (ff.hardLinkDirRecursive(path, detachedPath, mkDirMode) != 0) {
934                     if (ff.isCrossDeviceCopyError(ff.errno())) {
935                         // Cross drive operation. Make full copy to another device.
936                         if (ff.copyRecursive(path, detachedPath, mkDirMode) != 0) {
937                             LOG.critical().$("could not copy detached partition [errno=").$(ff.errno())
938                                     .$(", from=").$(path)
939                                     .$(", to=").$(detachedPath)
940                                     .I$();

942                         }
943                     } else {
944                         LOG.critical().$("could not create hard link to detached partition [errno=").$(ff.errno())
945                                 .$(", from=").$(path)
946                                 .$(", to=").$(detachedPath)
947                                 .I$();
948                         return AttachDetachStatus.DETACH_ERR_HARD_LINK;
949                     }
950                 }
951 
952                 // copy _meta, _cv and _txn to partition.detached _meta, _cv and _txn
953                 other.of(path).trimTo(rootLen).concat(META_FILE_NAME).$(); // exists already checked
954                 detachedPath.trimTo(detachedPathLen).concat(META_FILE_NAME).$();
955 
956                 attachDetachStatus = AttachDetachStatus.OK;
957                 if (-1 == copyOverwrite(detachedPath)) {
958                     attachDetachStatus = AttachDetachStatus.DETACH_ERR_COPY_META;
959                     LOG.critical().$("could not copy [errno=").$(ff.errno())
960                             .$(", from=").$(other)
961                             .$(", to=").$(detachedPath)
962                             .I$();
963                 } else {
964                     other.parent().concat(COLUMN_VERSION_FILE_NAME).$();
965                     detachedPath.parent().concat(COLUMN_VERSION_FILE_NAME).$();
966                     if (-1 == copyOverwrite(detachedPath)) {
967                         attachDetachStatus = AttachDetachStatus.DETACH_ERR_COPY_META;
968                         LOG.critical().$("could not copy [errno=").$(ff.errno())
969                                 .$(", from=").$(other)
970                                 .$(", to=").$(detachedPath)
971                                 .I$();
972                     } else {
973                         other.parent().concat(TXN_FILE_NAME).$();
974                         detachedPath.parent().concat(TXN_FILE_NAME).$();
975                         if (-1 == copyOverwrite(detachedPath)) {
976                             attachDetachStatus = AttachDetachStatus.DETACH_ERR_COPY_META;
977                             LOG.critical().$("could not copy [errno=").$(ff.errno())
978                                     .$(", from=").$(other)
979                                     .$(", to=").$(detachedPath)
980                                     .I$();
981                         }
982                     }
983                 }
984             }
985 
986             if (attachDetachStatus == AttachDetachStatus.OK) {
987                 // find out if we are removing min partition
988                 long nextMinTimestamp = minTimestamp;
989                 if (timestamp == txWriter.getPartitionTimestamp(0)) {
990                     other.of(path).trimTo(rootLen);
991                     nextMinTimestamp = readMinTimestamp(txWriter.getPartitionTimestamp(1));
992                 }
993 
994                 // all good, commit
995                 txWriter.beginPartitionSizeUpdate();
996                 txWriter.removeAttachedPartitions(timestamp);
997                 txWriter.setMinTimestamp(nextMinTimestamp);
998                 txWriter.finishPartitionSizeUpdate(nextMinTimestamp, txWriter.getMaxTimestamp());
999                 txWriter.bumpTruncateVersion();
1000 
1001                 columnVersionWriter.removePartition(timestamp);
1002                 columnVersionWriter.commit();
1003 
1004                 txWriter.setColumnVersion(columnVersionWriter.getVersion());
1005                 txWriter.commit(defaultCommitMode, denseSymbolMapWriters);
1006                 // return at the end of the method after removing partition directory
1007             } else {
1008                 // rollback detached copy
1009                 detachedPath.trimTo(detachedPathLen).slash().$();
1010                 if (ff.rmdir(detachedPath) != 0) {
1011                     LOG.error()
1012                             .$("could not rollback detached copy (rmdir) [errno=").$(ff.errno())
1013                             .$(", undo=").$(detachedPath)
1014                             .$(", original=").$(path)
1015                             .I$();
1016                 }
1017                 return attachDetachStatus;
1018             }
1019         } finally {
1020             path.trimTo(rootLen);
1021             other.trimTo(rootLen);
1022         }
1023         safeDeletePartitionDir(timestamp, partitionNameTxn);
1024         return AttachDetachStatus.OK;
1025     }

1438                             o3TimestampMemCpy.addressOf(0)

1446                     lagRowCount = 0L;
1447                     o3Columns = o3MemColumns;

1449                     o3TimestampMin = getTimestampIndexValue(timestampAddr, o3Lo);
1450                     o3TimestampMax = getTimestampIndexValue(timestampAddr, o3Hi - 1);

1469                                 );

1477                                 .$(", lagRowCount=").$(lagRowCount)
1478                                 .$(", rowLo=").$(o3Lo)
1479                                 .$(", rowHi=").$(o3Hi).I$();

1481                         lagRowCount = 0;
1482                     }
1483

1485                     processO3Block(
1486                             lagRowCount,
1487                             timestampIndex,
1488                             timestampAddr,
1489                             o3Hi,
1490                             o3TimestampMin,
1491                             o3TimestampMax,
1492                             copiedToMemory,
1493                             o3Lo
