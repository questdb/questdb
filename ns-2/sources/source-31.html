38     // we use the highest bit of every storage size (byte, short, int, long)
39     // to indicate null value. When a null value is cast down, nullity is
40     // preserved, i.e. highest bit remains set:
41     //     long nullLong = -1L;
42     //     short nullShort = (short) nullLong;
43     //     nullShort == nullLong;
44     // in addition, -1 is the first negative non geohash value.

71     public static void addNormalizedGeoPrefix(long hash, int prefixType, int columnType, final LongList prefixes) throws NumericException {
72         final int bits = ColumnType.getGeoHashBits(prefixType);
73         final int columnSize = ColumnType.sizeOf(columnType);
74         final int columnBits = ColumnType.getGeoHashBits(columnType);
75 
76         if (hash == NULL || bits > columnBits) {
77             throw NumericException.INSTANCE;
78         }
79 
80         final int shift = columnBits - bits;
81         long norm = hash << shift;
82         long mask = GeoHashes.bitmask(bits, shift);
83         mask |= 1L << (columnSize * 8 - 1); // set the most significant bit to ignore null from prefix matching
84 
85         prefixes.add(norm);
86         prefixes.add(mask);
87     }

111         // Below assertion can happen if there is corrupt metadata
112         // which should not happen in production code since reader and writer check table metadata
113         assert bits > 0 && bits <= ColumnType.GEO_HASH_MAX_BITS_LENGTH;

196         if (start == end) {
197             return NULL;
198         }
199         final int chars = Math.min(end - start, MAX_STRING_LENGTH);
200         int fromBits = 5 * chars;

205     }

207     public static long fromStringTruncatingNl(long lo, long hi, int bits) throws NumericException {
208         if (lo == hi) {
209             return NULL;
210         }
211         final int chars = Math.min((int) (hi - lo), MAX_STRING_LENGTH);
212         int actualBits = 5 * chars;
213         if (actualBits < bits || bits == 0) {
214             throw NumericException.INSTANCE;
215         }
216         long geohash = 0;
217         for (long p = lo, limit = p + chars; p < limit; p++) {
218             geohash = appendChar(geohash, (char) Unsafe.getUnsafe().getByte(p)); // base32
219         }
220         return widen(geohash, actualBits, bits);
221     }

224         if (!ColumnType.isGeoHash(columnType)) {
225             return 0;
226         }
227         final int bits = ColumnType.getGeoHashBits(columnType);
228         if (bits > 0 && bits % 5 == 0) {
229             // It's 5 bit per char. If it's integer number of chars value to be serialized as chars
230             return -bits / 5;
231         }
232         // Value to be serialized as bit array.
233         return bits;

278         final byte idx = encodeChar(c);
279         if (idx > -1) {

285     private static long fromBitString(CharSequence bits, int start, int limit) throws NumericException {
286         long result = 0;
287         for (int i = start; i < limit; i++) {
288             switch (bits.charAt(i)) {
289                 case '0':
290                     result = result << 1;
291                     break;
292                 case '1':
293                     result = (result << 1) | 1;
294                     break;
295                 default:
296                     throw NumericException.INSTANCE;
297             }
298         }
299         return result;
