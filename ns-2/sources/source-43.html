79     private static void parsePartitionDateVersion(StringSink fileNameSink, DirectLongList partitionList, CharSequence tableName, DateFormat partitionByFormat) {
80         int index = Chars.lastIndexOf(fileNameSink, '.');
81 
82         int len = fileNameSink.length();
83         if (index < 0) {
84             index = len;
85         }
86         try {
87             if (index < len) {
88                 long partitionVersion = Numbers.parseLong(fileNameSink, index + 1, len);
89                 // When reader locks transaction 100 it opens partition version .99 or lower.
90                 // Also, when there is no transaction version in the name, it is counted as -1.
91                 // By adding +1 here we kill 2 birds in with one stone, partition versions are aligned with
92                 // txn scoreboard reader locks and no need to add -1 which allows us to use 128bit
93                 // sort to sort 2 x 64bit unsigned integers
94                 partitionList.add(partitionVersion + 1);
95             } else {
96                 // This should be -1, but it is only possible to correctly sort 2 unsigned longs
97                 // as 128bit integer sort
98                 // Set 0 instead of -1 and revert it later on. There should be not possible to have .0 in the partition name
99                 partitionList.add(0);
100             }
101 
102             try {
103                 long partitionTs = partitionByFormat.parse(fileNameSink, 0, index, null);
104                 partitionList.add(partitionTs);
105             } catch (NumericException e) {
106                 if (!Chars.startsWith(tableName, WalUtils.WAL_NAME_BASE) && !Chars.equals(tableName, WalUtils.SEQ_DIR)) {
107                     LOG.error().$("unknown directory [table=").utf8(tableName).$(", dir=").utf8(fileNameSink).I$();
108                 }
109                 partitionList.setPos(partitionList.size() - 1); // remove partition version record
110             }
111         } catch (NumericException e) {
112             LOG.error().$("unknown directory [table=").utf8(tableName).$(", dir=").utf8(fileNameSink).I$();
113         }
114     }

116     private static void processDetachedPartition(
117             FilesFacade ff,
118             Path path,
119             int tableRootLen,
120             TxReader txReader,
121             TxnScoreboard txnScoreboard,
122             long partitionTimestamp,
123             int partitionBy,
124             DirectLongList partitionList,
125             int lo,
126             int hi
127     ) {
128         // Partition is dropped or not fully committed.
129         // It is only possible to delete when there are no readers
130         long lastTxn = txReader.getTxn();
131         for (int i = hi - 2, n = lo - 1; i > n; i -= 2) {
132             long nameTxn = partitionList.get(i);
133 
134             // If last committed transaction number is 4, TableWriter can write partition with ending .4 and .3
135             // If the version on disk is .2 (nameTxn == 3) can remove it if the lastTxn > 3, e.g. when nameTxn < lastTxn
136             boolean rangeUnlocked = nameTxn < lastTxn && txnScoreboard.isRangeAvailable(nameTxn, lastTxn);
137 
138             path.trimTo(tableRootLen);
139             TableUtils.setPathForPartition(path, partitionBy, partitionTimestamp, false);
140             TableUtils.txnPartitionConditionally(path, nameTxn - 1);
141             path.$();
142 
143             if (rangeUnlocked) {
144                 // nameTxn can be deleted
145                 // -1 here is to compensate +1 added when partition version parsed from folder name
146                 // See comments of why +1 added there in parsePartitionDateVersion()
147                 LOG.info().$("purging dropped partition directory [path=").utf8(path).I$();
148                 ff.unlinkOrRemove(path, LOG);
149                 lastTxn = nameTxn;
150             } else {
151                 LOG.info().$("cannot purge partition directory, locked for reading [path=").utf8(path).I$();
152                 break;
153             }
154         }
155     }

199     private static void processPartition0(
200             FilesFacade ff,
201             Path path,
202             int tableRootLen,
203             TxReader txReader,
204             TxnScoreboard txnScoreboard,
205             long partitionTimestamp,
206             int partitionBy,
207             DirectLongList partitionList,
208             int lo,
209             int hi
210     ) {
211         long lastCommittedPartitionName = txReader.getPartitionNameTxnByPartitionTimestamp(partitionTimestamp);
212         if (lastCommittedPartitionName > -1) {
213             assert hi <= partitionList.size();
214             // lo points to the beginning element in partitionList, hi next after last
215             // each partition folder represented by a pair in the partitionList (partition version, partition timestamp)
216             // Skip first pair, start from second and check if it can be deleted.
217             for (int i = lo + 2; i < hi; i += 2) {
218                 long nextNameVersion = Math.min(lastCommittedPartitionName + 1, partitionList.get(i));
219                 long previousNameVersion = partitionList.get(i - 2);
220 
221                 boolean rangeUnlocked = previousNameVersion < nextNameVersion
222                         && txnScoreboard.isRangeAvailable(previousNameVersion, nextNameVersion);
223 
224                 path.trimTo(tableRootLen);
225                 TableUtils.setPathForPartition(path, partitionBy, partitionTimestamp, false);
226                 TableUtils.txnPartitionConditionally(path, previousNameVersion - 1);
227                 path.$();
228 
229                 if (rangeUnlocked) {
230                     // previousNameVersion can be deleted
231                     // -1 here is to compensate +1 added when partition version parsed from folder name
232                     // See comments of why +1 added there in parsePartitionDateVersion()
233                     LOG.info().$("purging overwritten partition directory [path=").utf8(path).I$();
234                     ff.unlinkOrRemove(path, LOG);
235                 } else {
236                     LOG.info().$("cannot purge overwritten partition directory, locked for reading [path=").utf8(path).I$();
237                 }
238             }
239         }
240     }

242     private void discoverPartitions(
243             FilesFacade ff,
244             StringSink fileNameSink,
245             DirectLongList partitionList,
246             CharSequence root,
247             TableToken tableToken,
248             TxnScoreboard txnScoreboard,
249             TxReader txReader,
250             int partitionBy) {
251 
252         LOG.info().$("processing [table=").utf8(tableToken.getDirName()).I$();
253         Path path = Path.getThreadLocal(root).concat(tableToken);
254         int plimit = path.length();
255         partitionList.clear();
256         DateFormat partitionByFormat = PartitionBy.getPartitionDirFormatMethod(partitionBy);
257         long p = ff.findFirst(path.$());
258         if (p > 0) {
259             try {
260                 do {
261                     if (ff.isDirOrSoftLinkDirNoDots(path, plimit, ff.findName(p), ff.findType(p), fileNameSink)) {
262                         parsePartitionDateVersion(fileNameSink, partitionList, tableToken.getDirName(), partitionByFormat);
263                         path.trimTo(plimit).$();
264                     }
265                 } while (ff.findNext(p) > 0);
266             } finally {
267                 ff.findClose(p);
268             }
269         }
270 
271         // find duplicate partitions
272         assert partitionList.size() % 2 == 0;
273         Vect.sort128BitAscInPlace(partitionList.getAddress(), partitionList.size() / 2);
274 
275         long partitionTimestamp = Numbers.LONG_NaN;
276         int lo = 0;
277         int n = (int) partitionList.size();
278 
279         path.of(root).concat(tableToken);
280 
281         int tableRootLen = path.length();
282         try {
283             txnScoreboard.ofRO(path);
284             txReader.ofRO(path.trimTo(tableRootLen).concat(TXN_FILE_NAME).$(), partitionBy);
285             TableUtils.safeReadTxn(txReader, configuration.getMillisecondClock(), configuration.getSpinLockTimeout());
286 
287             for (int i = 0; i < n; i += 2) {
288                 long currentPartitionTs = partitionList.get(i + 1);
289                 if (currentPartitionTs != partitionTimestamp) {
290                     if (i > lo + 2 ||
291                             (i > 0 && txReader.getPartitionSizeByPartitionTimestamp(partitionTimestamp) < 0)) {
292                         processPartition(
293                                 ff,
294                                 path,
295                                 tableRootLen,
296                                 txReader,
297                                 txnScoreboard,
298                                 partitionTimestamp,
299                                 partitionBy,
300                                 partitionList,
301                                 lo,
302                                 i
303                         );
304                     }
305                     lo = i;
306                     partitionTimestamp = currentPartitionTs;
307                 }
308             }
309             // Tail
310             if (n > lo + 2 || txReader.getPartitionSizeByPartitionTimestamp(partitionTimestamp) < 0) {
311                 processPartition(
312                         ff,
313                         path,
314                         tableRootLen,
315                         txReader,
316                         txnScoreboard,
317                         partitionTimestamp,
318                         partitionBy,
319                         partitionList,
320                         lo,
321                         n
322                 );
323             }
324         } catch (CairoException ex) {
325             // It is possible that table is dropped while this async job was in the queue.
326             // so it can be not too bad. Log error and continue work on the queue
327             LOG.error()
328                     .$("could not purge partition open [table=`").utf8(tableToken.getDirName())
329                     .$("`, ex=").$(ex.getFlyweightMessage())
330                     .$(", errno=").$(ex.getErrno())
331                     .I$();
332             LOG.error().$(ex.getFlyweightMessage()).$();
333         } finally {
334             txReader.clear();
335             txnScoreboard.clear();
336         }
337         LOG.info().$("processed [table=").$(tableToken).I$();
338     }
