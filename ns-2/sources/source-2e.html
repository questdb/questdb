31 public class FullFwdDataFrameCursorFactory extends AbstractDataFrameCursorFactory {

39     @Override
40     public DataFrameCursor getCursor(SqlExecutionContext executionContext, int order) {
41         if (order == ORDER_ASC || order == ORDER_ANY) {
42             return cursor.of(getReader(executionContext));
43         }
44 
45         // Create backward scanning cursor when needed. Factory requesting backward cursor must
46         // still return records in ascending timestamp order.
47         if (bwdCursor == null) {
48             bwdCursor = new FullBwdDataFrameCursor();
49         }
50         return bwdCursor.of(getReader(executionContext));
51     }
52 
53     @Override
54     public int getOrder() {
55         return ORDER_ASC;
56     }
57 
58     @Override
59     public void toPlan(PlanSink sink) {
60         if (sink.getOrder() == ORDER_DESC) {
61             sink.type("Frame backward scan");
62         } else {
63             sink.type("Frame forward scan");
64         }
65         super.toPlan(sink);
