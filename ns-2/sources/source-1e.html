72     public void dumpTo(MemoryW mem) {
73         mem.putLong(OFFSET_VERSION_64, version);
74         boolean areaA = (version & 1L) == 0L;
75         final long offset = HEADER_SIZE;
76         mem.putLong(areaA ? OFFSET_OFFSET_A_64 : OFFSET_OFFSET_B_64, offset);
77         final long size = (long) (cachedList.size() / BLOCK_SIZE) * BLOCK_SIZE_BYTES;
78         mem.putLong(areaA ? OFFSET_SIZE_A_64 : OFFSET_SIZE_B_64, size);
79 
80         int i = 0;
81         long p = offset;
82         long lim = offset + size;
83 
84         while (p < lim) {
85             mem.putLong(p, cachedList.getQuick(i));
86             mem.putLong(p + COLUMN_INDEX_OFFSET * Long.BYTES, cachedList.getQuick(i + COLUMN_INDEX_OFFSET));
87             mem.putLong(p + COLUMN_NAME_TXN_OFFSET * Long.BYTES, cachedList.getQuick(i + COLUMN_NAME_TXN_OFFSET));
88             mem.putLong(p + COLUMN_TOP_OFFSET * Long.BYTES, cachedList.getQuick(i + COLUMN_TOP_OFFSET));
89             i += BLOCK_SIZE;
90             p += BLOCK_SIZE_BYTES;
91         }
92     }

107     /**
108      * Checks that column exists in the partition and returns the column top
109      *
110      * @param partitionTimestamp timestamp of the partition
111      * @param columnIndex        column index
112      * @return column top in the partition or -1 if column does not exist in the partition
113      */
114     public long getColumnTop(long partitionTimestamp, int columnIndex) {
115         // Check if there is explicit record for this partitionTimestamp / columnIndex combination
116         int recordIndex = getRecordIndex(partitionTimestamp, columnIndex);
117         if (recordIndex > -1L) {
118             return cachedList.getQuick(recordIndex + COLUMN_TOP_OFFSET);
119         }
120 
121         // Check if column has been already added before this partition
122         long columnTopDefaultPartition = getColumnTopPartitionTimestamp(columnIndex);
123         if (columnTopDefaultPartition <= partitionTimestamp) {
124             return 0;
125         }
126 
127         // This column does not exist in the partition
128         return -1L;
129     }

167     public int getRecordIndex(long partitionTimestamp, int columnIndex) {
168         int index = cachedList.binarySearchBlock(BLOCK_SIZE_MSB, partitionTimestamp, BinarySearch.SCAN_UP);
169         if (index > -1) {
170             final int sz = cachedList.size();
171             for (; index < sz && cachedList.getQuick(index) == partitionTimestamp; index += BLOCK_SIZE) {
172                 final long thisIndex = cachedList.getQuick(index + COLUMN_INDEX_OFFSET);
173 
174                 if (thisIndex == columnIndex) {
175                     return index;
176                 }
177 
178                 if (thisIndex > columnIndex) {
179                     break;
180                 }
181             }
182         }
183         return -1;
184     }

200     public void readSafe(MillisecondClock microsecondClock, long spinLockTimeout) {
201         final long tick = microsecondClock.getTicks();
202         while (true) {
203             long version = unsafeGetVersion();
204             if (version == this.version) {
205                 return;
206             }
207             Unsafe.getUnsafe().loadFence();
208 
209             final long offset;
210             final long size;
211 
212             final boolean areaA = (version & 1L) == 0;
213             if (areaA) {
214                 offset = mem.getLong(OFFSET_OFFSET_A_64);
215                 size = mem.getLong(OFFSET_SIZE_A_64);
216             } else {
217                 offset = mem.getLong(OFFSET_OFFSET_B_64);
218                 size = mem.getLong(OFFSET_SIZE_B_64);
219             }
220 
221             Unsafe.getUnsafe().loadFence();
222             if (version == unsafeGetVersion()) {
223                 mem.resize(offset + size);
224                 readUnsafe(offset, size, cachedList, mem);
225 
226                 Unsafe.getUnsafe().loadFence();
227                 if (version == unsafeGetVersion()) {
228                     this.version = version;
229                     LOG.debug().$("read clean version ").$(version).$(", offset ").$(offset).$(", size ").$(size).$();
230                     return;
231                 }
232             }
233 
234             if (microsecondClock.getTicks() - tick > spinLockTimeout) {
235                 LOG.error().$("Column Version read timeout [timeout=").$(spinLockTimeout).utf8("ms]").$();
236                 throw CairoException.critical(0).put("Column Version read timeout");
237             }
238             Os.pause();
239             LOG.debug().$("read dirty version ").$(version).$(", retrying").$();
240         }
241     }

243     private static void readUnsafe(long offset, long areaSize, LongList cachedList, MemoryR mem) {
244         long lim = offset + areaSize;
245         mem.extend(lim);
246         int i = 0;
247         long p = offset;
248 
249         assert areaSize % BLOCK_SIZE_BYTES == 0;
250 
251         cachedList.setPos((int) ((areaSize / BLOCK_SIZE_BYTES) * BLOCK_SIZE));
252 
253         while (p < lim) {
254             cachedList.setQuick(i, mem.getLong(p));
255             cachedList.setQuick(i + COLUMN_INDEX_OFFSET, mem.getLong(p + COLUMN_INDEX_OFFSET * Long.BYTES));
256             cachedList.setQuick(i + COLUMN_NAME_TXN_OFFSET, mem.getLong(p + COLUMN_NAME_TXN_OFFSET * Long.BYTES));
257             cachedList.setQuick(i + COLUMN_TOP_OFFSET, mem.getLong(p + COLUMN_TOP_OFFSET * Long.BYTES));
258             i += BLOCK_SIZE;
259             p += BLOCK_SIZE_BYTES;
260         }
261     }
