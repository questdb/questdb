32 public class ColumnIndexerJob extends AbstractQueueConsumerJob<ColumnIndexerTask> {
33 
34     public ColumnIndexerJob(MessageBus messageBus) {
35         super(messageBus.getIndexerQueue(), messageBus.getIndexerSubSequence());
36     }
37 
38     protected boolean doRun(int workerId, long cursor, RunStatus runStatus) {
39         final ColumnIndexerTask queueItem = queue.get(cursor);
40         // copy values and release queue item
41         final ColumnIndexer indexer = queueItem.indexer;
42         final long lo = queueItem.lo;
43         final long hi = queueItem.hi;
44         final long indexSequence = queueItem.sequence;
45         final SOCountDownLatch latch = queueItem.countDownLatch;
46         subSeq.done(cursor);
47 
48         // On the face of it main thread could have consumed same sequence as
49         // child workers. The reason it is undesirable is because all writers
50         // share the same queue and main thread end up indexing content for other writers.
51         // Using CAS allows main thread to steal only parts of its own job.
52         if (indexer.tryLock(indexSequence)) {
53             TableWriter.indexAndCountDown(indexer, lo, hi, latch);
54             return true;
55         }
56         // This is hard to test. Condition occurs when main thread successfully steals
57         // work from under nose of this worker.
58         return false;
59     }
