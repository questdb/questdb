73     public RowCursor initCursor(RowCursor rowCursor, int key, long minValue, long maxValue) {
74         Cursor cursor = null;
75         if (rowCursor != null && rowCursor != EmptyRowCursor.INSTANCE) {
76             cursor = (Cursor) rowCursor;
77             assert cursor.owner() == this;
78         }
79 
80         if (key == 0 && unIndexedNullCount > 0 && minValue < unIndexedNullCount) {
81             // we need to return some nulls and the whole set of actual index values
82             if (cursor == null) {
83                 cursor = new Cursor();
84             }
85             cursor.of(key, 0, maxValue, keyCount, minValue, unIndexedNullCount);
86             return cursor;
87         }
88 
89         if (key < keyCount) {
90             if (cursor == null) {
91                 cursor = new Cursor();
92             }
93             cursor.of(key, minValue, maxValue, keyCount, 0, 0);
94             return cursor;
95         }
96 
97         return EmptyRowCursor.INSTANCE;
98     }

110         @Override
111         public boolean hasNext() {
112             if (nullPos < nullCount) {
113                 next = nullPos++;
114                 return true;
115             }
116             if (position < valueCount) {
117                 long cellIndex = getValueCellIndex(position++);
118                 long result = valueMem.getLong(valueBlockOffset + cellIndex * 8);
119 
120                 if (result > maxValue) {
121                     valueCount = 0;
122                     return false;
123                 }
124 
125                 if (cellIndex == blockValueCountMod && position < valueCount) {
126                     // We are at edge of block right now, next value will be in next block.
127                     if (!jumpToNextValueBlock()) {
128                         // We've reached the memory boundary which means that no values left.
129                         valueCount = 0;
130                         return false;
131                     }
132                 }
133 
134                 this.next = result;
135                 return true;
136             }
137             return false;
138         }

145         private long getValueCellIndex(long absoluteValueIndex) {

149         private boolean jumpToNextValueBlock() {
150             long nextBlock = valueMem.getLong(valueBlockOffset + blockCapacity - BitmapIndexUtils.VALUE_BLOCK_FILE_RESERVED + 8);
151             if (nextBlock >= valueMem.size()) {
152                 return false;
153             }
154             valueBlockOffset = nextBlock;
155             return true;
156         }

167         void of(int key, long minValue, long maxValue, long keyCount, long nullPos, long nullCount) {
168             this.nullPos = nullPos;
169             this.nullCount = nullCount;
170             if (keyCount == 0) {
171                 valueCount = 0;
172             } else {
173                 assert key > -1 : "key must be positive integer: " + key;
174                 assert key < keyCount : "key must be within the last known boundary: " + key + ", " + keyCount;
175 
176                 long offset = BitmapIndexUtils.getKeyEntryOffset(key);
177                 // Read value count and first block offset atomically. In that we must orderly read value count first and
178                 // value count check last. If they match - everything we read between those holds true. We must retry
179                 // should these values do not match.
180                 long valueCount;
181                 long valueBlockOffset;
182                 final long deadline = clock.getTicks() + spinLockTimeoutUs;
183                 while (true) {
184                     valueCount = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT);
185 
186                     Unsafe.getUnsafe().loadFence();
187                     if (keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK) == valueCount) {
188                         valueBlockOffset = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_FIRST_VALUE_BLOCK_OFFSET);
189 
190                         Unsafe.getUnsafe().loadFence();
191                         if (keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT) == valueCount) {
192                             break;
193                         }
194                     }
195 
196                     if (clock.getTicks() > deadline) {
197                         LOG.error().$(INDEX_CORRUPT).$(" [timeout=").$(spinLockTimeoutUs).utf8("μs, key=").$(key).$(", offset=").$(offset).$(']').$();

199                     }
200                 }

205                     // the valueMem size boundary as a part of the value search. Luckily, it does that.
206                     BitmapIndexUtils.seekValueBlockLTR(valueCount, valueBlockOffset, valueMem, minValue, blockValueCountMod, SEEKER);
207                 } else {
