111     private void compactTableNameFile(Map<CharSequence, TableToken> nameTableTokenMap, int lastFileVersion, FilesFacade ff, Path path, long currentOffset) {
112         // compact the memory, remove deleted entries.
113         // write to the tmp file.
114         int pathRootLen = path.length();
115         path.concat(TABLE_REGISTRY_NAME_FILE).put(".tmp").$();
116 
117         try {
118             tableNameMemory.close(false);
119             tableNameMemory.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
120             tableNameMemory.putLong(0L);
121 
122             for (TableToken token : nameTableTokenMap.values()) {
123                 writeEntry(token, OPERATION_ADD);
124             }
125             tableNameMemory.sync(false);
126             long newAppendOffset = tableNameMemory.getAppendOffset();
127             tableNameMemory.close();
128 
129             // rename tmp to next version file, everyone will automatically switch to new file
130             Path path2 = Path.getThreadLocal2(configuration.getRoot())
131                     .concat(TABLE_REGISTRY_NAME_FILE).put('.').put(lastFileVersion + 1).$();
132             if (ff.rename(path, path2) == Files.FILES_RENAME_OK) {
133                 LOG.info().$("compacted tables file [path=").$(path2).$(']').$();
134                 lastFileVersion++;
135                 currentOffset = newAppendOffset;
136                 // best effort to remove old files, but we don't care if it fails
137                 path.trimTo(pathRootLen).concat(TABLE_REGISTRY_NAME_FILE).put('.').put(lastFileVersion - 1).$();
138                 ff.remove(path);
139             } else {
140                 // Not critical, if rename fails, compaction will be done next time
141                 LOG.error().$("could not rename tables file, tables file will not be compacted [from=").$(path)
142                         .$(", to=").$(path2).$(']').$();
143             }
144         } finally {
145             path.trimTo(pathRootLen).concat(TABLE_REGISTRY_NAME_FILE).put('.').put(lastFileVersion).$();
146             tableNameMemory.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
147             tableNameMemory.jumpTo(currentOffset);
148         }
149     }

151     private int findLastTablesFileVersion(FilesFacade ff, Path path) {
152         long findPtr = ff.findFirst(path.$());
153         if (findPtr == 0) {
154             throw CairoException.critical(0).put("database root directory does not exist at ").put(path);
155         }
156         try {
157             int lastVersion = 0;
158             do {
159                 long pUtf8NameZ = ff.findName(findPtr);
160                 if (ff.findType(findPtr) == DT_FILE) {
161                     nameSink.clear();
162                     Chars.utf8DecodeZ(pUtf8NameZ, nameSink);
163                     if (Chars.startsWith(nameSink, TABLE_REGISTRY_NAME_FILE) && nameSink.length() > TABLE_REGISTRY_NAME_FILE.length() + 1) {
164                         //noinspection CatchMayIgnoreException
165                         try {
166                             int version = Numbers.parseInt(nameSink, TABLE_REGISTRY_NAME_FILE.length() + 1, nameSink.length());
167                             if (version > lastVersion) {
168                                 lastVersion = version;
169                             }
170                         } catch (NumericException e) {
171                         }
172                     }
173                 }
174             } while (ff.findNext(findPtr) > 0);
175             return lastVersion;
176         } finally {
177             ff.findClose(findPtr);
178         }
179     }

181     private int readTableId(Path path, CharSequence dirName, FilesFacade ff) {
182         path.of(configuration.getRoot()).concat(dirName).concat(TableUtils.META_FILE_NAME).$();
183         int fd = ff.openRO(path);
184         if (fd < 1) {
185             return 0;
186         }
187 
188         try {
189             int tableId = ff.readNonNegativeInt(fd, TableUtils.META_OFFSET_TABLE_ID);
190             if (tableId < 0) {
191                 LOG.error().$("cannot read table id from metadata file [path=").$(path).I$();
192                 return 0;
193             }
194             byte isWal = (byte) (ff.readNonNegativeInt(fd, TableUtils.META_OFFSET_WAL_ENABLED) & 0xFF);
195             return isWal == 0 ? tableId : -tableId;
196         } finally {
197             ff.close(fd);
198         }
199     }

202             ConcurrentHashMap<TableToken> nameTableTokenMap,

204     ) {
205         Path path = Path.getThreadLocal(configuration.getRoot());

208         long findPtr = ff.findFirst(path.$());
209         StringSink sink = Misc.getThreadLocalBuilder();
210         try {
211             do {
212                 if (ff.isDirOrSoftLinkDirNoDots(path, plimit, ff.findName(findPtr), ff.findType(findPtr), sink)) {
213                     if (!reverseTableNameTokenMap.containsKey(sink)
214                             && TableUtils.exists(ff, path, configuration.getRoot(), sink) == TableUtils.TABLE_EXISTS) {
215

217                         int tableId;
218                         boolean isWal;
219                         String tableName;
220 
221                         try {
222                             tableId = readTableId(path, dirName, ff);
223                             isWal = tableId < 0;

226                         } catch (CairoException e) {
227                             if (e.errnoReadPathDoesNotExist()) {
228                                 // table is being removed.
229                                 continue;
230                             } else {
231                                 throw e;
232                             }
233                         } finally {
234                             tableNameRoMemory.close();
235                         }
236 
237                         if (tableName == null) {
238                             if (isWal) {
239                                 LOG.error().$("could not read table name, table will not be available [dirName=").utf8(dirName).I$();
240                                 continue;
241                             } else {
242                                 // Non-wal tables may not have _name file.
243                                 tableName = Chars.toString(TableUtils.getTableNameFromDirName(dirName));
244                             }
245                         }
246 
247                         if (tableId > -1L) {
248                             if (tableIds.contains(tableId)) {
249                                 LOG.critical().$("duplicate table id found, table will not be available " +
250                                         "[dirName=").utf8(dirName).$(", id=").$(tableId).I$();
251                                 continue;
252                             }
253 
254                             if (nameTableTokenMap.containsKey(tableName)) {
255                                 LOG.critical().$("duplicate table name found, table will not be available " +
256                                                 "[dirName=").utf8(dirName).$(", name=").utf8(tableName)
257                                         .$(", existingTableDir=").utf8(nameTableTokenMap.get(tableName).getDirName()).I$();
258                                 continue;
259                             }
260 
261                             TableToken token = new TableToken(tableName, dirName, tableId, isWal);
262                             nameTableTokenMap.put(tableName, token);
263                             reverseTableNameTokenMap.put(dirName, ReverseTableMapItem.of(token));
264                         }
265                     }
266                 }
267             } while (ff.findNext(findPtr) > 0);
268         } finally {
269             ff.findClose(findPtr);
270         }
271     }

273     private void reloadFromTablesFile(
274             ConcurrentHashMap<TableToken> nameTableTokenMap,
275             ConcurrentHashMap<ReverseTableMapItem> reverseTableNameTokenMap,
276             ObjList<TableToken> convertedTables
277     ) {
278         int lastFileVersion;
279         FilesFacade ff = configuration.getFilesFacade();
280         Path path = Path.getThreadLocal(configuration.getRoot());
281         path.of(configuration.getRoot());
282         int pathRootLen = path.length();
283 
284         MemoryMR memory = isLocked() ? tableNameMemory : tableNameRoMemory;
285         do {
286             lastFileVersion = findLastTablesFileVersion(ff, path.trimTo(pathRootLen).$());
287             path.trimTo(pathRootLen).concat(TABLE_REGISTRY_NAME_FILE).put('.').put(lastFileVersion).$();
288             try {
289                 memory.smallFile(ff, path, MemoryTag.MMAP_DEFAULT);
290                 LOG.info().$("reloading tables file [path=").utf8(path).I$();
291                 if (memory.size() >= 2 * Long.BYTES) {
292                     break;
293                 }
294             } catch (CairoException e) {
295                 if (!isLocked()) {
296                     if (e.errnoReadPathDoesNotExist()) {
297                         if (lastFileVersion == 0) {
298                             // This is RO mode and file and tables.d.0 does not exist.
299                             return;
300                         } else {
301                             // This is RO mode and file we want to read was just swapped to new one by the RW instance.
302                             continue;
303                         }
304                     }
305                 }
306                 throw e;
307             }
308         } while (true);
309 
310         long mapMem = memory.getLong(0);
311         long currentOffset = Long.BYTES;
312         memory.extend(mapMem);
313 
314         boolean compactRequired = false;
315         while (currentOffset < mapMem) {
316             int operation = memory.getInt(currentOffset);
317             currentOffset += Integer.BYTES;
318             String tableName = Chars.toString(memory.getStr(currentOffset));
319             currentOffset += Vm.getStorageLength(tableName);
320             String dirName = Chars.toString(memory.getStr(currentOffset));
321             currentOffset += Vm.getStorageLength(dirName);
322             int tableId = memory.getInt(currentOffset);
323             currentOffset += Integer.BYTES;
324             int tableType = memory.getInt(currentOffset);
325             currentOffset += Integer.BYTES;
326 
327             if (operation == OPERATION_REMOVE) {
328                 // remove from registry
329                 final TableToken token = nameTableTokenMap.remove(tableName);
330                 if (token != null) {
331                     reverseTableNameTokenMap.put(dirName, ReverseTableMapItem.ofDropped(token));
332                     compactRequired = true;
333                 }
334             } else {
335                 if (tableIds.contains(tableId)) {
336                     LOG.critical().$("duplicate table id found, table will not be accessible " +
337                             "[dirName=").$(dirName).$(", id=").$(tableId).$(']').$();
338                     continue;
339                 }
340 
341                 final TableToken token = new TableToken(tableName, dirName, tableId, tableType == TableUtils.TABLE_TYPE_WAL);
342                 nameTableTokenMap.put(tableName, token);
343                 if (!Chars.startsWith(token.getDirName(), token.getTableName())) {
344                     // This table is renamed, log system to real table name mapping
345                     LOG.advisory().$("renamed WAL table system name [table=").utf8(tableName).$(", dirName=").utf8(dirName).$();
346                 }
347 
348                 reverseTableNameTokenMap.put(token.getDirName(), ReverseTableMapItem.of(token));
349                 currentOffset += TABLE_NAME_ENTRY_RESERVED_LONGS * Long.BYTES;
350             }
351         }
352 
353         if (isLocked()) {
354             // Check that the table directories exist
355             for (TableToken token : nameTableTokenMap.values()) {
356                 if (TableUtils.exists(ff, path, configuration.getRoot(), token.getDirName()) != TableUtils.TABLE_EXISTS) {
357                     LOG.error().$("table directory directly removed from File System, table will not be available [path=").utf8(path)
358                             .$(", dirName=").utf8(token.getDirName()).
359                             $(", table=").utf8(token.getTableName())
360                             .I$();
361 
362                     nameTableTokenMap.remove(token.getTableName());
363                     reverseTableNameTokenMap.remove(token.getDirName());
364                     compactRequired = true;
365                 }
366             }
367 
368             if (convertedTables != null) {
369                 for (int i = 0, n = convertedTables.size(); i < n; i++) {
370                     final TableToken token = convertedTables.get(i);
371                     if (token.isWal()) {
372                         nameTableTokenMap.put(token.getTableName(), token);
373                         reverseTableNameTokenMap.put(token.getDirName(), ReverseTableMapItem.of(token));
374                     } else {
375                         nameTableTokenMap.remove(token.getTableName());
376                         reverseTableNameTokenMap.remove(token.getDirName());
377                     }
378                     compactRequired = true;
379                 }
380             }
381 
382             if (compactRequired) {
383                 path.trimTo(pathRootLen);
384                 LOG.info().$("compacting tables file").$();
385                 compactTableNameFile(nameTableTokenMap, lastFileVersion, ff, path, currentOffset);
386             } else {
387                 tableNameMemory.jumpTo(currentOffset);
388             }
389         } else {
390             tableNameRoMemory.close();
391         }
392     }

394     private void writeEntry(TableToken tableToken, int operation) {
395         if (!isLocked()) {
396             throw CairoException.critical(0).put("table registry is not locked");
397         }
398         tableNameMemory.putInt(operation);
399         tableNameMemory.putStr(tableToken.getTableName());
400         tableNameMemory.putStr(tableToken.getDirName());
401         tableNameMemory.putInt(tableToken.getTableId());
402         tableNameMemory.putInt(tableToken.isWal() ? TableUtils.TABLE_TYPE_WAL : TableUtils.TABLE_TYPE_NON_WAL);
403 
404         if (operation != OPERATION_REMOVE) {
405             for (int i = 0; i < TABLE_NAME_ENTRY_RESERVED_LONGS; i++) {
406                 tableNameMemory.putLong(0);
407             }
408         }
409         tableNameMemory.putLong(0L, tableNameMemory.getAppendOffset());
410     }
