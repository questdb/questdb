373     public void reconcileOpenPartitionsFrom(int partitionIndex, boolean forceTruncate) {
374         int txPartitionCount = txFile.getPartitionCount();
375         int txPartitionIndex = partitionIndex;
376         boolean changed = false;
377 
378         while (partitionIndex < partitionCount && txPartitionIndex < txPartitionCount) {
379             final int offset = partitionIndex * PARTITIONS_SLOT_SIZE;
380             final long txPartTs = txFile.getPartitionTimestamp(txPartitionIndex);
381             final long openPartitionTimestamp = openPartitionInfo.getQuick(offset);
382 
383             if (openPartitionTimestamp < txPartTs) {
384                 // Deleted partitions
385                 // This will decrement partitionCount
386                 deletePartition(partitionIndex);
387             } else if (openPartitionTimestamp > txPartTs) {
388                 // Insert partition
389                 insertPartition(partitionIndex, txPartTs);
390                 changed = true;
391                 txPartitionIndex++;
392                 partitionIndex++;
393             } else {
394                 // Refresh partition
395                 final long newPartitionSize = txFile.getPartitionSize(txPartitionIndex);
396                 final long txPartitionNameTxn = txFile.getPartitionNameTxn(partitionIndex);
397                 final long txPartitionColumnVersion = txFile.getPartitionColumnVersion(partitionIndex);
398                 final long openPartitionSize = openPartitionInfo.getQuick(offset + PARTITIONS_SLOT_OFFSET_SIZE);
399                 final long openPartitionNameTxn = openPartitionInfo.getQuick(offset + PARTITIONS_SLOT_OFFSET_NAME_TXN);
400                 final long openPartitionColumnVersion = openPartitionInfo.getQuick(offset + PARTITIONS_SLOT_OFFSET_COLUMN_VERSION);
401 
402                 if (!forceTruncate) {
403                     if (openPartitionNameTxn == txPartitionNameTxn && openPartitionColumnVersion == txPartitionColumnVersion) {
404                         if (openPartitionSize != newPartitionSize) {
405                             if (openPartitionSize > -1L) {
406                                 reloadPartition(partitionIndex, newPartitionSize, txPartitionNameTxn);
407                                 this.openPartitionInfo.setQuick(offset + PARTITIONS_SLOT_OFFSET_SIZE, newPartitionSize);
408                                 LOG.debug().$("updated partition size [partition=").$(openPartitionTimestamp).I$();
409                             }
410                             changed = true;
411                         }
412                     } else {
413                         reOpenPartition(offset, partitionIndex, txPartitionNameTxn);
414                         changed = true;
415                     }
416                 } else if (openPartitionSize > -1L && newPartitionSize > -1L) { // Don't force re-open if not yet opened
417                     reOpenPartition(offset, partitionIndex, txPartitionNameTxn);
418                 }
419                 txPartitionIndex++;
420                 partitionIndex++;
421             }
422         }
423 
424         // if while finished on txPartitionIndex == txPartitionCount condition
425         // remove deleted opened partitions
426         while (partitionIndex < partitionCount) {
427             deletePartition(partitionIndex);
428             changed = true;
429         }
430 
431         // if while finished on partitionIndex == partitionCount condition
432         // insert new partitions at the end
433         for (; partitionIndex < txPartitionCount; partitionIndex++) {
434             insertPartition(partitionIndex, txFile.getPartitionTimestamp(partitionIndex));
435             changed = true;
436         }
437 
438         if (forceTruncate) {
439             reloadAllSymbols();
440         } else if (changed) {
441             reloadSymbolMapCounts();
442         }
443     }

445     public boolean reload() {
446         if (acquireTxn()) {
447             return false;
448         }
449         try {
450             reloadSlow(true);
451             // partition reload will apply truncate if necessary
452             // applyTruncate for non-partitioned tables only
453             reconcileOpenPartitions(txPartitionVersion, txColumnVersion, txTruncateVersion);
454 
455             // Save transaction details which impact the reloading. Do not rely on txReader, it can be reloaded outside this method.
456             txPartitionVersion = this.txFile.getPartitionTableVersion();
457             txColumnVersion = this.txFile.getColumnVersion();
458             txTruncateVersion = this.txFile.getTruncateVersion();
459 
460             return true;
461         } catch (Throwable e) {
462             releaseTxn();
463             throw e;
464         }
465     }

492     private boolean acquireTxn() {
493         if (!txnAcquired) {
494             if (txnScoreboard.acquireTxn(txn)) {
495                 txnAcquired = true;
496             } else {
497                 return false;
498             }
499         }
500 
501         // txFile can also be reloaded in goPassive->checkSchedulePurgeO3Partitions
502         // if txFile txn doesn't much reader txn, reader has to be slow reloaded
503         if (txn == txFile.getTxn()) {
504             // We have to be sure last txn is acquired in Scoreboard
505             // otherwise writer can delete partition version files
506             // between reading txn file and acquiring txn in the Scoreboard.
507             Unsafe.getUnsafe().loadFence();
508             return txFile.getVersion() == txFile.unsafeReadVersion();
509         }
510         return false;
511     }

513     private void checkSchedulePurgeO3Partitions() {
514         long txnLocks = txnScoreboard.getActiveReaderCount(txn);
515         long partitionTableVersion = txFile.getPartitionTableVersion();
516         if (txnLocks == 0 && txFile.unsafeLoadAll() && txFile.getPartitionTableVersion() > partitionTableVersion) {
517             // Last lock for this txn is released and this is not latest txn number
518             // Schedule a job to clean up partition versions this reader may hold
519             if (TableUtils.schedulePurgeO3Partitions(messageBus, tableToken, partitionBy)) {
520                 return;
521             }
522 
523             LOG.error()
524                     .$("could not queue purge partition task, queue is full [")
525                     .$("dirName=").utf8(this.tableToken.getDirName())
526                     .$(", txn=").$(txn)
527                     .$(']').$();
528         }

632         LOG.debug().$("resizing columns file list [table=").utf8(tableToken.getTableName()).I$();
633         int capacity = partitionCount << columnCountShl;
634         final ObjList<MemoryMR> toColumns = new ObjList<>(capacity + 2);

641         toIndexReaders.setPos(capacity + 2);
642         final long pIndexBase = pTransitionIndex + 8;

644

647             final int fromBase = partitionIndex << this.columnCountShl;

918     private void reconcileOpenPartitions(long prevPartitionVersion, long prevColumnVersion, long prevTruncateVersion) {
919         // Reconcile partition full or partial will only update row count of last partition and append new partitions
920         boolean truncateHappened = this.txFile.getTruncateVersion() != prevTruncateVersion;
921         if (txFile.getPartitionTableVersion() == prevPartitionVersion && txFile.getColumnVersion() == prevColumnVersion && !truncateHappened) {
922             int partitionIndex = Math.max(0, partitionCount - 1);
923             final int txPartitionCount = txFile.getPartitionCount();
924             if (partitionIndex < txPartitionCount) {
925                 if (partitionIndex < partitionCount) {
926                     final int offset = partitionIndex * PARTITIONS_SLOT_SIZE;
927                     final long openPartitionSize = openPartitionInfo.getQuick(offset + PARTITIONS_SLOT_OFFSET_SIZE);
928                     // we check that open partition size is non-negative to avoid loading
929                     // partition that is not yet in memory
930                     if (openPartitionSize > -1L) {
931                         final long openPartitionNameTxn = openPartitionInfo.getQuick(offset + PARTITIONS_SLOT_OFFSET_NAME_TXN);
932                         final long txPartitionSize = txFile.getPartitionSize(partitionIndex);
933                         final long txPartitionNameTxn = txFile.getPartitionNameTxn(partitionIndex);
934 
935                         if (openPartitionNameTxn == txPartitionNameTxn) {
936                             if (openPartitionSize != txPartitionSize) {
937                                 reloadPartition(partitionIndex, txPartitionSize, txPartitionNameTxn);
938                                 this.openPartitionInfo.setQuick(offset + PARTITIONS_SLOT_OFFSET_SIZE, txPartitionSize);
939                                 LOG.debug().$("updated partition size [partition=").$(openPartitionInfo.getQuick(offset)).I$();
940                             }
941                         } else {
942                             openPartition0(partitionIndex);
943                         }
944                     }
945                     partitionIndex++;
946                 }
947                 for (; partitionIndex < txPartitionCount; partitionIndex++) {
948                     insertPartition(partitionIndex, txFile.getPartitionTimestamp(partitionIndex));
949                 }
950                 reloadSymbolMapCounts();
951             }
952             return;
953         }
954         reconcileOpenPartitionsFrom(0, truncateHappened);

981     private void reloadColumnAt(
982             int partitionIndex,
983             Path path,
984             ObjList<MemoryMR> columns,
985             LongList columnTops,
986             ObjList<BitmapIndexReader> indexReaders,
987             int columnBase,
988             int columnIndex,
989             long partitionRowCount
990     ) {
991         final int plen = path.length();
992         try {
993             final CharSequence name = metadata.getColumnName(columnIndex);
994             final int primaryIndex = getPrimaryColumnIndex(columnBase, columnIndex);
995             final int secondaryIndex = primaryIndex + 1;
996 
997             MemoryMR mem1 = columns.getQuick(primaryIndex);
998             MemoryMR mem2 = columns.getQuick(secondaryIndex);
999 
1000             final long partitionTimestamp = openPartitionInfo.getQuick(partitionIndex * PARTITIONS_SLOT_SIZE);
1001             int writerIndex = metadata.getWriterIndex(columnIndex);
1002             final int versionRecordIndex = columnVersionReader.getRecordIndex(partitionTimestamp, writerIndex);
1003             final long columnTop = versionRecordIndex > -1L ? columnVersionReader.getColumnTopByIndex(versionRecordIndex) : 0L;
1004             long columnTxn = versionRecordIndex > -1L ? columnVersionReader.getColumnNameTxnByIndex(versionRecordIndex) : -1L;
1005             if (columnTxn == -1L) {
1006                 // When column is added, column version will have txn number for the partition
1007                 // where it's added. It will also have the txn number in the [default] partition
1008                 columnTxn = columnVersionReader.getDefaultColumnNameTxn(writerIndex);
1009             }
1010             final long columnRowCount = partitionRowCount - columnTop;
1011             assert partitionRowCount < 0 || columnRowCount >= 0;
1012 
1013             // When column is added mid-table existence the top record is only
1014             // created in the current partition. Older partitions would simply have no
1015             // column file. This makes it necessary to check the partition timestamp in Column Version file
1016             // of when the column was added.
1017             if (columnRowCount > 0 && (versionRecordIndex > -1L || columnVersionReader.getColumnTopPartitionTimestamp(writerIndex) <= partitionTimestamp)) {
1018                 final int columnType = metadata.getColumnType(columnIndex);
1019 
1020                 if (ColumnType.isVariableLength(columnType)) {
1021                     long columnSize = columnRowCount * 8L + 8L;
1022                     TableUtils.iFile(path.trimTo(plen), name, columnTxn);
1023                     mem2 = openOrCreateMemory(path, columns, secondaryIndex, mem2, columnSize);
1024                     columnSize = mem2.getLong(columnRowCount * 8L);
1025                     TableUtils.dFile(path.trimTo(plen), name, columnTxn);
1026                     openOrCreateMemory(path, columns, primaryIndex, mem1, columnSize);
1027                 } else {
1028                     long columnSize = columnRowCount << ColumnType.pow2SizeOf(columnType);
1029                     TableUtils.dFile(path.trimTo(plen), name, columnTxn);
1030                     openOrCreateMemory(path, columns, primaryIndex, mem1, columnSize);
1031                     Misc.free(columns.getAndSetQuick(secondaryIndex, null));
1032                 }
1033 
1034                 columnTops.setQuick(columnBase / 2 + columnIndex, columnTop);
1035 
1036                 if (metadata.isColumnIndexed(columnIndex)) {
1037                     BitmapIndexReader indexReader = indexReaders.getQuick(primaryIndex);
1038                     if (indexReader != null) {
1039                         // name txn is -1 because the parent call sets up partition name for us
1040                         indexReader.of(configuration, path.trimTo(plen), name, columnTxn, columnTop, -1);
1041                     }
1042                 } else {
1043                     Misc.free(indexReaders.getAndSetQuick(primaryIndex, null));
1044                     Misc.free(indexReaders.getAndSetQuick(secondaryIndex, null));
1045                 }
1046             } else {
1047                 Misc.free(columns.getAndSetQuick(primaryIndex, NullMemoryMR.INSTANCE));
1048                 Misc.free(columns.getAndSetQuick(secondaryIndex, NullMemoryMR.INSTANCE));
1049                 // the appropriate index for NUllColumn will be created lazily when requested
1050                 // these indexes have state and may not be always required
1051                 Misc.free(indexReaders.getAndSetQuick(primaryIndex, null));
1052                 Misc.free(indexReaders.getAndSetQuick(secondaryIndex, null));
1053 
1054                 // Column is not present in the partition. Set column top to be the size of the partition.
1055                 columnTops.setQuick(columnBase / 2 + columnIndex, partitionRowCount);
1056             }
1057         } finally {
1058             path.trimTo(plen);
1059         }
1060     }

1069     private boolean reloadMetadata(long txnStructureVersion, long deadline, boolean reshuffleColumns) {
1070         // create transition index, which will help us reuse already open resources
1071         if (txnStructureVersion == metadata.getStructureVersion()) {
1072             return true;
1073         }
1074 
1075         while (true) {
1076             long pTransitionIndex;
1077             try {
1078                 pTransitionIndex = metadata.createTransitionIndex(txnStructureVersion);
1079                 if (pTransitionIndex < 0) {
1080                     if (clock.getTicks() < deadline) {
1081                         return false;
1082                     }
1083                     LOG.error().$("metadata read timeout [timeout=").$(configuration.getSpinLockTimeout()).utf8("ms, table=").$(tableToken.getTableName()).I$();
1084                     throw CairoException.critical(0).put("Metadata read timeout [table=").put(tableToken.getTableName()).put(']');
1085                 }
1086             } catch (CairoException ex) {
1087                 // This is temporary solution until we can get multiple version of metadata not overwriting each other
1088                 TableUtils.handleMetadataLoadException(tableToken.getTableName(), deadline, ex, configuration.getMillisecondClock(), configuration.getSpinLockTimeout());
1089                 continue;
1090             }
1091 
1092             try {
1093                 metadata.applyTransitionIndex();
1094                 if (reshuffleColumns) {
1095                     final int columnCount = metadata.getColumnCount();
1096 
1097                     int columnCountShl = getColumnBits(columnCount);
1098                     // when a column is added we cannot easily reshuffle columns in-place
1099                     // the reason is that we'd have to create gaps in columns list between
1100                     // partitions. It is possible in theory, but this could be an algo for
1101                     // another day.
1102                     if (columnCountShl > this.columnCountShl) {
1103                         createNewColumnList(columnCount, pTransitionIndex, columnCountShl);
1104                     } else {
1105                         reshuffleColumns(columnCount, pTransitionIndex);
1106                     }

1108                     reshuffleSymbolMapReaders(pTransitionIndex, columnCount);
1109                     this.columnCount = columnCount;
1110                     reloadSymbolMapCounts();
