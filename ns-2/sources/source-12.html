71     /**
72      * Searches ordered list of long values. Return value is either index behind matching value in the list or
73      * index of where values would be inserted in order to maintain ascending order of the list. When list
74      * contains duplicate values the index would be behind group of duplicate values. In this example list:
75      * <p>
76      * 1,1,2,2,2,2,5,5,5
77      * <p>
78      * when we search for 2, the index would be here:
79      * <p>
80      * 1,1,2,2,2,2,5,5,5
81      * ^
82      * <p>
83      * Same index will be returned when we search for value of 3.
84      * <p>
85      * This method is meant to search for the position of a rowid (value), 100% guaranteed to exist, otherwise
86      * it means the index is corrupt.
87      *
88      * @param memory    virtual memory instance
89      * @param offset    offset in virtual memory
90      * @param cellCount length of the available memory measured in 64-bit cells
91      * @param value     value we search of
92      * @return index directly behind the searched value or group of values if list contains duplicate values
93      * @throws CairoException when the index is corrupted
94      */
95     static long searchValueBlock(MemoryR memory, long offset, long cellCount, long value) {
96         // when block is "small", we just scan it linearly
97         if (cellCount < 64) {
98             // this will definitely exit because we had checked that at least the last value is greater than value
99             for (long i = offset, limit = memory.size(); i < limit; i += 8) {
100                 if (memory.getLong(i) > value) {
101                     return (i - offset) / 8;
102                 }
103             }
104         } else {
105             // use binary search on larger block
106             long low = 0;
107             long high = cellCount - 1;
108             long half;
109             long pivot;
110             do {
111                 half = (high - low) / 2;
112                 if (half == 0) {
113                     break;
114                 }
115                 pivot = memory.getLong(offset + (low + half) * 8);
116                 if (pivot <= value) {
117                     low += half;
118                 } else {
119                     high = low + half;
120                 }
121             } while (true);
122 
123             return low + 1;
124         }
125         throw CairoException.critical(0)
126                 .put("index is corrupt, rowid not found [offset=").put(offset)
127                 .put(", cellCount=").put(cellCount)
128                 .put(", value=").put(value)
129                 .put(']');
130     }

132     /**
133      * Seeks first block that contains first value, which is greater or equal to minValue.
134      * It starts from first block in the list and proceeds moving right until it finds
135      * block with low and high values surrounding minValue.
136      * <p>
137      * The found block is then binary searched to count values that are all greater or
138      * equal than minValue.
139      * <p>
140      * Unlike seekValueBlockRTL, this method does a boundary check over the valueMem size.
141      *
142      * @param totalCount            total count of values in all blocks;
143      *                              *important note*: initialCount may include blocks that exceed the file boundary
144      * @param firstValueBlockOffset offset of first block in linked list
145      * @param valueMem              value block memory
146      * @param minValue              lower limit for values
147      * @param blockValueCountMod    number of values in single block - 1
148      * @param seeker                interface that collects results of the search
149      */
150     static void seekValueBlockLTR(
151             long totalCount,
152             long firstValueBlockOffset,
153             MemoryR valueMem,
154             long minValue,
155             long blockValueCountMod,
156             ValueBlockSeeker seeker
157     ) {
158         long valueCount = totalCount;
159         long valueBlockOffset = firstValueBlockOffset;
160         if (firstValueBlockOffset >= valueMem.size()) {
161             // The very first block is beyond the memory boundary. Report that we didn't find the value by
162             // using the total count of values in the seeker call.
163             seeker.seek(totalCount, firstValueBlockOffset);
164             return;
165         }
166         if (valueCount > 0) {
167             long cellCount;
168             do {
169                 // check block range by peeking at first and last value
170                 if (valueCount > blockValueCountMod) {
171                     cellCount = blockValueCountMod + 1;
172                 } else {
173                     cellCount = valueCount;
174                 }
175                 final long hi = valueMem.getLong(valueBlockOffset + (cellCount - 1) * 8);
176 
177                 // can we skip this block?
178                 if (hi < minValue) {
179                     valueCount -= cellCount;
180                     // do we have next block?
181                     if (valueCount > 0) {
182                         final long nextBlockOffset = (blockValueCountMod + 1) * 8 + 8;
183                         valueBlockOffset = valueMem.getLong(valueBlockOffset + nextBlockOffset);
184                         if (valueBlockOffset >= valueMem.size()) {
185                             // We've reached the memory boundary. Report that we didn't find the value by
186                             // using the total count of values in the seeker call.
187                             seeker.seek(totalCount, valueBlockOffset);
188                             return;
189                         }
190                         continue;
191                     }
192                 }
193                 break;
194             } while (true);
195 
196             if (valueCount > 0) {
197                 // do we need to search this block?
198                 final long lo = valueMem.getLong(valueBlockOffset);
199                 if (minValue > lo) {
200                     // yes, we do
201                     valueCount -= searchValueBlock(valueMem, valueBlockOffset, cellCount, minValue - 1);
202                 }
203             }
204         }
205         seeker.seek(totalCount - valueCount, valueBlockOffset);
206     }

209      * Seeks index value block for rows <= maxValue. It starts from append block, which is the last block in
210      * the list, and proceeds moving left until it finds block with low and high values surrounding maxValue.
211      * This block is then binary searched to count values that are less or equal than maxValue.
212      * <p>
213      * List of value blocks is assumed to be ordered by value in ascending order.
214      *
215      * @param valueCount         total count of values in all blocks
216      * @param blockOffset        offset of last value block in chain of blocks
217      * @param valueMem           value block memory
218      * @param maxValue           upper limit for block values
219      * @param blockValueCountMod number of values in single block - 1
220      * @param seeker             interface that collects results of the search
221      */
222     static void seekValueBlockRTL(
223             long valueCount,
224             long blockOffset,
225             MemoryR valueMem,
226             long maxValue,
227             long blockValueCountMod,
228             ValueBlockSeeker seeker
229     ) {
230         long valueBlockOffset = blockOffset;
231         if (valueCount > 0) {
232             long prevBlockOffset = (blockValueCountMod + 1) * 8;
233             long cellCount;
234             do {
235                 // check block range by peeking at first and last value
236                 long lo = valueMem.getLong(valueBlockOffset);
237                 cellCount = (valueCount - 1 & blockValueCountMod) + 1;
238 
239                 // can we skip this block?
240                 if (lo > maxValue) {
241                     valueCount -= cellCount;
242                     // do we have previous block?
243                     if (valueCount > 0) {
244                         valueBlockOffset = valueMem.getLong(valueBlockOffset + prevBlockOffset);
245                         continue;
246                     }
247                 }
248                 break;
249             } while (true);
250 
251             if (valueCount > 0) {
252                 // do we need to search this block?
253                 long hi = valueMem.getLong(valueBlockOffset + (cellCount - 1) * 8);
254                 if (maxValue < hi) {
255                     // yes, we do
256                     valueCount -= cellCount - searchValueBlock(valueMem, valueBlockOffset, cellCount, maxValue);
257                 }
258             }
259         }
260         seeker.seek(valueCount, valueBlockOffset);
261     }
