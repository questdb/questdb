39      * Data frame low and high row will be within intervals inclusive of edges. Intervals

44      * @param timestampIndex index of timestamp column in the readr that is used by this cursor

50     @Override
51     public DataFrame next() {
52         // order of logical operations is important
53         // we are not calculating partition rages when intervals are empty
54         while (intervalsLo < intervalsHi && partitionLo < partitionHi) {
55             // We don't need to worry about column tops and null column because we
56             // are working with timestamp. Timestamp column cannot be added to existing table.
57             final int currentInterval = intervalsHi - 1;
58             final int currentPartition = partitionHi - 1;
59             long rowCount = reader.openPartition(currentPartition);
60             if (rowCount > 0) {
61 
62                 final MemoryR column = reader.getColumn(TableReader.getPrimaryColumnIndex(reader.getColumnBase(currentPartition), timestampIndex));
63                 final long intervalLo = intervals.getQuick(currentInterval * 2);
64                 final long intervalHi = intervals.getQuick(currentInterval * 2 + 1);
65 
66                 final long limitHi;
67                 if (partitionLimit == -1) {
68                     limitHi = rowCount - 1;
69                 } else {
70                     limitHi = partitionLimit - 1;
71                 }
72 
73                 final long partitionTimestampLo = column.getLong(0);
74 
75                 LOG.debug()
76                         .$("next [partition=").$(currentPartition)
77                         .$(", intervalLo=").microTime(intervalLo)
78                         .$(", intervalHi=").microTime(intervalHi)
79                         .$(", partitionLo=").microTime(partitionTimestampLo)
80                         .$(", limitHi=").$(limitHi)
81                         .$(", rowCount=").$(rowCount)
82                         .$(", currentInterval=").$(currentInterval)
83                         .$(']').$();
84 
85                 // interval is wholly above partition, skip partition
86                 if (partitionTimestampLo > intervalHi) {
87                     skipPartition(currentPartition);
88                     continue;
89                 }
90 
91                 // interval is wholly below partition, skip interval
92                 final long partitionTimestampHi = column.getLong(limitHi * Long.BYTES);
93                 if (partitionTimestampHi < intervalLo) {
94                     skipInterval(currentInterval, limitHi + 1);
95                     continue;
96                 }
97 
98                 // calculate intersection for inclusive intervals "intervalLo" and "intervalHi"
99                 final long lo;
100                 if (partitionTimestampLo < intervalLo) {
101                     lo = BinarySearch.find(column, intervalLo - 1, 0, limitHi, BinarySearch.SCAN_DOWN) + 1;
102                 } else {
103                     lo = 0;
104                 }
105 
106                 final long hi;
107                 if (partitionTimestampHi > intervalHi) {
108                     hi = BinarySearch.find(column, intervalHi, lo, limitHi, BinarySearch.SCAN_DOWN) + 1;
109                 } else {
110                     hi = limitHi + 1;
111                 }
112 
113                 if (lo == 0) {
114                     // interval yielded empty data frame, skip partition
115                     skipPartition(currentPartition);
116                 } else {
117                     // only fragment, need to skip to next interval
118                     skipInterval(currentInterval, lo);
119                 }
120 
121                 if (lo < hi) {
122                     dataFrame.partitionIndex = currentPartition;
123                     dataFrame.rowLo = lo;
124                     dataFrame.rowHi = hi;
125                     sizeSoFar += hi - lo;
126                     return dataFrame;
127                 }
128             } else {
129                 // partition was empty, just skip to next
130                 partitionLimit = -1;
131                 partitionHi = currentPartition;
132             }
133         }
134         return null;
135     }
