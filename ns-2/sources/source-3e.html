39     public static void runCallbackWithCol(O3CallbackTask task, long cursor, @Nullable Sequence subSeq) {
40         final int columnIndex = task.getColumnIndex();
41         final int columnType = task.getColumnType();
42         final long mergedTimestampsAddr = task.getMergedTimestampsAddr();
43         final long row1Count = task.getRow1Count();
44         final long row2Lo = task.getRow2Lo();
45         final long row2Hi = task.getRow2Hi();
46         final TableWriter.O3ColumnUpdateMethod callbackMethod = task.getWriterCallbackMethod();
47         final CountDownLatchSPI countDownLatchSPI = task.getCountDownLatchSPI();
48         if (subSeq != null) {
49             subSeq.done(cursor);
50         }
51 
52         try {
53             callbackMethod.run(
54                     columnIndex,
55                     columnType,
56                     mergedTimestampsAddr,
57                     row1Count,
58                     row2Lo,
59                     row2Hi
60             );
61         } finally {
62             countDownLatchSPI.countDown();
63         }
64     }
65 
66     @Override
67     protected boolean doRun(int workerId, long cursor, RunStatus runStatus) {
68         O3CallbackTask task = queue.get(cursor);
69         // copy task on stack so that publisher has fighting chance of
70         // publishing all it has to the queue
71 
72         final boolean locked = task.tryLock();
73         if (locked) {
74             runCallbackWithCol(task, cursor, subSeq);
75         } else {
76             subSeq.done(cursor);
77         }
78 
79         return true;
80     }
