136     public void commit(int commitMode, ObjList<? extends SymbolCountProvider> symbolCountProviders) {
137 
138         if (prevRecordStructureVersion == recordStructureVersion && prevRecordBaseOffset > 0) {
139 
140             // Optimisation for the case where commit appends rows to the last partition only
141             // In this case all to be changed is TX_OFFSET_MAX_TIMESTAMP_64 and TX_OFFSET_TRANSIENT_ROW_COUNT_64
142             writeBaseOffset = prevRecordBaseOffset;
143             putLong(TX_OFFSET_TXN_64, ++txn);
144             putLong(TX_OFFSET_SEQ_TXN_64, seqTxn);
145             putLong(TX_OFFSET_MAX_TIMESTAMP_64, maxTimestamp);
146             putLong(TX_OFFSET_TRANSIENT_ROW_COUNT_64, transientRowCount);
147 
148             // Store symbol counts. Unfortunately we cannot skip it in here
149             storeSymbolCounts(symbolCountProviders);

152             txMemBase.putLong(TX_BASE_OFFSET_VERSION_64, ++baseVersion);
153

155             this.readBaseOffset = writeBaseOffset;
156 
157             prevTransientRowCount = transientRowCount;
158             prevMinTimestamp = minTimestamp;
159             prevMaxTimestamp = maxTimestamp;
160 
161             prevRecordBaseOffset = lastRecordBaseOffset;
162             lastRecordBaseOffset = writeBaseOffset;
163         } else {
164             // Slow path, record structure changed

409     private int calculateWriteOffset(int areaSize) {
410         boolean currentIsA = (baseVersion & 1L) == 0L;
411         int currentOffset = currentIsA ? txMemBase.getInt(TX_BASE_OFFSET_A_32) : txMemBase.getInt(TX_BASE_OFFSET_B_32);
412         if (TX_BASE_HEADER_SIZE + areaSize <= currentOffset) {
413             return TX_BASE_HEADER_SIZE;
414         }
415         int currentSizeSymbols = currentIsA ? txMemBase.getInt(TX_BASE_OFFSET_SYMBOLS_SIZE_A_32) : txMemBase.getInt(TX_BASE_OFFSET_SYMBOLS_SIZE_B_32);
416         int currentSizePartitions = currentIsA ? txMemBase.getInt(TX_BASE_OFFSET_PARTITIONS_SIZE_A_32) : txMemBase.getInt(TX_BASE_OFFSET_PARTITIONS_SIZE_B_32);
417         int currentSize = calculateTxRecordSize(currentSizeSymbols, currentSizePartitions);
418         return currentOffset + currentSize;
419     }

430         symbolColumnCount = symbolCountProviders.size();
431 
432         writeAreaSize = calculateWriteSize();
433         writeBaseOffset = calculateWriteOffset(writeAreaSize);
434         putLong(TX_OFFSET_TXN_64, ++txn);
435         putLong(TX_OFFSET_TRANSIENT_ROW_COUNT_64, transientRowCount);
436         putLong(TX_OFFSET_FIXED_ROW_COUNT_64, fixedRowCount);
437         putLong(TX_OFFSET_MIN_TIMESTAMP_64, minTimestamp);
438         putLong(TX_OFFSET_MAX_TIMESTAMP_64, maxTimestamp);
439         putLong(TX_OFFSET_STRUCT_VERSION_64, structureVersion.get());
440         putLong(TX_OFFSET_DATA_VERSION_64, dataVersion);

465     private void finishABHeader(int areaOffset, int bytesSymbols, int bytesPartitions, int commitMode) {
466         boolean currentIsA = (baseVersion & 1L) == 0L;
467 
468         // When current is A, write to B
469         long offsetOffset = currentIsA ? TX_BASE_OFFSET_B_32 : TX_BASE_OFFSET_A_32;
470         long symbolSizeOffset = currentIsA ? TX_BASE_OFFSET_SYMBOLS_SIZE_B_32 : TX_BASE_OFFSET_SYMBOLS_SIZE_A_32;
471         long partitionsSizeOffset = currentIsA ? TX_BASE_OFFSET_PARTITIONS_SIZE_B_32 : TX_BASE_OFFSET_PARTITIONS_SIZE_A_32;
472 
473         txMemBase.putInt(offsetOffset, areaOffset);
474         txMemBase.putInt(symbolSizeOffset, bytesSymbols);
475         txMemBase.putInt(partitionsSizeOffset, bytesPartitions);
476 
477         Unsafe.getUnsafe().storeFence();
478         txMemBase.putLong(TX_BASE_OFFSET_VERSION_64, ++baseVersion);
479 
480         readRecordSize = calculateTxRecordSize(bytesSymbols, bytesPartitions);
481         readBaseOffset = areaOffset;
482         super.switchRecord(readBaseOffset, readRecordSize);
483 
484         if (commitMode != CommitMode.NOSYNC) {
485             txMemBase.sync(commitMode == CommitMode.ASYNC);
486         }
487     }

585     // It is possible that O3 commit will create partition just before
586     // the last one, leaving last partition row count 0 when doing ic().
587     // That's when the data from the last partition is moved to in-memory lag.
588     // One way to detect this is to check if index of the "last" partition is not
589     // last partition in the attached partition list.
590     boolean reconcileOptimisticPartitions() {
591         int lastPartitionTsIndex = attachedPartitions.size() - LONGS_PER_TX_ATTACHED_PARTITION + PARTITION_TS_OFFSET;
592         if (lastPartitionTsIndex > 0 && maxTimestamp < attachedPartitions.getQuick(lastPartitionTsIndex)) {
593             int maxTimestampPartitionIndex = getPartitionIndex(getLastPartitionTimestamp());
594             if (maxTimestampPartitionIndex < getPartitionCount() - 1) {
595                 // accumulate value, which we have to subtract
596                 // from fixedRowCount (total count of rows of non-active partitions)
597                 long rowCount = 0;
598                 for (int i = maxTimestampPartitionIndex, n = getPartitionCount() - 1; i < n; i++) {
599                     rowCount += getPartitionSize(i);
600                 }
601                 attachedPartitions.setPos((maxTimestampPartitionIndex + 1) * LONGS_PER_TX_ATTACHED_PARTITION);
602                 recordStructureVersion++;
