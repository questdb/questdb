75         executionContext.getCairoSecurityContext().checkWritePermission();
76         LOG.info().$("processing [dirName=").utf8(reader.getTableToken().getDirName()).I$();
77         fileNameSink = new StringSink();

85         path2 = Path.getThreadLocal2(configuration.getRoot()).concat(tableToken);

88         partitionBy = reader.getPartitionedBy();

93             Vect.sort3LongAscInPlace(tableFiles.getAddress(), tableFiles.size() / 3);

104     private void purgeColumnVersions(DirectLongList tableFiles, TableReader reader, CairoEngine engine) {
105         int columnIndex = -1;
106         int writerIndex = -1;
107         int tableId = reader.getMetadata().getTableId();
108         long truncateVersion = reader.getTxFile().getTruncateVersion();
109         TableReaderMetadata metadata = reader.getMetadata();
110         long updateTxn = reader.getTxn();
111         ColumnVersionReader columnVersionReader = reader.getColumnVersionReader();
112         purgeTask.clear();
113 
114         for (long i = 0, n = tableFiles.size(); i < n; i += 3) {
115             if (tableFiles.get(i) != columnIndex) {
116                 int newReaderIndex = (int) tableFiles.get(i);
117                 if (columnIndex != newReaderIndex) {
118                     if (columnIndex != -1 && purgeTask.getUpdatedColumnInfo().size() > 0) {
119                         if (!purgeExecution.purge(purgeTask, tableReader)) {
120                             queueColumnVersionPurge(purgeTask, engine);
121                         }
122                         purgeTask.clear();
123                     }
124 
125                     writerIndex = metadata.getWriterIndex(newReaderIndex);
126                     CharSequence columnName = metadata.getColumnName(newReaderIndex);
127                     int columnType = metadata.getColumnType(newReaderIndex);
128                     purgeTask.of(reader.getTableToken(), columnName, tableId, truncateVersion, columnType, partitionBy, updateTxn);
129 
130                 }
131             }
132 
133             columnIndex = (int) tableFiles.get(i);
134             long partitionTs = tableFiles.get(i + 1);
135             long columnVersion = tableFiles.get(i + 2);
136             long latestColumnNameTxn = columnVersionReader.getColumnNameTxn(partitionTs, writerIndex);
137             if (columnVersion != latestColumnNameTxn) {
138                 // Has to be deleted. Columns can have multiple files e.g. .i, .d, .k, .v
139                 if (!versionSetToDelete(purgeTask, partitionTs, columnVersion)) {
140                     long partitionNameTxn = reader.getTxFile().getPartitionNameTxnByPartitionTimestamp(partitionTs);
141                     purgeTask.appendColumnInfo(columnVersion, partitionTs, partitionNameTxn);
142                 }
143             }
144         }
145 
146         if (purgeTask.getUpdatedColumnInfo().size() > 0) {
147             if (!purgeExecution.purge(purgeTask, tableReader)) {
148                 queueColumnVersionPurge(purgeTask, engine);
149             }
150             purgeTask.clear();
151         }
152     }

154     private void queueColumnVersionPurge(ColumnPurgeTask purgeTask, CairoEngine engine) {
155         MessageBus messageBus = engine.getMessageBus();
156         LOG.info().$("scheduling column version purge [table=").$(purgeTask.getTableName())
157                 .$(", column=").$(purgeTask.getColumnName())
158                 .I$();
159 
160         Sequence pubSeq = messageBus.getColumnPurgePubSeq();
161         while (true) {
162             long cursor = pubSeq.next();
163             if (cursor > -1) {
164                 ColumnPurgeTask task = messageBus.getColumnPurgeQueue().get(cursor);
165                 task.copyFrom(purgeTask);
166                 pubSeq.done(cursor);
167                 return;
168             } else if (cursor == -1) {
169                 // Queue overflow
170                 throw CairoException.nonCritical().put("failed to schedule column version purge, queue is full. " +
171                                 "Please retry and consider increasing ").put(PropertyKey.CAIRO_SQL_COLUMN_PURGE_QUEUE_CAPACITY.getPropertyPath())
172                         .put(" configuration parameter");
173             }
174             Os.pause();
175         }
176     }

178     private boolean versionSetToDelete(ColumnPurgeTask purgeTask, long partitionTs, long columnVersion) {
179         // Brute force for now
180         LongList columnVersionToDelete = purgeTask.getUpdatedColumnInfo();
181         for (int i = 0, n = columnVersionToDelete.size(); i < n; i += ColumnPurgeTask.BLOCK_SIZE) {
182             long cv = columnVersionToDelete.getQuick(i + ColumnPurgeTask.OFFSET_COLUMN_VERSION);
183             long ts = columnVersionToDelete.getQuick(i + ColumnPurgeTask.OFFSET_PARTITION_TIMESTAMP);
184 
185             if (cv == columnVersion && ts == partitionTs) {
186                 return true;
187             }
188         }
189         return false;
190     }

192     private void visitTableFiles(long pUtf8NameZ, int type) {
193         if (type != DT_DIR) {
194             fileNameSink.clear();
195             Chars.utf8DecodeZ(pUtf8NameZ, fileNameSink);
196             if (notDots(fileNameSink)) {
197                 int dotIndex = Chars.indexOf(fileNameSink, '.');
198                 if (dotIndex > 0) {
199                     long columnVersion = -1;
200 
201                     CharSequence columnName = fileNameSink.subSequence(0, dotIndex);
202                     int name2Index = resolveName2Index(columnName, tableReader);
203                     if (name2Index < 0) {
204                         // Unknown file. Log the problem
205                         LOG.error().$("file does not belong to the table [name=").$(fileNameSink).$(", path=").$(path2).I$();
206                         return;
207                     }
208 
209                     int secondDot = Chars.indexOf(fileNameSink, dotIndex + 1, '.');
210                     int lo = secondDot + 1;
211                     if (lo < fileNameSink.length()) {
212                         try {
213                             columnVersion = Numbers.parseLong(fileNameSink, lo, fileNameSink.length());
214                         } catch (NumericException e) {
215                             // leave -1, default version
216                         }
217                     }
218 
219                     tableFiles.add(name2Index);
220                     tableFiles.add(partitionTimestamp);
221                     tableFiles.add(columnVersion);
222                 }
223             }
224         }
225     }

227     private void visitTablePartition(long pUtf8NameZ, int type) {
228         if (ff.isDirOrSoftLinkDirNoDots(path2, tablePathLen, pUtf8NameZ, type, fileNameSink)) {
229             path2.trimTo(tablePathLen).$();
230 
231             int dotIndex = Chars.indexOf(fileNameSink, '.');
232             if (dotIndex < 0) {
233                 dotIndex = fileNameSink.length();
234             }
235 
236             try {
237                 partitionTimestamp = getPartitionDirFormatMethod(partitionBy).parse(fileNameSink, 0, dotIndex, null);
238             } catch (NumericException ex) {
239                 // Directory is invalid partition name, continue
240                 LOG.error().$("skipping column version purge VACUUM, invalid partition directory name [name=").$(fileNameSink)
241                         .$(", path=").$(path2).I$();
242                 return;
243             }
244 
245             long partitionNameTxn = -1L;
246             if (dotIndex + 1 < fileNameSink.length()) {
247                 try {
248                     partitionNameTxn = Numbers.parseLong(fileNameSink, dotIndex + 1, fileNameSink.length());
249                 } catch (NumericException ex) {
250                     // Invalid partition name txn
251                     LOG.error().$("skipping column version purge VACUUM, invalid partition directory name [name=").$(fileNameSink)
252                             .$(", path=").$(path2).I$();
253                     return;
254                 }
255             }
256 
257             if (partitionNameTxn != tableReader.getTxFile().getPartitionNameTxnByPartitionTimestamp(partitionTimestamp)) {
258                 // This is partition version to be deleted by O3 partition purge
259                 return;
260             }
261 
262             path2.concat(pUtf8NameZ);
263             LOG.info().$("enumerating files at ").$(path2).$();
264             ff.iterateDir(path2.$(), visitTableFiles);
265         }
266     }
