46     @Override

48         final FilesFacade ff = configuration.getFilesFacade();
49         if (uniqueIdMem != 0) {
50             ff.munmap(uniqueIdMem, uniqueIdMemSize, MemoryTag.MMAP_DEFAULT);
51             uniqueIdMem = 0;
52         }
53         if (ff.close(uniqueIdFd)) {
54             uniqueIdFd = -1;
55         }
56     }

58     public long getNextId() {
59         long next;
60         long x = getCurrentId();
61         do {
62             next = x;
63             x = Os.compareAndSwap(uniqueIdMem, next, next + 1);
64         } while (next != x);
65         return next + 1;
66     }

72     public void open(Path path) {
73         close();
74         if (path == null) {
75             path = Path.getThreadLocal(configuration.getRoot());
76         }
77         final int rootLen = path.length();
78         try {
79             path.concat(uniqueIdFileName).$();
80             final FilesFacade ff = configuration.getFilesFacade();
81             uniqueIdFd = TableUtils.openFileRWOrFail(ff, path, configuration.getWriterFileOpenOpts());
82             uniqueIdMem = TableUtils.mapRW(ff, uniqueIdFd, uniqueIdMemSize, MemoryTag.MMAP_DEFAULT);
83         } catch (Throwable e) {
84             close();
85             throw e;
86         } finally {
87             path.trimTo(rootLen);
88         }
89     }
