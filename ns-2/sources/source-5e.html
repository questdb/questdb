137     private boolean switchPartition0() {
138         while (partitionIndex < partitionLimit) {
139             final long partitionSize = reader.openPartition(partitionIndex);
140             if (partitionSize > 0) {
141                 if (partitionIndex == partitionHi && recordHi > -1) {
142                     maxRecordIndex = recordHi - 1;
143                 } else {
144                     maxRecordIndex = partitionSize - 1;
145                 }

151         }
152         return false;
153     }
