186     public static long setSinkForPartition(CharSink path, int partitionBy, long timestamp, boolean calculatePartitionMax) {
187         int y, m, d;
188         boolean leap;
189         switch (partitionBy) {
190             case DAY:
191                 y = Timestamps.getYear(timestamp);
192                 leap = Timestamps.isLeapYear(y);
193                 m = Timestamps.getMonthOfYear(timestamp, y, leap);
194                 d = Timestamps.getDayOfMonth(timestamp, y, m, leap);
195                 TimestampFormatUtils.appendYear000(path, y);
196                 path.put('-');
197                 TimestampFormatUtils.append0(path, m);
198                 path.put('-');
199                 TimestampFormatUtils.append0(path, d);
200 
201                 if (calculatePartitionMax) {
202                     return Timestamps.yearMicros(y, leap)
203                             + Timestamps.monthOfYearMicros(m, leap)
204                             + (d - 1) * Timestamps.DAY_MICROS + 24 * Timestamps.HOUR_MICROS - 1;
205                 }
206                 return 0;
207             case MONTH:
208                 y = Timestamps.getYear(timestamp);
209                 leap = Timestamps.isLeapYear(y);
210                 m = Timestamps.getMonthOfYear(timestamp, y, leap);
211                 TimestampFormatUtils.appendYear000(path, y);
212                 path.put('-');
213                 TimestampFormatUtils.append0(path, m);
214 
215                 if (calculatePartitionMax) {
216                     return Timestamps.yearMicros(y, leap)
217                             + Timestamps.monthOfYearMicros(m, leap)
218                             + Timestamps.getDaysPerMonth(m, leap) * 24L * Timestamps.HOUR_MICROS - 1;
219                 }
220                 return 0;
221             case YEAR:
222                 y = Timestamps.getYear(timestamp);
223                 leap = Timestamps.isLeapYear(y);
224                 TimestampFormatUtils.appendYear000(path, y);
225                 if (calculatePartitionMax) {
226                     return Timestamps.addYear(Timestamps.yearMicros(y, leap), 1) - 1;
227                 }
228                 return 0;
229             case HOUR:
230                 y = Timestamps.getYear(timestamp);
231                 leap = Timestamps.isLeapYear(y);
232                 m = Timestamps.getMonthOfYear(timestamp, y, leap);
233                 d = Timestamps.getDayOfMonth(timestamp, y, m, leap);
234                 int h = Timestamps.getHourOfDay(timestamp);
235                 TimestampFormatUtils.appendYear000(path, y);
236                 path.put('-');
237                 TimestampFormatUtils.append0(path, m);
238                 path.put('-');
239                 TimestampFormatUtils.append0(path, d);
240                 path.put('T');
241                 TimestampFormatUtils.append0(path, h);
242 
243                 if (calculatePartitionMax) {
244                     return Timestamps.yearMicros(y, leap)
245                             + Timestamps.monthOfYearMicros(m, leap)
246                             + (d - 1) * Timestamps.DAY_MICROS + (h + 1) * Timestamps.HOUR_MICROS - 1;
247                 }
248                 return 0;
249             case WEEK:
250                 y = Timestamps.getIsoYear(timestamp);
251                 int w = Timestamps.getWeek(timestamp);
252                 TimestampFormatUtils.appendYear000(path, y);
253                 path.put("-W");
254                 TimestampFormatUtils.append0(path, w);
255 
256                 if (calculatePartitionMax) {
257                     return Timestamps.ceilWW(timestamp) - 1;
258                 }
259                 return 0;
260             default:
261                 path.put(DEFAULT_PARTITION_NAME);
262                 return Long.MAX_VALUE;
263         }
264     }
