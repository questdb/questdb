100     private boolean acquireTxn() {
101         // txFile can also be reloaded in goPassive->checkSchedulePurgeO3Partitions
102         // if txFile txn doesn't much reader txn, reader has to be slow reloaded
103         if (txn == txFile.getTxn()) {
104             // We have to be sure last txn is acquired in Scoreboard
105             // otherwise writer can delete partition version files
106             // between reading txn file and acquiring txn in the Scoreboard.
107             Unsafe.getUnsafe().loadFence();
108             return txFile.getVersion() == txFile.unsafeReadVersion();
109         }
110         return false;
111     }

113     private void readTxnSlow(long deadline) {
114         int count = 0;
115 
116         while (true) {
117             if (txFile.unsafeLoadAll()) {
118                 // good, very stable, congrats
119                 long txn = txFile.getTxn();
120                 this.txn = txn;
121 
122                 if (acquireTxn()) {
123                     this.rowCount = txFile.getFixedRowCount() + txFile.getTransientRowCount();
124                     LOG.debug()
125                             .$("new transaction [txn=").$(txn)
126                             .$(", transientRowCount=").$(txFile.getTransientRowCount())
127                             .$(", fixedRowCount=").$(txFile.getFixedRowCount())
128                             .$(", maxTimestamp=").$ts(txFile.getMaxTimestamp())
129                             .$(", attempts=").$(count)
130                             .$(", thread=").$(Thread.currentThread().getName())
131                             .$(']').$();
132                     break;
133                 }
134             }
135             // This is unlucky, sequences have changed while we were reading transaction data
136             // We must discard and try again
137             count++;
138             if (clock.getTicks() > deadline) {
139                 LOG.error().$("tx read timeout [timeout=").$(configuration.getSpinLockTimeout()).utf8("ms]").$();
140                 throw CairoException.critical(0).put("Transaction read timeout");
141             }
142             Os.pause();
143         }
144     }

146     private boolean reloadMetadata(long txnStructureVersion, long deadline) {
147         // create transition index, which will help us reuse already open resources
148         if (txnStructureVersion == getStructureVersion()) {
149             return true;
150         }
151 
152         while (true) {
153             long pTransitionIndex;
154             try {
155                 pTransitionIndex = createTransitionIndex(txnStructureVersion);
156                 if (pTransitionIndex < 0) {
157                     if (clock.getTicks() < deadline) {
158                         return false;
159                     }
160                     LOG.error().$("metadata read timeout [timeout=").$(configuration.getSpinLockTimeout()).utf8("ms]").$();
161                     throw CairoException.critical(0).put("Metadata read timeout");
162                 }
163             } catch (CairoException ex) {
164                 // This is temporary solution until we can get multiple version of metadata not overwriting each other
165                 TableUtils.handleMetadataLoadException(getTableToken().getTableName(), deadline, ex, configuration.getMillisecondClock(), configuration.getSpinLockTimeout());
166                 continue;
167             }
168 
169             try {
170                 applyTransitionIndex();
171                 return true;
172             } finally {
173                 TableUtils.freeTransitionIndex(pTransitionIndex);
174             }
175         }
176     }

178     private void reloadSlow() {
179         final long deadline = clock.getTicks() + configuration.getSpinLockTimeout();
180         do {
181             // Reload txn
182             readTxnSlow(deadline);
183             // Reload _meta if structure version updated, reload _cv if column version updated
184             // Start again if _meta with matching structure version cannot be loaded
185         } while (!reloadMetadata(txFile.getStructureVersion(), deadline));
186     }
