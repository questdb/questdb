67     private void createIndexFiles(CharSequence columnName, int indexValueBlockCapacity, int plen, long columnNameTxn) {
68         try {
69             BitmapIndexUtils.keyFileName(path.trimTo(plen), columnName, columnNameTxn);
70             try {
71                 LOG.info().$("writing ").utf8(path).$();
72                 ddlMem.smallFile(ff, path, MemoryTag.MMAP_TABLE_WRITER);
73                 BitmapIndexWriter.initKeyMemory(ddlMem, indexValueBlockCapacity);
74             } catch (CairoException e) {
75                 // looks like we could not create key file properly
76                 // lets not leave half-baked file sitting around
77                 LOG.error()
78                         .$("could not create index [name=").utf8(path)
79                         .$(", errno=").$(e.getErrno())
80                         .$(']').$();
81                 if (!ff.remove(path)) {
82                     LOG.error()
83                             .$("could not remove '").utf8(path).$("'. Please remove MANUALLY.")
84                             .$("[errno=").$(ff.errno())
85                             .$(']').$();
86                 }
87                 throw e;
88             } finally {
89                 // this close() closes the underlying file, but ddlMem object remains reusable
90                 ddlMem.close();
91             }
92             if (!ff.touch(BitmapIndexUtils.valueFileName(path.trimTo(plen), columnName, columnNameTxn))) {
93                 LOG.error().$("could not create index [name=").utf8(path).$(']').$();
94                 throw CairoException.critical(ff.errno()).put("could not create index [name=").put(path).put(']');
95             }
96             LOG.info().$("writing ").utf8(path).$();
97         } finally {
98             path.trimTo(plen);
99         }
100     }

102     private void removeFile(Path path) {
103         LOG.info().$("deleting ").utf8(path).$();
104         if (!ff.remove(this.path)) {
105             if (!ff.exists(this.path)) {
106                 // This is fine, index can be corrupt, rewriting is what we try to do here
107                 LOG.info().$("index file did not exist, file will be re-written [path=").utf8(path).I$();
108             } else {
109                 throw CairoException.critical(ff.errno()).put("cannot remove index file");
110             }
111         }
112     }

114     private void removeIndexFiles(CharSequence columnName, long columnNameTxn) {
115         final int plen = path.length();
116         BitmapIndexUtils.keyFileName(path.trimTo(plen), columnName, columnNameTxn);
117         removeFile(path);
118 
119         BitmapIndexUtils.valueFileName(path.trimTo(plen), columnName, columnNameTxn);
120         removeFile(path);
121     }

123     protected void doReindex(
124             ColumnVersionReader columnVersionReader,
125             int columnWriterIndex,
126             CharSequence columnName,
127             CharSequence partitionName,
128             long partitionNameTxn,
129             long partitionSize,
130             long partitionTimestamp,
131             int indexValueBlockCapacity
132     ) {
133         path.trimTo(rootLen).concat(partitionName);
134         TableUtils.txnPartitionConditionally(path, partitionNameTxn);
135         final int plen = path.length();
136 
137         if (ff.exists(path.$())) {
138             try (final MemoryMR roMem = indexMem) {
139                 long columnNameTxn = columnVersionReader.getColumnNameTxn(partitionTimestamp, columnWriterIndex);
140                 removeIndexFiles(columnName, columnNameTxn);
141                 TableUtils.dFile(path.trimTo(plen), columnName, columnNameTxn);
142 
143                 final long columnTop = columnVersionReader.getColumnTop(partitionTimestamp, columnWriterIndex);
144                 if (columnTop > -1L) {
145 
146                     if (partitionSize > columnTop) {
147                         LOG.info().$("indexing [path=").utf8(path).I$();
148                         createIndexFiles(columnName, indexValueBlockCapacity, plen, columnNameTxn);
149                         TableUtils.dFile(path.trimTo(plen), columnName, columnNameTxn);
150                         roMem.of(
151                                 ff,
152                                 path,
153                                 0,
154                                 (partitionSize - columnTop) * Integer.BYTES,
155                                 MemoryTag.MMAP_TABLE_WRITER
156                         );
157                         try {
158                             indexer.configureWriter(configuration, path.trimTo(plen), columnName, columnNameTxn, columnTop);
159                             indexer.index(roMem, columnTop, partitionSize);
160                         } finally {
161                             indexer.clear();
162                         }
163                     }
164                 } else {
165                     LOG.info().$("column is empty in partition [path=").$(path).I$();
166                 }
167             }
168         } else {
169             LOG.info().$("partition does not exist [path=").$(path).I$();
170         }
171     }
