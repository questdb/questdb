278             FilesFacade ff,
279             CharSequence root,
280             int mkDirMode,
281             MemoryMARW memory,
282             Path path,
283             CharSequence tableDir,
284             TableStructure structure,
285             int tableVersion,

287     ) {

290 
291         if (ff.mkdirs(path.slash$(), mkDirMode) != 0) {
292             throw CairoException.critical(ff.errno()).put("could not create [dir=").put(path).put(']');
293         }
294 
295         final int rootLen = path.length();
296 
297         final int dirFd = !ff.isRestrictedFileSystem() ? TableUtils.openRO(ff, path.$(), LOG) : 0;

299             mem.smallFile(ff, path.trimTo(rootLen).concat(META_FILE_NAME).$(), MemoryTag.MMAP_DEFAULT);

304             mem.putInt(structure.getPartitionBy());

306             assert timestampIndex == -1 ||

312             mem.putLong(structure.getO3MaxLag());

317             assert count > 0;
318

382     public static long createTransitionIndex(
383             MemoryR masterMeta,
384             AbstractRecordMetadata slaveMeta
385     ) {
386         int slaveColumnCount = slaveMeta.columnCount;
387         int masterColumnCount = masterMeta.getInt(META_OFFSET_COUNT);
388         final long pTransitionIndex;
389         final int size = 8 + masterColumnCount * 8;
390 
391         long index = pTransitionIndex = Unsafe.calloc(size, MemoryTag.NATIVE_TABLE_READER);
392         Unsafe.getUnsafe().putInt(index, size);
393         index += 8;
394 
395         // index structure is
396         // [action: int, copy from:int]
397 
398         // action: if -1 then current column in slave is deleted or renamed, else it's reused
399         // "copy from" >= 0 indicates that column is to be copied from slave position
400         // "copy from" < 0  indicates that column is new and should be taken from updated metadata position
401         // "copy from" == Integer.MIN_VALUE  indicates that column is deleted for good and should not be re-added from any source
402 
403         long offset = getColumnNameOffset(masterColumnCount);
404         int slaveIndex = 0;
405         int shiftLeft = 0;
406         for (int masterIndex = 0; masterIndex < masterColumnCount; masterIndex++) {
407             CharSequence name = masterMeta.getStr(offset);
408             offset += Vm.getStorageLength(name);
409             int masterColumnType = getColumnType(masterMeta, masterIndex);
410 
411             if (slaveIndex < slaveColumnCount) {
412                 int existingWriterIndex = slaveMeta.getWriterIndex(slaveIndex);
413                 if (existingWriterIndex > masterIndex) {
414                     // This column must be deleted so existing dense columns do not contain it
415                     assert masterColumnType < 0;
416                     continue;
417                 }
418                 assert existingWriterIndex == masterIndex;
419             }
420 
421             int outIndex = slaveIndex - shiftLeft;
422             if (masterColumnType < 0) {
423                 shiftLeft++; // Deleted in master
424                 if (slaveIndex < slaveColumnCount) {
425                     Unsafe.getUnsafe().putInt(index + slaveIndex * 8L, -1);
426                     Unsafe.getUnsafe().putInt(index + slaveIndex * 8L + 4, Integer.MIN_VALUE);
427                 }
428             } else {
429                 if (
430                         slaveIndex < slaveColumnCount
431                                 && isColumnIndexed(masterMeta, masterIndex) == slaveMeta.isColumnIndexed(slaveIndex)
432                                 && Chars.equals(name, slaveMeta.getColumnName(slaveIndex))
433                 ) {
434                     // reuse
435                     Unsafe.getUnsafe().putInt(index + outIndex * 8L + 4, slaveIndex);
436                     if (slaveIndex > outIndex) {
437                         // mark to do nothing with existing column, this may be overwritten later
438                         Unsafe.getUnsafe().putInt(index + slaveIndex * 8L + 4, Integer.MIN_VALUE);
439                     }
440                 } else {
441                     // new
442                     if (slaveIndex < slaveColumnCount) {
443                         // free
444                         Unsafe.getUnsafe().putInt(index + slaveIndex * 8L, -1);
445                     }
446                     Unsafe.getUnsafe().putInt(index + outIndex * 8L + 4, -masterIndex - 1);
447                 }
448             }
449             slaveIndex++;
450         }
451         Unsafe.getUnsafe().putInt(pTransitionIndex + 4, slaveIndex - shiftLeft);
452         return pTransitionIndex;
453     }

1035     public static void safeReadTxn(TxReader txReader, MillisecondClock clock, long spinLockTimeout) {
1036         long deadline = clock.getTicks() + spinLockTimeout;
1037         if (txReader.unsafeReadVersion() == txReader.getVersion()) {
1038             LOG.debug().$("checked clean txn, version ").$(txReader.getVersion()).$(", txn=").$(txReader.getTxn()).$();
1039             return;
1040         }

1042         while (true) {
1043             if (txReader.unsafeLoadAll()) {
1044                 LOG.debug().$("loaded clean txn, version ").$(txReader.getVersion())
1045                         .$(", offset=").$(txReader.getBaseOffset())
1046                         .$(", size=").$(txReader.getRecordSize())
1047                         .$(", txn=").$(txReader.getTxn()).$();
1048                 // All good, snapshot read
1049                 return;
1050             }
1051             // This is unlucky, sequences have changed while we were reading transaction data
1052             // We must discard and try again
1053             if (clock.getTicks() > deadline) {
1054                 LOG.error().$("tx read timeout [timeout=").$(spinLockTimeout).utf8("ms]").$();
1055                 throw CairoException.critical(0).put("Transaction read timeout");
1056             }
1057 
1058             LOG.debug().$("loaded __dirty__ txn, version ").$(txReader.getVersion()).$();
1059             Os.pause();
1060         }
