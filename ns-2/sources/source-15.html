184     public TableToken createTable(
185             CairoSecurityContext securityContext,
186             MemoryMARW mem,
187             Path path,
188             boolean ifNotExists,
189             TableStructure struct,
190             boolean keepLock
191     ) {
192         securityContext.checkWritePermission();
193         CharSequence tableName = struct.getTableName();
194         validNameOrThrow(tableName);
195 
196         int tableId = (int) tableIdGenerator.getNextId();
197         TableToken tableToken = lockTableName(tableName, tableId, struct.isWalEnabled());
198         if (tableToken == null) {
199             if (ifNotExists) {
200                 return null;
201             }
202             throw EntryUnavailableException.instance("table exists");
203         }
204 
205         try {
206             String lockedReason = lock(securityContext, tableToken, "createTable");
207             if (null == lockedReason) {
208                 boolean tableCreated = false;
209                 try {
210                     int status = TableUtils.exists(configuration.getFilesFacade(), path, configuration.getRoot(), tableToken.getDirName());
211                     if (status != TableUtils.TABLE_DOES_NOT_EXIST) {
212                         throw CairoException.nonCritical().put("name is reserved [table=").put(tableName).put(']');
213                     }
214                     createTableUnsafe(
215                             securityContext,
216                             mem,
217                             path,
218                             struct,
219                             tableToken,
220                             tableId
221                     );
222                     tableCreated = true;
223                 } finally {
224                     if (!keepLock) {
225                         unlockTableUnsafe(tableToken, null, tableCreated);
226                         LOG.info().$("unlocked [table=`").$(tableToken).$("`]").$();
227                     }
228                 }
229                 tableNameRegistry.registerName(tableToken);
230             } else {
231                 if (!ifNotExists) {
232                     throw EntryUnavailableException.instance(lockedReason);
233                 }
234             }
235         } catch (Throwable th) {
236             if (struct.isWalEnabled()) {
237                 // tableToken.getLoggingName() === tableName, table cannot be renamed while creation hasn't finished
238                 tableSequencerAPI.dropTable(tableToken, true);
239             }
240             throw th;
241         } finally {
242             tableNameRegistry.unlockTableName(tableToken);
243         }
244         return tableToken;
245     }

278     public void drop(
279             CairoSecurityContext securityContext,
280             Path path,
281             TableToken tableToken
282     ) {
283         securityContext.checkWritePermission();
284         verifyTableToken(tableToken);
285         if (tableToken.isWal()) {
286             if (tableNameRegistry.dropTable(tableToken)) {
287                 tableSequencerAPI.dropTable(tableToken, false);
288             } else {
289                 LOG.info().$("table is already dropped [table=").$(tableToken)
290                         .$(", dirName=").$(tableToken.getDirName()).I$();
291             }
292         } else {
293             CharSequence lockedReason = lock(securityContext, tableToken, "removeTable");
294             if (null == lockedReason) {
295                 try {
296                     path.of(configuration.getRoot()).concat(tableToken).$();
297                     int errno;
298                     if ((errno = configuration.getFilesFacade().rmdir(path)) != 0) {
299                         LOG.error().$("drop failed [tableName='").$(tableToken).$("', error=").$(errno).I$();
300                         throw CairoException.critical(errno).put("could not remove table [name=").put(tableToken)
301                                 .put(", dirName=").put(tableToken.getDirName()).put(']');
302                     }
303                 } finally {
304                     unlockTableUnsafe(tableToken, null, false);
305                 }
306 
307                 tableNameRegistry.dropTable(tableToken);
308                 return;
309             }
310             throw CairoException.nonCritical().put("Could not lock '").put(tableToken).put("' [reason='").put(lockedReason).put("']");
311         }

401     public TableReader getReader(
402             CairoSecurityContext securityContext,
403             TableToken tableToken,
404             long version
405     ) {
406         TableToken newTableToken = tableNameRegistry.getTableToken(tableToken.getTableName());
407         if (newTableToken == null) {
408             throw CairoException.tableDoesNotExist(tableToken.getTableName());
409         }
410         final int tableId = tableToken.getTableId();
411         if (tableId > -1 && newTableToken.getTableId() != tableId) {
412             throw TableReferenceOutOfDateException.of(tableToken, tableId, newTableToken.getTableId(), version, -1);
413         }
414 
415         TableReader reader = readerPool.get(tableToken);
416         if ((version > -1 && reader.getVersion() != version)
417                 || tableId > -1 && reader.getMetadata().getTableId() != tableId) {
418             TableReferenceOutOfDateException ex = TableReferenceOutOfDateException.of(tableToken, tableId, reader.getMetadata().getTableId(), version, reader.getVersion());
419             reader.close();
420             throw ex;
421         }
422         return reader;
423     }

597     public String lock(
598             CairoSecurityContext securityContext,
599             TableToken tableToken,
600             String lockReason
601     ) {
602         assert null != lockReason;
603         securityContext.checkWritePermission();
604         // busy metadata is same as busy reader from user perspective
605         String lockedReason = BUSY_READER;
606         if (metadataPool.lock(tableToken)) {
607             lockedReason = writerPool.lock(tableToken, lockReason);
608             if (lockedReason == null) {
609                 // not locked
610                 if (readerPool.lock(tableToken)) {
611                     LOG.info().$("locked [table=`").utf8(tableToken.getDirName()).$("`, thread=").$(Thread.currentThread().getId()).I$();
612                     return null;
613                 }
614                 writerPool.unlock(tableToken);
615                 lockedReason = BUSY_READER;
616             }
617             metadataPool.unlock(tableToken);
618         }
619         return lockedReason;
620     }

816         final FilesFacade ff = configuration.getFilesFacade();
817         final CharSequence root = configuration.getRoot();
818 
819         path.of(root).concat(srcTableToken).$();
820         TableToken dstTableToken = lockTableName(to, srcTableToken.getTableId(), false);
821 
822         if (dstTableToken == null || ff.exists(otherPath.of(root).concat(dstTableToken).$())) {
823             if (dstTableToken != null) {
824                 tableNameRegistry.unlockTableName(dstTableToken);
825             }
826             LOG.error().$("rename target exists [from='").utf8(tableName).$("', to='").utf8(otherPath.$()).I$();
827             throw CairoException.nonCritical().put("Rename target exists");
828         }
829 
830         try {
831             if (ff.rename(path, otherPath) != Files.FILES_RENAME_OK) {
832                 int error = ff.errno();
833                 LOG.error().$("could not rename [from='").utf8(path).$("', to='").utf8(otherPath).$("', error=").$(error).I$();
834                 throw CairoException.critical(error)
835                         .put("could not rename [from='").put(path)
836                         .put("', to='").put(otherPath)
837                         .put("', error=").put(error);
838             }
839             tableNameRegistry.registerName(dstTableToken);
840             return dstTableToken;
841         } finally {
842             tableNameRegistry.unlockTableName(dstTableToken);
843         }
844     }
