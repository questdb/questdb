163     // Process incoming queue and put it on priority queue with next timestamp to rerun
164     private boolean processInQueue() {
165         boolean useful = false;
166         long microTime = clock.getTicks();
167         while (true) {
168             long cursor = inSubSequence.next();
169             // -2 = there was a contest for queue index and this thread has lost
170             if (cursor < -1) {
171                 Os.pause();
172                 continue;
173             }
174             // -1 = queue is empty, all done
175             if (cursor < 0) {
176                 break;
177             }
178 
179             ColumnPurgeTask queueTask = inQueue.get(cursor);
180             ColumnPurgeRetryTask purgeTaskRun = taskPool.pop();
181             purgeTaskRun.copyFrom(queueTask, retryDelay, microTime + retryDelay);
182             purgeTaskRun.timestamp = microTime++;
183             inSubSequence.done(cursor);
184 
185             saveToStorage(purgeTaskRun);
186 
187             retryQueue.add(purgeTaskRun);
188             useful = true;
189         }
190 
191         if (useful) {
192             commit();
193         }
194         return useful;
195     }

198         try {
199             CompiledQuery reloadQuery = sqlCompiler.compile(
200                     "SELECT * FROM \"" + tableToken.getTableName() + "\" WHERE completed = null",
201                     sqlExecutionContext
202             );
203 
204             long microTime = clock.getTicks();
205             try (RecordCursorFactory recordCursorFactory = reloadQuery.getRecordCursorFactory()) {
206                 assert recordCursorFactory.supportsUpdateRowId(tableToken);
207                 int count = 0;
208

210                     Record rec = records.getRecord();
211                     long lastTs = 0;
212                     ColumnPurgeRetryTask taskRun = null;
213 
214                     CharSequenceObjHashMap<String> stringIntern = new CharSequenceObjHashMap<>();
215 
216                     while (records.hasNext()) {
217                         count++;
218                         long ts = rec.getTimestamp(0);
219                         if (ts != lastTs || taskRun == null) {

222                             } else {

224                             }

227                             String tableName = internStrObj(stringIntern, rec.getSym(TABLE_NAME_COLUMN));

230                             long truncateVersion = rec.getLong(TABLE_TRUNCATE_VERSION);
231                             int columnType = rec.getInt(COLUMN_TYPE_COLUMN);
232                             int partitionBY = rec.getInt(PARTITION_BY_COLUMN);
233                             long updateTxn = rec.getLong(UPDATE_TXN_COLUMN);

235

242                                     token,
243                                     columnName,
244                                     tableId,
245                                     truncateVersion,
246                                     columnType,
247                                     partitionBY,
248                                     updateTxn,
249                                     retryDelay,
250                                     microTime
251                             );
252                         }
253                         long columnVersion = rec.getLong(COLUMN_VERSION_COLUMN);
254                         long partitionTs = rec.getLong(PARTITION_TIMESTAMP_COLUMN);
255                         long partitionNameTxn = rec.getLong(PARTITION_NAME_COLUMN);
256                         taskRun.appendColumnInfo(columnVersion, partitionTs, partitionNameTxn, rec.getUpdateRowId());
257                     }
258                     if (taskRun != null) {
259                         columnPurgeOperator.purgeExclusive(taskRun);
260                         taskPool.push(taskRun);
261                     }
262                 }
263 
264                 if (count > 0) {

266                 }

271                 } catch (Throwable th) {

273                 }
274             }
275         } catch (SqlException e) {
276             LOG.error().$("failed to reload column version purge tasks").$((Throwable) e).$();
277         }

280     private boolean purge() {
281         boolean useful = false;
282         final long now = clock.getTicks() + 1;
283         while (retryQueue.size() > 0) {
284             ColumnPurgeRetryTask nextTask = retryQueue.peek();
285             if (nextTask.nextRunTimestamp < now) {
286                 retryQueue.poll();
287                 useful = true;
288                 if (!columnPurgeOperator.purge(nextTask)) {
289                     // Re-queue
290                     calculateNextTimestamp(nextTask, now);
291                     retryQueue.add(nextTask);
292                 } else {
293                     taskPool.push(nextTask);
294                 }
295             } else {
296                 // All reruns are in the future.
297                 return useful;
298             }
299         }
300         return useful;
301     }

303     private void saveToStorage(ColumnPurgeRetryTask cleanTask) {
304         if (writer != null) {
305             try {
306                 LongList updatedColumnInfo = cleanTask.getUpdatedColumnInfo();
307                 for (int i = 0, n = updatedColumnInfo.size(); i < n; i += ColumnPurgeTask.BLOCK_SIZE) {
308                     TableWriter.Row row = writer.newRow(cleanTask.timestamp);
309                     row.putSym(TABLE_NAME_COLUMN, cleanTask.getTableName().getDirName());
310                     row.putSym(COLUMN_NAME_COLUMN, cleanTask.getColumnName());
311                     row.putInt(TABLE_ID_COLUMN, cleanTask.getTableId());
312                     row.putLong(TABLE_TRUNCATE_VERSION, cleanTask.getTruncateVersion());
313                     row.putInt(COLUMN_TYPE_COLUMN, cleanTask.getColumnType());
314                     row.putInt(PARTITION_BY_COLUMN, cleanTask.getPartitionBy());
315                     row.putLong(UPDATE_TXN_COLUMN, cleanTask.getUpdateTxn());
316                     row.putLong(COLUMN_VERSION_COLUMN, updatedColumnInfo.getQuick(i + ColumnPurgeTask.OFFSET_COLUMN_VERSION));
317                     row.putTimestamp(PARTITION_TIMESTAMP_COLUMN, updatedColumnInfo.getQuick(i + ColumnPurgeTask.OFFSET_PARTITION_TIMESTAMP));
318                     row.putLong(PARTITION_NAME_COLUMN, updatedColumnInfo.getQuick(i + ColumnPurgeTask.OFFSET_PARTITION_NAME_TXN));
319                     row.append();
320                     updatedColumnInfo.setQuick(
321                             i + ColumnPurgeTask.OFFSET_UPDATE_ROW_ID,
322                             Rows.toRowID(writer.getPartitionCount() - 1, writer.getTransientRowCount() - 1)
323                     );
324                 }
325             } catch (Throwable th) {
326                 LOG.error().$("error saving to column version house keeping log, unable to insert")
327                         .$(", releasing writer and stop updating log [table=").$(tableToken)
328                         .$(", error=").$(th)
329                         .I$();
330                 writer = Misc.free(writer);
331             }
332         }

335     @Override
336     protected boolean runSerially() {
337         if (inErrorCount >= MAX_ERRORS) {
338             return false;
339         }
340 
341         try {
342             boolean useful = processInQueue();
343             boolean cleanupUseful = purge();
344             if (cleanupUseful) {
345                 LOG.debug().$("cleaned column version, outstanding tasks: ").$(retryQueue.size()).$();
346             }
347             inErrorCount = 0;
348             return cleanupUseful || useful;
349         } catch (Throwable th) {
350             LOG.error().$("failed to clean up column versions").$(th).$();
351             inErrorCount++;
352             if (inErrorCount == MAX_ERRORS) {
353                 if (retryQueue.size() > 0) {
354                     LOG.error().$("clean up column versions reached maximum error count and will be recycled. Some column version may be left behind.").$(th).$();
355                     retryQueue.clear();
356                     inErrorCount = 0;
357                 } else {
358                     LOG.error().$("clean up column versions reached maximum error count and will be DISABLED. Restart QuestDB to re-enable the job.").$(th).$();
359                     close();
360                 }
361             }
362             return false;
363         }
