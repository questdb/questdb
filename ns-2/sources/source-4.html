122     private void calculateRanges(LongList intervals) {
123         size = -1;
124         if (intervals.size() > 0) {
125             if (PartitionBy.isPartitioned(reader.getPartitionedBy())) {
126                 cullIntervals(intervals);
127                 if (initialIntervalsLo < initialIntervalsHi) {
128                     cullPartitions(intervals);
129                 }
130             } else {
131                 initialIntervalsLo = 0;
132                 initialIntervalsHi = intervals.size() / 2;
133                 initialPartitionLo = 0;
134                 initialPartitionHi = reader.getPartitionCount();
135             }
136             toTop();
137         }

141      * Best effort size calculation.
142      */
143     private long computeSize() {
144         int intervalsLo = this.intervalsLo;
145         int intervalsHi = this.intervalsHi;
146         int partitionLo = this.partitionLo;
147         int partitionHi = this.partitionHi;
148         long partitionLimit = this.partitionLimit;
149         long size = this.sizeSoFar;
150 
151         while (intervalsLo < intervalsHi && partitionLo < partitionHi) {
152             // We don't need to worry about column tops and null column because we
153             // are working with timestamp. Timestamp column cannot be added to existing table.
154             long rowCount;
155             try {
156                 rowCount = reader.openPartition(partitionLo);
157             } catch (DataUnavailableException e) {
158                 // The data is in cold storage, close the event and give up on size calculation.
159                 Misc.free(e.getEvent());
160                 return -1;
161             }
162             if (rowCount > 0) {
163 
164                 final MemoryR column = reader.getColumn(TableReader.getPrimaryColumnIndex(reader.getColumnBase(partitionLo), timestampIndex));
165                 final long intervalLo = intervals.getQuick(intervalsLo * 2);
166                 final long intervalHi = intervals.getQuick(intervalsLo * 2 + 1);
167 
168                 final long partitionTimestampLo = column.getLong(0);
169                 // interval is wholly above partition, skip interval
170                 if (partitionTimestampLo > intervalHi) {
171                     intervalsLo++;
172                     continue;
173                 }
174 
175                 final long partitionTimestampHi = column.getLong((rowCount - 1) * 8);
176                 // interval is wholly below partition, skip partition
177                 if (partitionTimestampHi < intervalLo) {
178                     partitionLimit = 0;
179                     partitionLo++;
180                     continue;
181                 }
182 
183                 // calculate intersection
184 
185                 long lo;
186                 if (partitionTimestampLo == intervalLo) {
187                     lo = 0;
188                 } else {
189                     lo = search(column, intervalLo, partitionLimit, rowCount, AbstractIntervalDataFrameCursor.SCAN_UP);
190                     if (lo < 0) {
191                         lo = -lo - 1;
192                     }
193                 }
194 
195                 long hi = search(column, intervalHi, lo, rowCount, AbstractIntervalDataFrameCursor.SCAN_DOWN);
196 
197                 if (hi < 0) {
198                     hi = -hi - 1;
199                 } else {
200                     // We have direct hit. Interval is inclusive of edges and we have to
201                     // bump to high bound because it is non-inclusive
202                     hi++;
203                 }
204 
205                 if (lo < hi) {
206                     size += (hi - lo);
207 
208                     // we do have whole partition of fragment?
209                     if (hi == rowCount) {
210                         // whole partition, will need to skip to next one
211                         partitionLimit = 0;
212                         partitionLo++;
213                     } else {
214                         // only fragment, need to skip to next interval
215                         partitionLimit = hi;
216                         intervalsLo++;
217                     }
218                     continue;
219                 }
220                 // interval yielded empty data frame
221                 partitionLimit = hi;
222                 intervalsLo++;
223             } else {
224                 // partition was empty, just skip to next
225                 partitionLo++;
226             }
227         }
228 
229         return this.size = size;
230     }

232     private void cullIntervals(LongList intervals) {
233         int intervalsLo = intervals.binarySearch(reader.getMinTimestamp(), BinarySearch.SCAN_UP);
234 
235         // not a direct hit
236         if (intervalsLo < 0) {
237             intervalsLo = -intervalsLo - 1;
238         }
239 
240         // normalise interval index
241         this.initialIntervalsLo = intervalsLo / 2;
242 
243         int intervalsHi = intervals.binarySearch(reader.getMaxTimestamp(), BinarySearch.SCAN_UP);
244         if (reader.getMaxTimestamp() == intervals.getQuick(intervals.size() - 1)) {
245             this.initialIntervalsHi = intervals.size() - 1;
246         }
247         if (reader.getMaxTimestamp() == intervals.getQuick(0)) {
248             this.initialIntervalsHi = 1;
249         }
250         if (intervalsHi < 0) {
251             intervalsHi = -intervalsHi - 1;
252             // when interval index is "even" we scored just between two interval
253             // in which case we chose previous interval
254             if (intervalsHi % 2 == 0) {
255                 this.initialIntervalsHi = intervalsHi / 2;
256             } else {
257                 this.initialIntervalsHi = intervalsHi / 2 + 1;
258             }
259         }
260     }

262     private void cullPartitions(LongList intervals) {
263         final long lo = intervals.getQuick(initialIntervalsLo * 2);
264         long intervalLo;
265         if (lo == Long.MIN_VALUE) {
266             intervalLo = reader.floorToPartitionTimestamp(reader.getMinTimestamp());
267         } else {
268             intervalLo = reader.floorToPartitionTimestamp(lo);
269         }
270         this.initialPartitionLo = reader.getMinTimestamp() < intervalLo ? reader.getPartitionIndexByTimestamp(intervalLo) : 0;
271         long intervalHi = reader.floorToPartitionTimestamp(intervals.getQuick((initialIntervalsHi - 1) * 2 + 1));
272         this.initialPartitionHi = Math.min(reader.getPartitionCount(), reader.getPartitionIndexByTimestamp(intervalHi) + 1);
273     }

275     protected static long search(MemoryR column, long value, long low, long high, int increment) {
276         while (low < high) {
277             long mid = (low + high - 1) >>> 1;
278             long midVal = column.getLong(mid * 8);
279 
280             if (midVal < value)
281                 low = mid + 1;
282             else if (midVal > value)
283                 high = mid;
284             else {
285                 // In case of multiple equal values, find the first
286                 mid += increment;
287                 while (mid > 0 && mid < high && midVal == column.getLong(mid * 8)) {
288                     mid += increment;
289                 }
290                 return mid - increment;
291             }
292         }
293         return -(low + 1);
294     }
