105     @Override
106     public int keyOf(CharSequence value) {
107         if (value != null) {
108             int hash = Hash.boundedHash(value, maxHash);
109             final RowCursor cursor = indexReader.getCursor(true, hash, 0, maxOffset - Long.BYTES);
110             while (cursor.hasNext()) {
111                 final long offsetOffset = cursor.next();
112                 if (Chars.equals(value, charMem.getStr(offsetMem.getLong(offsetOffset)))) {
113                     return SymbolMapWriter.offsetToKey(offsetOffset);
114                 }
115             }
116             return SymbolTable.VALUE_NOT_FOUND;
117         }
118         return SymbolTable.VALUE_IS_NULL;
119     }

125     public void of(CairoConfiguration configuration, Path path, CharSequence columnName, long columnNameTxn, int symbolCount) {
126         FilesFacade ff = configuration.getFilesFacade();
127         this.configuration = configuration;
128         this.path.of(path);
129         this.columnNameSink.clear();
130         this.columnNameSink.put(columnName);
131         this.columnNameTxn = columnNameTxn;
132         this.symbolCount = symbolCount;
133         this.maxOffset = SymbolMapWriter.keyToOffset(symbolCount);
134         final int plen = path.length();
135         try {
136             // this constructor does not create index. Index must exist,
137             // and we use "offset" file to store "header"
138             offsetFileName(path.trimTo(plen), columnName, columnNameTxn);
139             if (!ff.exists(path)) {
140                 LOG.error().$(path).$(" is not found").$();
141                 throw CairoException.critical(0).put("SymbolMap does not exist: ").put(path);
142             }
143 
144             // is there enough length in "offset" file for "header"?
145             long len = ff.length(path);
146             if (len < SymbolMapWriter.HEADER_SIZE) {
147                 LOG.error().$(path).$(" is too short [len=").$(len).$(']').$();
148                 throw CairoException.critical(0).put("SymbolMap is too short: ").put(path);
149             }
150 
151             // open "offset" memory and make sure we start appending from where
152             // we left off. Where we left off is stored externally to symbol map
153             final long offsetMemSize = SymbolMapWriter.keyToOffset(symbolCount) + Long.BYTES;
154             LOG.debug().$("offsetMem.of [columnName=").$(path).$(",offsetMemSize=").$(offsetMemSize).I$();
155             this.offsetMem.of(ff, path, offsetMemSize, offsetMemSize, MemoryTag.MMAP_INDEX_READER);
156             this.symbolCapacity = offsetMem.getInt(SymbolMapWriter.HEADER_CAPACITY);
157             assert this.symbolCapacity > 0;
158             this.cached = offsetMem.getBool(SymbolMapWriter.HEADER_CACHE_ENABLED);
159             this.nullValue = offsetMem.getBool(SymbolMapWriter.HEADER_NULL_FLAG);
160 
161             // index reader is used to identify attempts to store duplicate symbol value
162             this.indexReader.of(configuration, path.trimTo(plen), columnName, columnNameTxn, 0, -1);
163 
164             // this is the place where symbol values are stored
165             this.charMem.wholeFile(ff, charFileName(path.trimTo(plen), columnName, columnNameTxn), MemoryTag.MMAP_INDEX_READER);
166 
167             // move append pointer for symbol values in the correct place
168             this.charMem.extend(this.offsetMem.getLong(maxOffset));
169 
170             // we use index hash maximum equals to half of symbol capacity, which
171             // theoretically should require 2 value cells in index per hash
172             // we use 4 cells to compensate for occasionally unlucky hash distribution
173             this.maxHash = Math.max(Numbers.ceilPow2(symbolCapacity / 2) - 1, 1);
174             if (cached) {
175                 this.cache.setPos(symbolCapacity);
176             }
177             this.cache.clear();
178             LOG.debug().$("open [columnName=").$(path.trimTo(plen).concat(columnName).$()).$(", fd=").$(this.offsetMem.getFd()).$(", capacity=").$(symbolCapacity).$(']').$();
179         } catch (Throwable e) {
180             close();
181             throw e;
182         } finally {
183             path.trimTo(plen);
184         }
185     }

188     public void updateSymbolCount(int symbolCount) {
189         if (symbolCount > this.symbolCount) {
190             this.symbolCount = symbolCount;
191             this.maxOffset = SymbolMapWriter.keyToOffset(symbolCount);
192             // offset mem contains offsets of symbolCount + 1
193             // we need to make sure we have access to the last element
194             // which will indicate size of the char column
195             this.offsetMem.extend(maxOffset + Long.BYTES);
196             this.charMem.extend(this.offsetMem.getLong(maxOffset));
197         } else if (symbolCount < this.symbolCount) {
198             cache.remove(symbolCount + 1, this.symbolCount);
199             this.symbolCount = symbolCount;
200         }
201         // Refresh index reader to avoid memory remapping on keyOf() calls.
202         this.indexReader.of(configuration, path, columnNameSink, columnNameTxn, 0, -1);
203     }
