43     public static void setupWorkerPool(
44             WorkerPool workerPool,
45             CairoEngine cairoEngine,
46             @Nullable SqlExecutionCircuitBreakerConfiguration sqlExecutionCircuitBreakerConfiguration,
47             @Nullable FunctionFactoryCache functionFactoryCache
48     ) throws SqlException {
49         final MessageBus messageBus = cairoEngine.getMessageBus();
50         final int workerCount = workerPool.getWorkerCount();
51         final O3PartitionPurgeJob purgeDiscoveryJob = new O3PartitionPurgeJob(messageBus, workerPool.getWorkerCount());
52         workerPool.assign(purgeDiscoveryJob);
53 
54         // ColumnPurgeJob has expensive init (it creates a table), disable it in some tests.
55         if (!cairoEngine.getConfiguration().disableColumnPurgeJob()) {
56             final ColumnPurgeJob columnPurgeJob = new ColumnPurgeJob(cairoEngine, functionFactoryCache);
57             workerPool.freeOnExit(columnPurgeJob);
58             workerPool.assign(columnPurgeJob);
59         }
60 
61         workerPool.assign(new O3PartitionJob(messageBus));
62         workerPool.assign(new O3OpenColumnJob(messageBus));
63         workerPool.assign(new O3CopyJob(messageBus));
64         workerPool.assign(new O3CallbackJob(messageBus));
65         workerPool.freeOnExit(purgeDiscoveryJob);
66 
67         final MicrosecondClock microsecondClock = messageBus.getConfiguration().getMicrosecondClock();
68         final NanosecondClock nanosecondClock = messageBus.getConfiguration().getNanosecondClock();
69 
70         for (int i = 0; i < workerCount; i++) {
71             // create job per worker to allow each worker to have
72             // own shard walk sequence
73             final PageFrameReduceJob pageFrameReduceJob = new PageFrameReduceJob(
74                     messageBus,
75                     new Rnd(microsecondClock.getTicks(), nanosecondClock.getTicks()),
76                     sqlExecutionCircuitBreakerConfiguration
77             );
78             workerPool.assign(i, pageFrameReduceJob);
79             workerPool.freeOnExit(pageFrameReduceJob);
80         }
81     }
