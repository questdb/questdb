71     public void copyPartition(long partitionTimestamp, ColumnVersionReader source) {
72         LongList src = source.cachedList;
73         LongList dest = this.cachedList;
74 
75         int srcIndex = src.binarySearchBlock(BLOCK_SIZE_MSB, partitionTimestamp, BinarySearch.SCAN_UP);
76         if (srcIndex < 0) { // source does not have partition information
77             return;
78         }
79 
80         int index = dest.binarySearchBlock(BLOCK_SIZE_MSB, partitionTimestamp, BinarySearch.SCAN_UP);
81         if (index > -1L) {
82             // Wipe out all the information about this partition to replace with the new one.
83             removePartition(partitionTimestamp);
84             index = dest.binarySearchBlock(BLOCK_SIZE_MSB, partitionTimestamp, BinarySearch.SCAN_UP);
85         }
86 
87         if (index < 0) { // the cache does not contain this partition
88             index = -index - 1;
89             int srcEnd = src.binarySearchBlock(srcIndex, BLOCK_SIZE_MSB, partitionTimestamp, BinarySearch.SCAN_DOWN);
90             dest.insertFromSource(index, src, srcIndex, srcEnd + BLOCK_SIZE);
91         } else {
92             throw CairoException.critical(0)
93                     .put("invalid Column Version state ")
94                     .ts(partitionTimestamp)
95                     .put(" column version state, cannot update partition information");
96         }
97         hasChanges = true;
98     }

150     /**
151      * Adds or updates column version entry in the cached list. Entries from the cache are committed to disk via
152      * commit() call. In cache and on disk entries are maintained in ascending chronological order of partition
153      * timestamps and ascending column index order within each timestamp.
154      *
155      * @param timestamp   partition timestamp
156      * @param columnIndex column index
157      * @param txn         column file txn name
158      * @param columnTop   column top
159      */
160     public void upsert(long timestamp, int columnIndex, long txn, long columnTop) {
161         final int sz = cachedList.size();
162         int index = cachedList.binarySearchBlock(BLOCK_SIZE_MSB, timestamp, BinarySearch.SCAN_UP);
163         boolean insert = true;
164         if (index > -1) {
165             // brute force columns for this timestamp
166             while (index < sz && cachedList.getQuick(index) == timestamp) {
167                 final long thisIndex = cachedList.getQuick(index + COLUMN_INDEX_OFFSET);
168 
169                 if (thisIndex == columnIndex) {
170                     if (txn > -1) {
171                         cachedList.setQuick(index + COLUMN_NAME_TXN_OFFSET, txn);
172                     }
173                     cachedList.setQuick(index + COLUMN_TOP_OFFSET, columnTop);
174                     insert = false;
175                     break;
176                 }
177 
178                 if (thisIndex > columnIndex) {
179                     break;
180                 }
181 
182                 index += BLOCK_SIZE;
183             }
184         } else {
185             index = -index - 1;
186         }
187 
188         if (insert) {
189             if (index < sz) {
190                 cachedList.insert(index, BLOCK_SIZE);
191             } else {
192                 cachedList.setPos(Math.max(index + BLOCK_SIZE, sz + BLOCK_SIZE));
193             }
194             cachedList.setQuick(index, timestamp);
195             cachedList.setQuick(index + COLUMN_INDEX_OFFSET, columnIndex);
196             cachedList.setQuick(index + COLUMN_NAME_TXN_OFFSET, txn);
197             cachedList.setQuick(index + COLUMN_TOP_OFFSET, columnTop);
198         }
199         hasChanges = true;
200     }

202     public void upsertColumnTop(long partitionTimestamp, int columnIndex, long colTop) {
203         int recordIndex = getRecordIndex(partitionTimestamp, columnIndex);
204         if (recordIndex > -1L) {
205             cachedList.setQuick(recordIndex + COLUMN_TOP_OFFSET, colTop);
206             hasChanges = true;
207         } else {
208             // This is a 0 column top record we need to store it
209             // to mark that the column is written in O3 even before the partition the column was originally added
210             int defaultRecordIndex = getRecordIndex(COL_TOP_DEFAULT_PARTITION, columnIndex);
211             if (defaultRecordIndex >= 0) {
212                 long columnNameTxn = cachedList.getQuick(defaultRecordIndex + COLUMN_NAME_TXN_OFFSET);
213                 long defaultPartitionTimestamp = cachedList.getQuick(defaultRecordIndex + COLUMN_TOP_OFFSET);
214                 // Do not add 0 column top if the default partition
215                 if (defaultPartitionTimestamp > partitionTimestamp || colTop > 0) {
216                     upsert(partitionTimestamp, columnIndex, columnNameTxn, colTop);
217                 }
218             } else if (colTop > 0) {
219                 // Store non-zero column tops only, zero is default
220                 // for columns added on the table creation
221                 upsert(partitionTimestamp, columnIndex, -1L, colTop);
222             }
223         }
224     }

243     private long calculateWriteOffset(long areaSize) {
244         boolean currentIsA = isCurrentA();
245         long currentOffset = currentIsA ? getOffsetA() : getOffsetB();
246         currentOffset = Math.max(currentOffset, HEADER_SIZE);
247         if (HEADER_SIZE + areaSize <= currentOffset) {
248             return HEADER_SIZE;
249         }
250         long currentSize = currentIsA ? getSizeA() : getSizeB();
251         return currentOffset + currentSize;
252     }
253 
254     private void doCommit() {
255         int entryCount = cachedList.size() / BLOCK_SIZE;
256         long areaSize = calculateSize(entryCount);
257         long writeOffset = calculateWriteOffset(areaSize);
258         bumpFileSize(writeOffset + areaSize);
259         store(entryCount, writeOffset);
260         if (isCurrentA()) {
261             updateB(writeOffset, areaSize);
262         } else {
263             updateA(writeOffset, areaSize);
264         }
265 
266         Unsafe.getUnsafe().storeFence();
267         storeNewVersion();
268     }

282     private void store(int entryCount, long offset) {
283         for (int i = 0; i < entryCount; i++) {
284             int x = i * BLOCK_SIZE;
285             mem.putLong(offset, cachedList.getQuick(x));
286             mem.putLong(offset + 8, cachedList.getQuick(x + COLUMN_INDEX_OFFSET));
287             mem.putLong(offset + 16, cachedList.getQuick(x + COLUMN_NAME_TXN_OFFSET));
288             mem.putLong(offset + 24, cachedList.getQuick(x + COLUMN_TOP_OFFSET));
289             offset += BLOCK_SIZE * 8;
290         }
291     }
