94     public void of(CairoConfiguration configuration, Path path, CharSequence name, long columnNameTxn, long unIndexedNullCount, long partitionTxn) {
95         this.unIndexedNullCount = unIndexedNullCount;
96         TableUtils.txnPartitionConditionally(path, partitionTxn);
97         final int plen = path.length();
98         this.spinLockTimeoutUs = configuration.getSpinLockTimeout();
99 
100         try {
101             this.keyMem.wholeFile(configuration.getFilesFacade(), BitmapIndexUtils.keyFileName(path, name, columnNameTxn), MemoryTag.MMAP_INDEX_READER);
102             this.clock = configuration.getMillisecondClock();
103 
104             // key file should already be created at least with header
105             long keyMemSize = this.keyMem.size();
106             if (keyMemSize < BitmapIndexUtils.KEY_FILE_RESERVED) {
107                 LOG.error().$("file too short [corrupt] ").$(path).$();
108                 throw CairoException.critical(0).put("Index file too short: ").put(path);
109             }
110 
111             // verify header signature
112             if (this.keyMem.getByte(BitmapIndexUtils.KEY_RESERVED_OFFSET_SIGNATURE) != BitmapIndexUtils.SIGNATURE) {
113                 LOG.error().$("unknown format [corrupt] ").$(path).$();
114                 throw CairoException.critical(0).put("Unknown format: ").put(path);
115             }
116 
117             // Triple check atomic read. We read first and last sequences. If they match - there is a chance at stable
118             // read. Confirm start sequence hasn't changed after values read. If it has changed - retry the whole thing.
119             int blockValueCountMod;
120             int keyCount;
121             long valueMemSize;
122             final long deadline = clock.getTicks() + spinLockTimeoutUs;
123             while (true) {
124                 long seq = this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_SEQUENCE);
125 
126                 Unsafe.getUnsafe().loadFence();
127                 if (this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_SEQUENCE_CHECK) == seq) {
128 
129                     blockValueCountMod = this.keyMem.getInt(BitmapIndexUtils.KEY_RESERVED_OFFSET_BLOCK_VALUE_COUNT) - 1;
130                     keyCount = this.keyMem.getInt(BitmapIndexUtils.KEY_RESERVED_OFFSET_KEY_COUNT);
131                     valueMemSize = this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_VALUE_MEM_SIZE);
132 
133                     Unsafe.getUnsafe().loadFence();
134                     if (this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_SEQUENCE) == seq) {
135                         break;
136                     }
137                 }
138 
139                 if (clock.getTicks() > deadline) {
140                     LOG.error().$(INDEX_CORRUPT).$(" [timeout=").$(spinLockTimeoutUs).utf8("ms]").$();
141                     throw CairoException.critical(0).put(INDEX_CORRUPT);
142                 }
143 
144                 Os.pause();
145             }
146 
147             // Resize key memory eagerly to avoid doing that when searching keys.
148             keyMem.extend(BitmapIndexUtils.getKeyEntryOffset(keyCount));
149 
150             this.blockValueCountMod = blockValueCountMod;
151             this.blockCapacity = (blockValueCountMod + 1) * 8 + BitmapIndexUtils.VALUE_BLOCK_FILE_RESERVED;
152             this.keyCount = keyCount;
153             this.keyCountIncludingNulls = unIndexedNullCount > 0 ? keyCount + 1 : keyCount;
154             this.valueMem.of(configuration.getFilesFacade(), BitmapIndexUtils.valueFileName(path.trimTo(plen), name, columnNameTxn), valueMemSize, valueMemSize, MemoryTag.MMAP_INDEX_READER);
155         } catch (Throwable e) {
156             close();
157             throw e;
158         } finally {
159             path.trimTo(plen);
160         }
161     }

164         int keyCount;
165         final long deadline = clock.getTicks() + spinLockTimeoutUs;
166         while (true) {
167             long seq = this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_SEQUENCE);
168

170             if (this.keyMem.getLong(BitmapIndexUtils.KEY_RESERVED_OFFSET_SEQUENCE_CHECK) == seq) {

172

176                 }
177             }
178

185         }
