51     @Override

56     @Override
57     public TableToken getTableToken(String dirName, int tableId) {
58         ReverseTableMapItem rmi = reverseNameTokenMap.get(dirName);
59         if (rmi != null && !rmi.isDropped() && rmi.getToken().getTableId() == tableId) {
60             return rmi.getToken();
61         }
62         return null;
63     }

65     @Override
66     public void getTableTokens(ObjList<TableToken> bucket, boolean includeDropped) {
67         bucket.clear();
68         Iterator<ReverseTableMapItem> iterator = reverseNameTokenMap.values().iterator();
69         while (iterator.hasNext()) {
70             ReverseTableMapItem entry = iterator.next();
71             if (includeDropped || !entry.isDropped()) {
72                 bucket.add(entry.getToken());
73             }
74         }
75     }
