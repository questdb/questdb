71

88      * Adds key-value pair to index. If key already exists, value is appended to end of list of existing values. Otherwise,
89      * new value list is associated with the key.
90      * <p>
91      * Index is updated atomically as far as concurrent reading is concerned. Please refer to notes on classes that
92      * are responsible for reading bitmap indexes, such as {@link BitmapIndexBwdReader}.
93      *
94      * @param key   int key
95      * @param value long value
96      */
97     public void add(int key, long value) {
98         assert key > -1 : "key must be positive integer: " + key;
99         final long offset = BitmapIndexUtils.getKeyEntryOffset(key);
100         if (key < keyCount) {
101             // when key exists we have possible outcomes in regard to the values
102             // 1. last value block has space if value cell index is not the last in block
103             // 2. value block is full, and we have to allocate a new one
104             // 3. value count is 0. This means key was created as byproduct of adding sparse key value
105             // second option is supposed to be less likely because we attempt to
106             // configure block capacity to accommodate as many values as possible
107             long valueBlockOffset = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_LAST_VALUE_BLOCK_OFFSET);
108             long valueCount = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT);
109             int valueCellIndex = (int) (valueCount & blockValueCountMod);
110             if (valueCellIndex > 0) {
111                 // this is scenario #1: key exists and there is space in last block to add value
112                 // we don't need to allocate new block, just add value and update value count on key
113                 assert valueBlockOffset + blockCapacity <= valueMemSize;
114                 appendValue(offset, valueBlockOffset, valueCount, valueCellIndex, value);
115             } else if (valueCount == 0) {
116                 // this is scenario #3: we are effectively adding a new key and creating new block
117                 initValueBlockAndStoreValue(offset, value);
118             } else {
119                 // this is scenario #2: key exists but last block is full. We need to create new block and add value there
120                 assert valueBlockOffset + blockCapacity <= valueMemSize;
121                 addValueBlockAndStoreValue(offset, valueBlockOffset, valueCount, value);
122             }
123         } else {
124             // This is a new key. Because index can have sparse keys whenever we think "key exists" we must deal
125             // with holes left by this branch, which allocates new key. All key entries that have been
126             // skipped during creation of new key will have been initialized with zeroes. This includes counts and
127             // block offsets.
128             initValueBlockAndStoreValue(offset, value);
129             // here we also need to update key count
130             // we don't just increment key count, in case this addition creates sparse key set
131             updateKeyCount(key);
132         }
133     }

319      * Rolls values back. Remove values that are strictly greater than given maximum. Empty value blocks
320      * will also be removed as well as blank space at end of value memory.
321      *
322      * @param maxValue maximum value allowed in index.
323      */
324     public void rollbackValues(long maxValue) {
325 
326         long maxValueBlockOffset = 0;
327         for (int k = 0; k < keyCount; k++) {
328             long offset = BitmapIndexUtils.getKeyEntryOffset(k);
329             long valueCount = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT);
330 
331             // do we have anything for the key?
332             if (valueCount > 0) {
333                 long blockOffset = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_LAST_VALUE_BLOCK_OFFSET);
334                 BitmapIndexUtils.seekValueBlockRTL(valueCount, blockOffset, valueMem, maxValue, blockValueCountMod, SEEKER);
335 
336                 if (valueCount != seekValueCount || blockOffset != seekValueBlockOffset) {
337                     // set new value count
338                     keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT, seekValueCount);
339 
340                     if (blockOffset != seekValueBlockOffset) {
341                         Unsafe.getUnsafe().storeFence();
342                         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT + 16, seekValueBlockOffset);
343                         Unsafe.getUnsafe().storeFence();
344                     }
345                     keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK, seekValueCount);
346                 }
347 
348                 if (seekValueBlockOffset > maxValueBlockOffset) {
349                     maxValueBlockOffset = seekValueBlockOffset;
350                 }
351             }
352         }
353         valueMemSize = maxValueBlockOffset + blockCapacity;
354         updateValueMemSize();
355         setMaxValue(maxValue);
356     }

362     private void addValueBlockAndStoreValue(long offset, long valueBlockOffset, long valueCount, long value) {
363         long newValueBlockOffset = allocateValueBlockAndStore(value);
364 
365         // update block linkage before we increase count
366         // this is important to index readers, which will act on value count they read
367 
368         // we subtract 8 because we have just written long value
369         // update this block reference to previous block
370         valueMem.putLong(valueMemSize - BitmapIndexUtils.VALUE_BLOCK_FILE_RESERVED, valueBlockOffset);
371 
372         // update previous block' "next" block reference to this block
373         valueMem.putLong(valueBlockOffset + blockCapacity - BitmapIndexUtils.VALUE_BLOCK_FILE_RESERVED + 8, newValueBlockOffset);
374 
375         // update count and last value block offset for the key
376         // in atomic fashion
377         // we make sure count is always written _after_ new value block is added
378         Unsafe.getUnsafe().storeFence();
379         keyMem.putLong(offset, valueCount + 1);
380         Unsafe.getUnsafe().storeFence();
381 
382         // don't set first block offset here
383         // it would have been done when this key was first created
384 
385         // write last block offset because it changed in this scenario
386         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_LAST_VALUE_BLOCK_OFFSET, newValueBlockOffset);
387         Unsafe.getUnsafe().storeFence();
388 
389         // write count check
390         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK, valueCount + 1);
391         Unsafe.getUnsafe().storeFence();
392 
393         // we are done adding value to new block of values
394     }

396     private long allocateValueBlockAndStore(long value) {
397         long newValueBlockOffset = valueMemSize;
398 
399         // store our value
400         valueMem.putLong(newValueBlockOffset, value);
401 
402         // reserve memory for value block
403         valueMem.skip(blockCapacity);
404

406         valueMemSize += blockCapacity;

409         // so that index can be opened correctly next time it loads

414     private void appendValue(long offset, long valueBlockOffset, long valueCount, int valueCellIndex, long value) {
415         // first set value
416         valueMem.putLong(valueBlockOffset + valueCellIndex * 8L, value);
417         Unsafe.getUnsafe().storeFence();
418         // update count and last value block offset for the key
419         // in atomic fashion
420         keyMem.putLong(offset, valueCount + 1);
421         // write count check
422         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK, valueCount + 1);
423     }

425     private void initValueBlockAndStoreValue(long offset, long value) {
426         long newValueBlockOffset = allocateValueBlockAndStore(value);
427 
428         // don't need to update linkage, value count is less than block size
429         // index readers must not access linkage information in this case
430 
431         // now update key entry in atomic fashion
432         // update count and last value block offset for the key
433         // in atomic fashion
434         Unsafe.getUnsafe().storeFence();
435         keyMem.putLong(offset, 1);
436         Unsafe.getUnsafe().storeFence();
437 
438         // first and last blocks are the same
439         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_FIRST_VALUE_BLOCK_OFFSET, newValueBlockOffset);
440         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_LAST_VALUE_BLOCK_OFFSET, newValueBlockOffset);
441         Unsafe.getUnsafe().storeFence();
442 
443         // write count check
444         keyMem.putLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK, 1);
445         Unsafe.getUnsafe().storeFence();
446     }
