590     private static long createMergeIndex(
591             long srcDataTimestampAddr,
592             long sortedTimestampsAddr,
593             long mergeDataLo,
594             long mergeDataHi,
595             long mergeOOOLo,
596             long mergeOOOHi,
597             long indexSize
598     ) {
599         // Create "index" for existing timestamp column. When we reshuffle timestamps during merge we will
600         // have to go back and find data rows we need to move accordingly
601         final long index = Unsafe.malloc(indexSize, MemoryTag.NATIVE_O3);
602         try {
603             Vect.makeTimestampIndex(srcDataTimestampAddr, mergeDataLo, mergeDataHi, index);
604             long ptr = Vect.mergeTwoLongIndexesAsc(
605                     index,
606                     mergeDataHi - mergeDataLo + 1,
607                     sortedTimestampsAddr + mergeOOOLo * 16,
608                     mergeOOOHi - mergeOOOLo + 1
609             );
610             Unsafe.recordMemAlloc(indexSize, MemoryTag.NATIVE_O3);
611             return ptr;
612         } finally {
613             Unsafe.free(index, indexSize, MemoryTag.NATIVE_O3);
614         }
