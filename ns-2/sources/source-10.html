112         private final IndexFrame indexFrame = new IndexFrame();
113         protected long next;
114         protected long position;

116         private long maxValue;

120         @Override
121         public IndexFrame getNext() {
122             if (position < valueCount) {
123                 long cellIndex = getValueCellIndex(position);
124                 long address = valueMem.addressOf(valueBlockOffset + cellIndex * Long.BYTES);
125 
126                 long pageSize = Math.min(valueCount - position, blockValueCountMod - cellIndex + 1);
127                 position += pageSize;
128                 if (position < valueCount) {
129                     // we are at edge of block right now, next value will be in next block
130                     jumpToNextValueBlock();
131                 }
132 
133                 return indexFrame.of(address, pageSize);
134             }
135 
136             return IndexFrame.NULL_INSTANCE;
137         }

139         @Override

141             if (position < valueCount) {
142                 long cellIndex = getValueCellIndex(position++);
143                 long result = valueMem.getLong(valueBlockOffset + cellIndex * 8);
144 
145                 if (result > maxValue) {
146                     valueCount = 0;
147                     return false;

149

151                     // we are at edge of block right now, next value will be in next block
152                     jumpToNextValueBlock();
153                 }

185         void of(int key, long minValue, long maxValue, long keyCount) {
186             if (keyCount == 0) {
187                 valueCount = 0;
188             } else {
189                 assert key > -1 : "key must be positive integer: " + key;
190                 long offset = BitmapIndexUtils.getKeyEntryOffset(key);
191                 keyMem.extend(offset + BitmapIndexUtils.KEY_ENTRY_SIZE);
192                 // Read value count and last block offset atomically. In that we must orderly read value count first and
193                 // value count check last. If they match - everything we read between those holds true. We must retry
194                 // should these values do not match.
195                 long valueCount;
196                 long valueBlockOffset;
197                 long lastValueBlockOffset;
198                 final long deadline = clock.getTicks() + spinLockTimeoutUs;
199                 while (true) {
200                     valueCount = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT);
201 
202                     Unsafe.getUnsafe().loadFence();
203                     if (keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_COUNT_CHECK) == valueCount) {
204                         valueBlockOffset = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_FIRST_VALUE_BLOCK_OFFSET);
205                         lastValueBlockOffset = keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_LAST_VALUE_BLOCK_OFFSET);
206 
207                         Unsafe.getUnsafe().loadFence();
208                         if (keyMem.getLong(offset + BitmapIndexUtils.KEY_ENTRY_OFFSET_VALUE_COUNT) == valueCount) {
209                             break;
210                         }
211                     }
212 
213                     if (clock.getTicks() > deadline) {
214                         LOG.error().$(INDEX_CORRUPT).$(" [timeout=").$(spinLockTimeoutUs).utf8("μs, key=").$(key).$(", offset=").$(offset).$(']').$();
215                         throw CairoException.critical(0).put(INDEX_CORRUPT);
216                     }
217                 }
218 
219                 valueMem.extend(lastValueBlockOffset + blockCapacity);
220                 this.valueCount = valueCount;
221                 if (valueCount > 0) {
222                     BitmapIndexUtils.seekValueBlockLTR(valueCount, valueBlockOffset, valueMem, minValue, blockValueCountMod, SEEKER);
223                 } else {
224                     seekValue(valueCount, valueBlockOffset);
225                 }
226 
227                 this.maxValue = maxValue;
228             }
229         }
