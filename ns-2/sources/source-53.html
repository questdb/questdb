58     public SymbolMapWriter(
59             CairoConfiguration configuration,
60             Path path,
61             CharSequence name,
62             long columnNameTxn,
63             int symbolCount,
64             int symbolIndexInTxWriter,
65             @NotNull SymbolValueCountCollector valueCountCollector
66     ) {
67         final int plen = path.length();
68         try {
69             final FilesFacade ff = configuration.getFilesFacade();
70             final long mapPageSize = configuration.getMiscAppendPageSize();
71 
72             // this constructor does not create index. Index must exist,
73             // and we use "offset" file to store "header"
74             offsetFileName(path.trimTo(plen), name, columnNameTxn);
75             if (!ff.exists(path)) {
76                 LOG.error().$(path).$(" is not found").$();
77                 throw CairoException.critical(0).put("SymbolMap does not exist: ").put(path);
78             }
79 
80             // is there enough length in "offset" file for "header"?
81             long len = ff.length(path);
82             if (len < HEADER_SIZE) {
83                 LOG.error().$(path).$(" is too short [len=").$(len).$(']').$();
84                 throw CairoException.critical(0).put("SymbolMap is too short: ").put(path);
85             }
86 
87             // open "offset" memory and make sure we start appending from where
88             // we left off. Where we left off is stored externally to symbol map
89             this.offsetMem = Vm.getWholeMARWInstance(
90                     ff,
91                     path,
92                     mapPageSize,
93                     MemoryTag.MMAP_INDEX_WRITER,
94                     configuration.getWriterFileOpenOpts()
95             );
96             // formula for calculating symbol capacity needs to be in agreement with symbol reader
97             final int symbolCapacity = offsetMem.getInt(HEADER_CAPACITY);
98             assert symbolCapacity > 0;
99             final boolean useCache = offsetMem.getBool(HEADER_CACHE_ENABLED);
100             this.offsetMem.jumpTo(keyToOffset(symbolCount) + Long.BYTES);
101 
102             // index writer is used to identify attempts to store duplicate symbol value
103             // symbol table index stores int keys and long values, e.g. value = key * 2 storage size
104             this.indexWriter = new BitmapIndexWriter(
105                     configuration,
106                     path.trimTo(plen),
107                     name,
108                     columnNameTxn,
109                     configuration.getDataIndexKeyAppendPageSize(),
110                     configuration.getDataIndexKeyAppendPageSize() * 2
111             );
112 
113             // this is the place where symbol values are stored
114             this.charMem = Vm.getWholeMARWInstance(
115                     ff,
116                     charFileName(path.trimTo(plen), name, columnNameTxn),
117                     mapPageSize,
118                     MemoryTag.MMAP_INDEX_WRITER,
119                     configuration.getWriterFileOpenOpts()
120             );
121 
122             // move append pointer for symbol values in the correct place
123             jumpCharMemToSymbolCount(symbolCount);
124 
125             // we use index hash maximum equals to half of symbol capacity, which
126             // theoretically should require 2 value cells in index per hash
127             // we use 4 cells to compensate for occasionally unlucky hash distribution
128             this.maxHash = Math.max(Numbers.ceilPow2(symbolCapacity / 2) - 1, 1);
129 
130             if (useCache) {
131                 this.cache = new CharSequenceIntHashMap(symbolCapacity, 0.3, CharSequenceIntHashMap.NO_ENTRY_VALUE);
132             } else {
133                 this.cache = null;
134             }
135 
136             this.symbolIndexInTxWriter = symbolIndexInTxWriter;
137             this.valueCountCollector = valueCountCollector;
138             LOG.debug()
139                     .$("open [name=").$(path.trimTo(plen).concat(name).$())
140                     .$(", fd=").$(this.offsetMem.getFd())
141                     .$(", cache=").$(cache != null)
142                     .$(", capacity=").$(symbolCapacity)
143                     .I$();
144         } catch (Throwable e) {
145             close();
146             throw e;
147         } finally {
148             path.trimTo(plen);
149         }
150     }

272     private int lookupAndPut(CharSequence symbol, SymbolValueCountCollector countCollector) {
273         int hash = Hash.boundedHash(symbol, maxHash);
274         RowCursor cursor = indexWriter.getCursor(hash);
275         while (cursor.hasNext()) {
276             long offsetOffset = cursor.next();
277             if (Chars.equals(symbol, charMem.getStr(offsetMem.getLong(offsetOffset)))) {
278                 return offsetToKey(offsetOffset);
279             }
280         }
281         return put0(symbol, hash, countCollector);
282     }
283 
284     private int lookupPutAndCache(int index, CharSequence symbol, SymbolValueCountCollector countCollector) {
285         int result;
286         result = lookupAndPut(symbol, countCollector);
287         cache.putAt(index, symbol.toString(), result);
288         return result;
289     }

291     private int put0(CharSequence symbol, int hash, SymbolValueCountCollector countCollector) {
292         long offsetOffset = offsetMem.getAppendOffset() - Long.BYTES;
293         offsetMem.putLong(charMem.putStr(symbol));
294         indexWriter.add(hash, offsetOffset);
295         final int symIndex = offsetToKey(offsetOffset);
296         countCollector.collectValueCount(symbolIndexInTxWriter, symIndex + 1);
297         return symIndex;
298     }
