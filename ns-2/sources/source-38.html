33      * Cursor for data frames that chronologically intersect collection of intervals.
34      * Data frame low and high row will be within intervals inclusive of edges. Intervals

38      *                       edges.
39      * @param timestampIndex index of timestamp column in the readr that is used by this cursor
40      */

42         super(intervals, timestampIndex);
43     }

45     @Override
46     public DataFrame next() {
47         // order of logical operations is important
48         // we are not calculating partition rages when intervals are empty
49         while (intervalsLo < intervalsHi && partitionLo < partitionHi) {
50             // We don't need to worry about column tops and null column because we
51             // are working with timestamp. Timestamp column cannot be added to existing table.
52             long rowCount = reader.openPartition(partitionLo);
53             if (rowCount > 0) {
54 
55                 final MemoryR column = reader.getColumn(TableReader.getPrimaryColumnIndex(reader.getColumnBase(partitionLo), timestampIndex));
56                 final long intervalLo = intervals.getQuick(intervalsLo * 2);
57                 final long intervalHi = intervals.getQuick(intervalsLo * 2 + 1);
58 
59 
60                 final long partitionTimestampLo = column.getLong(0);
61                 // interval is wholly above partition, skip interval
62                 if (partitionTimestampLo > intervalHi) {
63                     intervalsLo++;
64                     continue;
65                 }
66 
67                 final long partitionTimestampHi = column.getLong((rowCount - 1) * Long.BYTES);
68                 // interval is wholly below partition, skip partition
69                 if (partitionTimestampHi < intervalLo) {
70                     partitionLimit = 0;
71                     partitionLo++;
72                     continue;
73                 }
74 
75                 // calculate intersection
76 
77                 long lo;
78                 if (partitionTimestampLo < intervalLo) {
79                     // IntervalLo is inclusive of value. We will look for bottom index of intervalLo - 1
80                     // and then do index + 1 to skip to top of where we need to be.
81                     // We are not scanning up on the exact value of intervalLo because it may not exist. In which case
82                     // the search function will scan up to top of the lower value.
83                     lo = BinarySearch.find(column, intervalLo - 1, partitionLimit, rowCount - 1, BinarySearch.SCAN_DOWN) + 1;
84                 } else {
85                     lo = 0;
86                 }
87 
88                 final long hi;
89                 if (partitionTimestampHi > intervalHi) {
90                     hi = BinarySearch.find(column, intervalHi, lo, rowCount - 1, BinarySearch.SCAN_DOWN) + 1;
91                 } else {
92                     hi = rowCount;
93                 }
94 
95                 if (lo < hi) {
96                     dataFrame.partitionIndex = partitionLo;
97                     dataFrame.rowLo = lo;
98                     dataFrame.rowHi = hi;
99                     sizeSoFar += (hi - lo);
100 
101                     // we do have whole partition of fragment?
102                     if (hi == rowCount) {
103                         // whole partition, will need to skip to next one
104                         partitionLimit = 0;
105                         partitionLo++;
106                     } else {
107                         // only fragment, need to skip to next interval
108                         partitionLimit = hi;
109                         intervalsLo++;
110                     }
111 
112                     return dataFrame;
113                 }
114                 // interval yielded empty data frame
115                 partitionLimit = hi;
116                 intervalsLo++;
117             } else {
118                 // partition was empty, just skip to next
119                 partitionLo++;
120             }
121         }
122         return null;
123     }
