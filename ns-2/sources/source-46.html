78     public void reindex(
79             @Nullable CharSequence partitionName,
80             @Nullable CharSequence columnName
81     ) {
82         try {
83             lock(ff);
84             path.concat(TableUtils.COLUMN_VERSION_FILE_NAME).$();
85             try (ColumnVersionReader columnVersionReader = new ColumnVersionReader().ofRO(ff, path)) {
86                 final long deadline = clock.getTicks() + configuration.getSpinLockTimeout();
87                 columnVersionReader.readSafe(clock, deadline);

175     private void reindex0(
176             ColumnVersionReader columnVersionReader,
177             @Nullable CharSequence partitionName, // will reindex all partitions if partition name is not provided
178             @Nullable CharSequence columnName // will reindex all columns if name is not provided
179     ) {
180         path.trimTo(rootLen).concat(TableUtils.META_FILE_NAME);
181         try (TableReaderMetadata metadata = new TableReaderMetadata(configuration)) {
182             metadata.load(path.$());
183             // Resolve column id if the column name specified
184             final int columnIndex;
185             if (columnName != null) {
186                 columnIndex = metadata.getColumnIndex(columnName);
187             } else {
188                 columnIndex = REBUILD_ALL_COLUMNS;
189             }
190 
191             path.trimTo(rootLen);
192             final int partitionBy = metadata.getPartitionBy();
193             final DateFormat partitionDirFormatMethod = PartitionBy.getPartitionDirFormatMethod(partitionBy);
194 
195             try (TxReader txReader = new TxReader(ff).ofRO(path.concat(TXN_FILE_NAME).$(), partitionBy)) {
196                 txReader.unsafeLoadAll();
197                 path.trimTo(rootLen);
198 
199                 if (PartitionBy.isPartitioned(partitionBy)) {
200                     // Resolve partition timestamp if partition name specified
201                     if (partitionName != null) {
202                         final long partitionTimestamp = PartitionBy.parsePartitionDirName(partitionName, partitionBy);
203                         int partitionIndex = txReader.findAttachedPartitionIndexByLoTimestamp(partitionTimestamp);
204                         if (partitionIndex > -1L) {
205                             reindexPartition(
206                                     metadata,
207                                     columnVersionReader,
208                                     txReader,
209                                     columnIndex,
210                                     partitionIndex,
211                                     partitionDirFormatMethod,
212                                     partitionTimestamp
213                             );
214                         }
215                     } else {
216                         for (int partitionIndex = txReader.getPartitionCount() - 1; partitionIndex > -1; partitionIndex--) {
217                             reindexPartition(
218                                     metadata,
219                                     columnVersionReader,
220                                     txReader,
221                                     columnIndex,
222                                     partitionIndex,
223                                     partitionDirFormatMethod,
224                                     txReader.getPartitionTimestamp(partitionIndex)
225                             );
226                         }
227                     }
228                 } else {
229                     // reindexing columns in non-partitioned table
230                     reindexOneOrAllColumns(
231                             metadata,
232                             columnVersionReader,
233                             columnIndex,
234                             partitionDirFormatMethod,
235                             -1L,
236                             0L,
237                             txReader.getTransientRowCount()
238                     );
239                 }
240             }
241         } finally {
242             path.trimTo(rootLen);
243         }
244     }

246     private void reindexOneOrAllColumns(
247             RecordMetadata metadata,
248             ColumnVersionReader columnVersionReader,
249             int columnIndex,
250             DateFormat partitionDirFormatMethod,
251             long partitionNameTxn,
252             long partitionTimestamp,
253             long partitionSize
254     ) {
255         boolean isIndexed = false;
256         tempStringSink.clear();
257         partitionDirFormatMethod.format(partitionTimestamp, null, null, tempStringSink);
258 
259         if (columnIndex == REBUILD_ALL_COLUMNS) {
260             for (int i = 0, n = metadata.getColumnCount(); i < n; i++) {
261                 if (isSupportedColumn(metadata, i)) {
262                     isIndexed = true;
263                     reindexColumn(
264                             columnVersionReader,
265                             metadata,
266                             i,
267                             tempStringSink,
268                             partitionNameTxn,
269                             partitionTimestamp,
270                             partitionSize
271                     );
272                 }
273             }
274             if (!isIndexed) {
275                 throw CairoException.nonCritical().put(unsupportedTableMessage);
276             }
277         } else {
278             if (isSupportedColumn(metadata, columnIndex)) {
279                 reindexColumn(
280                         columnVersionReader,
281                         metadata,
282                         columnIndex,
283                         tempStringSink,
284                         partitionNameTxn,
285                         partitionTimestamp,
286                         partitionSize
287                 );
288             } else {
289                 throw CairoException.nonCritical().put(unsupportedColumnMessage);
290             }
291         }
292     }
