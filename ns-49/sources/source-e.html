131         ensureSeparator();
132         long p = pUtf8NameZ;
133         while (true) {
134             byte b = Unsafe.getUnsafe().getByte(p++);
135             if (b == NULL) {

179     public Path of(LPSZ other) {
180         // This is different from of(CharSequence str) because
181         // another Path is already UTF8 encoded and cannot be treated as CharSequence.
182         // Copy binary array representation instead of trying to UTF8 encode it
183         int len = other.length();
184         if (headPtr == 0L) {
185             headPtr = Unsafe.malloc(len + 1, MemoryTag.NATIVE_PATH);
186             capacity = len;

193         }
194         tailPtr = headPtr + len;
