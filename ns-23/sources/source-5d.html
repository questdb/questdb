76     public static Function newInstance(int position, Function value, int toType, int fromType) throws SqlException {
77         int fromBits = ColumnType.getGeoHashBits(fromType);
78         int toBits = ColumnType.getGeoHashBits(toType);
79         int shift = fromBits - toBits;
80         if (shift > 0) {
81             if (value.isConstant()) {
82                 long val = GeoHashes.getGeoLong(fromType, value, null);
83                 // >> shift will take care of NULL value -1
84                 return Constants.getGeoHashConstantWithType(val >> shift, toType);
85             }
86 
87             final Function result = getCastGeoHashToGeoHashFunction(value, toType, fromType, shift);
88             if (result != null) {
89                 return result;
90             }
91         } else if (shift == 0) {
92             return value;
93         }
94 
95         // check if this is a null of different bit count
96         if (value.isConstant() && GeoHashes.getGeoLong(value.getType(), value, null) == GeoHashes.NULL) {
97             return Constants.getNullConstant(toType);
98         }
99 
100         switch (ColumnType.tagOf(toType)) {
101             case ColumnType.GEOBYTE:
102             case ColumnType.GEOSHORT:
103             case ColumnType.GEOINT:
104             case ColumnType.GEOLONG:
105                 throw SqlException.position(position)
106                         .put("CAST cannot narrow values from GEOHASH(")
107                         .put(fromBits)
108                         .put("b) to GEOHASH(")
109                         .put(toBits)
110                         .put("b)");
111             case ColumnType.STRING:
112                 switch (ColumnType.tagOf(fromType)) {
113                     case ColumnType.GEOBYTE:
114                         return getGeoByteToStrCastFunction(value, fromBits);
115                     case ColumnType.GEOSHORT:
116                         return getGeoShortToStrCastFunction(value, fromBits);
117                 }
118             default:
119                 throw SqlException.position(position)
120                         .put("cannot cast GEOHASH(")
121                         .put(fromBits)
122                         .put("b) to ")
123                         .put(ColumnType.nameOf(toType));
124         }
125     }

148     private static Function getCastGeoHashToGeoHashFunction(Function value, int toType, int fromType, int shift) {
149         switch (ColumnType.tagOf(fromType)) {
150             case ColumnType.GEOBYTE:
151                 if (ColumnType.tagOf(toType) == ColumnType.GEOBYTE) {
152                     return new CastByteFunc(shift, toType, value);
153                 }
154                 break;
155             case ColumnType.GEOSHORT:
156                 switch (ColumnType.tagOf(toType)) {
157                     case ColumnType.GEOBYTE:
158                         return new CastShortToByteFunc(shift, toType, value);
159                     case ColumnType.GEOSHORT:
160                         return new CastGeoShortFunction(shift, toType, value);
161                 }
162                 break;
163             case ColumnType.GEOINT:
164                 switch (ColumnType.tagOf(toType)) {
165                     case ColumnType.GEOBYTE:
166                         return new CastIntToByteFunc(shift, toType, value);
167                     case ColumnType.GEOSHORT:
168                         return new CastGeoIntToGeoShortFunction(shift, toType, value);
169                     case ColumnType.GEOINT:
170                         return new CastIntFunc(shift, toType, value);
171                 }
172                 break;
173             default:
174                 switch (ColumnType.tagOf(toType)) {
175                     case ColumnType.GEOBYTE:
176                         return new CastLongToByteFunc(shift, toType, value);
177                     case ColumnType.GEOSHORT:
178                         return new CastGeoLongToGeoShortFunction(shift, toType, value);
179                     case ColumnType.GEOINT:
180                         return new CastLongToIntFunc(shift, toType, value);
181                     case ColumnType.GEOLONG:
182                         return new CastLongFunc(shift, toType, value);
183                 }
184         }
185         return null;
186     }
