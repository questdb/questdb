75     private static long getGeoHashImpl(CharSequence value, int position, int typeBits) throws SqlException {
76         if (value == null || value.length() == 0) {
77             return GeoHashes.NULL;
78         }
79         int actualBits = value.length() * 5;
80         if (actualBits < typeBits) {
81             throw SqlException.position(position)
82                     .put("string is too short to cast to chosen GEOHASH precision [len=").put(value.length())
83                     .put(", precision=").put(typeBits).put(']');
84         }
85         // Don't parse full string, it can be over 12 chars and result in overflow
86         int parseChars = (typeBits - 1) / 5 + 1;
87         try {
88             long lvalue = GeoHashes.fromString(value, 0, parseChars);
89             return lvalue >>> (parseChars * 5 - typeBits);
90         } catch (NumericException e) {
91             // throw SqlException if string literal is invalid geohash
92             // runtime parsing errors will result in NULL geohash
93             throw SqlException.position(position).put("invalid GEOHASH");
94         }
95     }
