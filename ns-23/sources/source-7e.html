40 public class CastLongToGeoHashFunctionFactory implements FunctionFactory {
41     @Override
42     public String getSignature() {
43         // GeoHashes are of different lengths
44         // and can be cast to lower precision
45         // for example cast(cast('questdb' as geohash(6c)) as geohash(5c))
46         return "cast(Lg)";
47     }
48 
49     @Override
50     public Function newInstance(
51             int position,
52             ObjList<Function> args,
53             IntList argPositions,
54             CairoConfiguration configuration,
55             SqlExecutionContext sqlExecutionContext
56     ) {
57         final int targetType = args.getQuick(1).getType();
58         switch (ColumnType.tagOf(targetType)) {
59             case ColumnType.GEOBYTE:
60                 return new CastGeoByteFunc(targetType, args.getQuick(0));
61             case ColumnType.GEOSHORT:
62                 return new CastGeoShortFunc(targetType, args.getQuick(0));
63             case ColumnType.GEOINT:
64                 return new CastGeoIntFunc(targetType, args.getQuick(0));
65             default:
66                 return new CastGeoLongFunc(targetType, args.getQuick(0));
67         }
68     }
