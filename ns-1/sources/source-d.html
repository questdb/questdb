68     private void appendConfigRow(SqlCompiler compiler, TableWriter configWriter, Long256 id, boolean enabled) {
69         final TableWriter.Row row = configWriter.newRow();
70         if (id == null) {
71             final MicrosecondClock clock = compiler.getEngine().getConfiguration().getMicrosecondClock();
72             final NanosecondClock nanosecondClock = compiler.getEngine().getConfiguration().getNanosecondClock();
73             final long a = nanosecondClock.getTicks();
74             final long b = clock.getTicks();
75             row.putLong256(0, a, b, 0, 0);
76             LOG.info()
77                     .$("new instance [id=").$256(a, b, 0, 0)
78                     .$(", enabled=").$(enabled)
79                     .$(']').$();
80         } else {
81             row.putLong256(0, id);
82         }
83         row.putBool(1, enabled);
84         row.putSym(2, questDBVersion);
85         row.putSym(3, System.getProperty(OS_NAME));
86         final String packageStr = System.getenv().get(QDB_PACKAGE);
87         if (packageStr != null) {
88             row.putSym(4, packageStr);
89         }
90         row.append();
91         configWriter.commit();
92     }

108     private TableWriter updateTelemetryConfig(
109             SqlCompiler compiler,
110             SqlExecutionContextImpl sqlExecutionContext,
111             TableToken tableToken
112     ) throws SqlException {
113         final TableWriter configWriter = compiler.getEngine().getWriter(AllowAllCairoSecurityContext.INSTANCE, tableToken, "telemetryConfig");
114         final CompiledQuery cc = compiler.compile(TELEMETRY_CONFIG_TABLE_NAME + " LIMIT -1", sqlExecutionContext);
115         try (
116                 final RecordCursorFactory factory = cc.getRecordCursorFactory();
117                 final RecordCursor cursor = factory.getCursor(sqlExecutionContext)
118         ) {
119             final boolean enabled = telemetryConfiguration.getEnabled();
120             if (cursor.hasNext()) {
121                 final Record record = cursor.getRecord();
122                 final boolean _enabled = record.getBool(1);
123                 final Long256 l256 = record.getLong256A(0);
124                 final CharSequence _questDBVersion = record.getSym(2);
125 
126                 // if the configuration changed to enable or disable telemetry
127                 // we need to update the table to reflect that
128                 if (enabled != _enabled || !questDBVersion.equals(_questDBVersion)) {
129                     appendConfigRow(compiler, configWriter, l256, enabled);
130                     LOG.advisory()
131                             .$("instance config changes [id=").$256(l256.getLong0(), l256.getLong1(), 0, 0)
132                             .$(", enabled=").$(enabled)
133                             .$(']').$();
134                 } else {
135                     LOG.advisory()
136                             .$("instance [id=").$256(l256.getLong0(), l256.getLong1(), 0, 0)
137                             .$(", enabled=").$(enabled)
138                             .$(']').$();
139                 }
140             } else {
141                 // if there are no record for telemetry id we need to create one using clocks
142                 appendConfigRow(compiler, configWriter, null, enabled);
143             }
144         }
145         return configWriter;
146     }
