52     public TelemetryJob(CairoEngine engine, @Nullable FunctionFactoryCache functionFactoryCache) throws SqlException {
53         telemetry = engine.getTelemetry();

55         telemetryConfigLogger = new TelemetryConfigLogger(engine);
56 
57         try (final SqlCompiler compiler = new SqlCompiler(engine, functionFactoryCache, null)) {
58             final SqlExecutionContextImpl sqlExecutionContext = new SqlExecutionContextImpl(engine, 1);
59             sqlExecutionContext.with(AllowAllCairoSecurityContext.INSTANCE, null, null);
60 
61             telemetry.init(engine, compiler, sqlExecutionContext);
62             telemetryWal.init(engine, compiler, sqlExecutionContext);
63             telemetryConfigLogger.init(engine, compiler, sqlExecutionContext);
64         }
65     }
