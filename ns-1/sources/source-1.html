419     static CharSequenceObjHashMap<String> processArgs(String... args) {
420         final int n = args.length;
421         if (n == 0) {
422             throw new BootstrapException("Arguments expected, non provided");
423         }
424         CharSequenceObjHashMap<String> optHash = new CharSequenceObjHashMap<>();
425         for (int i = 0; i < n; i++) {
426             String arg = args[i];
427             if (arg.length() > 1 && arg.charAt(0) == '-') {
428                 if (i + 1 < n) {
429                     String nextArg = args[i + 1];
430                     if (nextArg.length() > 1 && nextArg.charAt(0) == '-') {
431                         optHash.put(arg, "");
432                     } else {
433                         optHash.put(arg, nextArg);
434                         i++;
435                     }
436                 } else {
437                     optHash.put(arg, "");
438                 }
439             } else {
440                 optHash.put("$" + i, arg);
441             }
442         }
443         return optHash;

486     void extractSite() throws IOException {
487         URL resource = ServerMain.class.getResource(PUBLIC_ZIP);
488         long thisVersion = Long.MIN_VALUE;
489         if (resource == null) {
490             log.infoW().$("Web Console build [").$(PUBLIC_ZIP).$("] not found").$();
491         } else {
492             thisVersion = resource.openConnection().getLastModified();
493         }
494 
495         final String publicDir = rootDirectory + Files.SEPARATOR + "public";
496         final byte[] buffer = new byte[1024 * 1024];
497 
498         boolean extracted = false;
499         final String oldVersionStr = getPublicVersion(publicDir);
500         final CharSequence dbVersion = buildInformation.getQuestDbVersion();
501         if (oldVersionStr == null) {
502             if (thisVersion != 0) {
503                 extractSite0(publicDir, buffer, Long.toString(thisVersion));
504             } else {
505                 extractSite0(publicDir, buffer, Chars.toString(dbVersion));
506             }
507             extracted = true;
508         } else {
509             // This is a hack to deal with RT package problem
510             // in this package "thisVersion" is always 0, and we need to fall back
511             // to the database version.
512             if (thisVersion == 0) {
513                 if (!Chars.equals(oldVersionStr, dbVersion)) {
514                     extractSite0(publicDir, buffer, Chars.toString(dbVersion));
515                     extracted = true;
516                 }
517             } else {
518                 // it is possible that old version is the database version
519                 // which means user might have switched from RT distribution to no-JVM on the same data dir
520                 // in this case we might fail to parse the version string
521                 try {
522                     final long oldVersion = Numbers.parseLong(oldVersionStr);
523                     if (thisVersion > oldVersion) {
524                         extractSite0(publicDir, buffer, Long.toString(thisVersion));
525                         extracted = true;
526                     }
527                 } catch (NumericException e) {
528                     extractSite0(publicDir, buffer, Long.toString(thisVersion));
529                     extracted = true;
530                 }
531             }
532         }
533         if (!extracted) {
534             log.infoW().$("Web Console is up to date").$();
535         }
536     }
