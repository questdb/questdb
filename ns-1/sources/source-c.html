49     private final boolean enabled;

59     public Telemetry(TelemetryTypeBuilder<T> builder, CairoConfiguration configuration) {
60         TelemetryType<T> type = builder.build(configuration);
61         final TelemetryConfiguration telemetryConfiguration = type.getTelemetryConfiguration(configuration);
62         enabled = telemetryConfiguration.getEnabled();
63         if (enabled) {
64             this.telemetryType = type;
65             clock = configuration.getMicrosecondClock();
66             telemetryQueue = new RingQueue<>(type.getTaskFactory(), telemetryConfiguration.getQueueCapacity());
67             telemetryPubSeq = new MPSequence(telemetryQueue.getCycle());
68             telemetrySubSeq = new SCSequence();
69             telemetryPubSeq.then(telemetrySubSeq).then(telemetryPubSeq);
70         }
71     }
72 
73     @Override
74     public void close() {
75         if (writer == null) {
76             return;
77         }
78 
79         consumeAll();
80         Misc.free(telemetryQueue);
81 
82         telemetryType.logStatus(writer, TelemetrySystemEvent.SYSTEM_DOWN, clock.getTicks());
83         writer = Misc.free(writer);
84     }
85 
86     public void consume(T task) {
87         task.writeTo(writer, clock.getTicks());
88     }
89 
90     public void consumeAll() {
91         if (enabled && telemetrySubSeq.consumeAll(telemetryQueue, taskConsumer)) {
92             writer.commit();
93         }
94     }

96     public void init(CairoEngine engine, SqlCompiler compiler, SqlExecutionContext sqlExecutionContext) throws SqlException {
97         if (!enabled) {
98             return;
99         }
100         String tableName = telemetryType.getTableName();
101         compiler.compile(telemetryType.getCreateSql(), sqlExecutionContext);
102         final TableToken tableToken = engine.getTableToken(tableName);
103         try {
104             writer = engine.getWriter(AllowAllCairoSecurityContext.INSTANCE, tableToken, "telemetry");
105         } catch (CairoException ex) {
106             LOG.error()
107                     .$("could not open [table=`").utf8(tableToken.getTableName())
108                     .$("`, ex=").$(ex.getFlyweightMessage())
109                     .$(", errno=").$(ex.getErrno())
110                     .$(']').$();
111         }
112 
113         telemetryType.logStatus(writer, TelemetrySystemEvent.SYSTEM_UP, clock.getTicks());
114     }
