111     static int getRunWhatsLeft(
112             int queuedCount,
113             int reclaimed,
114             int workerId,
115             ObjList<VectorAggregateEntry> activeEntries,
116             SOUnboundedCountDownLatch doneLatch,
117             Log log,
118             SqlExecutionCircuitBreaker circuitBreaker,
119             AtomicBooleanCircuitBreaker sharedCB
120     ) {
121         for (int i = activeEntries.size() - 1; i > -1 && doneLatch.getCount() > -queuedCount; i--) {
122             if (circuitBreaker.checkIfTripped()) {
123                 sharedCB.cancel();
124             }
125             if (activeEntries.getQuick(i).run(workerId)) {
126                 reclaimed++;
127             }
128         }
129 
130         log.info().$("waiting for parts [queuedCount=").$(queuedCount).$(']').$();
131         doneLatch.await(queuedCount);
132         return reclaimed;
133     }

190         private void buildFunctions() {
191             final int vafCount = vafList.size();
192             final RingQueue<VectorAggregateTask> queue = bus.getVectorAggregateQueue();
193             final Sequence pubSeq = bus.getVectorAggregatePubSeq();
194 
195             sharedCircuitBreaker.reset();
196             entryPool.clear();
197             activeEntries.clear();
198             int queuedCount = 0;
199             int ownCount = 0;
200             int reclaimed = 0;
201             int total = 0;
202 
203             doneLatch.reset();
204 
205             // check if this executed via worker pool
206             final Thread thread = Thread.currentThread();
207             final int workerId;
208             if (thread instanceof Worker) {
209                 workerId = ((Worker) thread).getWorkerId();
210             } else {
211                 workerId = 0;
212             }
213 
214             try {
215                 PageFrame frame;
216                 while ((frame = pageFrameCursor.next()) != null) {
217                     for (int i = 0; i < vafCount; i++) {
218                         final VectorAggregateFunction vaf = vafList.getQuick(i);
219                         final int columnIndex = vaf.getColumnIndex();
220                         // for functions like `count()`, that do not have arguments we are required to provide
221                         // count of rows in table in a form of "pageSize >> shr". Since `vaf` doesn't provide column
222                         // this code used column 0. Assumption here that column 0 is fixed size.
223                         // This assumption only holds because our aggressive algorithm for "top down columns", e.g.
224                         // the algorithm that forces page frame to provide only columns required by the select. At the time
225                         // of writing this code there is no way to return variable length column out of non-keyed aggregation
226                         // query. This might change if we introduce something like `first(string)`. When this happens we will
227                         // need to rethink our way of computing size for the count. This would be either type checking column
228                         // 0 and working out size differently or finding any fixed-size column and using that.
229                         final long pageAddress = columnIndex > -1 ? frame.getPageAddress(columnIndex) : 0;
230                         final long pageSize = columnIndex > -1 ? frame.getPageSize(columnIndex) : frame.getPageSize(0);
231                         final int colSizeShr = columnIndex > -1 ? frame.getColumnShiftBits(columnIndex) : frame.getColumnShiftBits(0);
232                         long seq = pubSeq.next();
233                         if (seq < 0) {
234                             circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
235                             // diy the func
236                             // vaf need to know which column it is hitting in the frame and will need to
237                             // aggregate between frames until done
238                             vaf.aggregate(pageAddress, pageSize, colSizeShr, workerId);
239                             ownCount++;
240                         } else {
241                             final VectorAggregateEntry entry = entryPool.next();
242                             // null pRosti means that we do not need keyed aggregation
243                             entry.of(
244                                     queuedCount++,
245                                     vaf,
246                                     null,
247                                     0,
248                                     pageAddress,
249                                     pageSize,
250                                     colSizeShr,
251                                     doneLatch,
252                                     null,
253                                     null,
254                                     sharedCircuitBreaker
255                             );
256                             activeEntries.add(entry);
257                             queue.get(seq).entry = entry;
258                             pubSeq.done(seq);
259                         }
260                         total++;
261                     }
262                 }
263 
264                 circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
265             } catch (DataUnavailableException e) {
266                 // We're not yet done, so no need to cancel the circuit breaker. 
267                 throw e;
268             } catch (Throwable e) {
269                 sharedCircuitBreaker.cancel();
270                 throw e;
271             } finally {
272                 // all done? great start consuming the queue we just published
273                 // how do we get to the end? If we consume our own queue there is chance we will be consuming
274                 // aggregation tasks not related to this execution (we work in concurrent environment)
275                 // To deal with that we need to have our own checklist.
276 
277                 // start at the back to reduce chance of clashing
278                 reclaimed = getRunWhatsLeft(
279                         queuedCount,
280                         reclaimed,
281                         workerId,
282                         activeEntries,
283                         doneLatch,
284                         LOG,
285                         circuitBreaker,
286                         sharedCircuitBreaker
287                 );
288             }
289 
290             toTop();
291 
292             LOG.info().$("done [total=").$(total)
293                     .$(", ownCount=").$(ownCount)
294                     .$(", reclaimed=").$(reclaimed)
295                     .$(", queuedCount=").$(queuedCount).I$();
296         }
297     }
