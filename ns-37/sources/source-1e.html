56     public boolean run(int workerId) {
57         if (tryLock()) {
58             if (!circuitBreaker.checkIfTripped()
59                     && (oomCounter == null || oomCounter.get() == 0)) {
60                 if (pRosti != null) {
61                     long oldSize = Rosti.getAllocMemory(pRosti[workerId]);
62                     if (!func.aggregate(pRosti[workerId], keyAddress, valueAddress, valueCount, columnSizeShr, workerId)) {
63                         oomCounter.incrementAndGet();
64                     }
65                     raf.updateMemoryUsage(pRosti[workerId], oldSize);
66                 } else {
67                     func.aggregate(valueAddress, valueCount, columnSizeShr, workerId);
68                 }
69             }
70             doneLatch.countDown();
71             return true;
72         }
73         return false;
74     }
