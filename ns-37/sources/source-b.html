67     public GroupByRecordCursorFactory(
68             CairoConfiguration configuration,
69             RecordCursorFactory base,
70             RecordMetadata metadata,
71             @Transient ColumnTypes columnTypes,
72             int workerCount,
73             @Transient ObjList<VectorAggregateFunction> vafList,
74             int keyColumnIndexInBase,
75             int keyColumnIndexInThisCursor,
76             @Transient IntList symbolTableSkewIndex
77     ) {
78         super(metadata);
79         this.workerCount = workerCount;
80         entryPool = new ObjectPool<>(VectorAggregateEntry::new, configuration.getGroupByPoolCapacity());
81         activeEntries = new ObjList<>(configuration.getGroupByPoolCapacity());
82         // columnTypes and functions must align in the following way:
83         // columnTypes[0] is the type of key, for now single key is supported
84         // functions.size = columnTypes.size - 1, functions do not have instance for key, only for values
85         // functions[0].type == columnTypes[1]
86         // ...
87         // functions[n].type == columnTypes[n+1]
88 
89         sharedCircuitBreaker = new AtomicBooleanCircuitBreaker();
90         this.base = base;
91         // first column is INT or SYMBOL
92         pRosti = new long[workerCount];
93         final int vafCount = vafList.size();
94         this.vafList = new ObjList<>(vafCount);
95         raf = configuration.getRostiAllocFacade();
96         for (int i = 0; i < workerCount; i++) {
97             long ptr = raf.alloc(columnTypes, configuration.getGroupByMapCapacity());
98             if (ptr == 0) {
99                 for (int k = i - 1; k > -1; k--) {
100                     raf.free(pRosti[k]);
101                 }
102                 throw new OutOfMemoryError();
103             }
104             pRosti[i] = ptr;
105 
106             // remember, single key for now
107             switch (ColumnType.tagOf(columnTypes.getColumnType(0))) {
108                 case ColumnType.INT:
109                     Unsafe.getUnsafe().putInt(Rosti.getInitialValueSlot(pRosti[i], 0), Numbers.INT_NaN);
110                     break;
111                 case ColumnType.SYMBOL:
112                     Unsafe.getUnsafe().putInt(Rosti.getInitialValueSlot(pRosti[i], 0), SymbolTable.VALUE_IS_NULL);
113                     break;
114                 default:
115             }
116 
117             // configure map with default values
118             // when our execution order is sum(x) then min(y) over the same map
119             // min(y) may not find any new keys slots(they will be created by first pass with sum(x))
120             // for aggregation function to continue, such slots have to be initialized to the
121             // appropriate value for the function.
122             for (int j = 0; j < vafCount; j++) {
123                 vafList.getQuick(j).initRosti(pRosti[i]);
124             }
125         }
126 
127         // all maps are the same at this point
128         // check where our keys are and pull them to front
129         final long pRosti = this.pRosti[0];
130         final long columnOffsets = Rosti.getValueOffsets(pRosti);
131 
132         // skew logic assumes single key, for multiple keys skew would be different
133 
134         final IntList columnSkewIndex = new IntList();
135         // key is in the middle, shift aggregates before the key one position left
136         addOffsets(columnSkewIndex, vafList, 0, keyColumnIndexInThisCursor, columnOffsets);
137 
138         // this is offset of the key column
139         columnSkewIndex.add(0);
140 
141         // add remaining aggregate columns as is
142         addOffsets(columnSkewIndex, vafList, keyColumnIndexInThisCursor, vafCount, columnOffsets);
143 
144         this.vafList.addAll(vafList);
145         keyColumnIndex = keyColumnIndexInBase;
146         if (symbolTableSkewIndex.size() > 0) {
147             final IntList symbolSkew = new IntList(symbolTableSkewIndex.size());
148             symbolSkew.addAll(symbolTableSkewIndex);
149             cursor = new RostiRecordCursor(pRosti, columnSkewIndex, symbolSkew);
150         } else {
151             cursor = new RostiRecordCursor(pRosti, columnSkewIndex, null);
152         }
153     }

324         private void buildRosti() {
325             final int vafCount = vafList.size();
326             final RingQueue<VectorAggregateTask> queue = bus.getVectorAggregateQueue();
327             final Sequence pubSeq = bus.getVectorAggregatePubSeq();
328 
329             sharedCircuitBreaker.reset();
330             entryPool.clear();
331             activeEntries.clear();
332             int queuedCount = 0;
333             int ownCount = 0;
334             int reclaimed = 0;
335             int total = 0;
336 
337             doneLatch.reset();
338 
339             // check if this is executed via worker pool
340             final Thread thread = Thread.currentThread();
341             final int workerId;
342             if (thread instanceof Worker) {
343                 workerId = ((Worker) thread).getWorkerId();
344             } else {
345                 workerId = pRosti.length - 1; // avoid clashing with other worker with id=0 in tests
346             }
347 
348             try {
349                 PageFrame frame;
350                 while ((frame = pageFrameCursor.next()) != null) {
351                     final long keyAddress = frame.getPageAddress(keyColumnIndex);
352                     for (int i = 0; i < vafCount; i++) {
353                         final VectorAggregateFunction vaf = vafList.getQuick(i);
354                         // when column index = -1 we assume that vector function does not have value
355                         // argument, and it can only derive count via memory size
356                         final int columnIndex = vaf.getColumnIndex();
357                         // for functions like `count()`, that do not have arguments we are required to provide
358                         // count of rows in table in a form of "pageSize >> shr". Since `vaf` doesn't provide column
359                         // this code used column 0. Assumption here that column 0 is fixed size.
360                         // This assumption only holds because our aggressive algorithm for "top down columns", e.g.
361                         // the algorithm that forces page frame to provide only columns required by the select. At the time
362                         // of writing this code there is no way to return variable length column out of non-keyed aggregation
363                         // query. This might change if we introduce something like `first(string)`. When this happens we will
364                         // need to rethink our way of computing size for the count. This would be either type checking column
365                         // 0 and working out size differently or finding any fixed-size column and using that.
366                         final long valueAddress = columnIndex > -1 ? frame.getPageAddress(columnIndex) : 0;
367                         final int pageColIndex = columnIndex > -1 ? columnIndex : 0;
368                         final int columnSizeShr = frame.getColumnShiftBits(pageColIndex);
369                         final long valueAddressSize = frame.getPageSize(pageColIndex);
370 
371                         long seq = pubSeq.next();
372                         if (seq < 0) {
373                             circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
374                             if (keyAddress == 0) {
375                                 vaf.aggregate(valueAddress, valueAddressSize, columnSizeShr, workerId);
376                             } else {
377                                 long oldSize = Rosti.getAllocMemory(pRosti[workerId]);
378                                 if (!vaf.aggregate(pRosti[workerId], keyAddress, valueAddress, valueAddressSize, columnSizeShr, workerId)) {
379                                     oomCounter.incrementAndGet();
380                                 }
381                                 raf.updateMemoryUsage(pRosti[workerId], oldSize);
382                             }
383                             ownCount++;
384                         } else {
385                             final VectorAggregateEntry entry = entryPool.next();
386                             if (keyAddress == 0) {
387                                 entry.of(
388                                         queuedCount++,
389                                         vaf,
390                                         null,
391                                         0,
392                                         valueAddress,
393                                         valueAddressSize,
394                                         columnSizeShr,
395                                         doneLatch,
396                                         oomCounter,
397                                         null,
398                                         sharedCircuitBreaker
399                                 );
400                             } else {
401                                 entry.of(
402                                         queuedCount++,
403                                         vaf,
404                                         pRosti,
405                                         keyAddress,
406                                         valueAddress,
407                                         valueAddressSize,
408                                         columnSizeShr,
409                                         doneLatch,
410                                         oomCounter,
411                                         raf,
412                                         sharedCircuitBreaker
413                                 );
414                             }
415                             activeEntries.add(entry);
416                             queue.get(seq).entry = entry;
417                             pubSeq.done(seq);
418                         }
419                         total++;
420                     }
421                 }
422             } catch (DataUnavailableException e) {
423                 // We're not yet done, so no need to cancel the circuit breaker. 
424                 throw e;
425             } catch (Throwable e) {
426                 sharedCircuitBreaker.cancel();
427                 throw e;
428             } finally {
429                 // all done? great start consuming the queue we just published
430                 // how do we get to the end? If we consume our own queue there is chance we will be consuming
431                 // aggregation tasks not related to this execution (we work in concurrent environment)
432                 // To deal with that we need to have our own checklist.
433 
434                 // Make sure we're consuming jobs even when we failed. We cannot close "rosti" when there are
435                 // tasks in flight.
436 
437                 // start at the back to reduce chance of clashing
438                 reclaimed = GroupByNotKeyedVectorRecordCursorFactory.getRunWhatsLeft(
439                         queuedCount,
440                         reclaimed,
441                         workerId,
442                         activeEntries,
443                         doneLatch,
444                         LOG,
445                         circuitBreaker,
446                         sharedCircuitBreaker
447                 );
448                 // we can't reallocate rosti until tasks are complete because some other thread could be using it
449                 if (sharedCircuitBreaker.isCanceled()) {
450                     resetRostiMemorySize();
451                 }
452             }
453 
454             if (oomCounter.get() > 0) {
455                 resetRostiMemorySize();
456                 throw new OutOfMemoryError();
457             }
458 
459             // merge maps only when cursor was fetched successfully
460             // otherwise assume error and save CPU cycles
461             pRostiBig = pRosti[0];
462             try {
463                 if (pRosti.length > 1) {
464                     LOG.debug().$("merging").$();
465 
466                     // due to uneven load distribution some rostis could be much bigger and some empty
467                     long size = raf.getSize(pRostiBig);
468                     for (int i = 1, n = pRosti.length; i < n; i++) {
469                         long curSize = raf.getSize(pRosti[i]);
470                         if (curSize > size) {
471                             size = curSize;
472                             pRostiBig = pRosti[i];
473                         }
474                     }
475 
476                     for (int j = 0; j < vafCount; j++) {
477                         final VectorAggregateFunction vaf = vafList.getQuick(j);
478                         for (int i = 0, n = pRosti.length; i < n; i++) {
479                             if (pRostiBig == pRosti[i] || raf.getSize(pRosti[i]) < 1) {
480                                 continue;
481                             }
482                             circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
483                             long oldSize = Rosti.getAllocMemory(pRostiBig);
484                             if (!vaf.merge(pRostiBig, pRosti[i])) {
485                                 resetRostiMemorySize();
486                                 throw new OutOfMemoryError();
487                             }
488                             raf.updateMemoryUsage(pRostiBig, oldSize);
489                         }
490 
491                         circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
492 
493                         // some wrapUp() methods can increase rosti size
494                         long oldSize = Rosti.getAllocMemory(pRostiBig);
495                         if (!vaf.wrapUp(pRostiBig)) {
496                             resetRostiMemorySize();
497                             throw new OutOfMemoryError();
498                         }
499                         raf.updateMemoryUsage(pRostiBig, oldSize);
500                     }
501                     circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
502                     for (int i = 0, n = pRosti.length; i < n; i++) {
503                         if (pRostiBig == pRosti[i]) {
504                             continue;
505                         }
506 
507                         if (!raf.reset(pRosti[i], ROSTI_MINIMIZED_SIZE)) {
508                             LOG.debug().$("couldn't minimize rosti memory [i=").$(i).$(",currentSize=").$(Rosti.getSize(pRosti[i])).I$();
509                         }
510                     }
511                 } else {
512                     circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
513                     for (int j = 0; j < vafCount; j++) {
514                         if (!vafList.getQuick(j).wrapUp(pRostiBig)) {
515                             resetRostiMemorySize();
516                             throw new OutOfMemoryError();
517                         }
518                     }
519                 }
520             } catch (Throwable t) {
521                 resetRostiMemorySize();
522                 throw t;
523             }
524 
525             toTop();
526 
527             LOG.info().$("done [total=").$(total)
528                     .$(", ownCount=").$(ownCount)
529                     .$(", reclaimed=").$(reclaimed)
530                     .$(", queuedCount=").$(queuedCount).I$();
531         }
