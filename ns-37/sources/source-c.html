40 public class KSumDoubleVectorAggregateFunction extends DoubleFunction implements VectorAggregateFunction {
41     private static final int COUNT_PADDING = Misc.CACHE_LINE_SIZE / Long.BYTES;
42     // We're using two double values per worker, hence +1 element in the padding.
43     private static final int SUM_PADDING = (Misc.CACHE_LINE_SIZE / Double.BYTES) + 1;
44     private final int columnIndex;
45     private final long[] count;
46     private final DistinctFunc distinctFunc;
47     private final KeyValueFunc keyValueFunc;
48     private final double[] sum;
49     private final int workerCount;
50     private int valueOffset;
51 
52     public KSumDoubleVectorAggregateFunction(int keyKind, int columnIndex, int workerCount) {
53         this.columnIndex = columnIndex;
54         this.sum = new double[workerCount * SUM_PADDING];
55         this.count = new long[workerCount * COUNT_PADDING];
56         this.workerCount = workerCount;
57         if (keyKind == GKK_HOUR_INT) {
58             this.distinctFunc = Rosti::keyedHourDistinct;
59             this.keyValueFunc = Rosti::keyedHourKSumDouble;
60         } else {
61             this.keyValueFunc = Rosti::keyedIntKSumDouble;
62             this.distinctFunc = Rosti::keyedIntDistinct;
63         }
64     }
65 
66     @Override
67     public void aggregate(long address, long addressSize, int columnSizeHint, int workerId) {
68         if (address != 0) {
69             // Kahan compensated summation
70             final double x = Vect.sumDoubleKahan(address, addressSize / Double.BYTES);
71             if (x == x) {
72                 final int sumOffset = workerId * SUM_PADDING;
73                 final double sum = this.sum[sumOffset];
74                 final double y = x - this.sum[sumOffset + 1]; // y = x - c
75                 final double t = sum + y; // t = sum + y
76                 this.sum[sumOffset + 1] = t - sum - y; // c = t - sum - y
77                 this.sum[sumOffset] = t; // sum = t
78                 this.count[workerId * COUNT_PADDING]++;
79             }
80         }
81     }
82 
83     @Override
84     public boolean aggregate(long pRosti, long keyAddress, long valueAddress, long valueAddressSize, int columnSizeShr, int workerId) {
85         if (valueAddress == 0) {
86             return distinctFunc.run(pRosti, keyAddress, valueAddressSize / Double.BYTES);
87         } else {
88             return keyValueFunc.run(pRosti, keyAddress, valueAddress, valueAddressSize / Double.BYTES, valueOffset);
89         }
90     }

103     @Override
104     public double getDouble(Record rec) {
105         double sum = 0;
106         long count = 0;
107         double c = 0;
108         for (int i = 0; i < workerCount; i++) {
109             double y = this.sum[i * SUM_PADDING] - c;
110             double t = sum + y;
111             c = t - sum - y;
112             sum = t;
113             count += this.count[i * COUNT_PADDING];
114         }
115         return count > 0 ? sum : Double.NaN;
116     }
