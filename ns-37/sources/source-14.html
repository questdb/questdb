41 public class MinIntVectorAggregateFunction extends IntFunction implements VectorAggregateFunction {
42 
43     public static final LongBinaryOperator MIN = (long l1, long l2) -> {
44         if (l1 == Numbers.INT_NaN) {
45             return l2;
46         }
47         if (l2 == Numbers.INT_NaN) {
48             return l1;
49         }
50         return Math.min(l1, l2);
51     };
52     private final LongAccumulator accumulator = new LongAccumulator(
53             MIN, Numbers.INT_NaN
54     );
55     private final int columnIndex;
56     private final DistinctFunc distinctFunc;
57     private final KeyValueFunc keyValueFunc;
58     private int valueOffset;
59 
60     public MinIntVectorAggregateFunction(int keyKind, int columnIndex, int workerCount) {
61         this.columnIndex = columnIndex;
62         if (keyKind == GKK_HOUR_INT) {
63             this.distinctFunc = Rosti::keyedHourDistinct;
64             this.keyValueFunc = Rosti::keyedHourMinInt;
65         } else {
66             this.distinctFunc = Rosti::keyedIntDistinct;
67             this.keyValueFunc = Rosti::keyedIntMinInt;
68         }
69     }
70 
71     @Override
72     public void aggregate(long address, long addressSize, int columnSizeHint, int workerId) {
73         if (address != 0) {
74             final int value = Vect.minInt(address, addressSize / Integer.BYTES);
75             if (value != Numbers.INT_NaN) {
76                 accumulator.accumulate(value);
77             }
78         }
79     }
80 
81     @Override
82     public boolean aggregate(long pRosti, long keyAddress, long valueAddress, long valueAddressSize, int columnSizeShr, int workerId) {
83         if (valueAddress == 0) {
84             return distinctFunc.run(pRosti, keyAddress, valueAddressSize / Integer.BYTES);
85         } else {

87         }
88     }
