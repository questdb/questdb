62     @Override
63     public void aggregate(long address, long addressSize, int columnSizeHint, int workerId) {
64         if (address != 0) {
65             final long count = addressSize / (Long.BYTES * 4);
66             Long256Impl value = sumLong256(partialSums.get(), address, count);
67             if (value != Long256Impl.NULL_LONG256) {
68                 lock.lock();
69                 try {
70                     Long256Util.add(sumA, value);
71                     this.count.increment();
72                 } finally {
73                     lock.unlock();
74                 }
75             }
76         }
77     }

159     private Long256Impl sumLong256(Long256Impl sum, long address, long count) {
160         boolean hasData = false;
161         long offset = 0;
162         sum.setAll(0, 0, 0, 0);
163         for (long i = 0; i < count; i++) {
164             final long l0 = Unsafe.getUnsafe().getLong(address + offset);
165             final long l1 = Unsafe.getUnsafe().getLong(address + offset + Long.BYTES);
166             final long l2 = Unsafe.getUnsafe().getLong(address + offset + Long.BYTES * 2);
167             final long l3 = Unsafe.getUnsafe().getLong(address + offset + Long.BYTES * 3);
168 
169             boolean isNull = l0 == Numbers.LONG_NaN &&
170                     l1 == Numbers.LONG_NaN &&
171                     l2 == Numbers.LONG_NaN &&
172                     l3 == Numbers.LONG_NaN;
173 
174             if (!isNull) {
175                 Long256Util.add(sum, l0, l1, l2, l3);
176                 hasData = true;
177             }
178             offset += 4 * Long.BYTES;
179         }
180         return hasData ? sum : Long256Impl.NULL_LONG256;
