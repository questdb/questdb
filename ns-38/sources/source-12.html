37 /**
38  * Nested Loop with filter join.
39  * Iterates on master factory in outer loop and on slave factory in inner loop
40  * and returns all row pairs matching filter plus all unmatched rows from master factory.
41  */
42 public class NestedLoopLeftJoinRecordCursorFactory extends AbstractRecordCursorFactory {
43     private final NestedLoopLeftRecordCursor cursor;
44     private final Function filter;

55     ) {

57         this.masterFactory = masterFactory;
58         this.slaveFactory = slaveFactory;

61     }

63     @Override

114             this.filter = filter;

117 
118         @Override
119         public Record getRecord() {
120             return record;
121         }

123         @Override
124         public boolean hasNext() {
125             while (true) {
126                 if (isMasterHasNextPending) {
127                     masterHasNext = masterCursor.hasNext();
128                     isMasterHasNextPending = false;
129                 }
130 
131                 if (!masterHasNext) {
132                     return false;
133                 }
134 
135                 while (slaveCursor.hasNext()) {
136                     if (filter.getBool(record)) {
137                         isMatch = true;
138                         return true;
139                     }
140                 }
141 
142                 if (!isMatch) {
143                     isMatch = true;
144                     record.hasSlave(false);
145                     return true;
146                 }
147 
148                 isMatch = false;
149                 slaveCursor.toTop();
150                 record.hasSlave(true);
151                 isMasterHasNextPending = true;
152             }
153         }
154

162             masterCursor.toTop();

165             isMatch = false;
166             isMasterHasNextPending = true;
167             record.hasSlave(true);
168         }
