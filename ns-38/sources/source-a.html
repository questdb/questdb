161         public boolean hasNext() {
162             if (!isMapBuilt) {
163                 buildMapOfSlaveRecords();
164                 isMapBuilt = true;
165             }
166 
167             if (slaveChainCursor != null && slaveChainCursor.hasNext()) {
168                 slaveCursor.recordAt(slaveRecord, slaveChainCursor.next());
169                 return true;
170             }
171 
172             if (masterCursor.hasNext()) {
173                 MapKey key = joinKeyMap.withKey();
174                 key.put(masterRecord, masterKeySink);
175                 MapValue value = key.findValue();
176                 if (value != null) {
177                     slaveChainCursor = slaveChain.getCursor(value.getLong(0));
178                     // we know cursor has values
179                     // advance to get first value
180                     slaveChainCursor.hasNext();
181                     slaveCursor.recordAt(slaveRecord, slaveChainCursor.next());
182                     record.hasSlave(true);
183                 } else {
184                     slaveChainCursor = null;
185                     record.hasSlave(false);
186                 }
187                 return true;
188             }
189             return false;
190         }

208         private void buildMapOfSlaveRecords() {
209             final Record record = slaveCursor.getRecord();
210             while (slaveCursor.hasNext()) {
211                 circuitBreaker.statefulThrowExceptionIfTripped();
212 
213                 MapKey key = joinKeyMap.withKey();
214                 key.put(record, slaveKeySink);
215                 MapValue value = key.createValue();
216                 if (value.isNew()) {
217                     final long offset = slaveChain.put(record.getRowId(), -1);
218                     value.putLong(0, offset);
219                     value.putLong(1, offset);
220                 } else {
221                     value.putLong(1, slaveChain.put(record.getRowId(), value.getLong(1)));
222                 }
223             }
224         }

227             if (!isOpen) {

238             slaveChainCursor = null;
239             isMapBuilt = false;
