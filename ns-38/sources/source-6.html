150         @Override
151         public boolean hasNext() {
152             if (!isMapBuilt) {
153                 buildMapOfSlaveRecords();
154                 isMapBuilt = true;
155             }
156 
157             if (slaveChainCursor != null && slaveChainCursor.hasNext()) {
158                 slaveCursor.recordAt(slaveRecord, slaveChainCursor.next());
159                 return true;
160             }
161 
162             while (masterCursor.hasNext()) {
163                 MapKey key = joinKeyMap.withKey();
164                 key.put(masterRecord, masterKeySink);
165                 MapValue value = key.findValue();
166                 if (value != null) {
167                     slaveChainCursor = slaveChain.getCursor(value.getLong(0));
168                     // we know cursor has values
169                     // advance to get the first value
170                     slaveChainCursor.hasNext();
171                     slaveCursor.recordAt(slaveRecord, slaveChainCursor.next());
172                     return true;
173                 }
174             }
175             return false;
176         }

194         private void buildMapOfSlaveRecords() {
195             final Record record = slaveCursor.getRecord();
196             while (slaveCursor.hasNext()) {
197                 circuitBreaker.statefulThrowExceptionIfTripped();
198 
199                 MapKey key = joinKeyMap.withKey();
200                 key.put(record, slaveKeySink);
201                 MapValue value = key.createValue();
202                 if (value.isNew()) {
203                     final long offset = slaveChain.put(record.getRowId(), -1);
204                     value.putLong(0, offset);
205                     value.putLong(1, offset);
206                 } else {
207                     value.putLong(1, slaveChain.put(record.getRowId(), value.getLong(1)));
208                 }
209             }
210         }

212         private void of(RecordCursor masterCursor, RecordCursor slaveCursor, SqlExecutionCircuitBreaker circuitBreaker) {
213             if (!isOpen) {
214                 isOpen = true;
215                 joinKeyMap.reopen();
216                 slaveChain.reopen();
217             }
218             this.masterCursor = masterCursor;
219             this.slaveCursor = slaveCursor;
220             this.circuitBreaker = circuitBreaker;
221             masterRecord = masterCursor.getRecord();
222             slaveRecord = slaveCursor.getRecordB();
223             record.of(masterRecord, slaveRecord);
224             slaveChainCursor = null;
225             isMapBuilt = false;
226         }
