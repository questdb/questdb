161     private int addAlias(CharSequence tableAlias, CharSequence columnName) {
162         int dot = Chars.indexOf(columnName, '.');
163         assert dot != -1 || tableAlias != null;
164 
165         // add column with its own alias
166         MapKey key = map.withKey();
167 
168         if (dot == -1) {
169             key.putStrLowerCase(tableAlias);
170         } else {
171             assert tableAlias == null;
172             key.putStrLowerCase(columnName, 0, dot);
173         }
174         key.putStrLowerCase(columnName, dot + 1, columnName.length());
175 
176         MapValue value = key.createValue();
177         if (!value.isNew()) {
178             throw CairoException.duplicateColumn(columnName, tableAlias);
179         }
180 
181         value.putInt(0, columnCount++);
182         return dot;
183     }

185     private void addToMap(CharSequence columnName, int dot, TableColumnMetadata cm) {
186         columnMetadata.add(cm);

193         if (value.isNew()) {

197             // we would treat this lookup as if column hadn't been found.
