46 /**
47  * Splice join compares time series that do not always align on timestamp. Consider
48  * the following: we have two engines, X and Y, that produce FX prices. We need to verify
49  * that prices are the same or similar. When engine X produces price A, we need to tell
50  * what was the last price produced by Y at the time of A and vice versa, when engine Y
51  * produced price B what was the last price produced by X at the time of B.
52  * <p>
53  * Splice join does exactly that. If engines produce prices for multiple currency pairs
54  * splice join optionally join on "tags", which in above example is currency pair field.
55  * The join result will match rows where value of currency pair is the same.
56  */

201         @Override
202         public boolean hasNext() {
203             if (dualRecord) {
204                 slaveRecordLeads();
205                 dualRecord = false;
206             }
207 
208             if (fetchMaster) {
209                 if (hasMasterPending) {
210                     hasMaster = masterCursor.hasNext();
211                     hasMasterPending = false;
212                 }
213                 if (hasMaster) {
214                     final MapKey key = joinKeyMap.withKey();
215                     key.put(masterRecord, masterKeySink);
216                     final MapValue value = key.createValue();
217                     masterKeyAddress = value.getAddress();
218                     if (value.isNew()) {
219                         value.putLong(VAL_MASTER_PREV, NULL_ROWID);
220                         value.putLong(VAL_SLAVE_PREV, NULL_ROWID);
221                         value.putLong(VAL_SLAVE_NEXT, NULL_ROWID);
222                     } else {
223                         // copy current to previous
224                         value.putLong(VAL_MASTER_PREV, value.getLong(VAL_MASTER_NEXT));
225                     }
226                     value.putLong(VAL_MASTER_NEXT, masterRecord.getRowId());
227                     masterTimestamp = masterRecord.getTimestamp(masterTimestampIndex);
228                 } else {
229                     masterTimestamp = Long.MAX_VALUE;
230                 }
231             }
232 
233             if (fetchSlave) {
234                 if (hasSlavePending) {
235                     hasSlave = slaveCursor.hasNext();
236                     hasSlavePending = false;
237                 }
238                 if (hasSlave) {
239                     final MapKey key = joinKeyMap.withKey();
240                     key.put(slaveRecord, slaveKeySink);
241                     final MapValue value = key.createValue();
242                     slaveKeyAddress = value.getAddress();
243                     if (value.isNew()) {
244                         value.putLong(VAL_MASTER_PREV, NULL_ROWID);
245                         value.putLong(VAL_MASTER_NEXT, NULL_ROWID);
246                         value.putLong(VAL_SLAVE_PREV, NULL_ROWID);
247                     } else {
248                         value.putLong(VAL_SLAVE_PREV, value.getLong(VAL_SLAVE_NEXT));
249                     }
250                     value.putLong(VAL_SLAVE_NEXT, slaveRecord.getRowId());
251                     slaveTimestamp = slaveRecord.getTimestamp(slaveTimestampIndex);
252                 } else {
253                     slaveTimestamp = Long.MAX_VALUE;
254                 }
255             }
256 
257             // all suspendable calls are done, so we can reset the pending flags
258             hasMasterPending = true;
259             hasSlavePending = true;
260 
261             if (masterTimestamp < slaveTimestamp) {
262                 masterRecordLeads();
263                 fetchMaster = true;
264                 fetchSlave = false;
265             } else if (masterTimestamp > slaveTimestamp) {
266                 slaveRecordLeads();
267                 fetchMaster = false;
268                 fetchSlave = true;
269             } else {
270                 if (slaveTimestamp == Long.MAX_VALUE) {
271                     return false;
272                 }
273                 if (masterKeyAddress == slaveKeyAddress) {
274                     record.of(masterRecord, slaveRecord);
275                 } else {
276                     masterRecordLeads();
277                     dualRecord = true;
278                 }
279                 fetchMaster = true;
280                 fetchSlave = true;
281             }
282             return true;
283         }

297         private void masterRecordLeads() {
298             // lookup previous slave
299             final long rowid = joinKeyMap.valueAt(masterKeyAddress).getLong(hasSlave ? VAL_SLAVE_PREV : VAL_SLAVE_NEXT);
300             if (rowid == NULL_ROWID) {
301                 record.of(masterRecord, nullSlaveRecord);
302             } else {
303                 slaveCursor.recordAt(slaveRecord2, rowid);
304                 record.of(masterRecord, slaveRecord2);
305             }
306         }

309             joinKeyMap.clear();

311             slaveKeyAddress = -1L;
312             masterTimestamp = -1L;
313             slaveTimestamp = -1L;
314             fetchMaster = true;
315             hasMasterPending = true;
316             fetchSlave = true;
317             hasSlavePending = true;
318             // wasn't there originally
319             dualRecord = false;
320         }

322         private void slaveRecordLeads() {
323             // lookup previous master
324             final long rowid = joinKeyMap.valueAt(slaveKeyAddress).getLong(hasMaster ? VAL_MASTER_PREV : VAL_MASTER_NEXT);
325             if (rowid == NULL_ROWID) {
326                 record.of(nullMasterRecord, slaveRecord);
327             } else {
328                 masterCursor.recordAt(masterRecord2, rowid);
329                 record.of(masterRecord2, slaveRecord);
330             }
331         }

333         void of(RecordCursor masterCursor, RecordCursor slaveCursor) {
334             if (!isOpen) {
335                 isOpen = true;
336                 joinKeyMap.reopen();
337             }
338             // avoid resetting these
339             if (this.masterCursor == null) {
340                 this.masterCursor = masterCursor;
341                 this.slaveCursor = slaveCursor;
342                 masterRecord = masterCursor.getRecord();
343                 slaveRecord = slaveCursor.getRecord();
344                 masterRecord2 = masterCursor.getRecordB();
345                 slaveRecord2 = slaveCursor.getRecordB();
346             }
347             resetState();
348         }
