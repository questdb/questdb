167         @Override
168         public boolean hasNext() {
169             if (isMasterHasNextPending) {
170                 masterHasNext = masterCursor.hasNext();
171                 isMasterHasNextPending = false;
172             }
173             if (masterHasNext) {
174                 final long masterTimestamp = masterRecord.getTimestamp(masterTimestampIndex);
175                 MapKey key;
176                 MapValue value;
177                 long slaveTimestamp = this.slaveTimestamp;
178                 if (slaveTimestamp <= masterTimestamp) {
179                     if (lastSlaveRowID != Numbers.LONG_NaN) {
180                         slaveCursor.recordAt(slaveRecord, lastSlaveRowID);
181                         key = joinKeyMap.withKey();
182                         key.put(slaveRecord, slaveKeySink);
183                         value = key.createValue();
184                         value.putLong(0, lastSlaveRowID);
185                     }
186 
187                     final Record rec = slaveCursor.getRecord();
188                     while (slaveCursor.hasNext()) {
189                         slaveTimestamp = rec.getTimestamp(slaveTimestampIndex);
190                         if (slaveTimestamp <= masterTimestamp) {
191                             key = joinKeyMap.withKey();
192                             key.put(rec, slaveKeySink);
193                             value = key.createValue();
194                             value.putLong(0, rec.getRowId());
195                         } else {
196                             break;
197                         }
198                     }
199 
200                     // now we have dangling slave record, which we need to hold on to
201                     this.slaveTimestamp = slaveTimestamp;
202                     this.lastSlaveRowID = rec.getRowId();
203                 }
204                 key = joinKeyMap.withKey();
205                 key.put(masterRecord, masterKeySink);
206                 value = key.findValue();
207                 if (value != null) {
208                     slaveCursor.recordAt(slaveRecord, value.getLong(0));
209                     record.hasSlave(true);
210                 } else {
211                     record.hasSlave(false);
212                 }
213 
214                 isMasterHasNextPending = true;
215                 return true;
216             }
217             return false;
218         }
