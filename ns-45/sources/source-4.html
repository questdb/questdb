156     public DateFormat compile(CharSequence pattern, int lo, int hi, boolean generic) {
157         this.lexer.of(pattern, lo, hi);
158 
159         IntList ops;
160         ObjList<String> delimiters;
161 
162         if (generic) {
163             ops = new IntList();
164             delimiters = new ObjList<>();
165         } else {
166             ops = this.ops;
167             delimiters = this.delimiters;
168             ops.clear();
169             delimiters.clear();
170         }
171 
172         while (this.lexer.hasNext()) {
173             final CharSequence cs = lexer.next();
174             final int op = opMap.get(cs);
175             switch (op) {
176                 case -1:
177                     makeLastOpGreedy(ops);
178                     delimiters.add(Chars.toString(cs));
179                     ops.add(-(delimiters.size()));
180                     break;
181                 case OP_AM_PM:
182                 case OP_TIME_ZONE_SHORT:
183                     makeLastOpGreedy(ops);
184                     // fall thru
185                 default:
186                     ops.add(op);
187                     break;
188             }
189         }
190 
191         // make last operation "greedy"
192         makeLastOpGreedy(ops);
193         return generic ? new GenericTimestampFormat(ops, delimiters) : compile(ops, delimiters);
