28 import io.questdb.std.ConcurrentHashMap;

31 import java.util.function.Function;

33 public class TimestampFormatFactory {
34     private final static ThreadLocal<TimestampFormatCompiler> tlCompiler = ThreadLocal.withInitial(TimestampFormatCompiler::new);
35     private static final Function<CharSequence, DateFormat> mapper = TimestampFormatFactory::map;
36     private final ConcurrentHashMap<DateFormat> cache = new ConcurrentHashMap<>();
37 
38     /**
39      * Retrieves cached data format, if already exists of creates and caches new one. Concurrent behaviour is
40      * backed by ConcurrentHashMap, making method calls thread-safe and largely non-blocking.
41      * <p>
42      * Input pattern does not have to be a string, but it does have to implement hashCode/equals methods
43      * correctly. No new objects created when pattern already exists.
44      *
45      * @param pattern can be mutable and is not stored if same pattern already in cache.
46      * @return compiled implementation of DateFormat
47      */
48     public DateFormat get(CharSequence pattern) {
49         return cache.computeIfAbsent(pattern, mapper);
50     }
51 
52     private static DateFormat map(CharSequence value) {
53         return tlCompiler.get().compile(value);
54     }
