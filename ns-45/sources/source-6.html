235     public static long compute(
236             DateLocale locale,
237             int era,
238             int year,
239             int month,
240             int week,
241             int day,
242             int hour,
243             int minute,
244             int second,
245             int millis,
246             int micros,
247             int timezone,
248             long offset,
249             int hourType
250     ) throws NumericException {
251 
252         if (era == 0) {
253             year = -(year - 1);
254         }
255 
256         boolean leap = Timestamps.isLeapYear(year);
257 
258         // wrong month
259         if (month < 1 || month > 12) {
260             throw NumericException.INSTANCE;
261         }
262 
263         switch (hourType) {
264             case HOUR_PM:
265                 hour += 12;
266             case HOUR_24:
267                 // wrong hour
268                 if (hour < 0 || hour > 23) {
269                     throw NumericException.INSTANCE;
270                 }
271                 break;
272             default:
273                 // wrong 12-hour clock hour
274                 if (hour < 0 || hour > 11) {
275                     throw NumericException.INSTANCE;
276                 }
277         }
278 
279         // wrong day of month
280         if (day < 1 || day > Timestamps.getDaysPerMonth(month, leap)) {
281             throw NumericException.INSTANCE;
282         }
283 
284         if (minute < 0 || minute > 59) {
285             throw NumericException.INSTANCE;
286         }
287 
288         if (second < 0 || second > 59) {
289             throw NumericException.INSTANCE;
290         }
291 
292         if ((week <= 0 && week != -1) || week > Timestamps.getWeeks(year)) {
293             throw NumericException.INSTANCE;
294         }
295 
296         // calculate year, month, and day of ISO week
297         if (week != -1) {
298             long firstDayOfIsoWeekMicros = Timestamps.yearMicros(year, Timestamps.isLeapYear(year)) +
299                     (week - 1) * Timestamps.WEEK_MICROS +
300                     Timestamps.getIsoYearDayOffset(year) * Timestamps.DAY_MICROS;
301             month = Timestamps.getMonthOfYear(firstDayOfIsoWeekMicros);
302             year += (week == 1 && Timestamps.getIsoYearDayOffset(year) < 0) ? -1 : 0;
303             day = Timestamps.getDayOfMonth(firstDayOfIsoWeekMicros, year, month, Timestamps.isLeapYear(year));
304         }
305 
306         long datetime = Timestamps.yearMicros(year, leap)
307                 + Timestamps.monthOfYearMicros(month, leap)
308                 + (day - 1) * Timestamps.DAY_MICROS
309                 + hour * Timestamps.HOUR_MICROS
310                 + minute * Timestamps.MINUTE_MICROS
311                 + second * Timestamps.SECOND_MICROS
312                 + (long) millis * Timestamps.MILLI_MICROS
313                 + micros;
314 
315         if (timezone > -1) {
316             datetime -= locale.getZoneRules(timezone, RESOLUTION_MICROS).getOffset(datetime, year, leap);
317         } else if (offset > Long.MIN_VALUE) {
318             datetime -= offset;
319         }
320 
321         return datetime;
322     }
