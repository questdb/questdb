88         if (months == 0) {
89             return micros;
90         }

716     public static long parseNanosAsMicrosGreedy(CharSequence sequence, final int p, int lim) throws NumericException {
717         if (lim == p) {
718             throw NumericException.INSTANCE;
719         }
720 
721         boolean negative = sequence.charAt(p) == '-';
722         int i = p;
723         if (negative) {
724             i++;
725         }
726 
727         if (i >= lim || Numbers.notDigit(sequence.charAt(i))) {
728             throw NumericException.INSTANCE;
729         }
730 
731         int val = 0;
732         for (; i < lim; i++) {
733             char c = sequence.charAt(i);
734 
735             if (Numbers.notDigit(c)) {
736                 break;
737             }
738 
739             // val * 10 + (c - '0')
740             int r = (val << 3) + (val << 1) - (c - '0');
741             if (r > val) {
742                 throw NumericException.INSTANCE;
743             }
744             val = r;
745         }
746 
747         final int len = i - p;
748 
749         if (len > 9 || val == Integer.MIN_VALUE && !negative) {
750             throw NumericException.INSTANCE;
751         }
752 
753         while (i - p < 9) {
754             val *= 10;
755             i++;
756         }
757 
758         val /= 1000;
759 
760         return Numbers.encodeLowHighInts(negative ? val : -val, len);

804                                     p--;
805                                 } else {
806                                     return Long.MIN_VALUE;

808                                 break;
809                         }
810                         p++;
811                         break;
812                     case STATE_UTC:
813                         if (p > hi - 2 || Chars.noMatch(in, p, p + 2, "tc", 0, 2)) {

817                         p += 2;
818                         break;

821                             return Long.MIN_VALUE;
822                         }
823                         state = STATE_SIGN;
824                         p += 2;

828                             case '+':
829                                 break;
830                             case '-':
831                                 negative = true;
832                                 break;
833                             default:
834                                 return Long.MIN_VALUE;
835                         }
836                         p++;
837                         state = STATE_HOUR;
838                         break;
839                     case STATE_HOUR:
840                         if (isDigit(c) && p < hi - 1) {
841                             hour = Numbers.parseInt(in, p, p + 2);
842                         } else {
843                             return Long.MIN_VALUE;
844                         }
845                         state = STATE_DELIM;
846                         p += 2;
847                         break;
848                     case STATE_DELIM:

851                             p++;
852                         } else if (isDigit(c)) {
853                             state = STATE_MINUTE;
854                         } else {
855                             return Long.MIN_VALUE;
856                         }
857                         break;
858                     case STATE_MINUTE:
859                         if (isDigit(c) && p < hi - 1) {
860                             minute = Numbers.parseInt(in, p, p + 2);
861                         } else {
862                             return Long.MIN_VALUE;
863                         }
864                         p += 2;
865                         state = STATE_END;
866                         break OUT;
867                     default:
868                         return Long.MIN_VALUE;
869                 }
870             }
871         } catch (NumericException e) {

873         }
874

876             case STATE_DELIM:
877             case STATE_END:
878                 if (hour > 23 || minute > 59) {
879                     return Long.MIN_VALUE;
880                 }

883             default:
884                 return Long.MIN_VALUE;
885         }
