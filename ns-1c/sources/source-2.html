226         private void buildRecordChain() {
227             // step #1: store source cursor in record list
228             // - add record list's row ids to all trees, which will put these row ids in necessary order
229             // for this we will be using out comparator, which helps tree compare long values
230             // based on record these values are addressing
231             final Record record = base.getRecord();
232             final Record chainRecord = recordChain.getRecord();
233             final Record chainRightRecord = recordChain.getRecordB();
234             if (orderedGroupCount > 0) {
235                 while (base.hasNext()) {
236                     recordChainOffset = recordChain.put(record, recordChainOffset);
237                     recordChain.recordAt(chainRecord, recordChainOffset);
238                     for (int i = 0; i < orderedGroupCount; i++) {
239                         circuitBreaker.statefulThrowExceptionIfTripped();
240                         orderedSources.getQuick(i).put(chainRecord, recordChain, chainRightRecord, comparators.getQuick(i));
241                     }
242                 }
243             } else {
244                 while (base.hasNext()) {
245                     circuitBreaker.statefulThrowExceptionIfTripped();
246                     recordChainOffset = recordChain.put(record, recordChainOffset);
247                 }
248             }
249 
250             // step #2: populate all analytic functions with records in order of respective tree
251             // run pass1 for all ordered functions
252             long offset;
253             if (orderedGroupCount > 0) {
254                 for (int i = 0; i < orderedGroupCount; i++) {
255                     final LongTreeChain tree = orderedSources.getQuick(i);
256                     final ObjList<AnalyticFunction> functions = orderedFunctions.getQuick(i);
257                     final LongTreeChain.TreeCursor cursor = tree.getCursor();
258                     final int functionCount = functions.size();
259                     while (cursor.hasNext()) {
260                         circuitBreaker.statefulThrowExceptionIfTripped();
261                         offset = cursor.next();
262                         recordChain.recordAt(chainRecord, offset);
263                         for (int j = 0; j < functionCount; j++) {
264                             functions.getQuick(j).pass1(chainRecord, offset, recordChain);
265                         }
266                     }
267                 }
268             }
269 
270             // run pass1 for all unordered functions
271             if (unorderedFunctions != null) {
272                 for (int j = 0, n = unorderedFunctions.size(); j < n; j++) {
273                     final AnalyticFunction f = unorderedFunctions.getQuick(j);
274                     recordChain.toTop();
275                     while (recordChain.hasNext()) {
276                         circuitBreaker.statefulThrowExceptionIfTripped();
277                         f.pass1(chainRecord, chainRecord.getRowId(), recordChain);
278                     }
279                 }
280             }
281 
282             recordChain.toTop();
283         }
