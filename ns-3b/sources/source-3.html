38 public abstract class AbstractDeferredTreeSetRecordCursorFactory extends AbstractTreeSetRecordCursorFactory {
39     protected final int columnIndex;
40     protected final ObjList<Function> deferredSymbolFuncs;
41     protected final IntHashSet deferredSymbolKeys;
42     // the following two instances are shared between factory and cursor
43     // factory will be resolving symbols for cursor and if successful
44     // symbol keys will be added to this hash set
45     protected final IntHashSet symbolKeys;
46 
47     public AbstractDeferredTreeSetRecordCursorFactory(
48             @NotNull CairoConfiguration configuration,
49             @NotNull RecordMetadata metadata,
50             @NotNull DataFrameCursorFactory dataFrameCursorFactory,
51             int columnIndex,
52             @Transient ObjList<Function> keyValueFuncs,
53             @Transient SymbolMapReader symbolMapReader
54     ) {
55         super(metadata, dataFrameCursorFactory, configuration);
56 
57         // we need two data structures, int hash set for symbol keys we can resolve here
58         // and CharSequence hash set for symbols we cannot resolve
59         // we could pass all symbols to factory to resolve, but this would lead to
60         // creating Strings for symbols that we may be able to avoid doing so
61 
62         final int nKeyValues = keyValueFuncs.size();
63         final IntHashSet symbolKeys = new IntHashSet(nKeyValues);
64         IntHashSet deferredSymbolKeys = null;
65         ObjList<Function> deferredFuncs = null;
66 
67         for (int i = 0; i < nKeyValues; i++) {
68             Function symbolFunc = keyValueFuncs.get(i);
69             int symbolKey = symbolFunc.isRuntimeConstant()
70                     ? SymbolTable.VALUE_NOT_FOUND
71                     : symbolMapReader.keyOf(symbolFunc.getStr(null));
72             if (symbolKey == SymbolTable.VALUE_NOT_FOUND) {
73                 if (deferredFuncs == null) {
74                     deferredFuncs = new ObjList<>();
75                     deferredSymbolKeys = new IntHashSet();
76                 }
77                 deferredFuncs.add(symbolFunc);
78             } else {
79                 symbolKeys.add(TableUtils.toIndexKey(symbolKey));
80             }
81         }
82 
83         this.columnIndex = columnIndex;
84         this.symbolKeys = symbolKeys;
85         this.deferredSymbolKeys = deferredSymbolKeys;
86         this.deferredSymbolFuncs = deferredFuncs;
87     }
88 
89     @Override
90     protected RecordCursor getCursorInstance(
91             DataFrameCursor dataFrameCursor,
92             SqlExecutionContext executionContext
93     ) throws SqlException {
94         if (deferredSymbolFuncs != null) {
95             deferredSymbolKeys.clear();
96             StaticSymbolTable symbolTable = dataFrameCursor.getSymbolTable(cursor.getColumnIndexes().getQuick(columnIndex));
97             for (int i = 0, n = deferredSymbolFuncs.size(); i < n; i++) {
98                 Function symbolFunc = deferredSymbolFuncs.get(i);
99                 final CharSequence symbol = symbolFunc.getStr(null);
100                 int symbolKey = symbolTable.keyOf(symbol);
101                 if (symbolKey != SymbolTable.VALUE_NOT_FOUND) {
102                     deferredSymbolKeys.add(TableUtils.toIndexKey(symbolKey));
103                 }
104             }
105         }
106         return super.getCursorInstance(dataFrameCursor, executionContext);
107     }
