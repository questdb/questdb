54     public FilterOnValuesRecordCursorFactory(
55             @NotNull RecordMetadata metadata,
56             @NotNull DataFrameCursorFactory dataFrameCursorFactory,
57             @NotNull @Transient ObjList<Function> keyValues,
58             int columnIndex,
59             @NotNull @Transient TableReader reader,
60             @Nullable Function filter,
61             int orderByMnemonic,
62             boolean followedOrderByAdvice,
63             int orderDirection,
64             int indexDirection,
65             @NotNull IntList columnIndexes
66     ) {
67         super(metadata, dataFrameCursorFactory);
68         final int nKeyValues = keyValues.size();
69         this.columnIndex = columnIndex;
70         this.filter = filter;
71         this.columnIndexes = columnIndexes;
72         this.orderDirection = orderDirection;
73         cursorFactories = new ObjList<>(nKeyValues);
74         cursorFactoriesIdx = new int[]{0};
75         final SymbolMapReader symbolMapReader = reader.getSymbolMapReader(columnIndex);
76         for (int i = 0; i < nKeyValues; i++) {
77             final Function symbol = keyValues.get(i);
78             if (symbol.isConstant()) {
79                 addSymbolKey(symbolMapReader.keyOf(symbol.getStr(null)), symbol, indexDirection);
80             } else {
81                 addSymbolKey(SymbolTable.VALUE_NOT_FOUND, symbol, indexDirection);
82             }
83         }
84         if (orderByMnemonic == OrderByMnemonic.ORDER_BY_INVARIANT) {
85             rowCursorFactory = new SequentialRowCursorFactory(cursorFactories, cursorFactoriesIdx);
86             cursor = new DataFrameRecordCursorImpl(rowCursorFactory, false, filter, columnIndexes);
87         } else {
88             rowCursorFactory = new HeapRowCursorFactory(cursorFactories, cursorFactoriesIdx);
89             cursor = new DataFrameRecordCursorImpl(rowCursorFactory, false, filter, columnIndexes);
90         }
91         this.followedOrderByAdvice = followedOrderByAdvice;
92     }

127     private void addSymbolKey(int symbolKey, Function symbolFunction, int indexDirection) {
128         final FunctionBasedRowCursorFactory rowCursorFactory;
129         if (filter == null) {
130             if (symbolKey == SymbolTable.VALUE_NOT_FOUND) {
131                 rowCursorFactory = new DeferredSymbolIndexRowCursorFactory(columnIndex, symbolFunction, cursorFactories.size() == 0, indexDirection);
132             } else {
133                 rowCursorFactory = new SymbolIndexRowCursorFactory(columnIndex, symbolKey, cursorFactories.size() == 0, indexDirection, symbolFunction);
134             }
135         } else {
136             if (symbolKey == SymbolTable.VALUE_NOT_FOUND) {
137                 rowCursorFactory = new DeferredSymbolIndexFilteredRowCursorFactory(
138                         columnIndex,
139                         symbolFunction,
140                         filter,
141                         cursorFactories.size() == 0,
142                         indexDirection,
143                         columnIndexes
144                 );
145             } else {
146                 rowCursorFactory = new SymbolIndexFilteredRowCursorFactory(
147                         columnIndex,
148                         symbolKey,
149                         filter,
150                         cursorFactories.size() == 0,
151                         indexDirection,
152                         columnIndexes,
153                         symbolFunction
154                 );
155             }
156         }
157         cursorFactories.add(rowCursorFactory);

160     private void findDuplicates() {
161         //bind variable actual values might repeat so remove adjacent duplicates
162         //go through the list and if duplicate is found push it to  end of list
163         int idx = 0;
164         int max = cursorFactories.size();
165 
166         OUT:
167         while (idx < max - 1) {
168             CharSequence symbol = symbol(idx);
169             idx++;
170 
171             while (equals(symbol, symbol(idx))) {
172                 FunctionBasedRowCursorFactory tmp = cursorFactories.get(idx);
173                 cursorFactories.remove(idx);
174                 cursorFactories.add(tmp);
175 
176                 idx++;
177                 max--;
178                 if (idx >= max) {
179                     break OUT;
180                 }
181             }
182         }
183         cursorFactoriesIdx[0] = max;
184     }
