107     private static class LatestByRecordCursor implements NoRandomAccessRecordCursor {
108 
109         // contains <[latest_by columns...], [row index, timestamp column]> pairs
110         private final Map latestByMap;
111         private final int timestampIndex;
112         private RecordCursor baseCursor;
113         private Record baseRecord;
114         private SqlExecutionCircuitBreaker circuitBreaker;
115         private long index = 0;
116         private boolean isMapBuilt;
117         private boolean isOpen;
118         private RecordSink recordSink;
119         private DirectLongList rowIndexes;
120         private long rowIndexesCapacityThreshold;
121         private long rowIndexesPos = 0;

123         public LatestByRecordCursor(Map latestByMap, int timestampIndex) {

129         @Override
130         public void close() {
131             if (isOpen) {

133                 Misc.free(baseCursor);
134                 if (rowIndexes != null) {
135                     rowIndexes.clear();
136                     if (rowIndexes.getCapacity() > rowIndexesCapacityThreshold) {
137                         // This call will shrink down the underlying array
138                         rowIndexes.setCapacity(rowIndexesCapacityThreshold);
139                     }
140                 }

142             }

155         @Override
156         public boolean hasNext() {
157             if (!isMapBuilt) {
158                 buildMap();
159                 toTop();
160                 isMapBuilt = true;
161             }
162 
163             if (rowIndexesPos == rowIndexes.size()) {
164                 return false;
165             }
166 
167             final long nextIndex = rowIndexes.get(rowIndexesPos);
168             while (baseCursor.hasNext()) {
169                 circuitBreaker.statefulThrowExceptionIfTripped();
170                 if (index++ == nextIndex) {
171                     rowIndexesPos++;
172                     return true;
173                 }
174             }
175             return false;
176         }

196             baseRecord = baseCursor.getRecord();

218         private void buildMap() {
219             final Record baseRecord = baseCursor.getRecord();
220             while (baseCursor.hasNext()) {
221                 circuitBreaker.statefulThrowExceptionIfTripped();
222 
223                 final MapKey key = latestByMap.withKey();
224                 recordSink.copy(baseRecord, key);
225                 final MapValue value = key.createValue();
226 
227                 if (value.isNew()) {
228                     value.putLong(RECORD_INDEX_VALUE_IDX, index);
229                     value.putTimestamp(TIMESTAMP_VALUE_IDX, baseRecord.getTimestamp(timestampIndex));
230                 } else {
231                     long prevTimestamp = value.getTimestamp(TIMESTAMP_VALUE_IDX);
232                     long newTimestamp = baseRecord.getTimestamp(timestampIndex);
233                     if (newTimestamp >= prevTimestamp) {
234                         value.putLong(RECORD_INDEX_VALUE_IDX, index);
235                         value.putTimestamp(TIMESTAMP_VALUE_IDX, newTimestamp);
236                     }
237                 }
238 
239                 index++;
240             }
241 
242             // Copy row indexes into the long list.
243             try (final RecordCursor mapCursor = latestByMap.getCursor()) {
244                 final MapRecord mapRecord = (MapRecord) mapCursor.getRecord();
245                 while (mapCursor.hasNext()) {
246                     circuitBreaker.statefulThrowExceptionIfTripped();
247                     final MapValue value = mapRecord.getValue();
248                     final long rowId = value.getLong(RECORD_INDEX_VALUE_IDX);
249                     rowIndexes.add(rowId);
250                 }
251             }
252 
253             // Sort the indexes, so that we can use them when iterating the base cursor.
254             rowIndexes.sortAsUnsigned();
255             // Map is no longer needed, deallocate native memory.
256             latestByMap.close();
257         }
