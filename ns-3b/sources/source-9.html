114     @Override
115     public boolean hasNext() {
116         // Check for the first hasNext call.
117         if (frameIndex == -1) {
118             fetchNextFrame();
119         }
120 
121         // Check for already reached row limit.
122         if (rowsRemaining < 0) {
123             return false;
124         }
125 
126         // We have rows in the current frame we still need to dispatch
127         if (frameRowIndex < frameRowCount) {
128             record.setRowIndex(rows.get(rowIndex()));
129             frameRowIndex++;
130             return checkLimit();
131         }
132 
133         // Release the previous queue item.
134         // There is no identity check here because this check
135         // had been done when 'cursor' was assigned.
136         collectCursor(false);
137 
138         // Do we have more frames?
139         if (frameIndex < frameLimit) {
140             fetchNextFrame();
141             if (frameRowCount > 0 && frameRowIndex < frameRowCount) {
142                 record.setRowIndex(rows.get(rowIndex()));
143                 frameRowIndex++;
144                 return checkLimit();
145             }
146         }
147 
148         if (!allFramesActive) {
149             throw CairoException.nonCritical().put(exceptionMessage).setInterruption(true);
150         }
151         return false;
152     }

172         // Check if we at the top already and there is nothing to do.
173         if (frameIndex == 0 && frameRowIndex == 0) {
174             return;

184     private boolean checkLimit() {
185         if (--rowsRemaining < 0) {
186             frameSequence.cancel();
187             return false;
188         }
189         return true;
190     }

192     private void collectCursor(boolean forceCollect) {
193         if (cursor > -1) {
194             frameSequence.collect(cursor, forceCollect);
195             // It is necessary to clear 'cursor' value
196             // because we updated frameIndex and loop can exit due to lack of frames.
197             // Non-update of 'cursor' could cause double-free.
198             cursor = -1;
199         }
200     }
201 
202     private void fetchNextFrame() {
203         if (frameLimit == -1) {
204             frameSequence.prepareForDispatch();
205             frameLimit = frameSequence.getFrameCount() - 1;
206         }
207 
208         try {
209             do {
210                 cursor = frameSequence.next();
211                 if (cursor > -1) {
212                     PageFrameReduceTask task = frameSequence.getTask(cursor);
213                     LOG.debug()
214                             .$("collected [shard=").$(frameSequence.getShard())
215                             .$(", frameIndex=").$(task.getFrameIndex())
216                             .$(", frameCount=").$(frameSequence.getFrameCount())
217                             .$(", frameId=").$(frameSequence.getId())
218                             .$(", active=").$(frameSequence.isActive())
219                             .$(", cursor=").$(cursor)
220                             .I$();
221                     allFramesActive &= frameSequence.isActive();
222                     rows = task.getRows();
223                     frameRowCount = rows.size();
224                     frameIndex = task.getFrameIndex();
225                     frameRowIndex = 0;
226                     if (frameRowCount > 0 && frameSequence.isActive()) {
227                         record.setFrameIndex(task.getFrameIndex());
228                         break;
229                     } else {
230                         // Force reset frame size if frameSequence was canceled or failed.
231                         frameRowCount = 0;
232                         collectCursor(false);
233                     }
234                 } else if (cursor == -2) {
235                     break; // No frames to filter
236                 } else {
237                     Os.pause();
238                 }
239             } while (frameIndex < frameLimit);
240         } catch (Throwable e) {
241             LOG.critical().$("unexpected error [ex=").$(e).I$();
242             throw CairoException.nonCritical().put(exceptionMessage).setInterruption(true);
243         }
244     }
245 
246     private long rowIndex() {
247         return hasDescendingOrder ? (frameRowCount - frameRowIndex - 1) : frameRowIndex;
248     }
