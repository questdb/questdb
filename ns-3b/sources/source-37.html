113     private void addFoundKey(int symbolKey, BitmapIndexReader indexReader, int partitionIndex, long rowLo, long rowHi) {
114         int index = found.keyIndex(symbolKey);
115         if (index > -1) {
116             RowCursor cursor = indexReader.getCursor(false, symbolKey, rowLo, rowHi);
117             while (cursor.hasNext()) {
118                 final long row = cursor.next();
119                 recordA.setRecordIndex(row);
120                 if (filter.getBool(recordA)) {
121                     rows.add(Rows.toRowID(partitionIndex, row));
122                     found.addAt(index, symbolKey);
123                     break;
124                 }
125             }
126         }
127     }

129     private void buildTreeMap() {
130         if (keyCount < 0) {
131             keyCount = symbolKeys.size();
132             if (deferredSymbolKeys != null) {
133                 keyCount += deferredSymbolKeys.size();
134             }
135         }
136 
137         DataFrame frame;
138         // frame metadata is based on TableReader, which is "full" metadata
139         // this cursor works with subset of columns, which warrants column index remap
140         int frameColumnIndex = columnIndexes.getQuick(columnIndex);
141         while ((frame = dataFrameCursor.next()) != null && found.size() < keyCount) {
142             circuitBreaker.statefulThrowExceptionIfTripped();
143             final int partitionIndex = frame.getPartitionIndex();
144             final BitmapIndexReader indexReader = frame.getBitmapIndexReader(frameColumnIndex, BitmapIndexReader.DIR_BACKWARD);
145             final long rowLo = frame.getRowLo();
146             final long rowHi = frame.getRowHi() - 1;
147             this.recordA.jumpTo(partitionIndex, 0);
148 
149             for (int i = 0, n = symbolKeys.size(); i < n; i++) {
150                 int symbolKey = symbolKeys.get(i);
151                 addFoundKey(symbolKey, indexReader, partitionIndex, rowLo, rowHi);
152             }
153             if (deferredSymbolKeys != null) {
154                 for (int i = 0, n = deferredSymbolKeys.size(); i < n; i++) {
155                     int symbolKey = deferredSymbolKeys.get(i);
156                     if (!symbolKeys.contains(symbolKey)) {
157                         addFoundKey(symbolKey, indexReader, partitionIndex, rowLo, rowHi);
158                     }
159                 }
160             }
161         }
162 
163         rows.sortAsUnsigned();
164         index = 0;
165         lim = rows.size();
166     }
