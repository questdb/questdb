98     private void lookupDeferredSymbols(DataFrameCursor dataFrameCursor, SqlExecutionContext executionContext) throws SqlException {
99         // If symbol values are restricted by a list in the query by syntax
100         // sym in ('val1', 'val2', 'val3')
101         // or similar we need to resolve string values into int symbol keys to search the table faster.
102         // Resolve values to int keys and save them in cursor.getSymbolKeys() set.
103         if (includedSymbolFuncs != null) {
104             IntHashSet symbolKeys = cursor.getIncludedSymbolKeys();
105             symbolKeys.clear();
106             StaticSymbolTable symbolMapReader = dataFrameCursor.getSymbolTable(frameSymbolIndex);
107             for (int i = 0, n = includedSymbolFuncs.size(); i < n; i++) {
108                 Function symbolFunc = includedSymbolFuncs.getQuick(i);
109                 symbolFunc.init(dataFrameCursor, executionContext);
110                 int key = symbolMapReader.keyOf(symbolFunc.getStr(null));
111                 if (key != SymbolTable.VALUE_NOT_FOUND) {
112                     symbolKeys.add(key);
113                 }
114             }
115         }
116         // Do the same with not in keys.
117         if (excludedSymbolFuncs != null) {
118             IntHashSet symbolKeys = cursor.getExcludedSymbolKeys();
119             symbolKeys.clear();
120             StaticSymbolTable symbolMapReader = dataFrameCursor.getSymbolTable(frameSymbolIndex);
121             for (int i = 0, n = excludedSymbolFuncs.size(); i < n; i++) {
122                 Function symbolFunc = excludedSymbolFuncs.getQuick(i);
123                 symbolFunc.init(dataFrameCursor, executionContext);
124                 int key = symbolMapReader.keyOf(symbolFunc.getStr(null));
125                 if (key != SymbolTable.VALUE_NOT_FOUND) {
126                     symbolKeys.add(key);
127                 }
128             }
129         }
130     }
