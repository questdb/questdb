127     @Override
128     public RecordCursor getCursor(SqlExecutionContext executionContext) throws SqlException {
129         long rowsRemaining;
130         int baseOrder = base.hasDescendingOrder() ? ORDER_DESC : ORDER_ASC;
131         final int order;
132         if (limitLoFunction != null) {
133             limitLoFunction.init(frameSequence.getSymbolTableSource(), executionContext);
134             rowsRemaining = limitLoFunction.getLong(null);
135             // on negative limit we will be looking for positive number of rows
136             // while scanning table from the highest timestamp to the lowest
137             if (rowsRemaining > -1) {
138                 order = baseOrder;
139             } else {
140                 order = reverse(baseOrder);
141                 rowsRemaining = -rowsRemaining;
142             }
143         } else {
144             rowsRemaining = Long.MAX_VALUE;
145             order = baseOrder;
146         }
147 
148         if (order == ORDER_DESC && rowsRemaining != Long.MAX_VALUE) {
149             if (rowsRemaining > maxNegativeLimit) {
150                 throw SqlException.position(limitLoPos).put("absolute LIMIT value is too large, maximum allowed value: ").put(maxNegativeLimit);
151             }
152             if (negativeLimitRows == null) {
153                 negativeLimitRows = new DirectLongList(maxNegativeLimit, MemoryTag.NATIVE_OFFLOAD);
154             }
155             negativeLimitCursor.of(execute(executionContext, collectSubSeq, order), rowsRemaining, negativeLimitRows);
156             return negativeLimitCursor;
157         }
158 
159         cursor.of(execute(executionContext, collectSubSeq, order), rowsRemaining);
160         return cursor;
161     }

214     private static void filter(
215             int workerId,
216             @NotNull PageAddressCacheRecord record,
217             @NotNull PageFrameReduceTask task,
218             @NotNull SqlExecutionCircuitBreaker circuitBreaker,
219             @Nullable PageFrameSequence<?> stealingFrameSequence
220     ) {
221         final DirectLongList rows = task.getRows();
222         final DirectLongList columns = task.getColumns();
223         final long frameRowCount = task.getFrameRowCount();
224         final AsyncJitFilterAtom atom = task.getFrameSequence(AsyncJitFilterAtom.class).getAtom();
225         final PageAddressCache pageAddressCache = task.getPageAddressCache();
226 
227         rows.clear();
228 
229         if (pageAddressCache.hasColumnTops(task.getFrameIndex())) {
230             // Use Java-based filter in case of a page frame with column tops.
231             final boolean owner = stealingFrameSequence != null && stealingFrameSequence == task.getFrameSequence();
232             final int filterId = atom.acquireFilter(workerId, owner, circuitBreaker);
233             final Function filter = atom.getFilter(filterId);
234             try {
235                 for (long r = 0; r < frameRowCount; r++) {
236                     record.setRowIndex(r);
237                     if (filter.getBool(record)) {
238                         rows.add(r);
239                     }
240                 }
241                 return;
242             } finally {
243                 atom.releaseFilter(filterId);
244             }
245         }
246 
247         // Use JIT-compiled filter.
248 
249         final long columnCount = pageAddressCache.getColumnCount();
250         if (columns.getCapacity() < columnCount) {
251             columns.setCapacity(columnCount);
252         }
253         columns.clear();
254         for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
255             columns.add(pageAddressCache.getPageAddress(task.getFrameIndex(), columnIndex));
256         }
257 
258         final long rowCount = task.getFrameRowCount();
259         if (rows.getCapacity() < rowCount) {
260             rows.setCapacity(rowCount);
261         }
262 
263         long hi = atom.compiledFilter.call(
264                 columns.getAddress(),
265                 columns.size(),
266                 atom.bindVarMemory.getAddress(),
267                 atom.bindVarFunctions.size(),
268                 rows.getAddress(),
269                 rowCount,
270                 0
271         );
272         rows.setPos(hi);
273 
274         // Pre-touch fixed-size columns, if asked.
275         atom.preTouchColumns(record, rows);
276     }

324         private void prepareBindVarMemory(SymbolTableSource symbolTableSource, SqlExecutionContext executionContext) throws SqlException {
325             //don't trigger memory allocation if there are no variables 
326             if (bindVarFunctions.size() > 0) {
327                 bindVarMemory.truncate();
328                 for (int i = 0, n = bindVarFunctions.size(); i < n; i++) {
329                     Function function = bindVarFunctions.getQuick(i);
330                     writeBindVarFunction(function, symbolTableSource, executionContext);
331                 }

335         private void writeBindVarFunction(
336                 Function function,
337                 SymbolTableSource symbolTableSource,
338                 SqlExecutionContext executionContext
339         ) throws SqlException {
340             final int columnType = function.getType();
341             final int columnTypeTag = ColumnType.tagOf(columnType);
342             switch (columnTypeTag) {
343                 case ColumnType.BOOLEAN:
344                     bindVarMemory.putLong(function.getBool(null) ? 1 : 0);
345                     return;
346                 case ColumnType.BYTE:
347                     bindVarMemory.putLong(function.getByte(null));
348                     return;
349                 case ColumnType.GEOBYTE:
350                     bindVarMemory.putLong(function.getGeoByte(null));
351                     return;
352                 case ColumnType.SHORT:
353                     bindVarMemory.putLong(function.getShort(null));
354                     return;
355                 case ColumnType.GEOSHORT:
356                     bindVarMemory.putLong(function.getGeoShort(null));
357                     return;
358                 case ColumnType.CHAR:
359                     bindVarMemory.putLong(function.getChar(null));
360                     return;
361                 case ColumnType.INT:
362                     bindVarMemory.putLong(function.getInt(null));
363                     return;
364                 case ColumnType.GEOINT:
365                     bindVarMemory.putLong(function.getGeoInt(null));
366                     return;
367                 case ColumnType.SYMBOL:
368                     assert function instanceof CompiledFilterSymbolBindVariable;
369                     function.init(symbolTableSource, executionContext);
370                     bindVarMemory.putLong(function.getInt(null));
371                     return;
372                 case ColumnType.FLOAT:
373                     // compiled filter function will read only the first word
374                     bindVarMemory.putFloat(function.getFloat(null));
375                     bindVarMemory.putFloat(Float.NaN);
376                     return;
377                 case ColumnType.LONG:
378                     bindVarMemory.putLong(function.getLong(null));
379                     return;
380                 case ColumnType.GEOLONG:
381                     bindVarMemory.putLong(function.getGeoLong(null));
382                     return;
383                 case ColumnType.DATE:
384                     bindVarMemory.putLong(function.getDate(null));
385                     return;
386                 case ColumnType.TIMESTAMP:
387                     bindVarMemory.putLong(function.getTimestamp(null));
388                     return;
389                 case ColumnType.DOUBLE:
390                     bindVarMemory.putDouble(function.getDouble(null));
391                     return;
392                 default:
393                     throw SqlException.position(0).put("unsupported bind variable type: ").put(ColumnType.nameOf(columnTypeTag));
394             }
395         }
