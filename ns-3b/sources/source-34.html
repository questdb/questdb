107     @Override
108     public void of(DataFrameCursor dataFrameCursor, SqlExecutionContext executionContext) throws SqlException {
109         this.dataFrameCursor = dataFrameCursor;
110         recordA.of(dataFrameCursor.getTableReader());
111         recordB.of(dataFrameCursor.getTableReader());
112         circuitBreaker = executionContext.getCircuitBreaker();
113         dataFrameCursor.toTop();
114         foundSize = 0;
115         foundKeys.clear();
116         rowIds.clear();
117         if (filter != null) {
118             filter.init(this, executionContext);
119             filter.toTop();
120         }
121         if (restrictedByIncludedValues) {
122             if (includedSymbolKeys.size() > 0) {
123                 // Find only restricted set of symbol keys
124                 rowIds.setCapacity(includedSymbolKeys.size());
125             }
126         } else if (restrictedByExcludedValues) {
127             // Find all, but excluded set of symbol keys
128             StaticSymbolTable symbolTable = dataFrameCursor.getSymbolTable(columnIndexes.getQuick(columnIndex));
129             int distinctSymbols = symbolTable.getSymbolCount();
130             if (symbolTable.containsNullValue()) {
131                 distinctSymbols++;
132             } else if (excludedSymbolKeys.contains(SymbolTable.VALUE_IS_NULL)) {
133                 // The excluded set contains a null while the symbol table doesn't.
134                 // Increment the counter to avoid miscalculation.
135                 distinctSymbols++;
136             }
137             distinctSymbols = Math.max(0, distinctSymbols - excludedSymbolKeys.size());
138             rowIds.setCapacity(distinctSymbols);
139         } else {
140             // Find latest by all distinct symbol values
141             StaticSymbolTable symbolTable = dataFrameCursor.getSymbolTable(columnIndexes.getQuick(columnIndex));
142             int distinctSymbols = symbolTable.getSymbolCount();
143             if (symbolTable.containsNullValue()) {
144                 distinctSymbols++;
145             }
146             rowIds.setCapacity(distinctSymbols);
147         }
148         areRecordsFound = false;
149     }

210     private void findRecords() {
211         // Find all record IDs and save in rowIds in descending order.
212         // Then return row by row in ascending timestamp order
213         // since most of the time factory is supposed to return in ASC timestamp order.
214         // It can be optimised later on to not buffer row IDs and return in desc order.
215         if (restrictedByIncludedValues) {
216             if (includedSymbolKeys.size() > 0) {
217                 // Find only restricted set of symbol keys
218                 if (filter != null) {
219                     findRestrictedWithFilter();
220                 } else {
221                     findRestrictedNoFilter();
222                 }
223             }
224         } else if (restrictedByExcludedValues) {
225             // Find all, but excluded set of symbol keys
226             int distinctSymbols = (int) rowIds.getCapacity();
227             if (filter != null) {
228                 findRestrictedExcludedOnlyWithFilter(distinctSymbols);
229             } else {
230                 findRestrictedExcludedOnlyNoFilter(distinctSymbols);
231             }
232         } else {
233             // Find latest by all distinct symbol values
234             int distinctSymbols = (int) rowIds.getCapacity();
235             if (distinctSymbols > 0) {
236                 if (filter != null) {
237                     findAllWithFilter(distinctSymbols);
238                 } else {
239                     findAllNoFilter(distinctSymbols);
240                 }
241             }
242         }
243     }
