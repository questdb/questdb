36 /**
37  * Returns rows from current data frame in table (physical) order :
38  * - fetches first record index per cursor into priority queue
39  * - then returns record with smallest index and adds next record from related cursor into queue
40  * until all cursors are exhausted .
41  */
42 public class HeapRowCursorFactory implements RowCursorFactory {
43     private final HeapRowCursor cursor;
44     private final ObjList<? extends RowCursorFactory> cursorFactories;
45     // used to skip some cursor factories if values repeat
46     private final int[] cursorFactoriesIdx;
47     private final ObjList<RowCursor> cursors;
48 
49     public HeapRowCursorFactory(ObjList<? extends RowCursorFactory> cursorFactories, int[] cursorFactoriesIdx) {
50         this.cursorFactories = cursorFactories;
51         this.cursors = new ObjList<>();
52         this.cursor = new HeapRowCursor();
53         this.cursorFactoriesIdx = cursorFactoriesIdx;
54     }
55 
56     @Override
57     public RowCursor getCursor(DataFrame dataFrame) {
58         for (int i = 0, n = cursorFactories.size(); i < n; i++) {
59             cursors.extendAndSet(i, cursorFactories.getQuick(i).getCursor(dataFrame));
60         }
61         cursor.of(cursors, cursorFactoriesIdx[0]);
62         return cursor;
63     }
64 
65     @Override
66     public void init(TableReader tableReader, SqlExecutionContext sqlExecutionContext) throws SqlException {
67         RowCursorFactory.init(cursorFactories, tableReader, sqlExecutionContext);
68     }
69 
70     @Override
71     public boolean isEntity() {
72         return false;
73     }
74 
75     @Override
76     public void prepareCursor(TableReader tableReader) {
77         RowCursorFactory.prepareCursor(cursorFactories, tableReader);
78     }
79 
80     @Override
81     public void toPlan(PlanSink sink) {
82         sink.type("Table-order scan");
83         for (int i = 0, n = cursorFactories.size(); i < n; i++) {
84             sink.child(cursorFactories.getQuick(i));
85         }
86     }
87 }
