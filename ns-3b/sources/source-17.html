200         private void buildFactories() {
201             final StaticSymbolTable symbolTable = delegate.getDataFrameCursor().getSymbolTable(columnIndex);
202             final Record record = baseCursor.getRecord();
203             while (baseCursor.hasNext()) {
204                 final CharSequence symbol = func.get(record, 0);
205                 int symbolKey = symbolTable.keyOf(symbol);
206                 if (symbolKey != SymbolTable.VALUE_NOT_FOUND) {
207                     final int targetIndex = targetFactories.keyIndex(symbolKey);
208                     if (targetIndex > -1) {
209                         final RowCursorFactory rowCursorFactory;
210                         final int index = factories.keyIndex(symbolKey);
211                         if (index < 0) {
212                             rowCursorFactory = factories.valueAtQuick(index);
213                         } else {
214                             // we could be constantly re-hashing factories, which is why
215                             // we cannot reliably tell that one of them could be using cursor that
216                             // belongs to index reader
217                             if (filter == null) {
218                                 rowCursorFactory = new SymbolIndexRowCursorFactory(
219                                         columnIndex,
220                                         symbolKey,
221                                         false,
222                                         BitmapIndexReader.DIR_FORWARD,
223                                         null
224                                 );
225                             } else {

236                         }
