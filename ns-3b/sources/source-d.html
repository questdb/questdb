94         if (reenterDataFrame) {

96         }
97         DataFrame dataFrame = dataFrameCursor.next();
98         if (dataFrame != null) {

135     private TableReaderPageFrame computeFrame(final long partitionLo, final long partitionHi) {
136         final int base = reader.getColumnBase(reenterPartitionIndex);
137 
138         // we may need to split this data frame either along "top" lines, or along
139         // max page frame sizes; to do this, we calculate min top value from given position
140         long adjustedLo = Math.max(partitionLo, partitionHi - currentPageFrameRowLimit);
141         for (int i = 0; i < columnCount; i++) {
142             final int columnIndex = columnIndexes.getQuick(i);
143             long top = reader.getColumnTop(base, columnIndex);
144             if (top > adjustedLo && top < partitionHi) {
145                 adjustedLo = top;
146             }
147         }
148 
149         for (int i = 0; i < columnCount; i++) {
150             final int columnIndex = columnIndexes.getQuick(i);
151             final int readerColIndex = TableReader.getPrimaryColumnIndex(base, columnIndex);
152             final MemoryR col = reader.getColumn(readerColIndex);
153             // when the entire column is NULL we make it skip the whole of the data frame
154             final long top = col instanceof NullMemoryMR ? partitionHi : reader.getColumnTop(base, columnIndex);
155             final long partitionLoAdjusted = adjustedLo - top;
156             final long partitionHiAdjusted = partitionHi - top;
157             final int sh = columnSizes.getQuick(i);
158 
159             if (partitionHiAdjusted > 0) {
160                 if (sh > -1) {
161                     // this assumes reader uses single page to map the whole column
162                     // non-negative sh means fixed length column
163                     long address = col.getPageAddress(0);
164                     long addressSize = partitionHiAdjusted << sh;
165                     long offset = partitionLoAdjusted << sh;
166                     columnPageAddress.setQuick(i * 2, address + offset);
167                     pageSizes.setQuick(i * 2, addressSize - offset);
168                 } else {
169                     final MemoryR fixCol = reader.getColumn(readerColIndex + 1);
170                     long fixAddress = fixCol.getPageAddress(0);
171                     long fixAddressSize = partitionHiAdjusted << 3;
172                     long fixOffset = partitionLoAdjusted << 3;
173 
174                     long varAddress = col.getPageAddress(0);
175                     long varAddressSize = Unsafe.getUnsafe().getLong(fixAddress + fixAddressSize);
176 
177                     columnPageAddress.setQuick(i * 2, varAddress);
178                     columnPageAddress.setQuick(i * 2 + 1, fixAddress + fixOffset);
179                     pageSizes.setQuick(i * 2, varAddressSize);
180                     pageSizes.setQuick(i * 2 + 1, fixAddressSize - fixOffset);
181                 }
182             } else {
183                 columnPageAddress.setQuick(i * 2, 0);
184                 columnPageAddress.setQuick(i * 2 + 1, 0);
185                 pageSizes.setQuick(i * 2, (partitionHiAdjusted - partitionLoAdjusted) << (sh > -1 ? sh : 3));
186                 pageSizes.setQuick(i * 2 + 1, 0);
187             }
188         }
189 
190         // it is possible that all columns in data frame are empty, but it doesn't mean
191         // the data frame size is 0; sometimes we may want to imply nulls
192         if (partitionLo < adjustedLo) {
193             this.reenterPartitionLo = partitionLo;
194             this.reenterPartitionHi = adjustedLo;
195             this.reenterDataFrame = true;
196         } else {
197             this.reenterDataFrame = false;
198         }
199 
200         frame.partitionLo = adjustedLo;
201         frame.partitionHi = partitionHi;
202         frame.partitionIndex = reenterPartitionIndex;
203         return frame;
204     }
