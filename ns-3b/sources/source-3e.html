121         private boolean selectPoolEntry() {
122             for (; ; ) {
123                 if (poolEntry == null) {
124                     // either we just started the iteration or the last Entry did not have
125                     // anything chained. let's advance in the CHM iterator
126                     assert allocationIndex == 0;
127                     if (!iterator.hasNext()) {
128                         return false;
129                     }
130                     Map.Entry<CharSequence, AbstractMultiTenantPool.Entry<ReaderPool.R>> mapEntry = iterator.next();
131                     poolEntry = mapEntry.getValue();
132                     return true;
133                 } else if (allocationIndex == ReaderPool.ENTRY_SIZE) {
134                     // we exhausted all slots in the current Entry
135                     // let's see if there is another Entry chained
136                     poolEntry = poolEntry.getNext();
137                     allocationIndex = 0;
138                 } else {
139                     // the current entry still has slots to inspect
140                     return true;
141                 }
142             }
143         }
144 
145         private boolean tryAdvance() {
146             TableReader reader;
147             do {
148                 if (!selectPoolEntry()) {
149                     return false;
150                 }
151                 // Volatile read because the owner is set by other thread(s).
152                 // It ensures everything the other thread wrote to other fields before (volatile) writing the 'owner'
153                 // will become visible too (Synchronizes-With Order).
154                 // This does not imply the individual fields will be consistent. Why? There are other threads modifying
155                 // entries concurrently with us reading them. So e.g. the 'reader' field can be changed after we read
156                 // 'owner'. So what's the point of volatile read? It ensures we don't read arbitrary stale data.
157                 owner = poolEntry.getOwnerVolatile(allocationIndex);
158                 timestamp = poolEntry.getReleaseOrAcquireTime(allocationIndex);
159                 reader = poolEntry.getTenant(allocationIndex);
160                 allocationIndex++;
161             } while (reader == null);
162             txn = reader.getTxn();
163             tableToken = reader.getTableToken();
164             return true;
165         }
