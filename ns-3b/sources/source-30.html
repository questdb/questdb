99     private void findRecord() {
100         DataFrame frame;
101         // frame metadata is based on TableReader, which is "full" metadata
102         // this cursor works with subset of columns, which warrants column index remap
103         int frameColumnIndex = columnIndexes.getQuick(columnIndex);
104         while ((frame = dataFrameCursor.next()) != null) {
105             circuitBreaker.statefulThrowExceptionIfTripped();
106             final int partitionIndex = frame.getPartitionIndex();
107             final BitmapIndexReader indexReader = frame.getBitmapIndexReader(frameColumnIndex, BitmapIndexReader.DIR_BACKWARD);
108             final long rowLo = frame.getRowLo();
109             final long rowHi = frame.getRowHi() - 1;
110             this.recordA.jumpTo(partitionIndex, 0);
111 
112             RowCursor cursor = indexReader.getCursor(false, symbolKey, rowLo, rowHi);
113             while (cursor.hasNext()) {
114                 circuitBreaker.statefulThrowExceptionIfTripped();
115                 recordA.setRecordIndex(cursor.next());
116                 if (filter.getBool(recordA)) {
117                     isRecordFound = true;
118                     return;
119                 }
120             }
121         }
