101     private long countSymbolCombinations() {
102         long combinations = 1;
103         for (int i = 0, n = partitionByColumnIndexes.size(); i < n; i++) {
104             int symbolCount = partitionBySymbolCounts != null ? partitionBySymbolCounts.getQuick(i) : Integer.MAX_VALUE;
105             assert symbolCount > 0;
106             int columnIndex = partitionByColumnIndexes.getQuick(i);
107             StaticSymbolTable symbolMapReader = dataFrameCursor.getSymbolTable(columnIndexes.getQuick(columnIndex));
108             int distinctSymbols = symbolMapReader.getSymbolCount();
109             if (symbolMapReader.containsNullValue()) {
110                 distinctSymbols++;
111             }
112             try {
113                 combinations = Math.multiplyExact(combinations, Math.min(symbolCount, distinctSymbols));
