71         if (deferredSymbolKeys != null) {
72             // We need to clean up the map when there are deferred keys since

82         }
83     }

92         DataFrame frame;

94             final int partitionIndex = frame.getPartitionIndex();

99             for (long row = rowHi; row >= rowLo; row--) {
