62         if (!areCursorsPrepared) {
63             rowCursorFactory.prepareCursor(dataFrameCursor.getTableReader());
64             areCursorsPrepared = true;
65         }
66 
67         try {
68             if (rowCursor != null && rowCursor.hasNext()) {
69                 recordA.setRecordIndex(rowCursor.next());
70                 return true;
71             }
72 
73             DataFrame dataFrame;
74             while ((dataFrame = dataFrameCursor.next()) != null) {
75                 rowCursor = rowCursorFactory.getCursor(dataFrame);
76                 if (rowCursor.hasNext()) {
77                     recordA.jumpTo(dataFrame.getPartitionIndex(), rowCursor.next());
78                     return true;
79                 }
80             }
81         } catch (NoMoreFramesException ignore) {
82             return false;
83         }
84 
85         return false;
86     }

122         }
123         return false;
124     }
