36 /**
37  * Returns rows from current data frame in order of cursors list :
38  * - first fetches and returns all records from first cursor
39  * - then from second cursors,
40  * until all cursors are exhausted .
41  */
42 public class SequentialRowCursorFactory implements RowCursorFactory {
43     private final SequentialRowCursor cursor;
44     private final ObjList<? extends RowCursorFactory> cursorFactories;
45     private final int[] cursorFactoriesIdx;
46     private final ObjList<RowCursor> cursors;
47 
48     public SequentialRowCursorFactory(ObjList<? extends RowCursorFactory> cursorFactories, int[] cursorFactoriesIdx) {
49         this.cursorFactories = cursorFactories;
50         cursors = new ObjList<>();
51         cursor = new SequentialRowCursor();
52         this.cursorFactoriesIdx = cursorFactoriesIdx;
53     }
54 
55     @Override
56     public RowCursor getCursor(DataFrame dataFrame) {
57         for (int i = 0, n = cursorFactoriesIdx[0]; i < n; i++) {
58             cursors.extendAndSet(i, cursorFactories.getQuick(i).getCursor(dataFrame));
59         }
60         cursor.init();
61         return cursor;
62     }
63 
64     @Override
65     public void init(TableReader tableReader, SqlExecutionContext sqlExecutionContext) throws SqlException {
66         RowCursorFactory.init(cursorFactories, tableReader, sqlExecutionContext);
67     }
68 
69     @Override
70     public boolean isEntity() {
71         return false;
72     }
73 
74     @Override
75     public void prepareCursor(TableReader tableReader) {
76         RowCursorFactory.prepareCursor(cursorFactories, tableReader);
77     }
78 
79     @Override
80     public void toPlan(PlanSink sink) {
81         sink.type("Cursor-order scan");//postgres uses 'Append' node  
82         for (int i = 0, n = cursorFactories.size(); i < n; i++) {
83             sink.child(cursorFactories.getQuick(i));
84         }
85     }

87     private class SequentialRowCursor implements RowCursor {
88         private RowCursor currentCursor;
89         private int cursorIndex = 0;
90 
91         @Override
92         public boolean hasNext() {
93             boolean hasNext = currentCursor.hasNext();
94             if (hasNext) {
95                 return true;
96             }
97 
98             while (cursorIndex < cursorFactoriesIdx[0] - 1) {
99                 currentCursor = cursors.getQuick(++cursorIndex);
100                 if (currentCursor.hasNext()) {
101                     return true;
102                 }
103             }
104 
105             return false;
106         }
107 
108         @Override
109         public long next() {
110             return currentCursor.next();
111         }
112 
113         private void init() {
114             cursorIndex = 0;
115             if (cursorIndex < cursorFactoriesIdx[0]) {
116                 currentCursor = cursors.getQuick(cursorIndex);
117             }
118         }
119     }
