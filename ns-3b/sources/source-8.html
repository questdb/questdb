40 /**
41  * Used to handle the LIMIT -N clause with the descending timestamp order case. To do so, this cursor
42  * accumulates the row ids in a buffer and only then starts the iteration. That's necessary to preserve
43  * the timestamp-based order in the result set. The buffer is filled in from bottom to top.
44  * <p>
45  * Here is an illustration of the described problem:
46  * <pre>
47  * row iteration order    frames                      frame iteration order
48  *         |            [ row 0   <- frame 0 start           /\
49  *         |              row 1                              |
50  *         |              row 2 ] <- frame 0 end             |
51  *         |            [ row 3   <- frame 1 start           |
52  *        \/              row 4 ] <- frame 1 end             |
53  * </pre>
54  */
55 class AsyncFilteredNegativeLimitRecordCursor implements RecordCursor {

67     // It is typically copied from LIMIT clause on SQL statement.

116         if (frameIndex == -1) {

150     private void fetchAllFrames() {
151         if (frameLimit == -1) {
152             frameSequence.prepareForDispatch();
153             frameLimit = frameSequence.getFrameCount() - 1;
154         }
155 
156         do {
157             final long cursor = frameSequence.next();
158             if (cursor > -1) {
159                 PageFrameReduceTask task = frameSequence.getTask(cursor);
160                 LOG.debug()
161                         .$("collected [shard=").$(frameSequence.getShard())
162                         .$(", frameIndex=").$(task.getFrameIndex())
163                         .$(", frameCount=").$(frameSequence.getFrameCount())
164                         .$(", active=").$(frameSequence.isActive())
165                         .$(", cursor=").$(cursor)
166                         .I$();
167 
168                 final DirectLongList frameRows = task.getRows();
169                 final long frameRowCount = frameRows.size();
170                 frameIndex = task.getFrameIndex();
171 
172                 if (frameRowCount > 0 && rowCount < rowLimit + 1 && frameSequence.isActive()) {
173                     // Copy rows into the buffer.
174                     for (long i = frameRowCount - 1; i > -1 && rowCount < rowLimit; i--, rowCount++) {
175                         rows.set(--rowIndex, Rows.toRowID(frameIndex, frameRows.get(i)));
176                     }
177 
178                     if (rowCount >= rowLimit) {
179                         frameSequence.cancel();
180                     }
181                 }
182 
183                 frameSequence.collect(cursor, false);
184             } else if (cursor == -2) {
185                 break; // No frames to filter.
186             } else {
187                 Os.pause();
188             }
189         } while (frameIndex < frameLimit);
190     }

203         }
204     }
