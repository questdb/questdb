54     public FilterOnExcludedValuesRecordCursorFactory(
55             @NotNull RecordMetadata metadata,
56             @NotNull DataFrameCursorFactory dataFrameCursorFactory,
57             @NotNull @Transient ObjList<Function> keyValues,
58             int columnIndex,
59             @Nullable Function filter,
60             int orderByMnemonic,
61             boolean followedOrderByAdvice,
62             int indexDirection,
63             @NotNull IntList columnIndexes,
64             int maxSymbolNotEqualsCount
65     ) {
66         super(metadata, dataFrameCursorFactory);
67         this.indexDirection = indexDirection;
68         this.maxSymbolNotEqualsCount = maxSymbolNotEqualsCount;
69         final int nKeyValues = keyValues.size();
70         boolean dynamicValues = false;
71         for (int i = 0; i < nKeyValues; i++) {
72             if (!keyValues.getQuick(i).isConstant()) {
73                 dynamicValues = true;
74                 break;
75             }
76         }
77         dynamicExcludedKeys = dynamicValues;
78         keyExcludedValueFunctions.addAll(keyValues);
79         this.columnIndex = columnIndex;
80         this.filter = filter;
81         cursorFactoriesIdx = new int[]{0};
82         cursorFactories = new ObjList<>(nKeyValues);
83         if (orderByMnemonic == OrderByMnemonic.ORDER_BY_INVARIANT) {
84             cursor = new DataFrameRecordCursorImpl(new SequentialRowCursorFactory(cursorFactories, cursorFactoriesIdx), false, filter, columnIndexes);
85         } else {

97     public void recalculateIncludedValues(TableReader tableReader) {
98         cursorFactoriesIdx[0] = cursorFactories.size();
99         excludedKeys.clear();
100         if (dynamicExcludedKeys) {
101             // In case of bind variable excluded values that may change between
102             // query executions the sets have to be subtracted from scratch.
103             includedKeys.clear();
104             cursorFactoriesIdx[0] = 0;
105         }
106 
107         final SymbolMapReaderImpl symbolMapReader = (SymbolMapReaderImpl) tableReader.getSymbolMapReader(columnIndex);
108 
109         // Generate excluded key set.
110         for (int i = 0, n = keyExcludedValueFunctions.size(); i < n; i++) {
111             final CharSequence value = keyExcludedValueFunctions.getQuick(i).getStr(null);
112             excludedKeys.add(symbolMapReader.keyOf(value));
113         }
114 
115         // Append new keys to the included set filtering out the excluded ones.
116         // Note: both includedKeys and cursorFactories are guaranteed to be monotonically
117         // growing in terms of the collection size.
118         for (int k = 0, n = symbolMapReader.getSymbolCount(); k < n; k++) {
119             if (!excludedKeys.contains(k) && includedKeys.add(k)) {
120                 upsertRowCursorFactory(k);
121             }
122         }
123 
124         if (symbolMapReader.containsNullValue()
125                 && !excludedKeys.contains(SymbolTable.VALUE_IS_NULL) && includedKeys.add(SymbolTable.VALUE_IS_NULL)) {
126             // If the table contains null values and they're not excluded, we need to include
127             // them to the result set to match the behavior of the NOT IN() SQL function.
128             upsertRowCursorFactory(SymbolTable.VALUE_IS_NULL);
129         }
130     }

146     private void upsertRowCursorFactory(int symbolKey) {
147         if (cursorFactoriesIdx[0] < cursorFactories.size()) {
148             // Reuse the existing factory.
149             cursorFactories.get(cursorFactoriesIdx[0]).of(symbolKey);
150             cursorFactoriesIdx[0]++;
151             return;
152         }
153 
154         // Create a new factory.
155         final SymbolFunctionRowCursorFactory rowCursorFactory;
156         if (filter == null) {
157             rowCursorFactory = new SymbolIndexRowCursorFactory(
158                     columnIndex,
159                     symbolKey,
160                     false,
161                     indexDirection,
162                     null
163             );
164         } else {
165             rowCursorFactory = new SymbolIndexFilteredRowCursorFactory(
166                     columnIndex,
167                     symbolKey,
168                     filter,
169                     false,
170                     indexDirection,
171                     columnIndexes,
172                     null
173             );
174         }
175         cursorFactories.add(rowCursorFactory);
176         cursorFactoriesIdx[0]++;
177     }

186     @Override
187     protected RecordCursor getCursorInstance(
188             DataFrameCursor dataFrameCursor,
189             SqlExecutionContext executionContext
190     ) throws SqlException {
191         TableReader reader = dataFrameCursor.getTableReader();
192         if (reader.getSymbolMapReader(columnIndex).getSymbolCount() > maxSymbolNotEqualsCount) {
193             throw TableReferenceOutOfDateException.of(reader.getTableToken().getTableName());
194         }
195         Function.init(keyExcludedValueFunctions, reader, executionContext);
196         recalculateIncludedValues(reader);
197         cursor.of(dataFrameCursor, executionContext);
198         if (filter != null) {
199             filter.init(cursor, executionContext);
200         }
201         return cursor;
202     }
