34 public final class TableWriterMetricsRecordCursorFactory extends AbstractRecordCursorFactory {
35     private static final int COMMITTED_ROWS_COLUMN_INDEX = 3;
36     private static final RecordMetadata METADATA;
37     private static final long METRICS_DISABLED_VALUE = -1;
38     private static final int O3_COMMITS_COLUMN_INDEX = 1;
39     private static final int PHYSICALLY_WRITTEN_ROWS_COLUMN_INDEX = 4;
40     private static final int ROLLBACKS_COLUMN_INDEX = 2;
41     private static final int TOTAL_COMMITS_COLUMN_INDEX = 0;
42     private static final int TOTAL_NUMBER_OF_METRIC = PHYSICALLY_WRITTEN_ROWS_COLUMN_INDEX + 1;
43     private static final String[] KEYS = new String[TOTAL_NUMBER_OF_METRIC];
44     private final StringLongTuplesRecordCursor cursor = new StringLongTuplesRecordCursor();
45     private final long[] values = new long[TOTAL_NUMBER_OF_METRIC];
46 
47     public TableWriterMetricsRecordCursorFactory() {
48         super(METADATA);
49     }
50 
51     @Override
52     public RecordCursor getCursor(SqlExecutionContext executionContext) {
53         Metrics metrics = executionContext.getCairoEngine().getMetrics();
54         if (metrics.isEnabled()) {
55             TableWriterMetrics tableWriterMetrics = metrics.tableWriter();
56             values[TOTAL_COMMITS_COLUMN_INDEX] = tableWriterMetrics.getCommitCount();
57             values[O3_COMMITS_COLUMN_INDEX] = tableWriterMetrics.getO3CommitCount();
58             values[ROLLBACKS_COLUMN_INDEX] = tableWriterMetrics.getRollbackCount();
59             values[COMMITTED_ROWS_COLUMN_INDEX] = tableWriterMetrics.getCommittedRows();
60             values[PHYSICALLY_WRITTEN_ROWS_COLUMN_INDEX] = tableWriterMetrics.getPhysicallyWrittenRows();
61         } else {
62             values[TOTAL_COMMITS_COLUMN_INDEX] = METRICS_DISABLED_VALUE;
63             values[O3_COMMITS_COLUMN_INDEX] = METRICS_DISABLED_VALUE;
64             values[ROLLBACKS_COLUMN_INDEX] = METRICS_DISABLED_VALUE;
65             values[COMMITTED_ROWS_COLUMN_INDEX] = METRICS_DISABLED_VALUE;
66             values[PHYSICALLY_WRITTEN_ROWS_COLUMN_INDEX] = METRICS_DISABLED_VALUE;
67         }
68         cursor.of(KEYS, values);
69         return cursor;
70     }
71 
72     @Override
73     public boolean recordCursorSupportsRandomAccess() {
74         return false;
75     }
76 
77     @Override
78     public void toPlan(PlanSink sink) {
79         sink.type("table_writer_metrics");
80     }
