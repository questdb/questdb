109     @Override
110     public RecordCursor getCursor(SqlExecutionContext executionContext) throws SqlException {
111         long rowsRemaining;
112         int baseOrder = base.hasDescendingOrder() ? ORDER_DESC : ORDER_ASC;
113         final int order;
114         if (limitLoFunction != null) {
115             limitLoFunction.init(frameSequence.getSymbolTableSource(), executionContext);
116             rowsRemaining = limitLoFunction.getLong(null);
117             // on negative limit we will be looking for positive number of rows
118             // while scanning table from the highest timestamp to the lowest
119             if (rowsRemaining > -1) {
120                 order = baseOrder;
121             } else {
122                 order = reverse(baseOrder);
123                 rowsRemaining = -rowsRemaining;
124             }
125         } else {
126             rowsRemaining = Long.MAX_VALUE;
127             order = baseOrder;
128         }
129 
130         if (order == ORDER_DESC && rowsRemaining != Long.MAX_VALUE) {
131             if (rowsRemaining > maxNegativeLimit) {
132                 throw SqlException.position(limitLoPos).put("absolute LIMIT value is too large, maximum allowed value: ").put(maxNegativeLimit);
133             }
134             if (negativeLimitRows == null) {
135                 negativeLimitRows = new DirectLongList(maxNegativeLimit, MemoryTag.NATIVE_OFFLOAD);
136             }
137             negativeLimitCursor.of(execute(executionContext, collectSubSeq, order), rowsRemaining, negativeLimitRows);
138             return negativeLimitCursor;
139         }
140 
141         cursor.of(execute(executionContext, collectSubSeq, order), rowsRemaining);
142         return cursor;
143     }

161     public void toPlan(PlanSink sink) {
162         sink.type("Async Filter");
163         //calc order and limit if possible  
164         long rowsRemaining;
165         int baseOrder = base.hasDescendingOrder() ? ORDER_DESC : ORDER_ASC;
166         int order;
167         if (limitLoFunction != null) {
168             try {
169                 limitLoFunction.init(frameSequence.getSymbolTableSource(), sink.getExecutionContext());
170                 rowsRemaining = limitLoFunction.getLong(null);
171             } catch (Exception e) {
172                 rowsRemaining = Long.MAX_VALUE;
173             }
174             if (rowsRemaining > -1) {
175                 order = baseOrder;
176             } else {
177                 order = reverse(baseOrder);
178                 rowsRemaining = -rowsRemaining;
179             }
180         } else {
181             rowsRemaining = Long.MAX_VALUE;
182             order = baseOrder;
183         }
184         if (rowsRemaining != Long.MAX_VALUE) {
185             sink.attr("limit").val(rowsRemaining);
186         }
187         sink.attr("filter").val(filterAtom);

197     private static void filter(
198             int workerId,
199             @NotNull PageAddressCacheRecord record,
200             @NotNull PageFrameReduceTask task,
201             @NotNull SqlExecutionCircuitBreaker circuitBreaker,
202             @Nullable PageFrameSequence<?> stealingFrameSequence
203     ) {
204         final DirectLongList rows = task.getRows();
205         final long frameRowCount = task.getFrameRowCount();
206         final AsyncFilterAtom atom = task.getFrameSequence(AsyncFilterAtom.class).getAtom();
207 
208         rows.clear();
209 
210         final boolean owner = stealingFrameSequence != null && stealingFrameSequence == task.getFrameSequence();
211         final int filterId = atom.acquireFilter(workerId, owner, circuitBreaker);
212         final Function filter = atom.getFilter(filterId);
213         try {
214             for (long r = 0; r < frameRowCount; r++) {
215                 record.setRowIndex(r);
216                 if (filter.getBool(record)) {
217                     rows.add(r);
218                 }
219             }
220         } finally {
221             atom.releaseFilter(filterId);
222         }
223 
224         // Pre-touch fixed-size columns, if asked.
225         atom.preTouchColumns(record, rows);
226     }
