39 public class DeferredSymbolIndexFilteredRowCursorFactory implements FunctionBasedRowCursorFactory {

44 
45     public DeferredSymbolIndexFilteredRowCursorFactory(
46             int columnIndex,
47             Function symbolFunction,
48             Function filter,
49             boolean cachedIndexReaderCursor,
50             int indexDirection,
51             IntList columnIndexes
52     ) {

56     }

58     @Override

60         if (symbolKey == SymbolTable.VALUE_NOT_FOUND) {
61             return EmptyRowCursor.INSTANCE;
62         }
63         return cursor.of(dataFrame);
64     }
65 
66     @Override

69     }

74     }
75

79     }
80 
81     @Override
82     public boolean isUsingIndex() {
83         return true;
84     }
85 
86     @Override
87     public void prepareCursor(TableReader tableReader) {
88         symbolKey = tableReader.getSymbolMapReader(columnIndex).keyOf(symbolFunction.getStr(null));
89         if (symbolKey != SymbolTable.VALUE_NOT_FOUND) {
90             cursor.of(symbolKey);
91             cursor.prepare(tableReader);
92         }
93     }
94
