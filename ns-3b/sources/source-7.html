54     public AsyncFilterAtom(
55             @NotNull CairoConfiguration configuration,
56             @NotNull Function filter,
57             @Nullable ObjList<Function> perWorkerFilters,
58             @Nullable IntList preTouchColumnTypes
59     ) {
60         this.rnd = new Rnd(
61                 configuration.getNanosecondClock().getTicks(),
62                 configuration.getMicrosecondClock().getTicks()
63         );
64         this.filter = filter;
65         this.perWorkerFilters = perWorkerFilters;
66         if (perWorkerFilters != null) {
67             perWorkerLocks = new AtomicIntegerArray(perWorkerFilters.size());
68         } else {
69             perWorkerLocks = null;
70         }
71         this.preTouchColumnTypes = preTouchColumnTypes;
72     }

74     public int acquireFilter(int workerId, boolean owner, SqlExecutionCircuitBreaker circuitBreaker) {
75         if (perWorkerFilters == null) {
76             return -1;
77         }
78         if (workerId == -1 && owner) {
79             // Owner thread is free to use the original filter anytime.
80             return -1;
81         }
82         final int size = perWorkerFilters.size();
83         workerId = workerId == -1 ? rnd.nextInt(size) : workerId;
84         while (true) {
85             for (int i = 0; i < size; i++) {
86                 int id = (i + workerId) % size;
87                 if (perWorkerLocks.compareAndSet(id, 0, 1)) {
88                     return id;
89                 }
90             }
91             circuitBreaker.statefulThrowExceptionIfTripped();
92             Os.pause();
93         }
94     }

135     /**
136      * Pre-touches column values for the filtered rows, if the feature is configured.
137      * <p>
138      * The idea is to access the memory to page fault and, thus, warm up the pages
139      * in parallel, on multiple threads, instead of relying on the "query owner" thread
140      * to do it later serially.
141      *
142      * @param record record to use
143      * @param rows   rows to pre-touch
144      */
145     public void preTouchColumns(PageAddressCacheRecord record, DirectLongList rows) {
146         if (!preTouchEnabled || preTouchColumnTypes == null) {
147             return;
148         }
149         // We use a LongAdder as a black hole to make sure that the JVM JIT compiler keeps the load instructions in place.
150         long sum = 0;
151         for (long p = 0; p < rows.size(); p++) {
152             long r = rows.get(p);
153             record.setRowIndex(r);
154             for (int i = 0; i < preTouchColumnTypes.size(); i++) {
155                 int columnType = preTouchColumnTypes.getQuick(i);
156                 switch (ColumnType.tagOf(columnType)) {
157                     case ColumnType.BOOLEAN:
158                         sum += record.getBool(i) ? 1 : 0;
159                         break;
160                     case ColumnType.BYTE:
161                         sum += record.getByte(i);
162                         break;
163                     case ColumnType.SHORT:
164                         sum += record.getShort(i);
165                         break;
166                     case ColumnType.INT:
167                     case ColumnType.SYMBOL: // We're interested in pre-touching pages, so we read the symbol key only.
168                         sum += record.getInt(i);
169                         break;
170                     case ColumnType.LONG:
171                     case ColumnType.DATE:
172                     case ColumnType.TIMESTAMP:
173                         sum += record.getLong(i);
174                         break;
175                     case ColumnType.FLOAT:
176                         sum += record.getFloat(i);
177                         break;
178                     case ColumnType.DOUBLE:
179                         sum += record.getDouble(i);
180                         break;
181                     case ColumnType.LONG256:
182                         Long256 l256 = record.getLong256A(i);
183                         sum += l256.getLong0();
184                         sum += l256.getLong1();
185                         sum += l256.getLong2();
186                         sum += l256.getLong3();
187                         break;
188                     case ColumnType.GEOBYTE:
189                         sum += record.getGeoByte(i);
190                         break;
191                     case ColumnType.GEOSHORT:
192                         sum += record.getGeoShort(i);
193                         break;
194                     case ColumnType.GEOINT:
195                         sum += record.getGeoInt(i);
196                         break;
197                     case ColumnType.GEOLONG:
198                         sum += record.getGeoLong(i);
199                         break;
200                     case ColumnType.STRING:
201                         CharSequence cs = record.getStr(i);
202                         if (cs != null && cs.length() > 0) {
203                             // Touch the first page of the string contents only.
204                             sum += cs.charAt(0);
205                         }
206                         break;
207                     case ColumnType.BINARY:
208                         BinarySequence bs = record.getBin(i);
209                         if (bs != null && bs.length() > 0) {
210                             // Touch the first page of the binary contents only.
211                             sum += bs.byteAt(0);
212                         }
213                         break;
214                     case ColumnType.UUID:
215                         sum += record.getLong128Hi(i);
216                         sum += record.getLong128Lo(i);
217                         break;
218                 }
219             }
220         }
221         // Flush the accumulated sum to the black hole.
222         PRE_TOUCH_BLACK_HOLE.add(sum);
223     }
