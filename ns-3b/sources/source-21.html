135     private void buildTreeMap() {
136         int taskCount;
137         if (keyCount < 0) {
138             keyCount = getSymbolTable(columnIndex).getSymbolCount() + 1;
139             final long chunkSize = getChunkSize(keyCount, workerCount);
140             taskCount = getTaskCount(keyCount, chunkSize);
141             rows.setCapacity(keyCount);
142             GeoHashNative.iota(rows.getAddress(), rows.getCapacity(), 0);
143 
144             argumentsAddress = LatestByArguments.allocateMemoryArray(taskCount);
145             for (long i = 0; i < taskCount; ++i) {
146                 final long klo = i * chunkSize;
147                 final long khi = Long.min(klo + chunkSize, keyCount);
148                 final long argsAddress = argumentsAddress + i * LatestByArguments.MEMORY_SIZE;
149                 LatestByArguments.setRowsAddress(argsAddress, rows.getAddress());
150                 LatestByArguments.setRowsCapacity(argsAddress, rows.getCapacity());
151                 LatestByArguments.setKeyLo(argsAddress, klo);
152                 LatestByArguments.setKeyHi(argsAddress, khi);
153                 LatestByArguments.setRowsSize(argsAddress, 0);
154             }
155 
156             sharedCircuitBreaker.reset();
157         } else {
158             final long chunkSize = getChunkSize(keyCount, workerCount);
159             taskCount = getTaskCount(keyCount, chunkSize);
160         }
161 
162         int hashColumnIndex = -1;
163         int hashColumnType = ColumnType.UNDEFINED;
164         long prefixesAddress = 0;
165         long prefixesCount = 0;
166 
167         if (prefixes.size() > 2) {
168             hashColumnIndex = (int) prefixes.get(0);
169             hashColumnType = (int) prefixes.get(1);
170             prefixesAddress = prefixes.getAddress() + 2 * Long.BYTES;
171             prefixesCount = prefixes.size() - 2;
172         }
173 
174         // frame metadata is based on TableReader, which is "full" metadata
175         // this cursor works with subset of columns, which warrants column index remap
176         int frameColumnIndex = columnIndexes.getQuick(columnIndex);
177 
178         final RingQueue<LatestByTask> queue = bus.getLatestByQueue();
179         final Sequence pubSeq = bus.getLatestByPubSeq();
180         final Sequence subSeq = bus.getLatestBySubSeq();
181         final TableReader reader = dataFrameCursor.getTableReader();
182 
183         DataFrame frame;
184         long foundRowCount = 0;
185         int queuedCount = 0;
186         try {
187             while ((frame = dataFrameCursor.next()) != null && foundRowCount < keyCount) {
188                 doneLatch.reset();
189                 final BitmapIndexReader indexReader = frame.getBitmapIndexReader(frameColumnIndex, BitmapIndexReader.DIR_BACKWARD);
190 
191                 final long rowLo = frame.getRowLo();
192                 final long rowHi = frame.getRowHi() - 1;
193 
194                 final long keyBaseAddress = indexReader.getKeyBaseAddress();
195                 final long keysMemorySize = indexReader.getKeyMemorySize();
196                 final long valueBaseAddress = indexReader.getValueBaseAddress();
197                 final long valuesMemorySize = indexReader.getValueMemorySize();
198                 final int valueBlockCapacity = indexReader.getValueBlockCapacity();
199                 final long unIndexedNullCount = indexReader.getUnIndexedNullCount();
200                 final int partitionIndex = frame.getPartitionIndex();
201 
202                 long hashColumnAddress = 0;
203 
204                 // hashColumnIndex can be -1 for latest by part only (no prefixes to match)
205                 if (hashColumnIndex > -1) {
206                     final int columnBase = reader.getColumnBase(partitionIndex);
207                     final int primaryColumnIndex = TableReader.getPrimaryColumnIndex(columnBase, hashColumnIndex);
208                     final MemoryR column = reader.getColumn(primaryColumnIndex);
209                     hashColumnAddress = column.getPageAddress(0);
210                 }
211 
212                 // -1 must be dead case here
213                 final int hashesColumnSize = ColumnType.isGeoHash(hashColumnType) ? getPow2SizeOfGeoHashType(hashColumnType) : -1;
214 
215                 queuedCount = 0;
216                 for (long i = 0; i < taskCount; ++i) {
217                     final long argsAddress = argumentsAddress + i * LatestByArguments.MEMORY_SIZE;
218                     final long found = LatestByArguments.getRowsSize(argsAddress);
219                     final long keyHi = LatestByArguments.getKeyHi(argsAddress);
220                     final long keyLo = LatestByArguments.getKeyLo(argsAddress);
221

223                     if (found >= keyHi - keyLo) {
224                         continue;
225                     }
226                     // Update hash column address with current frame value
227                     LatestByArguments.setHashesAddress(argsAddress, hashColumnAddress);
228 
229                     final long seq = pubSeq.next();
230                     if (seq < 0) {

233                                 keyBaseAddress,
234                                 keysMemorySize,
235                                 valueBaseAddress,
236                                 valuesMemorySize,
237                                 argsAddress,
238                                 unIndexedNullCount,
239                                 rowHi,

241                                 partitionIndex,
242                                 valueBlockCapacity,
243                                 hashColumnAddress,
244                                 hashesColumnSize,
245                                 prefixesAddress,
246                                 prefixesCount

251                                 keysMemorySize,
252                                 valueBaseAddress,
253                                 valuesMemorySize,
254                                 argsAddress,
255                                 unIndexedNullCount,
256                                 rowHi,

268                         queuedCount++;

270                 }
271 
272                 // process our own queue
273                 // this should fix deadlock with 1 worker configuration
274                 while (doneLatch.getCount() > -queuedCount) {
275                     circuitBreaker.statefulThrowExceptionIfTrippedNoThrottle();
276                     long seq = subSeq.next();
277                     if (seq > -1) {
278                         queue.get(seq).run();
279                         subSeq.done(seq);
280                     }
281                 }
282

284 
285                 foundRowCount = 0; // Reset found counter
286                 for (int i = 0; i < taskCount; i++) {
287                     final long address = argumentsAddress + i * LatestByArguments.MEMORY_SIZE;
288                     foundRowCount += LatestByArguments.getRowsSize(address);
289                 }
290             }
291         } catch (DataUnavailableException e) {
292             // We're not yet done, so no need to cancel the circuit breaker. 
293             throw e;
294         } catch (Throwable t) {
295             sharedCircuitBreaker.cancel();
296             throw t;
297         } finally {
298             processTasks(queuedCount);
299             if (sharedCircuitBreaker.isCanceled()) {
300                 LatestByArguments.releaseMemoryArray(argumentsAddress, taskCount);
301                 argumentsAddress = 0;
302             }
303         }
304 
305         long rowCount = 0;
306         if (argumentsAddress > 0) {
307             rowCount = GeoHashNative.slideFoundBlocks(argumentsAddress, taskCount);
308             LatestByArguments.releaseMemoryArray(argumentsAddress, taskCount);
309             argumentsAddress = 0;
310         }
311         aLimit = rowCount;
312         aIndex = indexShift;
313         postProcessRows();
314     }

320     private void processTasks(int queuedCount) {
321         final RingQueue<LatestByTask> queue = bus.getLatestByQueue();
322         final Sequence subSeq = bus.getLatestBySubSeq();
323         while (doneLatch.getCount() > -queuedCount) {
324             long seq = subSeq.next();
325             if (seq > -1) {
326                 if (circuitBreaker.checkIfTripped()) {
327                     sharedCircuitBreaker.cancel();
328                 }
329                 queue.get(seq).run();
330                 subSeq.done(seq);
331             }
332         }
333         doneLatch.await(queuedCount);
334     }
