302         for (int i = 0; i < len; i++) {
303             byte b = Unsafe.getUnsafe().getByte(ptr + i);

311     private byte[] receiveChallengeBytes() {
312         int n = 0;
313         for (; ; ) {
314             int rc = lineChannel.receive(ptr + n, capacity - n);
315             if (rc < 0) {
316                 int errno = lineChannel.errno();
317                 close();
318                 throw new LineSenderException("disconnected during authentication").errno(errno);
319             }
320             int eol = findEOL(ptr + n, rc);
321             if (eol != -1) {
322                 n += eol;
323                 break;
324             }
325             n += rc;
326             if (n == capacity) {
327                 close();
328                 throw new LineSenderException("challenge did not fit into buffer");
329             }
330         }
331         int sz = n;
332         byte[] challengeBytes = new byte[sz];
333         for (n = 0; n < sz; n++) {
334             challengeBytes[n] = Unsafe.getUnsafe().getByte(ptr + n);
335         }
336         return challengeBytes;
337     }

383         int len = (int) (ptr - lineStart);
384         if (len == 0) {

408     protected byte[] signAndEncode(PrivateKey privateKey, byte[] challengeBytes) {
409         // protected for testing
410         byte[] rawSignature;
411         try {
412             Signature sig = Signature.getInstance(AuthDb.SIGNATURE_TYPE_DER);
413             sig.initSign(privateKey);
414             sig.update(challengeBytes);
415             rawSignature = sig.sign();
416         } catch (InvalidKeyException ex) {
417             close();
418             throw new LineSenderException("invalid key", ex);
419         } catch (SignatureException ex) {
420             close();
421             throw new LineSenderException("cannot sign challenge", ex);
422         } catch (NoSuchAlgorithmException ex) {
423             close();
424             throw new LineSenderException("unsupported signing algorithm", ex);
425         }
426         return Base64.getEncoder().encode(rawSignature);
427     }
