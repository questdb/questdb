92      * Waits for frame sequence completion, fetches remaining pieces of the
93      * frame sequence from the queues. This method is not thread safe.
94      */
95     public void await() {
96         LOG.debug()
97                 .$("awaiting completion [shard=").$(shard)
98                 .$(", id=").$(id)
99                 .$(", frameCount=").$(frameCount)
100                 .I$();
101 
102         final MCSequence pageFrameReduceSubSeq = messageBus.getPageFrameReduceSubSeq(shard);
103         while (!done) {
104             if (dispatchStartFrameIndex == collectedFrameIndex + 1) {
105                 // We know that all frames were collected. We're almost done.
106                 if (!done) {
107                     // Looks like not all the frames were dispatched, so no one reached the very last frame and
108                     // reset the sequence via calling PageFrameReduceTask#collected(). Let's do it ourselves.
109                     reset();
110                 }
111                 break;
112             }
113 
114             // We were asked to steal work from the reduce queue and beyond, as much as we can.
115             boolean nothingProcessed = true;
116             try {
117                 nothingProcessed = PageFrameReduceJob.consumeQueue(reduceQueue, pageFrameReduceSubSeq, record, circuitBreaker, this);
118             } catch (Throwable e) {
119                 LOG.error()
120                         .$("await error [id=").$(id)
121                         .$(", ex=").$(e)
122                         .I$();
123             }
124 
125             if (nothingProcessed) {
126                 long cursor = collectSubSeq.next();
127                 if (cursor > -1) {
128                     // Discard collected items.
129                     final PageFrameReduceTask task = reduceQueue.get(cursor);
130                     if (task.getFrameSequence() == this) {
131                         assert id == task.getFrameSequenceId() : "ids mismatch: " + id + ", " + task.getFrameSequenceId();
132                         collectedFrameIndex = task.getFrameIndex();
133                         task.collected(true);
134                     }
135                     collectSubSeq.done(cursor);
136                 } else {
137                     Os.pause();
138                 }
139             }
140         }
141 
142         // It could be the case that one of the workers reduced a page frame, then marked the task as done,
143         // but haven't incremented reduce counter yet. In this case, we wait for the desired counter value.
144         while (reduceCounter.get() != dispatchStartFrameIndex) {
145             Os.pause();
146         }
147     }

256     /**
257      * This method is not thread safe. It's always invoked on a single "query owner" thread.
258      * <p>
259      * Returns a cursor that points either to the reduce queue or to the local reduce task.
260      * The caller of this method should avoid accessing the reduce queue directly and,
261      * instead, should use getTask and collect methods. <code>Long.MAX_VALUE</code> is the
262      * reserved cursor value for the local reduce task case.
263      *
264      * @return the next cursor value, or -1 value if the cursor failed and the caller
265      * should retry, or -2 if there are no frames to dispatch
266      */
267     public long next() {
268         if (frameCount == 0) {
269             return -2;
270         }
271 
272         assert collectedFrameIndex < frameCount - 1;
273         while (true) {
274             long cursor = collectSubSeq.next();
275             if (cursor > -1) {
276                 PageFrameReduceTask task = reduceQueue.get(cursor);
277                 PageFrameSequence<?> thatFrameSequence = task.getFrameSequence();
278                 if (thatFrameSequence == this) {
279                     assert id == task.getFrameSequenceId() : "ids mismatch: " + id + ", " + task.getFrameSequenceId();
280                     return cursor;
281                 } else {
282                     // Not our task, nothing to collect. Go for another spin.
283                     collectSubSeq.done(cursor);
284                 }
285             } else if (cursor == -1) {
286                 if (dispatch()) {
287                     // We have dispatched something, so let's try to collect it.
288                     continue;
289                 }
290                 if (dispatchStartFrameIndex == collectedFrameIndex + 1) {
291                     // We haven't dispatched anything, and we have collected everything
292                     // that was dispatched previously in this loop iteration. Use the
293                     // local task to avoid being blocked in case of full reduce queue.
294                     workLocally();
295                     return LOCAL_TASK_CURSOR;
296                 }
297                 return -1;
298             } else {
299                 Os.pause();
300             }
301         }
302     }

304     public PageFrameSequence<T> of(
305             RecordCursorFactory base,
306             SqlExecutionContext executionContext,
307             SCSequence collectSubSeq,
308             T atom,
309             int order
310     ) throws SqlException {
311         sqlExecutionContext = executionContext;
312         startTime = clock.getTicks();
313         circuitBreakerFd = executionContext.getCircuitBreaker().getFd();
314 
315         initRecord(executionContext.getCircuitBreaker());
316 
317         final Rnd rnd = executionContext.getAsyncRandom();
318         try {
319             // pass one to cache page addresses
320             // this has to be separate pass to ensure there no cache reads
321             // while cache might be resizing
322             pageAddressCache.of(base.getMetadata());
323 
324             assert pageFrameCursor == null;
325             pageFrameCursor = base.getPageFrameCursor(executionContext, order);
326             this.atom = atom;
327             this.collectSubSeq = collectSubSeq;
328             id = ID_SEQ.incrementAndGet();
329             done = false;
330             valid.set(true);
331             reduceCounter.set(0);
332             shard = rnd.nextInt(messageBus.getPageFrameReduceShardCount());
333             reduceQueue = messageBus.getPageFrameReduceQueue(shard);
334 
335             // It is essential to init the atom after we prepared sequence for dispatch.
336             // If atom is to fail, we will be releasing whatever we prepared.
337             atom.init(pageFrameCursor, executionContext);
338         } catch (Throwable e) {
339             pageFrameCursor = Misc.freeIfCloseable(pageFrameCursor);
340             throw e;
341         }
342         return this;
343     }

391         PageFrame frame;
392         while ((frame = pageFrameCursor.next()) != null) {
393             pageAddressCache.add(frameCount++, frame);
394             frameRowCounts.add(frame.getPartitionHi() - frame.getPartitionLo());
395         }

412     /**
413      * This method is re-enterable. It has to be in case queue capacity is smaller than number of frames to
414      * be dispatched. When it is the case, frame count published so far is stored in the `frameSequence`.
415      * This method has no responsibility to deal with "collect" stage hence it deals with everything to
416      * unblock the collect stage.
417      *
418      * @return true if at least one task was dispatched or reduced; false otherwise
419      */
420     private boolean dispatch() {
421         boolean idle = true;
422         boolean dispatched = false;
423 
424         // the sequence used to steal worker jobs
425         final MCSequence reduceSubSeq = messageBus.getPageFrameReduceSubSeq(shard);
426         final MPSequence reducePubSeq = messageBus.getPageFrameReducePubSeq(shard);
427 
428         long cursor;
429         int i = dispatchStartFrameIndex;
430         dispatchStartFrameIndex = frameCount;
431         OUT:
432         for (; i < frameCount; i++) {
433             // We cannot process work on this thread. If we do the consumer will
434             // never get the executions results. Consumer only picks ready to go
435             // tasks from the queue.
436 
437             while (true) {
438                 cursor = reducePubSeq.next();
439                 if (cursor > -1) {
440                     reduceQueue.get(cursor).of(this, i);
441                     LOG.debug()
442                             .$("dispatched [shard=").$(shard)
443                             .$(", id=").$(getId())
444                             .$(", frameIndex=").$(i)
445                             .$(", frameCount=").$(frameCount)
446                             .$(", cursor=").$(cursor)
447                             .I$();
448                     reducePubSeq.done(cursor);
449                     dispatched = true;
450                     break;
451                 } else if (cursor == -1) {
452                     idle = false;
453                     // start stealing work to unload the queue
454                     if (stealWork(reduceQueue, reduceSubSeq, record, circuitBreaker)) {
455                         continue;
456                     }
457                     dispatchStartFrameIndex = i;
458                     break OUT;
459                 } else {
460                     Os.pause();
461                 }
462             }
463         }
464 
465         // Reduce counter is here to provide safe backoff point
466         // for job stealing code. It is needed because queue is shared
467         // and there is possibility of never ending stealing if we don't
468         // specifically count only our items
469 
470         // join the gang to consume published tasks
471         while (reduceCounter.get() < frameCount) {
472             idle = false;
473             if (stealWork(reduceQueue, reduceSubSeq, record, circuitBreaker)) {
474                 if (isActive()) {
475                     continue;
476                 }
477             }
478             break;
479         }
480 
481         if (idle) {
482             stealWork(reduceQueue, reduceSubSeq, record, circuitBreaker);
483         }
484 
485         return dispatched;
486     }

515     private void workLocally() {
516         assert dispatchStartFrameIndex < frameCount;
517 
518         if (localTask == null) {
519             localTask = localTaskPool.pop();
520         }
521         localTask.of(this, dispatchStartFrameIndex++);
522 
523         try {
524             LOG.debug()
525                     .$("reducing locally [shard=").$(shard)
526                     .$(", id=").$(id)
527                     .$(", frameIndex=").$(localTask.getFrameIndex())
528                     .$(", frameCount=").$(frameCount)
529                     .$(", active=").$(isActive())
530                     .I$();
531             if (isActive()) {
532                 PageFrameReduceJob.reduce(record, circuitBreaker, localTask, this, this);
533             }
534         } catch (Throwable e) {
535             cancel();
536             throw e;
537         } finally {
538             reduceCounter.incrementAndGet();
539         }
