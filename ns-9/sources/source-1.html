55     // Each thread should be assigned own instance of this job, making the code effectively
56     // single threaded. Such assignment is necessary for threads to have their own shard walk sequence.
57     public PageFrameReduceJob(
58             MessageBus bus,
59             Rnd rnd,
60             @Nullable SqlExecutionCircuitBreakerConfiguration sqlExecutionCircuitBreakerConfiguration
61     ) {
62         this.messageBus = bus;
63         this.shardCount = messageBus.getPageFrameReduceShardCount();
64         this.shards = new int[shardCount];
65         // fill shards[] with shard indexes
66         for (int i = 0; i < shardCount; i++) {
67             shards[i] = i;
68         }
69 
70         // shuffle shard indexes such that each job has its own
71         // pass order over the shared queues
72         int currentIndex = shardCount;
73         int randomIndex;
74         while (currentIndex != 0) {
75             randomIndex = (int) Math.floor(rnd.nextDouble() * currentIndex);
76             currentIndex--;
77 
78             final int tmp = shards[currentIndex];
79             shards[currentIndex] = shards[randomIndex];
80             shards[randomIndex] = tmp;
81         }
82 
83         this.record = new PageAddressCacheRecord();
84         if (sqlExecutionCircuitBreakerConfiguration != null) {
85             this.circuitBreaker = new NetworkSqlExecutionCircuitBreaker(sqlExecutionCircuitBreakerConfiguration, MemoryTag.NATIVE_CB1);
86         } else {
87             this.circuitBreaker = NetworkSqlExecutionCircuitBreaker.NOOP_CIRCUIT_BREAKER;
88         }
89     }

92      * Reduces single queue item when item is available. Return value is inverted as in
93      * true when queue item is not available, false otherwise. Item is reduced using the
94      * reducer method provided with each queue item.
95      *
96      * @param queue                 page frame queue instance
97      * @param subSeq                subscriber sequence
98      * @param record                instance of record that can be positioned on the frame and each row in that frame
99      * @param circuitBreaker        circuit breaker instance
100      * @param stealingFrameSequence page frame sequence that is stealing work, it is used to identify if
101      *                              page frame sequence is stealing its own work or someone else's
102      * @return inverted value of queue processing status; true if nothing was processed.
103      */

105             RingQueue<PageFrameReduceTask> queue,

144     @Override
145     public boolean run(int workerId, @NotNull RunStatus runStatus) {
146         // there is job instance per thread, the worker id must never change
147         // for this job
148         boolean useful = false;
149         for (int i = 0; i < shardCount; i++) {
150             final int shard = shards[i];
151             useful = !consumeQueue(
152                     workerId,
153                     messageBus.getPageFrameReduceQueue(shard),
154                     messageBus.getPageFrameReduceSubSeq(shard),
155                     record,
156                     circuitBreaker,
157                     null // this is correct worker processing tasks rather than PageFrameSequence
158                     // helping to steal work
159             ) || useful;
160         }
161         return useful;
162     }

164     private static boolean consumeQueue(
165             int workerId,
166             RingQueue<PageFrameReduceTask> queue,
167             MCSequence subSeq,
168             PageAddressCacheRecord record,
169             SqlExecutionCircuitBreaker circuitBreaker,
170             @Nullable PageFrameSequence<?> stealingFrameSequence
171     ) {
172         // loop is required to deal with CAS errors, cursor == -2
173         do {
174             final long cursor = subSeq.next();
175             if (cursor > -1) {
176                 final PageFrameReduceTask task = queue.get(cursor);
177                 final PageFrameSequence<?> frameSequence = task.getFrameSequence();
178                 try {
179                     LOG.debug()
180                             .$("reducing [shard=").$(frameSequence.getShard())
181                             .$(", id=").$(frameSequence.getId())
182                             .$(", frameIndex=").$(task.getFrameIndex())
183                             .$(", frameCount=").$(frameSequence.getFrameCount())
184                             .$(", active=").$(frameSequence.isActive())
185                             .$(", cursor=").$(cursor)
186                             .I$();
187                     if (frameSequence.isActive()) {
188                         reduce(workerId, record, circuitBreaker, task, frameSequence, stealingFrameSequence);
189                     }
190                 } catch (Throwable e) {
191                     frameSequence.cancel();
192                     throw e;
193                 } finally {
194                     subSeq.done(cursor);
195                     // Reduce counter has to be incremented only when we make
196                     // sure that the task is available for consumers.
197                     frameSequence.getReduceCounter().incrementAndGet();
198                 }
199                 return false;
200             } else if (cursor == -1) {
201                 // queue is empty, we should yield or help
202                 break;
203             }
204             Os.pause();
205         } while (true);
206         return true;
207     }

209     private static void reduce(
210             int workerId,
211             PageAddressCacheRecord record,
212             SqlExecutionCircuitBreaker circuitBreaker,
213             PageFrameReduceTask task,
214             PageFrameSequence<?> frameSequence,
215             PageFrameSequence<?> stealingFrameSequence
216     ) {
217         // we deliberately hold the queue item because
218         // processing is daisy-chained. If we were to release item before
219         // finishing reduction, next step (job) will be processing an incomplete task
220         if (!circuitBreaker.checkIfTripped(frameSequence.getStartTime(), frameSequence.getCircuitBreakerFd())) {
221             record.of(frameSequence.getSymbolTableSource(), frameSequence.getPageAddressCache());
222             record.setFrameIndex(task.getFrameIndex());
223             assert !frameSequence.done;
224             frameSequence.getReducer().reduce(workerId, record, task, circuitBreaker, stealingFrameSequence);
225         } else {
226             frameSequence.cancel();
227         }
228     }
