107     void collected(boolean forceCollect) {
108         final long frameCount = frameSequence.getFrameCount();
109         // We have to reset capacity only on max all queue items
110         // What we are avoiding here is resetting capacity on 1000 frames given our queue size
111         // is 32 items. If our particular producer resizes queue items to 10x of the initial size
112         // we let these sizes stick until produce starts to wind down.
113         if (forceCollect || frameIndex >= frameCount - pageFrameQueueCapacity) {
114             resetCapacities();
115         }
116 
117         // we assume that frame indexes are published in ascending order
118         // and when we see the last index, we would free up the remaining resources
119         if (frameIndex + 1 == frameCount) {
120             frameSequence.reset();
121         }
122 
123         frameSequence = null;
124     }
