43     public DateLocaleFactory(TimeZoneRuleFactory timeZoneRuleFactory) {
44         this.timeZoneRuleFactory = timeZoneRuleFactory;
45         for (java.util.Locale l : java.util.Locale.getAvailableLocales()) {
46             String tag = l.toLanguageTag();
47             if ("und".equals(tag)) {
48                 tag = "";
49             }

52     }

58     public DateLocale getLocale(CharSequence id) {
59         DateLocale dateLocale = dateLocales.get(id);
60         if (dateLocale == null) {
61             return null;
62         }
63         if (dateLocale != dummyLocale) {
64             return dateLocale;
65         }
66         return dateLocales.compute(id, computeDateLocaleBiFunc);
67     }
68 
69     private DateLocale computeDateLocale(CharSequence key, DateLocale val) {
70         if (val != dummyLocale) {
71             // Someone was faster than us.
72             return val;
73         }
74         Locale locale = Locale.forLanguageTag(key.toString());
75         return new DateLocale(new DateFormatSymbols(locale), timeZoneRuleFactory);
76     }
