121     private static void defineToken(String token, int pos, IntObjHashMap<ObjList<CharSequence>> map) {
122         if (token == null || token.length() == 0) {
123             return;
124         }
125 
126         char c0 = Character.toUpperCase(token.charAt(0));
127         int index = map.keyIndex(c0);
128         ObjList<CharSequence> l;
129         if (index > -1) {
130             l = new ObjList<>();
131             map.putAt(index, c0, l);
132         } else {
133             l = map.valueAtQuick(index);
134         }
135         l.add(((char) pos) + token.toUpperCase());
136     }

138     private static long findToken(CharSequence content, int lo, int hi, IntObjHashMap<ObjList<CharSequence>> map) throws NumericException {
139 
140         if (lo >= hi) {
141             throw NumericException.INSTANCE;
142         }
143 
144         char c = Character.toUpperCase(content.charAt(lo));
145 
146         ObjList<CharSequence> l = map.get(c);
147         if (l == null) {
148             throw NumericException.INSTANCE;
149         }
150 
151         for (int i = 0, sz = l.size(); i < sz; i++) {
152             CharSequence txt = l.get(i);
153             int n = txt.length() - 1;
154             boolean match = n <= hi - lo;
155             if (match) {
156                 for (int k = 1; k < n; k++) {
157                     if (Character.toUpperCase(content.charAt(lo + k)) != txt.charAt(k + 1)) {
158                         match = false;
159                         break;
160                     }
161                 }
162             }
163 
164             if (match) {
165                 return Numbers.encodeLowHighInts(txt.charAt(0), n);
166             }
167         }
168 
169         throw NumericException.INSTANCE;
170     }

179     private void indexZones(String[][] zones, TimeZoneRuleFactory timeZoneRuleFactory) {
180         CharSequenceHashSet cache = new CharSequenceHashSet();
181         // this is a workaround a problem where UTC timezone comes nearly last
182         // in this array, which gives way to Antarctica/Troll take its place
183 
184         if (cache.add("UTC")) {
185             int index = timeZoneRuleFactory.getTimeZoneRulesIndex("UTC");
186             if (index != -1) {
187                 defineToken("UTC", index, this.zones);
188             }
189         }
190 
191         // end of workaround
192 
193         for (int i = 0, n = zones.length; i < n; i++) {
194             String[] zNames = zones[i];
195             String key = zNames[0];
196 
197             int index = timeZoneRuleFactory.getTimeZoneRulesIndex(key);
198             if (index == -1) {
199                 continue;
200             }
201 
202             for (int k = 0, m = zNames.length; k < m; k++) {
203                 String name = zNames[k];
204                 // we already added this name, skip
205                 if (cache.add(name)) {
206                     defineToken(name, index, this.zones);
207                 }
208             }
209         }
210         sort(this.zones);
211     }
