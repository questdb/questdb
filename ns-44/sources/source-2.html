183     private long getDSTFromRule(int year, boolean leap, int i) {
184         int offsetBefore;
185         TransitionRule zr = rules.getQuick(i);
186         offsetBefore = zr.offsetBefore;
187 
188         int dom = zr.dom;
189         int month = zr.month;
190 
191         int dow = zr.dow;
192         long date;
193         if (dom < 0) {
194             date = toEpoch(
195                     year,
196                     leap,
197                     month,
198                     getDaysPerMonth(month, leap) + 1 + dom,
199                     zr.hour,
200                     zr.minute
201             ) + zr.second * multiplier;
202             if (dow > -1) {
203                 date = previousOrSameDayOfWeek(date, dow);
204             }
205         } else {
206             assert month > 0;
207             date = toEpoch(year, leap, month, dom, zr.hour, zr.minute) + zr.second * multiplier;
208             if (dow > -1) {
209                 date = nextOrSameDayOfWeek(date, dow);
210             }
211         }
212 
213         if (zr.midnightEOD) {
214             date = addDays(date, 1);
215         }
216 
217         switch (zr.timeDef) {
218             case TransitionRule.UTC:
219                 date += (offsetBefore - ZoneOffset.UTC.getTotalSeconds()) * multiplier;
220                 break;
221             case TransitionRule.STANDARD:
222                 date += (offsetBefore - zr.standardOffset) * multiplier;
223                 break;
224             default:  // WALL
225                 break;
226         }
227 
228         // go back to epoch epoch
229         date -= offsetBefore * multiplier;
230         return date;
231     }
