40 public class TimeZoneRuleFactory {

42     public static final TimeZoneRuleFactory INSTANCE = new TimeZoneRuleFactory();
43     public static final int RESOLUTION_MICROS = 1;

45     private final ObjList<TimeZoneRules> ruleList = new ObjList<>();
46     private final CharSequenceIntHashMap ruleMap = new CharSequenceIntHashMap();
47 
48     public TimeZoneRuleFactory() {
49         int index = 0;
50         for (String z : ZoneRulesProvider.getAvailableZoneIds()) {
51             final ZoneRules rules = ZoneRulesProvider.getRules(z, true);
52             ruleList.add(new TimeZoneRulesMillis(rules));
53             ruleList.add(new TimeZoneRulesMicros(rules));
54             ruleMap.put(z, index++);
55         }
56 
57         for (Map.Entry<String, String> e : ZoneId.SHORT_IDS.entrySet()) {
58             String key = e.getKey();
59             String alias = e.getValue();
60 
61             // key already added somehow?
62             int i = ruleMap.get(key);
63             if (i == -1) {
64                 // no, good, add
65                 i = ruleMap.get(alias);
66                 if (i == -1) {
67                     // this could be fixed offset, try parsing value as one
68                     long offset = Dates.parseOffset(alias, 0, alias.length());
69                     if (offset != Long.MIN_VALUE) {
70                         ruleList.add(new FixedTimeZoneRule(Numbers.decodeLowInt(offset) * Dates.MINUTE_MILLIS));
71                         ruleList.add(new FixedTimeZoneRule(Numbers.decodeLowInt(offset) * Timestamps.MINUTE_MICROS));
72                         ruleMap.put(key, index++);
73                     }
74                 } else {
75                     ruleMap.put(key, i);
76                 }
77             }
78         }
79     }
80 
81     public int getTimeZoneRulesIndex(CharSequence id) {

83     }
