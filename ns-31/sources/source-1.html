51     public static String escapeSpecialChars(CharSequence pattern, CharSequence prev) {
52         int len = pattern.length();
53 
54         StringSink sink = Misc.getThreadLocalBuilder();
55         for (int i = 0; i < len; i++) {
56             char c = pattern.charAt(i);
57             if (c == '_')
58                 sink.put('.');
59             else if (c == '%')
60                 sink.put(".*?");
61             else if ("[](){}.*+?$^|#\\".indexOf(c) != -1) {
62                 sink.put("\\");
63                 sink.put(c);
64             } else
65                 sink.put(c);
66         }
67 
68         if (Chars.equalsNc(sink, prev)) {
69             return null;
70         }
71         return Chars.toString(sink);
72     }
