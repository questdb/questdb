111         public CharSequence getStr(Record rec, StringBufferSink sink) {
112             CharSequence cs = value.getStr(rec);
113             if (cs == null) {
114                 return null;
115             }
116 
117             matcher.reset(cs);
118             sink.clear();
119 
120             boolean result = matcher.find();
121             if (!result) {
122                 sink.buffer.append(cs);
123             } else {
124                 do {
125                     if (sink.length() > maxLength) {
126                         throw CairoException.nonCritical()
127                                 .put("breached memory limit set for ").put(SIGNATURE)
128                                 .put(" [maxLength=").put(maxLength).put(']');
129                     }
130                     matcher.appendReplacement(sink.buffer, replacementStr);
131                     result = matcher.find();
132                 } while (result);
133                 matcher.appendTail(sink.buffer);
134             }
135             return sink;
136         }
