155         @Override
156         public boolean hasNext() {
157             if (state == INIT_FIRST_TIMESTAMP) {
158                 // first iteration to get initial timestamp value
159                 if (baseCursor.hasNext()) {
160                     prevTimestamp = record.getTimestamp(timestampIndex);
161                     prevRowId = record.getRowId();
162                     state = REUSE_CURRENT;
163                 } else {
164                     // edge case - base cursor is empty, avoid calling hasNext() again
165                     state = NO_ROWS;
166                 }
167             }
168 
169             if (state == COMPUTE_NEXT) {
170                 while (baseCursor.hasNext()) {
171                     circuitBreaker.statefulThrowExceptionIfTripped();
172                     final long timestamp = record.getTimestamp(timestampIndex);
173                     if (timestamp != prevTimestamp) {
174                         prevTimestamp = timestamp;
175                         prevRowId = record.getRowId();
176                         return true;
177                     }
178 
179                     if (checkIfNotDupe()) {
180                         return true;
181                     }
182                 }
183                 return false;
184             }
185 
186             boolean next = (state == REUSE_CURRENT);
187             state = COMPUTE_NEXT;
188             return next;
189         }

225         private boolean checkIfNotDupe() {
226             MapKey key;
227             if (prevRowId != -1) {
228                 // jump recordB to the prev record of the base cursor
229                 baseCursor.recordAt(recordB, prevRowId);
230 
231                 dataMap.clear();
232                 key = dataMap.withKey();
233                 recordSink.copy(recordB, key);
234                 // we have already returned this record
235                 // and map should be empty
236                 key.create();
237                 prevRowId = -1;
238             }
239 
240             key = dataMap.withKey();
241             // now copy the current record to the map
242             recordSink.copy(record, key);
243             // record is not a duplicate
244             return key.create();
245         }
