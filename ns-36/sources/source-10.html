73

75                 // some functions may need more than one column in values,

78                 GroupByFunction func = (GroupByFunction) function;
79                 func.pushValueTypes(valueTypes);
80                 groupByFunctions.add(func);
81                 groupByFunctionPositions.add(columnAst.position);
82             }
83         }

87             @NotNull QueryModel model,
88             RecordMetadata metadata,
89             @NotNull ListColumnFilter listColumnFilter,
90             ObjList<GroupByFunction> groupByFunctions,
91             @Transient IntList groupByFunctionPositions,
92             ObjList<Function> recordFunctions,
93             @Transient IntList recordFunctionPositions,
94             GenericRecordMetadata groupByMetadata,
95             ArrayColumnTypes keyTypes,
96             int keyColumnIndex,
97             boolean timestampUnimportant,
98             int timestampIndex
99     ) throws SqlException {
100 
101         recordFunctionPositions.clear();
102 
103         // Process group-by functions first to get the idea of
104         // how many map values we will have.
105         // Map value count is needed to calculate offsets for
106         // map key columns.
107 
108         ObjList<QueryColumn> columns = model.getColumns();
109         int valueColumnIndex = 0;
110         int inferredKeyColumnCount = 0;
111 
112         // when we have same column several times in a row
113         // we only add it once to map keys
114         int lastIndex = -1;
115         for (int i = 0, n = columns.size(); i < n; i++) {
116             final QueryColumn column = columns.getQuick(i);
117             final ExpressionNode node = column.getAst();
118             final int type;
119 
120             if (node.type == ExpressionNode.LITERAL) {
121                 // this is key
122                 int index = metadata.getColumnIndexQuiet(node.token);
123                 if (index == -1) {
124                     throw SqlException.invalidColumn(node.position, node.token);
125                 }
126

128                 if (index != timestampIndex || timestampUnimportant) {
129                     if (lastIndex != index) {
130                         listColumnFilter.add(index + 1);
131                         keyTypes.add(type);
132                         keyColumnIndex++;
133                         lastIndex = index;
134                     }
135 
136                     final Function fun;
137                     switch (ColumnType.tagOf(type)) {
138                         case ColumnType.BOOLEAN:

144                         case ColumnType.SHORT:
145                             fun = ShortColumn.newInstance(keyColumnIndex - 1);
146                             break;
147                         case ColumnType.CHAR:
148                             fun = CharColumn.newInstance(keyColumnIndex - 1);
149                             break;
150                         case ColumnType.INT:
151                             fun = IntColumn.newInstance(keyColumnIndex - 1);
152                             break;
153                         case ColumnType.LONG:
154                             fun = LongColumn.newInstance(keyColumnIndex - 1);
155                             break;
156                         case ColumnType.FLOAT:
157                             fun = FloatColumn.newInstance(keyColumnIndex - 1);
158                             break;
159                         case ColumnType.DOUBLE:
160                             fun = DoubleColumn.newInstance(keyColumnIndex - 1);
161                             break;
162                         case ColumnType.STRING:
163                             fun = StrColumn.newInstance(keyColumnIndex - 1);

262     public static void validateGroupByColumns(@NotNull QueryModel model, int inferredKeyColumnCount) throws SqlException {
263         final ObjList<ExpressionNode> groupByColumns = model.getGroupBy();
264         int explicitKeyColumnCount = groupByColumns.size();
265         if (explicitKeyColumnCount == 0) {
266             return;
267         }
268 
269         final QueryModel nested = model.getNestedModel();
270         QueryModel chooseModel = model;
271         while (chooseModel != null && chooseModel.getSelectModelType() != QueryModel.SELECT_MODEL_CHOOSE && chooseModel.getSelectModelType() != QueryModel.SELECT_MODEL_NONE) {
272             chooseModel = chooseModel.getNestedModel();
273         }
274 
275         for (int i = 0; i < explicitKeyColumnCount; i++) {
276             final ExpressionNode key = groupByColumns.getQuick(i);
277             switch (key.type) {
278                 case ExpressionNode.LITERAL:
279                     final int dotIndex = Chars.indexOf(key.token, '.');
280 
281                     if (dotIndex > -1) {
282                         int aliasIndex = model.getModelAliasIndex(key.token, 0, dotIndex);
283                         if (aliasIndex > -1) {
284                             // we should now check against main model
285                             int refColumn = model.getAliasToColumnMap().keyIndex(key.token);
286                             if (refColumn > -1) {
287                                 // a.x not found, look for "x"
288                                 refColumn = model.getAliasToColumnMap().keyIndex(key.token, dotIndex + 1, key.token.length());
289                             }
290 
291                             if (refColumn > -1) {
292                                 throw SqlException.$(key.position, "group by column does not match any key column is select statement");
293                             }
294 
295                         } else {
296 
297                             // the table alias could be referencing join model
298                             // we need to descend to first NONE model and see if that can resolve columns we are
299                             // looking for
300 
301                             if (chooseModel != null && chooseModel.getColumnNameToAliasMap().keyIndex(key.token) < 0) {
302                                 continue;
303                             }
304                             throw SqlException.$(key.position, "invalid column reference");
305                         }
306                     } else {
307                         int refColumn = model.getAliasToColumnMap().keyIndex(key.token);
308                         if (refColumn > -1) {
309                             throw SqlException.$(key.position, "group by column does not match any key column is select statement");
310                         }
311                         QueryColumn qc = model.getAliasToColumnMap().valueAt(refColumn);
312                         if (qc.getAst().type != ExpressionNode.LITERAL && qc.getAst().type != ExpressionNode.CONSTANT) {
313                             throw SqlException.$(key.position, "group by column references aggregate expression");
314                         }
315                     }
316                     break;
317                 case ExpressionNode.BIND_VARIABLE:
318                     throw SqlException.$(key.position, "bind variable is not allowed here");
319                 case ExpressionNode.FUNCTION:
320                 case ExpressionNode.OPERATION:
321                 case ExpressionNode.CONSTANT:
322                     final ObjList<QueryColumn> availableColumns = nested.getTopDownColumns();
323                     boolean invalid = true;
324                     for (int j = 0, n = availableColumns.size(); j < n; j++) {
325                         final QueryColumn qc = availableColumns.getQuick(j);
326                         if (qc.getAst().type == key.type) {
327                             if (ExpressionNode.compareNodesGroupBy(key, qc.getAst(), chooseModel)) {
328                                 invalid = false;
329                                 break;
330                             }
331                         }
332                     }
333                     if (invalid) {
334                         throw SqlException.$(key.position, "group by expression does not match anything select in statement");
335                     }
336                     break;
337                 default:
338                     throw SqlException.$(key.position, "unsupported type of expression");
339             }
340         }
341 
342         if (explicitKeyColumnCount < inferredKeyColumnCount) {
343             throw SqlException.$(model.getModelPosition(), "not enough columns in group by");
344         }
345     }
