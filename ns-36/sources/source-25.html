303             if (!isMapInitialized) {
304                 if (!initMap()) {
305                     // managed cursor has no data, nothing to do
306                     return;
307                 }
308                 isMapInitialized = true;
309             }

312                 fillMap();

314             }

317

319                 final RecordCursor mapCursor = recordKeyMap.getCursor();

323

329                             // to timestamp after 'sample' to begin with

333                                 if (value.getByte(0) == 0) {
334                                     interpolateBoundaryRange(x1, x2, mapRecord);

339                             }

343             }
344 
345             // find gaps by checking each of the unique keys against every sample
346             long sample;
347             long prevSample;
348             for (sample = prevSample = loSample; sample < hiSample; prevSample = sample, sample = sampler.nextTimestamp(sample)) {

351                 while (mapCursor.hasNext()) {

357                         // gap is at 'sample', so potential X-value is at 'prevSample'

470         private void fillMap() {
471             // Evaluate group-by functions.
472             // On every change of timestamp sample value we
473             // check group for gaps and fill them with placeholder
474             // entries. Values for these entries will be interpolated later.
475 
476             if (prevSample == -1) {
477                 // we have data in cursor, so we can grab first value
478                 final boolean good = managedCursor.hasNext();
479                 assert good;
480                 long timestamp = managedRecord.getTimestamp(timestampIndex);
481                 sampler.setStart(timestamp);
482                 prevSample = sampler.round(timestamp);
483                 loSample = prevSample; // the lowest timestamp value
484             }
485 
486             do {
487                 circuitBreaker.statefulThrowExceptionIfTripped();
488 
489                 if (!isHasNextPending) {
490                     // this seems inefficient, but we only double-sample
491                     // very first record and nothing else
492                     long sample = sampler.round(managedRecord.getTimestamp(timestampIndex));
493                     if (sample != prevSample) {
494                         // before we continue with next interval
495                         // we need to fill gaps in current interval
496                         // we will go over unique keys and attempt to
497                         // find them in data map with current timestamp
498 
499                         fillGaps(prevSample, sample);
500                         prevSample = sample;
501                         GroupByUtils.toTop(groupByFunctions);
502                     }
503 
504                     // same data group - evaluate group-by functions
505                     MapKey key = dataMap.withKey();
506                     mapSink.copy(managedRecord, key);
507                     key.putLong(sample);
508 
509                     MapValue value = key.createValue();
510                     if (value.isNew()) {
511                         value.putByte(0, (byte) 0); // not a gap
512                         for (int i = 0; i < groupByFunctionCount; i++) {
513                             groupByFunctions.getQuick(i).computeFirst(value, managedRecord);
514                         }
515                     } else {
516                         for (int i = 0; i < groupByFunctionCount; i++) {
517                             groupByFunctions.getQuick(i).computeNext(value, managedRecord);
518                         }
519                     }
520                 }
521 
522                 isHasNextPending = true;
523                 boolean hasNext = managedCursor.hasNext();
524                 isHasNextPending = false;
525 
526                 if (!hasNext) {
527                     hiSample = sampler.nextTimestamp(prevSample);
528                     break;
529                 }
530             } while (true);
531 
532             // fill gaps if any at the end of base cursor
533             fillGaps(prevSample, hiSample);
534         }

557         private boolean initMap() {
558             // Collect map of unique key values.
559             // using these values we will fill gaps in main
560             // data before jumping to another timestamp.
561             // This will allow maintaining chronological order of
562             // main data map.
563             //
564             // At the same time check if cursor has data.
565             while (managedCursor.hasNext()) {
566                 circuitBreaker.statefulThrowExceptionIfTripped();
567 
568                 final MapKey key = recordKeyMap.withKey();
569                 mapSink.copy(managedRecord, key);
570                 key.createValue();
571             }
572 
573             // no data, nothing to do
574             if (recordKeyMap.size() == 0) {
575                 return false;
576             }
577 
578             // toTop() guarantees that we get
579             // the same data as previous while() loop
580             // there is no data
581             managedCursor.toTop();
582             return true;
583         }

585         private void interpolate(long lo, long hi, Record mapRecord, long x1, long x2, MapValue x1Value, MapValue x2value) {
586             computeYPoints(x1Value, x2value);
587             for (long x = lo; x < hi; x = sampler.nextTimestamp(x)) {
588                 final MapValue result = findDataMapValue3(mapRecord, x);
589                 assert result != null && result.getByte(0) == 1;
590                 for (int i = 0; i < groupByTwoPointFunctionCount; i++) {
591                     GroupByFunction function = groupByTwoPointFunctions.getQuick(i);
592                     InterpolationUtil.interpolateGap(function, result, sampler.getBucketSize(), x1Value, x2value);
593                 }
594                 for (int i = 0; i < groupByScalarFunctionCount; i++) {
595                     GroupByFunction function = groupByScalarFunctions.getQuick(i);
596                     interpolatorFunctions.getQuick(i).interpolateAndStore(function, result, x, x1, x2, yData + i * 16L, yData + i * 16L + 8);
597                 }
598                 result.putByte(0, (byte) 0); // fill the value, change flag from 'gap' to 'fill'
599             }
600         }

602         private void interpolateBoundaryRange(long x1, long x2, Record record) {
603             // interpolating boundary
