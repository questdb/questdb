38      * Creates a GroupByFunctionUpdater instance capturing the provided group by functions.
39      * The generated class will have fields GroupByFunction f0, GroupByFunction f1, GroupByFunction f2 ... GroupByFunction fn
40      * for each group by function from the provided list.
41      * <p>
42      * The generated class will have the following methods:
43      * <ul>
44      * <li>updateNew(MapValue value, Record record) - calls f0, f1, f2 ... fn.computeFirst(value, record) for each group by function</li>
45      * <li>updateExisting(MapValue value, Record record) - calls f0, f1, f2 ... fn.computeNext(value, record) for each group by function</li>
46      * <li>updateEmpty(MapValue value) - calls f0, f1, f2 ... fn.setEmpty(value) for each group by function</li>
47      * <li>setFunctions(ObjList&lt;GroupByFunction&gt; groupByFunctions) - sets the group by functions to the fields. This method is called by the factory and should not be called by the caller.</li>
48      * </ul>

51      * @param groupByFunctions list of group by functions
52      * @return GroupByFunctionUpdater instance

124     private static void generateSetFunction(BytecodeAssembler asm, int functionSize, int firstFieldIndex, int setFunctionsIndex, int setFunctionsSigIndex, int getIndex) {
125         asm.startMethod(setFunctionsIndex, setFunctionsSigIndex, 3, 3);
126         for (int i = 0; i < functionSize; i++) {
127             asm.aload(0);
128             asm.aload(1);

132         }
