89     private long addYears(long timestamp, int bucket) {
90         if (bucket == 0) {
91             return timestamp;
92         }
93         final int y = Timestamps.getYear(timestamp);
94         final boolean leap = Timestamps.isLeapYear(y + bucket);
95         final int maxDay = Math.min(startDay, Timestamps.getDaysPerMonth(startMonth, leap)) - 1;
96         return Timestamps.yearMicros(y + bucket, leap)
97                 + Timestamps.monthOfYearMicros(startMonth, leap)
98                 + maxDay * Timestamps.DAY_MICROS
99                 + startHour * Timestamps.HOUR_MICROS
100                 + startMin * Timestamps.MINUTE_MICROS
101                 + startSec * Timestamps.SECOND_MICROS
102                 + startMillis * Timestamps.MILLI_MICROS
103                 + startMicros;
