175     private void buildFirstLastIndex(
176             int[] firstLastIndex,
177             int[] queryToFrameColumnMapping,
178             RecordMetadata metadata,
179             ObjList<QueryColumn> columns,
180             int timestampIndex,
181             boolean[] isKeyColumn
182     ) throws SqlException {
183         for (int i = 0, n = firstLastIndex.length; i < n; i++) {
184             QueryColumn column = columns.getQuick(i);
185             ExpressionNode ast = column.getAst();
186             int resultSetColumnType = getMetadata().getColumnType(i);
187             if (ast.rhs != null) {
188                 if (SqlKeywords.isLastKeyword(ast.token)) {
189                     firstLastIndex[i] = LAST_OUT_INDEX;
190                 } else if (SqlKeywords.isFirstKeyword(ast.token)) {
191                     firstLastIndex[i] = FIRST_OUT_INDEX;
192                 } else {
193                     throw SqlException.$(ast.position, "expected first() or last() functions but got ").put(ast.token);
194                 }
195                 int underlyingColIndex = metadata.getColumnIndex(ast.rhs.token);
196                 queryToFrameColumnMapping[i] = underlyingColIndex;
197 
198                 int underlyingType = metadata.getColumnType(underlyingColIndex);
199                 if (underlyingType != resultSetColumnType || ColumnType.pow2SizeOf(resultSetColumnType) > 3) {
200                     throw SqlException.$(ast.position, "column \"")
201                             .put(metadata.getColumnName(underlyingColIndex))
202                             .put("\": first(), last() is not supported on data type ")
203                             .put(ColumnType.nameOf(underlyingType))
204                             .put(" ");
205                 }
206             } else {
207                 int underlyingColIndex = metadata.getColumnIndex(ast.token);
208                 isKeyColumn[i] = true;
209                 queryToFrameColumnMapping[i] = underlyingColIndex;
210                 if (underlyingColIndex == timestampIndex) {
211                     groupByTimestampIndex = i;
212                 }
213             }
214         }

391         // This method evaluates RecordCursor state
392         // by using state machine.
393         // Possible important states are:
394         // - START
395         // - FETCH_NEXT_DATA_FRAME
396         // - FETCH_NEXT_INDEX_FRAME
397         // - SEARCH
398         // - RETURN_LAST_ROW
399         // - DONE
400         // State machine can switch states non-linear until DONE reached.
401         private int getNextState(int state) {
402             // This method should not change this.state field
403             // but instead produce next state given incoming state parameter.
404             // This way the method is enforced to change state and less the caller to loop forever.
405             switch (state) {
406                 case STATE_START:
407                     prevSamplePeriodOffset = samplePeriodIndexOffset = 0;
408                     crossRowState = NONE;
409                     samplePeriodStart = Numbers.LONG_NaN;
410                     // Fall through to STATE_FETCH_NEXT_DATA_FRAME;
411 
412                 case STATE_FETCH_NEXT_DATA_FRAME:
413                     currentFrame = pageFrameCursor.next();
414                     if (currentFrame != null) {
415                         record.switchFrame();
416 
417                         // Switch to new data frame
418                         frameNextRowId = dataFrameLo = currentFrame.getPartitionLo();
419                         dataFrameHi = dataFrameLo + currentFrame.getPageSize(timestampIndex) / Long.BYTES;
420 
421                         // Re-fetch index cursor to correctly position it to frameNextRowId
422                         BitmapIndexReader symbolIndexReader = currentFrame.getBitmapIndexReader(groupBySymbolColIndex, BitmapIndexReader.DIR_FORWARD);
423                         indexCursor = symbolIndexReader.getFrameCursor(groupBySymbolKey, dataFrameLo, dataFrameHi);
424 
425                         // Fall through to STATE_FETCH_NEXT_INDEX_FRAME;
426                     } else {
427                         return STATE_RETURN_LAST_ROW;
428                     }
429 
430                 case STATE_FETCH_NEXT_INDEX_FRAME:
431                     indexFrame = indexCursor.getNext();
432                     indexFramePosition = 0;
433 
434                     long indexFrameAddress = indexFrame.getAddress();
435                     if (indexFrame.getSize() == 0) {
436                         if (indexFrameAddress != 0 || groupBySymbolKey != FILTER_KEY_IS_NULL) {
437                             // No rows in index for this dataframe left, go to next data frame
438                             frameNextRowId = dataFrameHi;
439                             // Jump back to fetch next data frame
440                             return STATE_FETCH_NEXT_DATA_FRAME;
441                         }
442                         // Special case - searching with `where symbol = null` on the partition where this column has not been added
443                         // Effectively all rows in data frame are the match to the symbol filter
444                         // Fall through, search code will figure that this is special case
445                     }
446 
447                     if (samplePeriodStart == Numbers.LONG_NaN) {
448                         long rowId = indexFrameAddress > 0 ? Unsafe.getUnsafe().getLong(indexFrameAddress) : dataFrameLo;
449                         long offsetTimestampColumnAddress = currentFrame.getPageAddress(timestampIndex) - dataFrameLo * Long.BYTES;
450                         samplePeriodStart = Unsafe.getUnsafe().getLong(offsetTimestampColumnAddress + rowId * Long.BYTES);
451                         startFrom(samplePeriodStart);
452                     }
453                     // Fall to STATE_SEARCH;
454 
455                 case STATE_OUT_BUFFER_FULL:
456                 case STATE_SEARCH:
457                     int outPosition = crossRowState == NONE ? 0 : 1;
458                     long offsetTimestampColumnAddress = currentFrame.getPageAddress(timestampIndex) - dataFrameLo * Long.BYTES;
459                     long iFrameAddress = indexFrame.getAddress();
460                     long iFrameSize = indexFrame.getSize();
461                     long lastIndexRowId = iFrameAddress > 0
462                             ? Unsafe.getUnsafe().getLong(iFrameAddress + (iFrameSize - 1) * Long.BYTES)
463                             : Long.MAX_VALUE;
464                     long lastInDataRowId = Math.min(lastIndexRowId, dataFrameHi - 1);
465                     long lastInDataTimestamp = Unsafe.getUnsafe().getLong(offsetTimestampColumnAddress + lastInDataRowId * Long.BYTES);
466                     int samplePeriodCount = fillSamplePeriodsUntil(lastInDataTimestamp);
467 
468                     rowsFound = BitmapIndexUtilsNative.findFirstLastInFrame(
469                             outPosition,
470                             frameNextRowId,
471                             dataFrameHi,
472                             offsetTimestampColumnAddress,
473                             dataFrameLo,
474                             iFrameAddress,
475                             iFrameSize,
476                             indexFramePosition,
477                             samplePeriodAddress.getAddress(),
478                             samplePeriodCount,
479                             samplePeriodIndexOffset,
480                             rowIdOutAddress.getAddress(),
481                             pageSize);
482 
483                     boolean firstRowLastRowIdUpdated = rowsFound < 0;
484                     rowsFound = Math.abs(rowsFound);
485 
486                     // If first row last() RowId is updated
487                     // re-copy last values to crossFrameRow
488                     checkSaveLastValues(firstRowLastRowIdUpdated);
489 
490                     int lastOutIndex = rowsFound << ITEMS_PER_OUT_ARRAY_SHIFT;
491                     prevSamplePeriodOffset = samplePeriodIndexOffset;
492                     indexFramePosition = (int) rowIdOutAddress.get(lastOutIndex + FIRST_OUT_INDEX);
493                     frameNextRowId = rowIdOutAddress.get(lastOutIndex + LAST_OUT_INDEX);
494                     samplePeriodIndexOffset = rowIdOutAddress.get(lastOutIndex + TIMESTAMP_OUT_INDEX);
495                     samplePeriodStart = samplePeriodAddress.get(samplePeriodIndexOffset - prevSamplePeriodOffset);
496 
497                     // decide what to do next
498                     int newState;
499                     if (frameNextRowId >= dataFrameHi) {
500                         // Data frame exhausted. Next time start from fetching new data frame
501                         newState = STATE_FETCH_NEXT_DATA_FRAME;
502                     } else if (indexFramePosition >= iFrameSize) {
503                         // Index frame exhausted. Next time start from fetching new index frame
504                         newState = iFrameAddress > 0 ? STATE_FETCH_NEXT_INDEX_FRAME : STATE_FETCH_NEXT_DATA_FRAME;
505                     } else if (rowsFound == pageSize - 1) {
506                         // output rows filled the output buffers or
507                         newState = STATE_OUT_BUFFER_FULL;
508                     } else if (samplePeriodIndexOffset - prevSamplePeriodOffset == maxSamplePeriodSize - 1) {
509                         //  search came to the end of sample by periods
510                         // re-fill periods and search again
511                         newState = STATE_SEARCH;
512                     } else {
513                         // Data frame exhausted. Next time start from fetching new data frame
514                         newState = STATE_FETCH_NEXT_DATA_FRAME;
515                     }
516 
517                     if (rowsFound > 1) {
518                         record.of(currentRow = 0);
519                         newState = -newState;
520                     }
521 
522                     // No rows to iterate, return where to continue from (fetching next data or index frame)
523                     return newState;
524 
525                 case STATE_RETURN_LAST_ROW:
526                     if (crossRowState != NONE) {
527                         record.of(currentRow = 0);
528                         // Signal there is a row by returning negative value
529                         return -STATE_DONE;
530                     }
531                     // Fall through to STATE_DONE;
532 
533                 case STATE_DONE:
534                     return STATE_DONE;
535 
536                 default:
537                     throw new UnsupportedOperationException("Invalid state " + state);
538             }
539         }

542             // Check if values from the last found sample by row have to be saved in crossFrameRow
543             checkCrossRowAfterFoundBufferIterated();
544             rowsFound = 0;
545 
546             while (state != STATE_DONE) {
547                 state = getNextState(state);
548                 if (state < 0) {

550                     return true;

552             }
