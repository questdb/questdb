110     static Function createPlaceHolderFunction(IntList recordFunctionPositions, int index, int type) throws SqlException {
111         switch (ColumnType.tagOf(type)) {
112             case ColumnType.INT:
113                 return IntConstant.NULL;
114             case ColumnType.LONG:
115                 return LongConstant.NULL;
116             case ColumnType.FLOAT:
117                 return FloatConstant.NULL;
118             case ColumnType.DOUBLE:
119                 return DoubleConstant.NULL;
120             case ColumnType.BYTE:
121                 return ByteConstant.ZERO;
122             case ColumnType.SHORT:
123                 return ShortConstant.ZERO;
124             case ColumnType.GEOBYTE:
125                 return GeoByteConstant.NULL;
126             case ColumnType.GEOSHORT:
127                 return GeoShortConstant.NULL;
128             case ColumnType.GEOINT:
129                 return GeoIntConstant.NULL;
130             case ColumnType.GEOLONG:
131                 return GeoLongConstant.NULL;
132             case ColumnType.UUID:
133                 return UuidConstant.NULL;
134             default:
135                 throw SqlException.$(recordFunctionPositions.getQuick(index), "Unsupported type: ").put(ColumnType.nameOf(type));
136         }
137     }

139     @NotNull
140     static ObjList<Function> createPlaceholderFunctions(
141             ObjList<Function> recordFunctions,
142             IntList recordFunctionPositions

144         final ObjList<Function> placeholderFunctions = new ObjList<>();
145         for (int i = 0, n = recordFunctions.size(); i < n; i++) {
146             Function function = recordFunctions.getQuick(i);
147             if (function instanceof GroupByFunction) {
148                 placeholderFunctions.add(createPlaceHolderFunction(recordFunctionPositions, i, function.getType()));
149             } else {
150                 placeholderFunctions.add(function);
151             }
152         }
153         return placeholderFunctions;
154     }
