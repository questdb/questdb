77     protected void parseParams(RecordCursor base, SqlExecutionContext executionContext) throws SqlException {
78         // factory guarantees that base cursor is not empty

83         final CharSequence tz = timezoneNameFunc.getStr(null);
84         if (tz != null) {
85             try {

94                 } else {
95                     // here timezone is in numeric offset format

110             if (val == Numbers.LONG_NaN) {
111                 // bad value for offset
112                 throw SqlException.$(offsetFuncPos, "invalid offset: ").put(offset);
113             }
114             fixedOffset = Numbers.decodeLowInt(val) * MINUTE_MICROS;
115         } else {
116             fixedOffset = Long.MIN_VALUE;

118     }
