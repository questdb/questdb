31 public final class LongNullUtils {
32     private static final long[] LONG_NULLs = new long[ColumnType.MAX];
33 
34     public static long getLongNull(int type) {
35         return LONG_NULLs[ColumnType.tagOf(type)];
36     }
37 
38     static {
39         for (int i = 0, n = LONG_NULLs.length; i < n; i++) {
40             switch (i) {
41                 case ColumnType.LONG:
42                 case ColumnType.TIMESTAMP:
43                 case ColumnType.DATE:
44                     LONG_NULLs[i] = Numbers.LONG_NaN;
45                     break;
46 
47                 case ColumnType.INT:
48                 case ColumnType.SYMBOL:
49                     LONG_NULLs[i] = Numbers.INT_NaN;
50                     break;
51 
52                 case ColumnType.FLOAT:
53                     LONG_NULLs[i] = Float.floatToIntBits(Float.NaN);
54                     break;
55 
56                 case ColumnType.DOUBLE:
57                     LONG_NULLs[i] = Double.doubleToLongBits(Double.NaN);
58                     break;
59 
60                 case ColumnType.GEOBYTE:
61                 case ColumnType.GEOSHORT:
62                 case ColumnType.GEOINT:
63                 case ColumnType.GEOLONG:
64                     LONG_NULLs[i] = GeoHashes.NULL;
65                     break;
66 
67                 default:
68                     LONG_NULLs[i] = 0L;
69                     break;
70             }
71         }
72     }
