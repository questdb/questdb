132     protected long adjustDst(long timestamp, @Nullable MapValue mapValue, long nextSampleTimestamp) {
133         final long utcTimestamp = timestamp - tzOffset;
134         if (utcTimestamp < nextDstUtc) {
135             return timestamp;
136         }
137         final long newTzOffset = rules.getOffset(utcTimestamp);
138         prevDst = nextDstUtc;
139         nextDstUtc = rules.getNextDST(utcTimestamp);
140         // check if DST takes this timestamp back "before" the nextSampleTimestamp
141         if (timestamp - (tzOffset - newTzOffset) < nextSampleTimestamp) {
142             // time moved backwards, we need to check if we should be collapsing this
143             // hour into previous period or not
144             updateValueWhenClockMovesBack(mapValue);
145             nextSampleLocalEpoch = timestampSampler.round(timestamp);
146             localEpoch = nextSampleLocalEpoch;
147             sampleLocalEpoch += (newTzOffset - tzOffset);
148             tzOffset = newTzOffset;
149             return Long.MIN_VALUE;
150         }
151         kludge(newTzOffset);
152 
153         // time moved forward, we need to make sure we move our sample boundary
154         return utcTimestamp + newTzOffset;
155     }

211     protected boolean notKeyedLoop(MapValue mapValue) {
212         if (!isNotKeyedLoopInitialized) {
213             sampleLocalEpoch = localEpoch;
214             nextSampleLocalEpoch = localEpoch;
215             // looks like we need to populate key map
216             // at the start of this loop 'lastTimestamp' will be set to timestamp
217             // of first record in base cursor
218             groupByFunctionsUpdater.updateNew(mapValue, baseRecord);
219             isNotKeyedLoopInitialized = true;
220         }
221 
222         long next = timestampSampler.nextTimestamp(localEpoch);
223         while (baseCursor.hasNext()) {
224             long timestamp = getBaseRecordTimestamp();
225             if (timestamp < next) {
226                 circuitBreaker.statefulThrowExceptionIfTripped();
227 
228                 adjustDstInFlight(timestamp - tzOffset);
229                 groupByFunctionsUpdater.updateExisting(mapValue, baseRecord);
230             } else {
231                 // timestamp changed, make sure we keep the value of 'lastTimestamp'
232                 // unchanged. Timestamp columns uses this variable
233                 // When map is exhausted we would assign 'next' to 'lastTimestamp'
234                 // and build another map
235                 timestamp = adjustDst(timestamp, mapValue, next);
236                 if (timestamp != Long.MIN_VALUE) {
237                     nextSamplePeriod(timestamp);
238                     isNotKeyedLoopInitialized = false;
239                     return true;
240                 }
241             }
242         }
243         // opportunity, after we stream map that's it
244         baseRecord = null;
245         isNotKeyedLoopInitialized = false;
246         return true;
247     }
