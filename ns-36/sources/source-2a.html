33 public final class TimestampSamplerFactory {
34 
35     /**
36      * Parses strings such as '10m', '3M', '5d', '12h', 'y', '35s'
37      *
38      * @param cs       the key
39      * @param position position in SQL text to report error against
40      * @return instance of appropriate TimestampSampler
41      * @throws SqlException when input string is invalid
42      */
43     public static TimestampSampler getInstance(CharSequence cs, int position) throws SqlException {
44         int k = -1;
45 
46         if (cs == null) {
47             throw SqlException.$(position, "missing interval");
48         }
49 
50         final int len = cs.length();
51 
52         // look for end of digits
53         for (int i = 0; i < len; i++) {
54             char c = cs.charAt(i);
55             if (c < '0' || c > '9') {
56                 k = i;
57                 break;
58             }
59         }
60 
61         if (k == -1) {
62             throw SqlException.$(position + len, "expected interval qualifier");
63         }
64 
65         // expect 1 letter qualifier
66         if (k + 1 < len) {
67             throw SqlException.$(position + k, "expected single letter qualifier");
68         }
69 
70         try {
71             final int n;
72             if (k == 0) {
73                 n = 1;
74             } else {
75                 n = Numbers.parseInt(cs, 0, k);
76                 if (n == 0) {
77                     throw SqlException.$(position, "zero is not a valid sample value");
78                 }
79             }
80 
81             return createTimestampSampler(n, cs.charAt(k), position + k);
82         } catch (NumericException ignore) {
83             // we are parsing a pre-validated number
84             // but we have to deal with checked exception anyway
85             assert false;
86         }
87 
88         throw SqlException.$(position + k, "unsupported interval qualifier");
89     }
90 
91     public static TimestampSampler getInstance(long period, CharSequence units, int position) throws SqlException {
92         if (units.length() == 1) {
93             return createTimestampSampler(period, units.charAt(0), position);
94         }
95         // Just in case SqlParser will allow this in the future
96         throw SqlException.$(position, "expected one character interval qualifier");
97     }

99     @NotNull
100     private static TimestampSampler createTimestampSampler(long interval, char timeUnit, int position) throws SqlException {
101         switch (timeUnit) {
102             case 'U':
103                 // micros
104                 return new MicroTimestampSampler(interval);
105             case 'T':
106                 // millis
107                 return new MicroTimestampSampler(Timestamps.MILLI_MICROS * interval);
108             case 's':
109                 // seconds
110                 return new MicroTimestampSampler(Timestamps.SECOND_MICROS * interval);
111             case 'm':
112                 // minutes
113                 return new MicroTimestampSampler(Timestamps.MINUTE_MICROS * interval);
114             case 'h':
115                 // hours
116                 return new MicroTimestampSampler(Timestamps.HOUR_MICROS * interval);
117             case 'd':
118                 // days
119                 return new MicroTimestampSampler(Timestamps.DAY_MICROS * interval);
120             case 'M':
121                 // months
122                 return new MonthTimestampSampler((int) interval);
123             case 'y':
124                 return new YearTimestampSampler((int) interval);
125             default:
126                 // Just in case SqlParser will allow this in the future
127                 throw SqlException.$(position, "unsupported interval qualifier");
128 
129         }
130     }
