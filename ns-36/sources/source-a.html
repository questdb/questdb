48     public DistinctRecordCursorFactory(
49             CairoConfiguration configuration,
50             RecordCursorFactory base,
51             @Transient @NotNull EntityColumnFilter columnFilter,
52             @Transient @NotNull BytecodeAssembler asm
53     ) {
54         super(base.getMetadata());
55         final RecordMetadata metadata = base.getMetadata();
56         // sink will be storing record columns to map key
57         columnFilter.of(metadata.getColumnCount());
58         mapSink = RecordSinkFactory.getInstance(asm, metadata, columnFilter, false);
59         this.base = base;
60         cursor = new DistinctRecordCursor(configuration, metadata);
61     }

69     public RecordCursor getCursor(SqlExecutionContext executionContext) throws SqlException {
70         final RecordCursor baseCursor = base.getCursor(executionContext);
71         try {
72             cursor.of(baseCursor, mapSink, executionContext.getCircuitBreaker());
73             return cursor;
74         } catch (Throwable e) {

141         public boolean hasNext() {
142             while (baseCursor.hasNext()) {
143                 circuitBreaker.statefulThrowExceptionIfTripped();
144                 MapKey key = dataMap.withKey();
145                 recordSink.copy(record, key);

149             }

159             if (!isOpen) {
