118     static ObjList<String> getKeys(ObjList<Function> recordFunctions, RecordMetadata metadata) {
119         ObjList<String> keyFuncs = null;

146         public GroupByRecordCursor(

148                 GroupByFunctionsUpdater groupByFunctionsUpdater,
149                 @Transient @NotNull ArrayColumnTypes keyTypes,
150                 @Transient @NotNull ArrayColumnTypes valueTypes,
151                 CairoConfiguration configuration
152         ) {
153             super(functions);

155             this.groupByFunctionsUpdater = groupByFunctionsUpdater;
156             this.isOpen = true;

159         @Override
160         public void close() {
161             if (isOpen) {
162                 isOpen = false;
163                 Misc.free(dataMap);
164                 Misc.clearObjList(groupByFunctions);
165                 super.close();
166             }
167         }
168 
169         @Override
170         public boolean hasNext() {
171             if (!isDataMapBuilt) {
172                 final Record baseRecord = managedCursor.getRecord();
173                 while (managedCursor.hasNext()) {
174                     circuitBreaker.statefulThrowExceptionIfTripped();
175                     final MapKey key = dataMap.withKey();
176                     mapSink.copy(baseRecord, key);
177                     MapValue value = key.createValue();
178                     if (value.isNew()) {
179                         groupByFunctionsUpdater.updateNew(value, baseRecord);
180                     } else {
181                         groupByFunctionsUpdater.updateExisting(value, baseRecord);
182                     }
183                 }
184                 super.of(dataMap.getCursor());
185                 isDataMapBuilt = true;
186             }
187             return super.hasNext();
188         }
189 
190         public void of(RecordCursor managedCursor, SqlExecutionCircuitBreaker circuitBreaker) {
191             if (!isOpen) {
192                 isOpen = true;
193                 dataMap.reopen();
194             }
195             this.circuitBreaker = circuitBreaker;
196             this.managedCursor = managedCursor;
197             isDataMapBuilt = false;
198         }
