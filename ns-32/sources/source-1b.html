56     static void copyConstants(
57             ObjList<Function> args,
58             IntList argPositions,
59             ObjList<String> symbols
60     ) throws SqlException {
61         for (int i = 0, n = args.size(); i < n; i++) {
62             final Function f = args.getQuick(i);
63             if (f.isConstant()) {
64                 final int typeTag = ColumnType.tagOf(f.getType());
65                 if (typeTag == ColumnType.STRING || typeTag == ColumnType.NULL) {
66                     symbols.add(Chars.toString(f.getStr(null)));
67                     continue;
68                 }
69                 if (typeTag == ColumnType.CHAR) {
70                     symbols.add(String.valueOf(f.getChar(null)));
71                     continue;
72                 }
73             }
74             throw SqlException.$(argPositions.getQuick(i), "STRING constant expected");
75         }
76     }
