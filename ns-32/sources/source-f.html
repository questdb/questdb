44 public class RndGeoHashFunctionFactory implements FunctionFactory {
45 
46     @Override
47     public String getSignature() {
48         return "rnd_geohash(i)";
49     }
50 
51     @Override
52     public Function newInstance(int position,
53                                 ObjList<Function> args,
54                                 IntList argPositions,
55                                 CairoConfiguration configuration,
56                                 SqlExecutionContext sqlExecutionContext) throws SqlException {
57         int bits = args.getQuick(0).getInt(null);
58         if (bits < 1 || bits > ColumnType.GEO_HASH_MAX_BITS_LENGTH) {
59             throw SqlException.$(argPositions.getQuick(0), "precision must be in [1..60] range");
60         }
61         final int type = ColumnType.getGeoHashTypeWithBits(bits);
62         switch (ColumnType.tagOf(type)) {
63             case ColumnType.GEOBYTE:
64                 return new RndByteFunction(type);
65             case ColumnType.GEOSHORT:
66                 return new RndShortFunction(type);
67             case ColumnType.GEOINT:
68                 return new RndIntFunction(type);
69             default:
70                 return new RndLongFunction(type);
71         }
72     }
