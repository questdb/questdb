44     RndStringMemory(String signature, int count, int lo, int hi, int position, CairoConfiguration configuration) throws SqlException {
45         this.count = count;
46         this.lo = lo;
47         this.hi = hi;
48 
49         final int pageSize = configuration.getRndFunctionMemoryPageSize();
50         final int maxPages = configuration.getRndFunctionMemoryMaxPages();
51         final long memLimit = (long) maxPages * pageSize;
52         // check against worst case, the highest possible mem usage
53         final long requiredMem = count * (Vm.getStorageLength(hi) + Long.BYTES);
54         if (requiredMem > memLimit) {
55             throw SqlException.position(position)
56                     .put("breached memory limit set for ").put(signature)
57                     .put(" [pageSize=").put(pageSize)
58                     .put(", maxPages=").put(maxPages)
59                     .put(", memLimit=").put(memLimit)
60                     .put(", requiredMem=").put(requiredMem)
61                     .put(']');
62         }
63 
64         final int idxPages = count * 8 / pageSize + 1;
65         strMem = Vm.getARInstance(pageSize, maxPages - idxPages, MemoryTag.NATIVE_FUNC_RSS);
66         idxMem = Vm.getARInstance(pageSize, idxPages, MemoryTag.NATIVE_FUNC_RSS);
67     }

87     private void initFixedLength(Rnd rnd) {
88         final long storageLength = Vm.getStorageLength(lo);
89         for (int i = 0; i < count; i++) {
90             final long o = strMem.putStr(rnd.nextChars(lo));
91             idxMem.putLong(o - storageLength);
92         }
93     }

95     private void initVariableLength(Rnd rnd) {
96         for (int i = 0; i < count; i++) {
97             final int len = lo + rnd.nextPositiveInt() % (hi - lo + 1);
98             final long o = strMem.putStr(rnd.nextChars(len));
99             idxMem.putLong(o - Vm.getStorageLength(len));
100         }
101     }
