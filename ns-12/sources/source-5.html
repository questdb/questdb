443                     .put('}');

530     private void doQueryRecord(HttpChunkedResponseSocket socket, int columnCount) {
531         queryState = QUERY_RECORD;
532         for (; columnIndex < columnCount; columnIndex++) {
533             socket.bookmark();
534             if (columnIndex > 0) {
535                 socket.put(',');
536             }
537 
538             int columnIdx = columnSkewList.size() > 0 ? columnSkewList.getQuick(columnIndex) : columnIndex;
539             int columnType = columnTypesAndFlags.getQuick(2 * columnIndex);
540             switch (ColumnType.tagOf(columnType)) {
541                 case ColumnType.BOOLEAN:
542                     putBooleanValue(socket, record, columnIdx);
543                     break;
544                 case ColumnType.BYTE:
545                     putByteValue(socket, record, columnIdx);
546                     break;
547                 case ColumnType.DOUBLE:
548                     putDoubleValue(socket, record, columnIdx);
549                     break;
550                 case ColumnType.FLOAT:
551                     putFloatValue(socket, record, columnIdx);
552                     break;
553                 case ColumnType.INT:
554                     putIntValue(socket, record, columnIdx);
555                     break;
556                 case ColumnType.LONG:
557                     putLongValue(socket, record, columnIdx, quoteLargeNum);
558                     break;
559                 case ColumnType.DATE:
560                     putDateValue(socket, record, columnIdx);
561                     break;
562                 case ColumnType.TIMESTAMP:
563                     putTimestampValue(socket, record, columnIdx);
564                     break;
565                 case ColumnType.SHORT:
566                     putShortValue(socket, record, columnIdx);
567                     break;
568                 case ColumnType.CHAR:
569                     putCharValue(socket, record, columnIdx);
570                     break;
571                 case ColumnType.STRING:
572                     putStrValue(socket, record, columnIdx);
573                     break;
574                 case ColumnType.SYMBOL:
575                     putSymValue(socket, record, columnIdx);
576                     break;
577                 case ColumnType.BINARY:
578                     putBinValue(socket);
579                     break;
580                 case ColumnType.LONG256:
581                     putLong256Value(socket, record, columnIdx);
582                     break;
583                 case ColumnType.GEOBYTE:
584                     putGeoHashStringByteValue(socket, record, columnIdx, columnTypesAndFlags.getQuick(2 * columnIndex + 1));
585                     break;
586                 case ColumnType.GEOSHORT:
587                     putGeoHashStringShortValue(socket, record, columnIdx, columnTypesAndFlags.getQuick(2 * columnIndex + 1));
588                     break;
589                 case ColumnType.GEOINT:
590                     putGeoHashStringIntValue(socket, record, columnIdx, columnTypesAndFlags.getQuick(2 * columnIndex + 1));
591                     break;
592                 case ColumnType.GEOLONG:
593                     putGeoHashStringLongValue(socket, record, columnIdx, columnTypesAndFlags.getQuick(2 * columnIndex + 1));
594                     break;
595                 case ColumnType.RECORD:
596                     putRecValue(socket);
597                     break;
598                 case ColumnType.NULL:
599                     socket.put("null");
600                     break;
601                 case ColumnType.LONG128:
602                     throw new UnsupportedOperationException();
603                 case ColumnType.UUID:
604                     putUuidValue(socket, record, columnIdx);
605                     break;
606                 default:
607                     assert false : "Not supported type in output " + ColumnType.nameOf(columnType);
608                     socket.put("null"); // To make JSON valid
609                     break;
610             }
611         }
612     }

806     boolean of(
807             RecordCursorFactory factory,
808             boolean queryCacheable,
809             SqlExecutionContextImpl sqlExecutionContext
810     ) throws PeerDisconnectedException, PeerIsSlowToReadException, SqlException {
811         this.recordCursorFactory = factory;
812         this.queryCacheable = queryCacheable;
813         this.queryJitCompiled = factory.usesCompiledFilter();
814         // Enable column pre-touch in REST API only when LIMIT K,N is not specified since when limit is defined
815         // we do a no-op loop over the cursor to calculate the total row count and pre-touch only slows things down.
816         sqlExecutionContext.setColumnPreTouchEnabled(stop == Long.MAX_VALUE);
817         this.cursor = factory.getCursor(sqlExecutionContext);
818         final RecordMetadata metadata = factory.getMetadata();
819         this.queryTimestampIndex = metadata.getTimestampIndex();
820         HttpRequestHeader header = httpConnectionContext.getRequestHeader();
821         DirectByteCharSequence columnNames = header.getUrlParam("cols");
822         int columnCount;
823         columnSkewList.clear();
824         if (columnNames != null) {
825             columnsQueryParameter.clear();
826             try {
827                 TextUtil.utf8Decode(columnNames.getLo(), columnNames.getHi(), columnsQueryParameter);
828             } catch (Utf8Exception e) {
829                 info().$("utf8 error when decoding column list '").$(columnNames).$('\'').$();
830                 HttpChunkedResponseSocket socket = getHttpConnectionContext().getChunkedResponseSocket();
831                 JsonQueryProcessor.header(socket, "", 400);
832                 socket.put('{')
833                         .putQuoted("error").put(':').putQuoted("utf8 error in column list")
834                         .put('}');
835                 socket.sendChunk(true);
836                 return false;
837             }
838 
839             columnCount = 1;
840             int start = 0;
841             int comma = 0;
842             while (comma > -1) {
843                 comma = Chars.indexOf(columnsQueryParameter, start, ',');
844                 if (comma > -1) {
845                     if (addColumnToOutput(metadata, columnsQueryParameter, start, comma)) {
846                         return false;
847                     }
848                     start = comma + 1;
849                     columnCount++;
850                 } else {
851                     int hi = columnsQueryParameter.length();
852                     if (addColumnToOutput(metadata, columnsQueryParameter, start, hi)) {
853                         return false;
854                     }
855                 }
856             }
857         } else {
858             columnCount = metadata.getColumnCount();
859             for (int i = 0; i < columnCount; i++) {
860                 addColumnTypeAndName(metadata, i);
861             }
862         }
863         this.columnCount = columnCount;
864         return true;
865     }
