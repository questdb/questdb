269         int pad = value == null ? w : w - value.length();
270         replicate(b, ' ', pad);
271

275             } else {

291     private static void resumeJson(TextImportProcessorState state, HttpChunkedResponseSocket socket) throws PeerDisconnectedException, PeerIsSlowToReadException {
292         final TextLoaderCompletedState completeState = state.completeState;
293         final RecordMetadata metadata = completeState.getMetadata();
294         final LongList errors = completeState.getColumnErrorCounts();
295 
296 
297         switch (state.responseState) {
298             case RESPONSE_PREFIX:
299                 long totalRows = completeState.getParsedLineCount();
300                 long importedRows = completeState.getWrittenLineCount();
301                 socket.put('{')
302                         .putQuoted("status").put(':').putQuoted("OK").put(',')
303                         .putQuoted("location").put(':').encodeUtf8AndQuote(completeState.getTableName()).put(',')
304                         .putQuoted("rowsRejected").put(':').put(totalRows - importedRows + completeState.getErrorLineCount()).put(',')
305                         .putQuoted("rowsImported").put(':').put(importedRows).put(',')
306                         .putQuoted("header").put(':').put(completeState.isForceHeaders()).put(',');
307                 if (completeState.getWarnings() != TextLoadWarning.NONE) {
308                     final int warningFlags = completeState.getWarnings();
309                     socket.putQuoted("warnings").put(':').put('[');
310                     boolean isFirst = true;
311                     if ((warningFlags & TextLoadWarning.TIMESTAMP_MISMATCH) != TextLoadWarning.NONE) {
312                         isFirst = false;
313                         socket.putQuoted("Existing table timestamp column is used");
314                     }
315                     if ((warningFlags & PARTITION_TYPE_MISMATCH) != TextLoadWarning.NONE) {
316                         if (!isFirst) socket.put(',');
317                         socket.putQuoted("Existing table PartitionBy is used");
318                     }
319                     socket.put(']').put(',');
320                 }
321                 socket.putQuoted("columns").put(':').put('[');
322                 state.responseState = RESPONSE_COLUMN;
323                 // fall through
324             case RESPONSE_COLUMN:
325                 if (metadata != null) {
326                     final int columnCount = metadata.getColumnCount();
327                     for (; state.columnIndex < columnCount; state.columnIndex++) {
328                         socket.bookmark();
329                         if (state.columnIndex > 0) {
330                             socket.put(',');
331                         }
332                         socket.put('{').
333                                 putQuoted("name").put(':').putQuoted(metadata.getColumnName(state.columnIndex)).put(',').
334                                 putQuoted("type").put(':').putQuoted(ColumnType.nameOf(metadata.getColumnType(state.columnIndex))).put(',').
335                                 putQuoted("size").put(':').put(ColumnType.sizeOf(metadata.getColumnType(state.columnIndex))).put(',').
336                                 putQuoted("errors").put(':').put(errors.getQuick(state.columnIndex));
337                         socket.put('}');
338                     }
339                 }
340                 state.responseState = RESPONSE_SUFFIX;
341                 // fall through
342             case RESPONSE_SUFFIX:
343                 socket.bookmark();
344                 socket.put(']').put('}');
345                 state.responseState = RESPONSE_COMPLETE;
346                 socket.sendChunk(true);
347                 break;
348             case RESPONSE_DONE:
349                 state.responseState = RESPONSE_COMPLETE;
350                 socket.done();
351                 break;
352             default:
353                 break;
354         }
355     }

357     private static void resumeText(TextImportProcessorState state, HttpChunkedResponseSocket socket) throws PeerDisconnectedException, PeerIsSlowToReadException {
358         final TextLoaderCompletedState textLoaderCompletedState = state.completeState;
359         final RecordMetadata metadata = textLoaderCompletedState.getMetadata();
360         LongList errors = textLoaderCompletedState.getColumnErrorCounts();
361 
362 
363         switch (state.responseState) {
364             case RESPONSE_PREFIX:
365                 sep(socket);
366                 socket.put('|');
367                 pad(socket, TO_STRING_COL1_PAD, "Location:");
368                 pad(socket, TO_STRING_COL2_PAD, textLoaderCompletedState.getTableName());
369                 pad(socket, TO_STRING_COL3_PAD, "Pattern");
370                 pad(socket, TO_STRING_COL4_PAD, "Locale");
371                 pad(socket, TO_STRING_COL5_PAD, "Errors").put(Misc.EOL);
372 
373 
374                 socket.put('|');
375                 pad(socket, TO_STRING_COL1_PAD, "Partition by");
376                 pad(socket, TO_STRING_COL2_PAD, PartitionBy.toString(textLoaderCompletedState.getPartitionBy()));
377                 pad(socket, TO_STRING_COL3_PAD, "");
378                 pad(socket, TO_STRING_COL4_PAD, "");
379                 if (hasFlag(textLoaderCompletedState.getWarnings(), PARTITION_TYPE_MISMATCH)) {
380                     pad(socket, TO_STRING_COL5_PAD, OVERRIDDEN_FROM_TABLE);
381                 } else {
382                     pad(socket, TO_STRING_COL5_PAD, "");
383                 }
384                 socket.put(Misc.EOL);
385 
386                 socket.put('|');
387                 pad(socket, TO_STRING_COL1_PAD, "Timestamp");
388                 pad(socket, TO_STRING_COL2_PAD, textLoaderCompletedState.getTimestampCol() == null ? "NONE" : textLoaderCompletedState.getTimestampCol());
389                 pad(socket, TO_STRING_COL3_PAD, "");
390                 pad(socket, TO_STRING_COL4_PAD, "");
391                 if (hasFlag(textLoaderCompletedState.getWarnings(), TIMESTAMP_MISMATCH)) {
392                     pad(socket, TO_STRING_COL5_PAD, OVERRIDDEN_FROM_TABLE);
393                 } else {
394                     pad(socket, TO_STRING_COL5_PAD, "");
395                 }
396                 socket.put(Misc.EOL);
397 
398                 sep(socket);
399 
400                 socket.put('|');
401                 pad(socket, TO_STRING_COL1_PAD, "Rows handled");
402                 pad(socket, TO_STRING_COL2_PAD, textLoaderCompletedState.getParsedLineCount() + textLoaderCompletedState.getErrorLineCount());
403                 pad(socket, TO_STRING_COL3_PAD, "");
404                 pad(socket, TO_STRING_COL4_PAD, "");
405                 pad(socket, TO_STRING_COL5_PAD, "").put(Misc.EOL);
406                 socket.put('|');
407                 pad(socket, TO_STRING_COL1_PAD, "Rows imported");
408                 pad(socket, TO_STRING_COL2_PAD, textLoaderCompletedState.getWrittenLineCount());
409                 pad(socket, TO_STRING_COL3_PAD, "");
410                 pad(socket, TO_STRING_COL4_PAD, "");
411                 pad(socket, TO_STRING_COL5_PAD, "").put(Misc.EOL);
412                 sep(socket);
413 
414                 state.responseState = RESPONSE_COLUMN;
415                 // fall through
416             case RESPONSE_COLUMN:
417 
418                 if (metadata != null) {
419                     final int columnCount = metadata.getColumnCount();
420 
421                     for (; state.columnIndex < columnCount; state.columnIndex++) {
422                         socket.bookmark();
423                         socket.put('|');
424                         pad(socket, TO_STRING_COL1_PAD, state.columnIndex);
425                         pad(socket, TO_STRING_COL2_PAD, metadata.getColumnName(state.columnIndex));
426                         if (!metadata.isColumnIndexed(state.columnIndex)) {
427                             pad(socket, TO_STRING_COL3_PAD + TO_STRING_COL4_PAD + 3, ColumnType.nameOf(metadata.getColumnType(state.columnIndex)));
428                         } else {
429                             StringSink sink = Misc.getThreadLocalBuilder();
430                             sink.put("(idx/").put(metadata.getIndexValueBlockCapacity(state.columnIndex)).put(") ");
431                             sink.put(ColumnType.nameOf(metadata.getColumnType(state.columnIndex)));
432                             pad(socket, TO_STRING_COL3_PAD + TO_STRING_COL4_PAD + 3, sink);
433                         }
434                         pad(socket, TO_STRING_COL5_PAD, errors.getQuick(state.columnIndex));
435                         socket.put(Misc.EOL);
436                     }
437                 }
438                 state.responseState = RESPONSE_SUFFIX;
439                 // fall through
440             case RESPONSE_SUFFIX:
441                 socket.bookmark();
442                 sep(socket);
443                 state.responseState = RESPONSE_COMPLETE;
444                 socket.sendChunk(true);
445                 break;
446             case RESPONSE_DONE:
447                 state.responseState = RESPONSE_COMPLETE;
448                 socket.done();
449                 break;
450             default:
451                 break;
452         }
453     }

461     private void doResumeSend(
462             TextImportProcessorState state,
463             HttpChunkedResponseSocket socket
464     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException {
465         try {
466             if (state.errorMessage != null) {
467                 resumeError(state, socket);
468             } else if (state.json) {
469                 resumeJson(state, socket);
470             } else {
471                 resumeText(state, socket);
472             }
473         } catch (NoSpaceLeftInResponseBufferException ignored) {
474             if (socket.resetToBookmark()) {
475                 socket.sendChunk(false);
476             } else {
477                 // what we have here is out unit of data, column value or query
478                 // is larger that response content buffer
479                 // all we can do in this scenario is to log appropriately
480                 // and disconnect socket
481                 socket.shutdownWrite();
482                 throw ServerDisconnectException.INSTANCE;
483             }
484         }
485 
486         state.clear();
487     }

529     private void sendResponse(HttpConnectionContext context)
530             throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException {
531         final TextImportProcessorState state = LV.get(context);
532         final HttpChunkedResponseSocket socket = context.getChunkedResponseSocket();
533 
534         // Copy written state to state, text loader, parser can be closed before re-attempt to send the response
535         state.snapshotStateAndCloseWriter();
536         if (state.state == TextImportProcessorState.STATE_OK) {
537             if (state.json) {
538                 socket.status(200, CONTENT_TYPE_JSON);
539             } else {
540                 socket.status(200, CONTENT_TYPE_TEXT);
541             }
542             socket.sendHeader();
543             doResumeSend(state, socket);
544         } else {
545             sendErr(context, state.stateMessage, context.getChunkedResponseSocket());
546         }
547     }
