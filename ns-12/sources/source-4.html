142     public void execute0(
143             JsonQueryProcessorState state
144     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException, QueryPausedException {
145 
146         OperationFuture fut = state.getOperationFuture();
147         final HttpConnectionContext context = state.getHttpConnectionContext();
148         circuitBreaker.resetTimer();
149 
150         if (fut == null) {
151             metrics.jsonQuery().markStart();
152             state.startExecutionTimer();
153             // do not set random for new request to avoid copying random from previous request into next one
154             // the only time we need to copy random from state is when we resume request execution
155             sqlExecutionContext.with(context.getCairoSecurityContext(), null, null, context.getFd(), circuitBreaker.of(context.getFd()));
156             if (state.getStatementTimeout() > 0L) {
157                 circuitBreaker.setTimeout(state.getStatementTimeout());
158             } else {
159                 circuitBreaker.resetMaxTimeToDefault();
160             }
161             state.info().$("exec [q='").utf8(state.getQuery()).$("']").$();
162         }
163 
164         try {
165             if (fut != null) {
166                 retryQueryExecution(state, fut);
167                 return;
168             }
169 
170             final RecordCursorFactory factory = QueryCache.getThreadLocalInstance().poll(state.getQuery());
171             if (factory != null) {
172                 try {
173                     sqlExecutionContext.storeTelemetry(CompiledQuery.SELECT, TelemetryOrigin.HTTP_JSON);
174                     executeCachedSelect(
175                             state,
176                             factory,
177                             configuration.getKeepAliveHeader()
178                     );
179                 } catch (TableReferenceOutOfDateException e) {
180                     LOG.info().$(e.getFlyweightMessage()).$();
181                     Misc.free(factory);
182                     compileQuery(state);
183                 }
184             } else {
185                 // new query
186                 compileQuery(state);
187             }
188         } catch (SqlException | ImplicitCastException e) {
189             sqlError(context.getChunkedResponseSocket(), state, e, configuration.getKeepAliveHeader());
190             readyForNextRequest(context);
191         } catch (EntryUnavailableException e) {
192             LOG.info().$("[fd=").$(context.getFd()).$("] resource busy, will retry").$();
193             throw RetryOperationException.INSTANCE;
194         } catch (DataUnavailableException e) {
195             LOG.info().$("[fd=").$(context.getFd()).$("] data is in cold storage, will retry").$();
196             throw QueryPausedException.instance(e.getEvent(), sqlExecutionContext.getCircuitBreaker());
197         } catch (CairoError | CairoException e) {
198             internalError(context.getChunkedResponseSocket(), context.getLastRequestBytesSent(), e.getFlyweightMessage(), e, state, context.getMetrics());
199             readyForNextRequest(context);
200         } catch (PeerIsSlowToReadException | PeerDisconnectedException | QueryPausedException e) {
201             // re-throw the exception
202             throw e;
203         } catch (Throwable e) {
204             state.critical().$("Uh-oh. Error!").$(e).$();
205             throw ServerDisconnectException.INSTANCE;
206         }
207     }

292     private static void doResumeSend(
293             JsonQueryProcessorState state,
294             HttpConnectionContext context,
295             SqlExecutionContext sqlExecutionContext
296     ) throws PeerDisconnectedException, PeerIsSlowToReadException, QueryPausedException {
297         if (state.noCursor()) {
298             return;
299         }
300 
301         LOG.debug().$("resume [fd=").$(context.getFd()).I$();
302 
303         final HttpChunkedResponseSocket socket = context.getChunkedResponseSocket();
304         while (true) {
305             try {
306                 state.resume(socket);
307                 break;
308             } catch (DataUnavailableException e) {
309                 socket.resetToBookmark();
310                 throw QueryPausedException.instance(e.getEvent(), sqlExecutionContext.getCircuitBreaker());
311             } catch (NoSpaceLeftInResponseBufferException ignored) {
312                 if (socket.resetToBookmark()) {
313                     socket.sendChunk(false);
314                 } else {
315                     // what we have here is out unit of data, column value or query
316                     // is larger that response content buffer
317                     // all we can do in this scenario is to log appropriately
318                     // and disconnect socket
319                     state.logBufferTooSmall();
320                     throw PeerDisconnectedException.INSTANCE;
321                 }
322             }
323         }
324         // reached the end naturally?
325         readyForNextRequest(context);
326     }

412     private void compileQuery(
413             JsonQueryProcessorState state
414     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException, QueryPausedException, SqlException {
415         boolean recompileStale = true;
416         for (int retries = 0; recompileStale; retries++) {
417             try {
418                 final long nanos = nanosecondClock.getTicks();
419                 final CompiledQuery cc = compiler.compile(state.getQuery(), sqlExecutionContext);
420                 sqlExecutionContext.storeTelemetry(cc.getType(), TelemetryOrigin.HTTP_JSON);
421                 state.setCompilerNanos(nanosecondClock.getTicks() - nanos);
422                 state.setQueryType(cc.getType());
423                 queryExecutors.getQuick(cc.getType()).execute(
424                         state,
425                         cc,
426                         configuration.getKeepAliveHeader()
427                 );
428                 recompileStale = false;
429             } catch (TableReferenceOutOfDateException e) {
430                 if (retries == TableReferenceOutOfDateException.MAX_RETRY_ATTEMPS) {
431                     throw e;
432                 }
433                 LOG.info().$(e.getFlyweightMessage()).$();
434                 // will recompile
435             }
436         }
437     }

561     private void executeUpdate(
562             JsonQueryProcessorState state,
563             CompiledQuery cq,
564             CharSequence keepAliveHeader
565     ) throws PeerDisconnectedException, PeerIsSlowToReadException, SqlException {
566         circuitBreaker.resetTimer();
567         OperationFuture fut = null;
568         boolean isAsyncWait = false;
569         try {
570             fut = cq.execute(sqlExecutionContext, state.getEventSubSequence(), true);
571             int waitResult = fut.await(getAsyncWriterStartTimeout(state));
572             if (waitResult != OperationFuture.QUERY_COMPLETE) {
573                 isAsyncWait = true;
574                 state.setOperationFuture(fut);
575                 throw EntryUnavailableException.instance("retry update table wait");
576             }
577             // All good, finished update
578             final long updatedCount = fut.getAffectedRowsCount();
579             metrics.jsonQuery().markComplete();
580             sendUpdateConfirmation(state, keepAliveHeader, updatedCount);
581         } finally {

609     private boolean parseUrl(
610             JsonQueryProcessorState state,
611             CharSequence keepAliveHeader
612     ) throws PeerDisconnectedException, PeerIsSlowToReadException {
613         // Query text.
614         final HttpRequestHeader header = state.getHttpConnectionContext().getRequestHeader();
615         final DirectByteCharSequence query = header.getUrlParam("query");
616         if (query == null || query.length() == 0) {
617             state.info().$("Empty query header received. Sending empty reply.").$();
618             sendException(state.getHttpConnectionContext().getChunkedResponseSocket(), 0, "No query text", query, keepAliveHeader);
619             return false;
620         }
621 
622         // Url Params.
623         long skip = 0;
624         long stop = Long.MAX_VALUE;
625 
626         CharSequence limit = header.getUrlParam("limit");
627         if (limit != null) {
628             int sepPos = Chars.indexOf(limit, ',');
629             try {
630                 if (sepPos > 0) {
631                     skip = Numbers.parseLong(limit, 0, sepPos) - 1;
632                     if (sepPos + 1 < limit.length()) {
633                         stop = Numbers.parseLong(limit, sepPos + 1, limit.length());
634                     }
635                 } else {
636                     stop = Numbers.parseLong(limit);
637                 }
638             } catch (NumericException ex) {
639                 // Skip or stop will have default value.
640             }
641         }
642         if (stop < 0) {
643             stop = 0;
644         }
645 
646         if (skip < 0) {
647             skip = 0;
648         }
649 
650         if ((stop - skip) > configuration.getMaxQueryResponseRowLimit()) {
651             stop = skip + configuration.getMaxQueryResponseRowLimit();
652         }
653 
654         try {
655             state.configure(header, query, skip, stop);
656         } catch (Utf8Exception e) {
657             state.info().$("Bad UTF8 encoding").$();
658             sendException(state.getHttpConnectionContext().getChunkedResponseSocket(), 0, "Bad UTF8 encoding in query text", query, keepAliveHeader);
659             return false;
660         }
661         return true;
662     }

664     private void retryQueryExecution(
665             JsonQueryProcessorState state,
666             OperationFuture fut
667     ) throws PeerIsSlowToReadException, PeerDisconnectedException, ServerDisconnectException, QueryPausedException, SqlException {
668         final int waitResult;
669         try {
670             waitResult = fut.await(0);
671         } catch (TableReferenceOutOfDateException e) {
672             state.freeAsyncOperation();
673             compileQuery(state);
674             return;
675         }
676 
677         if (waitResult != OperationFuture.QUERY_COMPLETE) {
678             long timeout = state.getStatementTimeout() > 0 ? state.getStatementTimeout() : asyncCommandTimeout;
679             if (state.getExecutionTimeNanos() / 1_000_000L < timeout) {
680                 // Schedule a retry
681                 state.info().$("waiting for update query [instance=").$(fut.getInstanceId()).I$();
682                 throw EntryUnavailableException.instance("wait for update query");
683             } else {
684                 state.freeAsyncOperation();
685                 throw SqlTimeoutException.timeout("Query timeout. Please add HTTP header 'Statement-Timeout' with timeout in ms");
686             }
687         } else {
688             // Done
689             state.freeAsyncOperation();
690             if (state.getQueryType() == CompiledQuery.UPDATE) {
691                 sendUpdateConfirmation(state, configuration.getKeepAliveHeader(), fut.getAffectedRowsCount());
692             } else {
693                 // Alter, sends ddl:OK
694                 sendConfirmation(state, configuration.getKeepAliveHeader());
695             }
696         }
697     }
