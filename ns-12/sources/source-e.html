100     public void execute(
101             HttpConnectionContext context,
102             TextQueryProcessorState state
103     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException, QueryPausedException {
104         try {
105             boolean isExpRequest = isExpUrl(context.getRequestHeader().getUrl());
106 
107             circuitBreaker.resetTimer();
108             state.recordCursorFactory = QueryCache.getThreadLocalInstance().poll(state.query);
109             state.setQueryCacheable(true);
110             sqlExecutionContext.with(
111                     context.getCairoSecurityContext(),
112                     null,
113                     null,
114                     context.getFd(),
115                     circuitBreaker.of(context.getFd())
116             );
117             if (state.recordCursorFactory == null) {
118                 final CompiledQuery cc = compiler.compile(state.query, sqlExecutionContext);
119                 if (cc.getType() == CompiledQuery.SELECT) {
120                     state.recordCursorFactory = cc.getRecordCursorFactory();
121                 } else if (isExpRequest) {
122                     throw SqlException.$(0, "/exp endpoint only accepts SELECT");
123                 }
124                 info(state).$("execute-new [q=`").utf8(state.query)
125                         .$("`, skip: ").$(state.skip)
126                         .$(", stop: ").$(state.stop)
127                         .I$();
128                 sqlExecutionContext.storeTelemetry(cc.getType(), TelemetryOrigin.HTTP_TEXT);
129             } else {
130                 info(state).$("execute-cached [q=`").utf8(state.query)
131                         .$("`, skip: ").$(state.skip)
132                         .$(", stop: ").$(state.stop)
133                         .I$();
134                 sqlExecutionContext.storeTelemetry(CompiledQuery.SELECT, TelemetryOrigin.HTTP_TEXT);
135             }
136 
137             if (state.recordCursorFactory != null) {
138                 try {
139                     boolean runQuery = true;
140                     for (int retries = 0; runQuery; retries++) {
141                         try {
142                             state.cursor = state.recordCursorFactory.getCursor(sqlExecutionContext);
143                             runQuery = false;
144                         } catch (TableReferenceOutOfDateException e) {
145                             if (retries == TableReferenceOutOfDateException.MAX_RETRY_ATTEMPS) {
146                                 throw e;
147                             }
148                             info(state).$(e.getFlyweightMessage()).$();
149                             state.recordCursorFactory = Misc.free(state.recordCursorFactory);
150                             final CompiledQuery cc = compiler.compile(state.query, sqlExecutionContext);
151                             if (cc.getType() != CompiledQuery.SELECT && isExpRequest) {
152                                 throw SqlException.$(0, "/exp endpoint only accepts SELECT");
153                             }
154 
155                             state.recordCursorFactory = cc.getRecordCursorFactory();
156                         }
157                     }
158                     state.metadata = state.recordCursorFactory.getMetadata();
159                     header(context.getChunkedResponseSocket(), state, 200);
160                     doResumeSend(context);
161                 } catch (CairoException e) {
162                     state.setQueryCacheable(e.isCacheable());
163                     internalError(context.getChunkedResponseSocket(), context.getLastRequestBytesSent(), e, state);
164                 } catch (CairoError e) {
165                     internalError(context.getChunkedResponseSocket(), context.getLastRequestBytesSent(), e, state);
166                 }
167             } else {
168                 headerNoContentDisposition(context.getChunkedResponseSocket());
169                 sendConfirmation(context.getChunkedResponseSocket());
170                 readyForNextRequest(context);
171             }
172         } catch (SqlException | ImplicitCastException e) {
173             syntaxError(context.getChunkedResponseSocket(), state, e);
174             readyForNextRequest(context);
175         } catch (CairoException | CairoError e) {
176             internalError(context.getChunkedResponseSocket(), context.getLastRequestBytesSent(), e, state);
177             readyForNextRequest(context);
178         }
179     }

277     private void doResumeSend(
278             HttpConnectionContext context
279     ) throws PeerDisconnectedException, PeerIsSlowToReadException, QueryPausedException {
280         TextQueryProcessorState state = LV.get(context);
281         if (state == null || state.cursor == null) {
282             return;
283         }
284 
285         // copy random during query resume
286         sqlExecutionContext.with(context.getCairoSecurityContext(), null, state.rnd, context.getFd(), circuitBreaker.of(context.getFd()));
287         LOG.debug().$("resume [fd=").$(context.getFd()).I$();
288 
289         if (!state.pausedQuery) {
290             context.resumeResponseSend();
291         } else {
292             state.pausedQuery = false;
293         }
294 
295         final HttpChunkedResponseSocket socket = context.getChunkedResponseSocket();
296         final int columnCount = state.metadata.getColumnCount();
297 
298         OUT:
299         while (true) {
300             try {
301                 SWITCH:
302                 switch (state.queryState) {
303                     case JsonQueryProcessorState.QUERY_PREFIX:
304                     case JsonQueryProcessorState.QUERY_METADATA:
305                         state.columnIndex = 0;
306                         state.queryState = JsonQueryProcessorState.QUERY_METADATA;
307                         while (state.columnIndex < columnCount) {
308                             if (state.columnIndex > 0) {
309                                 socket.put(state.delimiter);
310                             }
311                             socket.putQuoted(state.metadata.getColumnName(state.columnIndex));
312                             state.columnIndex++;
313                             socket.bookmark();
314                         }
315                         socket.put(Misc.EOL);
316                         state.queryState = JsonQueryProcessorState.QUERY_RECORD_START;
317                         socket.bookmark();
318                         // fall through
319                     case JsonQueryProcessorState.QUERY_RECORD_START:
320                         if (state.record == null) {
321                             // check if cursor has any records
322                             Record record = state.cursor.getRecord();
323                             while (true) {
324                                 if (state.cursor.hasNext()) {
325                                     state.count++;
326 
327                                     if (state.countRows && state.count > state.stop) {
328                                         continue;
329                                     }
330 
331                                     if (state.count > state.skip) {
332                                         state.record = record;
333                                         break;
334                                     }
335                                 } else {
336                                     state.queryState = JsonQueryProcessorState.QUERY_SUFFIX;
337                                     break SWITCH;
338                                 }
339                             }
340                         }
341 
342                         if (state.count > state.stop) {
343                             state.queryState = JsonQueryProcessorState.QUERY_SUFFIX;
344                             break;
345                         }
346 
347                         state.queryState = JsonQueryProcessorState.QUERY_RECORD;
348                         state.columnIndex = 0;
349                         // fall through
350                     case JsonQueryProcessorState.QUERY_RECORD:
351                         while (state.columnIndex < columnCount) {
352                             if (state.columnIndex > 0) {
353                                 socket.put(state.delimiter);
354                             }
355                             putValue(socket, state.metadata.getColumnType(state.columnIndex), state.record, state.columnIndex);
356                             state.columnIndex++;
357                             socket.bookmark();
358                         }
359 
360                         state.queryState = JsonQueryProcessorState.QUERY_RECORD_SUFFIX;
361                         // fall through
362                     case JsonQueryProcessorState.QUERY_RECORD_SUFFIX:
363                         socket.put(Misc.EOL);
364                         state.record = null;
365                         state.queryState = JsonQueryProcessorState.QUERY_RECORD_START;
366                         socket.bookmark();
367                         break;
368                     case JsonQueryProcessorState.QUERY_SUFFIX:
369                         sendDone(socket, state);
370                         break OUT;
371                     default:
372                         break OUT;
373                 }
374             } catch (DataUnavailableException e) {
375                 socket.resetToBookmark();
376                 throw QueryPausedException.instance(e.getEvent(), sqlExecutionContext.getCircuitBreaker());
377             } catch (NoSpaceLeftInResponseBufferException ignored) {
378                 if (socket.resetToBookmark()) {
379                     socket.sendChunk(false);
380                 } else {
381                     // what we have here is out unit of data, column value or query
382                     // is larger that response content buffer
383                     // all we can do in this scenario is to log appropriately
384                     // and disconnect socket
385                     info(state).$("Response buffer is too small, state=").$(state.queryState).$();
386                     throw PeerDisconnectedException.INSTANCE;
387                 }
388             }
389         }
390         // reached the end naturally?
391         readyForNextRequest(context);
392     }

419     private boolean parseUrl(
420             HttpChunkedResponseSocket socket,
421             HttpRequestHeader request,
422             TextQueryProcessorState state
423     ) throws PeerDisconnectedException, PeerIsSlowToReadException {
424         // Query text.
425         final DirectByteCharSequence query = request.getUrlParam("query");
426         if (query == null || query.length() == 0) {
427             info(state).$("Empty query request received. Sending empty reply.").$();
428             sendException(socket, 0, "No query text", state);
429             return false;
430         }
431 
432         // URL params.
433         long skip = 0;
434         long stop = Long.MAX_VALUE;
435 
436         CharSequence limit = request.getUrlParam("limit");
437         if (limit != null) {
438             int sepPos = Chars.indexOf(limit, ',');
439             try {
440                 if (sepPos > 0) {
441                     skip = Numbers.parseLong(limit, 0, sepPos);
442                     if (sepPos + 1 < limit.length()) {
443                         stop = Numbers.parseLong(limit, sepPos + 1, limit.length());
444                     }
445                 } else {
446                     stop = Numbers.parseLong(limit);
447                 }
448             } catch (NumericException ex) {
449                 // Skip or stop will have default value.
450             }
451         }
452         if (stop < 0) {
453             stop = 0;
454         }
455 
456         if (skip < 0) {
457             skip = 0;
458         }
459 
460         if ((stop - skip) > configuration.getMaxQueryResponseRowLimit()) {
461             stop = skip + configuration.getMaxQueryResponseRowLimit();
462         }
463 
464         state.query.clear();
465         try {
466             TextUtil.utf8Decode(query.getLo(), query.getHi(), state.query);
467         } catch (Utf8Exception e) {
468             info(state).$("Bad UTF8 encoding").$();
469             sendException(socket, 0, "Bad UTF8 encoding in query text", state);
470             return false;
471         }
472         CharSequence fileName = request.getUrlParam("filename");
473         state.fileName = null;
474         if (fileName != null && fileName.length() > 0) {
475             state.fileName = fileName.toString();
476         }
477 
478         DirectByteCharSequence delimiter = request.getUrlParam("delimiter");
479         state.delimiter = ',';
480 
481         if (delimiter != null && delimiter.length() == 1) {
482             state.delimiter = delimiter.charAt(0);
483         }
484 
485         state.skip = skip;
486         state.count = 0L;
487         state.stop = stop;
488         state.noMeta = Chars.equalsNc("true", request.getUrlParam("nm"));
489         state.countRows = Chars.equalsNc("true", request.getUrlParam("count"));
490         return true;
491     }

493     private void putValue(HttpChunkedResponseSocket socket, int type, Record rec, int col) {
494         switch (ColumnType.tagOf(type)) {
495             case ColumnType.BOOLEAN:
496                 socket.put(rec.getBool(col));
497                 break;
498             case ColumnType.BYTE:
499                 socket.put(rec.getByte(col));
500                 break;
501             case ColumnType.DOUBLE:
502                 double d = rec.getDouble(col);
503                 if (d == d) {
504                     socket.put(d, doubleScale);
505                 }
506                 break;
507             case ColumnType.FLOAT:
508                 float f = rec.getFloat(col);
509                 if (f == f) {
510                     socket.put(f, floatScale);
511                 }
512                 break;
513             case ColumnType.INT:
514                 final int i = rec.getInt(col);
515                 if (i > Integer.MIN_VALUE) {
516                     Numbers.append(socket, i);
517                 }
518                 break;
519             case ColumnType.LONG:
520                 long l = rec.getLong(col);
521                 if (l > Long.MIN_VALUE) {
522                     socket.put(l);
523                 }
524                 break;
525             case ColumnType.DATE:
526                 l = rec.getDate(col);
527                 if (l > Long.MIN_VALUE) {
528                     socket.put('"').putISODateMillis(l).put('"');
529                 }
530                 break;
531             case ColumnType.TIMESTAMP:
532                 l = rec.getTimestamp(col);
533                 if (l > Long.MIN_VALUE) {
534                     socket.put('"').putISODate(l).put('"');
535                 }
536                 break;
537             case ColumnType.SHORT:
538                 socket.put(rec.getShort(col));
539                 break;
540             case ColumnType.CHAR:
541                 char c = rec.getChar(col);
542                 if (c > 0) {
543                     socket.put(c);
544                 }
545                 break;
546             case ColumnType.NULL:
547             case ColumnType.BINARY:
548             case ColumnType.RECORD:
549                 break;
550             case ColumnType.STRING:
551                 putStringOrNull(socket, rec.getStr(col));
552                 break;
553             case ColumnType.SYMBOL:
554                 putStringOrNull(socket, rec.getSym(col));
555                 break;
556             case ColumnType.LONG256:
557                 rec.getLong256(col, socket);
558                 break;
559             case ColumnType.GEOBYTE:
560                 putGeoHashStringValue(socket, rec.getGeoByte(col), type);
561                 break;
562             case ColumnType.GEOSHORT:
563                 putGeoHashStringValue(socket, rec.getGeoShort(col), type);
564                 break;
565             case ColumnType.GEOINT:
566                 putGeoHashStringValue(socket, rec.getGeoInt(col), type);
567                 break;
568             case ColumnType.GEOLONG:
569                 putGeoHashStringValue(socket, rec.getGeoLong(col), type);
570                 break;
571             case ColumnType.UUID:
572                 putUuidOrNull(socket, rec.getLong128Lo(col), rec.getLong128Hi(col));
573                 break;
574             case ColumnType.LONG128:
575                 throw new UnsupportedOperationException();
576             default:
577                 assert false;
578         }
579     }

619     protected void header(
620             HttpChunkedResponseSocket socket,
621             TextQueryProcessorState state,
622             int statusCode
623     ) throws PeerDisconnectedException, PeerIsSlowToReadException {
