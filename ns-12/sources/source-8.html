69     @Override
70     public void onRequestComplete(HttpConnectionContext context) throws PeerDisconnectedException, PeerIsSlowToReadException {
71         final HttpRequestHeader headers = context.getRequestHeader();
72         CharSequence url = headers.getUrl();
73         logInfoWithFd(context).$("incoming [url=").$(url).$(']').$();
74         if (Chars.contains(url, "..")) {
75             logInfoWithFd(context).$("URL abuse: ").$(url).$();
76             sendStatusWithDefaultMessage(context, 404);
77         } else {
78             PrefixedPath path = prefixedPath.rewind();
79 
80             if (Chars.equals(url, '/')) {
81                 path.concat(indexFileName);
82             } else {
83                 path.concat(url);
84             }
85 
86             path.$();
87 
88             if (ff.exists(path)) {
89                 send(context, path, headers.getUrlParam("attachment") != null);
90             } else {
91                 logInfoWithFd(context).$("not found [path=").$(path).$(']').$();
92                 sendStatusWithDefaultMessage(context, 404);
93             }
94         }

99         LOG.debug().$("resumeSend").$();
100         StaticContentProcessorState state = LV.get(context);
101 
102         if (state == null || state.fd == -1) {
103             return;
104         }
105 
106         context.resumeResponseSend();
107 
108         final HttpRawSocket socket = context.getRawResponseSocket();
109         long address = socket.getBufferAddress();
110         int size = socket.getBufferSize();
111 
112         long l;
113         // todo: check what happens when this code cannot read file
114         while (state.bytesSent < state.sendMax && (l = ff.read(state.fd, address, size, state.bytesSent)) > 0) {
115             if (l + state.bytesSent > state.sendMax) {
116                 l = state.sendMax - state.bytesSent;
117             }
118             state.bytesSent += l;
119             socket.send((int) l);
120         }

127     private void send(HttpConnectionContext context, LPSZ path, boolean asAttachment) throws PeerDisconnectedException, PeerIsSlowToReadException {
128         int n = Chars.lastIndexOf(path, '.');
129         if (n == -1) {
130             logInfoWithFd(context).$("missing extension [file=").$(path).$(']').$();
131             sendStatusWithDefaultMessage(context, 404);
132             return;
133         }
134 
135         HttpRequestHeader headers = context.getRequestHeader();
136         CharSequence contentType = mimeTypes.valueAt(mimeTypes.keyIndex(path, n + 1, path.length()));
137         CharSequence val;
138         if ((val = headers.getHeader("Range")) != null) {
139             sendRange(context, val, path, contentType, asAttachment);
140             return;
141         }
142 
143         int l;
144         // attempt not to send file when remote side already has
145         // up-to-date version of the same
146         if ((val = headers.getHeader("If-None-Match")) != null
147                 && (l = val.length()) > 2
148                 && val.charAt(0) == '"'
149                 && val.charAt(l - 1) == '"') {
150             try {
151                 long that = Numbers.parseLong(val, 1, l - 1);
152                 if (that == ff.getLastModified(path)) {
153                     context.simpleResponse().sendStatus(304);
154                     return;
155                 }
156             } catch (NumericException e) {
157                 LOG.info().$("bad 'If-None-Match' [value=").$(val).$(']').$();
158                 sendStatusWithDefaultMessage(context, 400);
159                 return;
160             }
161         }
162 
163         sendVanilla(context, path, contentType, asAttachment);
164     }

166     private void sendRange(
167             HttpConnectionContext context,
168             CharSequence range,
169             LPSZ path,
170             CharSequence contentType,
171             boolean asAttachment) throws PeerDisconnectedException, PeerIsSlowToReadException {
172         if (rangeParser.of(range)) {
173 
174             StaticContentProcessorState state = LV.get(context);
175             if (state == null) {
176                 LV.set(context, state = new StaticContentProcessorState());
177             }
178 
179             state.fd = ff.openRO(path);
180             if (state.fd == -1) {
181                 LOG.info().$("Cannot open file: ").$(path).$();
182                 sendStatusWithDefaultMessage(context, 404);
183                 return;
184             }
185 
186             state.bytesSent = 0;
187 
188             final long length = ff.length(path);
189             final long lo = rangeParser.getLo();
190             final long hi = rangeParser.getHi();
191             if (lo > length || (hi != Long.MAX_VALUE && hi > length) || lo > hi) {
192                 sendStatusWithDefaultMessage(context, 416);
193             } else {
194                 state.bytesSent = lo;
195                 state.sendMax = hi == Long.MAX_VALUE ? length : hi;
196 
197                 final HttpResponseHeader header = context.getResponseHeader();
198                 header.status(httpProtocolVersion, 206, contentType, state.sendMax - lo);
199                 if (asAttachment) {
200                     header.put("Content-Disposition: attachment; filename=\"").put(FileNameExtractorCharSequence.get(path)).put('\"').put(Misc.EOL);
201                 }
202                 header.put("Accept-Ranges: bytes").put(Misc.EOL);
203                 header.put("Content-Range: bytes ").put(lo).put('-').put(state.sendMax).put('/').put(length).put(Misc.EOL);
204                 header.put("ETag: ").put(ff.getLastModified(path)).put(Misc.EOL);
205                 if (keepAliveHeader != null) {
206                     header.put(keepAliveHeader);
207                 }
208                 header.send();
209                 resumeSend(context);
210             }
211         } else {
212             sendStatusWithDefaultMessage(context, 416);
213         }
214     }

216     private void sendVanilla(
217             HttpConnectionContext context,
218             LPSZ path, CharSequence contentType,
219             boolean asAttachment
220     ) throws PeerDisconnectedException, PeerIsSlowToReadException {
221         int fd = ff.openRO(path);
222         if (fd == -1) {
223             LOG.info().$("Cannot open file: ").$(path).$('(').$(ff.errno()).$(')').$();
224             sendStatusWithDefaultMessage(context, 404);
225         } else {
226             StaticContentProcessorState h = LV.get(context);
227             if (h == null) {
228                 LV.set(context, h = new StaticContentProcessorState());
229             }
230             h.fd = fd;
231             h.bytesSent = 0;
232             final long length = ff.length(path);
233             h.sendMax = length;
234 
235             final HttpResponseHeader header = context.getResponseHeader();
236             header.status(httpProtocolVersion, 200, contentType, length);
237             if (asAttachment) {
238                 header.put("Content-Disposition: attachment; filename=\"").put(FileNameExtractorCharSequence.get(path)).put("\"").put(Misc.EOL);
239             }
240             header.put("ETag: ").put('"').put(ff.getLastModified(path)).put('"').put(Misc.EOL);
241             header.setKeepAlive(keepAliveHeader);
242             header.send();
243             resumeSend(context);
244         }
245     }
