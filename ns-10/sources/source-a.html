141     private void deflate() {
142         if (!compressedHeaderDone) {
143             int len = Zip.gzipHeaderLen;
144             Vect.memcpy(compressOutBuffer.getWriteAddress(len), Zip.gzipHeader, len);
145             compressOutBuffer.onWrite(len);
146             compressedHeaderDone = true;
147         }
148 
149         int nInAvailable = (int) buffer.getReadNAvailable();
150         if (nInAvailable > 0) {
151             long inAddress = buffer.getReadAddress();
152             LOG.debug().$("Zip.setInput [inAddress=").$(inAddress).$(", nInAvailable=").$(nInAvailable).$(']').$();
153             buffer.write64BitZeroPadding();
154             Zip.setInput(z_streamp, inAddress, nInAvailable);
155         }
156 
157         int ret;
158         int len;
159         // compress input until we run out of either input or output
160         do {
161             int sz = (int) compressOutBuffer.getWriteNAvailable() - 8;
162             long p = compressOutBuffer.getWriteAddress(0);
163             LOG.debug().$("deflate starting [p=").$(p).$(", sz=").$(sz).$(", chunkedRequestDone=").$(chunkedRequestDone).$(']').$();
164             ret = Zip.deflate(z_streamp, p, sz, chunkedRequestDone);
165             len = sz - Zip.availOut(z_streamp);
166             compressOutBuffer.onWrite(len);
167             if (ret < 0) {
168                 // This is not an error, zlib just couldn't do any work with the input/output buffers it was provided.
169                 // This happens often (will depend on output buffer size) when there is no new input and zlib has finished generating
170                 // output from previously provided input
171                 if (ret != Zip.Z_BUF_ERROR || len != 0) {
172                     throw HttpException.instance("could not deflate [ret=").put(ret);
173                 }
174             }
175 
176             int availIn = Zip.availIn(z_streamp);
177             int nInConsumed = nInAvailable - availIn;
178             if (nInConsumed > 0) {
179                 this.crc = Zip.crc32(this.crc, buffer.getReadAddress(), nInConsumed);
180                 this.total += nInConsumed;
181                 buffer.onRead(nInConsumed);
182                 nInAvailable = availIn;
183             }
184 
185             LOG.debug().$("deflate finished [ret=").$(ret).$(", len=").$(len).$(", availIn=").$(availIn).$(']').$();
186         } while (len == 0 && nInAvailable > 0);
187 
188         if (nInAvailable == 0) {
189             buffer.clearAndPrepareToWriteToBuffer();
190         }
191 
192         if (len == 0) {
193             compressedOutputReady = false;
194             return;
195         }
196         compressedOutputReady = true;
197 
198         // this is ZLib error, can't continue
199         if (len < 0) {
200             throw ZLibException.INSTANCE;
201         }
202 
203         // trailer
204         boolean finished = chunkedRequestDone && ret == Zip.Z_STREAM_END;
205         if (finished) {
206             long p = compressOutBuffer.getWriteAddress(0);
207             Unsafe.getUnsafe().putInt(p, crc); // crc
208             Unsafe.getUnsafe().putInt(p + 4, (int) total); // total
209             compressOutBuffer.onWrite(8);
210             compressionComplete = true;
211         }
212         compressOutBuffer.prepareToReadFromBuffer(true, finished);
213     }

243     private void sendBuffer(ChunkBuffer sendBuf) throws PeerDisconnectedException, PeerIsSlowToReadException {
244         int nSend = (int) sendBuf.getReadNAvailable();
245         while (nSend > 0) {
246             int n = nf.send(fd, sendBuf.getReadAddress(), nSend);
247             if (n < 0) {
248                 // disconnected
249                 LOG.error()
250                         .$("disconnected [errno=").$(nf.errno())
251                         .$(", fd=").$(fd)
252                         .$(']').$();
253                 throw PeerDisconnectedException.INSTANCE;
254             }
255             if (n == 0) {
256                 // test how many times we tried to send before parking up
257                 throw PeerIsSlowToReadException.INSTANCE;
258             } else {
259                 dumpBuffer(sendBuf.getReadAddress(), n);
260                 sendBuf.onRead(n);
261                 nSend -= n;
262                 totalBytesSent += n;
263             }
264         }
265         assert sendBuf.getReadNAvailable() == 0;
266         sendBuf.clearAndPrepareToWriteToBuffer();
267     }
