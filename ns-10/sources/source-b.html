59     public HttpServer(HttpMinServerConfiguration configuration, MessageBus messageBus, Metrics metrics, WorkerPool pool) {
60         this.workerCount = pool.getWorkerCount();
61         this.selectors = new ObjList<>(workerCount);
62 
63         for (int i = 0; i < workerCount; i++) {
64             selectors.add(new HttpRequestProcessorSelectorImpl());
65         }
66 
67         this.httpContextFactory = new HttpContextFactory(configuration.getHttpContextConfiguration(), metrics);
68         this.dispatcher = IODispatchers.create(
69                 configuration.getDispatcherConfiguration(),
70                 httpContextFactory
71         );
72         pool.assign(dispatcher);
73         this.rescheduleContext = new WaitProcessor(configuration.getWaitProcessorConfiguration());
74         pool.assign(this.rescheduleContext);
75 
76         for (int i = 0; i < workerCount; i++) {
77             final int index = i;
78 
79             final SCSequence queryCacheEventSubSeq = new SCSequence();
80             final FanOut queryCacheEventFanOut = messageBus.getQueryCacheEventFanOut();
81             queryCacheEventFanOut.and(queryCacheEventSubSeq);
82 
83             pool.assign(i, new Job() {
84                 private final HttpRequestProcessorSelector selector = selectors.getQuick(index);
85                 private final IORequestProcessor<HttpConnectionContext> processor =
86                         (operation, context) -> context.handleClientOperation(operation, selector, rescheduleContext);
87 
88                 @Override
89                 public boolean run(int workerId, @NotNull RunStatus runStatus) {
90                     long seq = queryCacheEventSubSeq.next();
91                     if (seq > -1) {
92                         // Queue is not empty, so flush query cache.
93                         LOG.info().$("flushing HTTP server query cache [worker=").$(workerId).$(']').$();
94                         QueryCache queryCache = QueryCache.getWeakThreadLocalInstance();
95                         if (queryCache != null) {
96                             queryCache.clear();
97                         }
98                         queryCacheEventSubSeq.done(seq);
99                     }
100 
101                     boolean useful = dispatcher.processIOQueue(processor);
102                     useful |= rescheduleContext.runReruns(selector);
103 
104                     return useful;
105                 }
106             });
107 
108             // http context factory has thread local pools
109             // therefore we need each thread to clean their thread locals individually
110             pool.assignThreadLocalCleaner(i, () -> {
111                 httpContextFactory.freeThreadLocal();
112                 Misc.free(QueryCache.getWeakThreadLocalInstance());
113             });
114 
115             pool.freeOnExit(() -> {
116                 messageBus.getQueryCacheEventFanOut().remove(queryCacheEventSubSeq);
117                 queryCacheEventSubSeq.clear();
118             });
119         }
120     }

204     public void bind(HttpRequestProcessorFactory factory, boolean useAsDefault) {
205         final String url = factory.getUrl();
206         assert url != null;
207         for (int i = 0; i < workerCount; i++) {
208             HttpRequestProcessorSelectorImpl selector = selectors.getQuick(i);
209             if (HttpServerConfiguration.DEFAULT_PROCESSOR_URL.equals(url)) {
210                 selector.defaultRequestProcessor = factory.newInstance();
211             } else {
212                 final HttpRequestProcessor processor = factory.newInstance();
213                 selector.processorMap.put(url, processor);
214                 if (useAsDefault) {
215                     selector.defaultRequestProcessor = processor;
216                 }
217             }
218         }
219     }
