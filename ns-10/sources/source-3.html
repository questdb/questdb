200         boolean keepGoing;

202             case IOOperation.READ:
203                 keepGoing = handleClientRecv(selector, rescheduleContext);

267         if (pendingRetry) {
268             pendingRetry = false;
269             HttpRequestProcessor processor = getHttpRequestProcessor(selector);
270             try {
271                 LOG.info().$("retrying query [fd=").$(fd).$(']').$();
272                 processor.onRequestRetry(this);
273                 if (multipartParserState.multipartRetry) {
274                     if (continueConsumeMultipart(
275                             fd,
276                             multipartParserState.start,
277                             multipartParserState.buf,
278                             multipartParserState.bufRemaining,
279                             (HttpMultipartContentListener) processor,
280                             processor,
281                             retryRescheduleContext

285                     } else {
286                         LOG.info().$("retry success but import not finished [fd=").$(fd).$(']').$();
287                     }
288                 } else {
289                     busyRcvLoop(selector, rescheduleContext);
290                 }
291             } catch (RetryOperationException e2) {
292                 pendingRetry = true;
293                 return false;
294             } catch (PeerDisconnectedException ignore) {

304                         .$(Thread.currentThread().getId()).$(']').$();
305                 processor.parkRequest(this, true);
306                 resumeProcessor = processor;
307                 suspendEvent = e.getEvent();
308                 dispatcher.registerChannel(this, IOOperation.WRITE);
309             } catch (ServerDisconnectException e) {
310                 LOG.info().$("kicked out [fd=").$(fd).$(']').$();
311                 dispatcher.disconnect(this, DISCONNECT_REASON_KICKED_OUT_AT_RERUN);
312             }
313         }
314         return true;

343             HttpRequestProcessor processor,
344             long headerEnd,
345             int read,
346             boolean newRequest,
347             RescheduleContext rescheduleContext
348     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException, QueryPausedException {
349         if (newRequest) {
350             processor.onHeadersReady(this);

352         }

380     private boolean continueConsumeMultipart(
381             int fd,
382             long start,
383             long buf,
384             int bufRemaining,
385             HttpMultipartContentListener multipartListener,
386             HttpRequestProcessor processor,
387             RescheduleContext rescheduleContext
388     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException, QueryPausedException {
389         boolean keepGoing = false;
390 
391         if (buf > start) {
392             try {
393                 if (parseMultipartResult(start, buf, bufRemaining, multipartListener, processor, rescheduleContext)) {
394                     return true;
395                 }
396 
397                 buf = start = recvBuffer;
398                 bufRemaining = recvBufferSize;
399             } catch (TooFewBytesReceivedException e) {
400                 start = multipartContentParser.getResumePtr();
401             }
402         }
403 
404         long spinsRemaining = multipartIdleSpinCount;
405 
406         while (true) {
407             final int n = nf.recv(fd, buf, bufRemaining);
408             if (n < 0) {
409                 dispatcher.disconnect(this, DISCONNECT_REASON_PEER_DISCONNECT_AT_MULTIPART_RECV);
410                 break;
411             }
412 
413             if (n == 0) {
414                 // Text loader needs as big of a data chunk as possible
415                 // to analyse columns and delimiters correctly. To make sure we
416                 // can deliver large data chunk we have to implement mini-Nagle
417                 // algorithm by accumulating small data chunks client could be
418                 // sending into our receive buffer. To make sure we don't
419                 // sit around accumulating for too long we have spin limit
420                 if (spinsRemaining-- > 0) {
421                     continue;
422                 }
423 
424                 // do we have anything in the buffer?
425                 if (buf > start) {
426                     try {
427                         if (parseMultipartResult(start, buf, bufRemaining, multipartListener, processor, rescheduleContext)) {
428                             keepGoing = true;
429                             break;
430                         }
431 
432                         buf = start = recvBuffer;
433                         bufRemaining = recvBufferSize;
434                         continue;
435                     } catch (TooFewBytesReceivedException e) {
436                         start = multipartContentParser.getResumePtr();
437                         shiftReceiveBufferUnprocessedBytes(start, (int) (buf - start));
438                         dispatcher.registerChannel(this, IOOperation.READ);
439                         break;
440                     }
441                 }
442 
443                 LOG.debug().$("peer is slow [multipart]").$();
444                 dispatcher.registerChannel(this, IOOperation.READ);
445                 break;
446             }
447 
448             LOG.debug().$("multipart recv [len=").$(n).$(']').$();
449 
450             dumpBuffer(buf, n);
451 
452             bufRemaining -= n;
453             buf += n;
454 
455             if (bufRemaining == 0) {
456                 try {
457                     if (buf - start > 1) {
458                         if (parseMultipartResult(start, buf, bufRemaining, multipartListener, processor, rescheduleContext)) {
459                             keepGoing = true;
460                             break;
461                         }
462                     }
463 
464                     buf = start = recvBuffer;
465                     bufRemaining = recvBufferSize;
466                 } catch (TooFewBytesReceivedException e) {
467                     start = multipartContentParser.getResumePtr();
468                     int unprocessedSize = (int) (buf - start);
469                     // Shift to start
470                     if (unprocessedSize < recvBufferSize) {
471                         start = multipartContentParser.getResumePtr();
472                         shiftReceiveBufferUnprocessedBytes(start, unprocessedSize);
473                         dispatcher.registerChannel(this, IOOperation.READ);
474                     } else {
475                         // Header does not fit receive buffer
476                         failProcessor(processor, BufferOverflowException.INSTANCE, DISCONNECT_REASON_MULTIPART_HEADER_TOO_BIG);
477                     }
478                     break;
479                 }
480             }
481         }
482         return keepGoing;
483     }

528     private boolean handleClientRecv(HttpRequestProcessorSelector selector, RescheduleContext rescheduleContext) {
529         boolean busyRecv = true;
530         try {
531             final int fd = this.fd;
532             // this is address of where header ended in our receive buffer
533             // we need to being processing request content starting from this address
534             long headerEnd = recvBuffer;
535             int read = 0;
536             final boolean newRequest = headerParser.isIncomplete();
537             if (newRequest) {
538                 while (headerParser.isIncomplete()) {
539                     // read headers
540                     read = nf.recv(fd, recvBuffer, recvBufferSize);
541                     LOG.debug().$("recv [fd=").$(fd).$(", count=").$(read).$(']').$();
542                     if (read < 0) {
543                         LOG.debug()
544                                 .$("done [fd=").$(fd)
545                                 .$(", errno=").$(nf.errno())
546                                 .$(']').$();
547                         // peer disconnect
548                         dispatcher.disconnect(this, DISCONNECT_REASON_PEER_DISCONNECT_AT_HEADER_RECV);
549                         return false;
550                     }
551 
552                     if (read == 0) {
553                         // client is not sending anything
554                         dispatcher.registerChannel(this, IOOperation.READ);
555                         return false;
556                     }
557 
558                     dumpBuffer(recvBuffer, read);
559                     headerEnd = headerParser.parse(recvBuffer, recvBuffer + read, true);
560                 }
561             }
562 
563             final CharSequence url = headerParser.getUrl();
564             if (url == null) {
565                 throw HttpException.instance("missing URL");
566             }
567             HttpRequestProcessor processor = getHttpRequestProcessor(selector);
568 
569             final boolean multipartRequest = Chars.equalsNc("multipart/form-data", headerParser.getContentType());
570             final boolean multipartProcessor = processor instanceof HttpMultipartContentListener;
571 
572             if (allowDeflateBeforeSend && Chars.contains(headerParser.getHeader("Accept-Encoding"), "gzip")) {
573                 responseSink.setDeflateBeforeSend(true);
574             }
575 
576             try {
577                 if (multipartRequest && !multipartProcessor) {
578                     // bad request - multipart request for processor that doesn't expect multipart
579                     busyRecv = rejectRequest("Bad request. non-multipart GET expected.");
580                 } else if (!multipartRequest && multipartProcessor) {
581                     // bad request - regular request for processor that expects multipart
582                     busyRecv = rejectRequest("Bad request. Multipart POST expected.");
583                 } else if (multipartProcessor) {
584                     busyRecv = consumeMultipart(fd, processor, headerEnd, read, newRequest, rescheduleContext);
585                 } else {
586 
587                     // Do not expect any more bytes to be sent to us before
588                     // we respond back to client. We will disconnect the client when
589                     // they abuse protocol. In addition, we will not call processor
590                     // if client has disconnected before we had a chance to reply.
591                     read = nf.recv(fd, recvBuffer, 1);
592                     if (read != 0) {
593                         dumpBuffer(recvBuffer, read);
594                         LOG.info().$("disconnect after request [fd=").$(fd).$(']').$();
595                         dispatcher.disconnect(this, DISCONNECT_REASON_KICKED_OUT_AT_EXTRA_BYTES);
596                         busyRecv = false;
597                     } else {
598                         processor.onHeadersReady(this);
599                         LOG.debug().$("good [fd=").$(fd).$(']').$();
600                         processor.onRequestComplete(this);
601                         resumeProcessor = null;
602                         clear();
603                     }
604                 }
605             } catch (RetryOperationException e) {
606                 pendingRetry = true;
607                 scheduleRetry(processor, rescheduleContext);
608                 busyRecv = false;
609             } catch (PeerDisconnectedException e) {
610                 dispatcher.disconnect(this, DISCONNECT_REASON_PEER_DISCONNECT_AT_RECV);
611                 busyRecv = false;
612             } catch (ServerDisconnectException e) {
613                 LOG.info().$("kicked out [fd=").$(fd).$(']').$();
614                 dispatcher.disconnect(this, DISCONNECT_REASON_KICKED_OUT_AT_RECV);
