50     public WaitProcessor(WaitProcessorConfiguration configuration) {

52         maxWaitCapMs = configuration.getMaxWaitCapMs();
53         exponentialWaitMultiplier = configuration.getExponentialWaitMultiplier();

83     public boolean runReruns(HttpRequestProcessorSelector selector) {
84         boolean useful = false;
85 
86         while (true) {
87             Retry retry = getNextRerun();
88             if (retry != null) {
89                 useful = true;
90                 if (!retry.tryRerun(selector, this)) {
91                     try {
92                         reschedule(retry, retry.getAttemptDetails().attempt + 1, retry.getAttemptDetails().waitStartTimestamp);
93                     } catch (RetryFailedOperationException e) {
94                         retry.fail(selector, e);
95                     }
96                 }

98                 return useful;
99             }
100         }

103     private static int compareRetriesInQueue(Retry r1, Retry r2) {
104         // r1, r2 are always not null, null retries are not queued
105         RetryAttemptAttributes a1 = r1.getAttemptDetails();
106         RetryAttemptAttributes a2 = r2.getAttemptDetails();
107         return Long.compare(a1.nextRunTimestamp, a2.nextRunTimestamp);
108     }
109 
110     private long calculateNextTimestamp(RetryAttemptAttributes attemptAttributes) {
111         if (attemptAttributes.attempt == 0) {
112             // First retry after fixed time of 2ms
113             return attemptAttributes.lastRunTimestamp + 2;
114         }
115 
116         // 'exponentialWaitMultiplier' times wait time starting until it is 'maxWaitCapMs' sec
117         long totalWait = attemptAttributes.lastRunTimestamp - attemptAttributes.waitStartTimestamp;
118         return Math.min(maxWaitCapMs, Math.max(4L, (long) (totalWait * exponentialWaitMultiplier))) + attemptAttributes.lastRunTimestamp;
119     }

121     private @Nullable Retry getNextRerun() {
122         long cursor = outSubSequence.next();
123         // -2 = there was a contest for queue index and this thread has lost
124         if (cursor < 0) {
125             return null;
126         }
127 
128         RetryHolder retryHolder = outQueue.get(cursor);
129         Retry r = retryHolder.retry;
130         retryHolder.retry = null;
131         outSubSequence.done(cursor);
132         return r;
133     }

135     // Process incoming queue and put it on priority queue with next timestamp to rerun
136     private boolean processInQueue() {
137         boolean any = false;
138         while (true) {
139             long cursor = inSubSequence.next();
140             // -2 = there was a contest for queue index and this thread has lost
141             if (cursor < -1) {
142                 Os.pause();
143                 continue;
144             }
145 
146             // -1 = queue is empty. All done.
147             if (cursor < 0) {
148                 return any;
149             }
150 
151             Retry retry;
152             try {
153                 RetryHolder toRun = inQueue.get(cursor);
154                 retry = toRun.retry;
155                 toRun.retry = null;
156             } finally {
157                 inSubSequence.done(cursor);
158             }
159             retry.getAttemptDetails().nextRunTimestamp = calculateNextTimestamp(retry.getAttemptDetails());
160 
161             nextRerun.add(retry);
162             any = true;
163         }
164     }

166     private void reschedule(Retry retry, int attempt, long waitStartMs) {
167         long now = clock.getTicks();
168         retry.getAttemptDetails().attempt = attempt;
169         retry.getAttemptDetails().lastRunTimestamp = now;
170         retry.getAttemptDetails().waitStartTimestamp = attempt == 0 ? now : waitStartMs;
171 
172         while (true) {
173             long cursor = inPubSequence.next();
174             // -2 = there was a contest for queue index and this thread has lost
175             if (cursor < -1) {
176                 Os.pause();
177                 continue;
178             }
179 
180             // -1 = queue is full. It means there are already too many retries waiting
181             // Send error to client.
182             if (cursor < 0) {
183                 throw RetryFailedOperationException.INSTANCE;
184             }
185 
186             RetryHolder retryHolder = inQueue.get(cursor);
187             retryHolder.retry = retry;
188             inPubSequence.done(cursor);
189             return;
190         }
191     }

193     private boolean sendToOutQueue() {
194         boolean useful = false;
195         final long now = clock.getTicks();
196         while (nextRerun.size() > 0) {
197             Retry next = nextRerun.peek();
198             if (next.getAttemptDetails().nextRunTimestamp <= now) {
199                 useful = true;
200                 Retry retry = nextRerun.poll();
201                 if (!sendToOutQueue(retry)) {
202                     nextRerun.add(retry);
203                     return true;
204                 }
205             } else {
206                 // All reruns are in the future.
207                 return useful;
208             }
209         }
210         return useful;
211     }
212 
213     private boolean sendToOutQueue(Retry retry) {
214         while (true) {
215             long cursor = outPubSequence.next();
216             // -2 = there was a contest for queue index and this thread has lost
217             if (cursor < -1) {
218                 Os.pause();
219                 continue;
220             }
221 
222             if (cursor < 0) {
223                 // Cannot put to out queue. It is full. Release job and retry next run.
224                 return false;
225             }
226 
227             RetryHolder retryHolderOut = outQueue.get(cursor);
228             retryHolderOut.retry = retry;
229             outPubSequence.done(cursor);
230             return true;
231         }
232     }
