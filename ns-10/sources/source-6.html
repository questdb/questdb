101     public boolean parse(
102             long lo,
103             long hi,
104             HttpMultipartContentListener listener
105     ) throws PeerDisconnectedException, PeerIsSlowToReadException, ServerDisconnectException {
106         long _lo = lo;
107         long ptr = lo;
108         while (ptr < hi) {
109             switch (state) {
110                 case BODY_BROKEN:
111                     _lo = ptr;
112                     state = BODY;
113                     break;
114                 case START_PARSING:
115                     state = START_BOUNDARY;
116                     // fall through
117                 case START_BOUNDARY:
118                     boundaryPtr = 2;
119                     // fall through
120                 case PARTIAL_START_BOUNDARY:
121                     switch (matchBoundary(ptr, hi)) {
122                         case BOUNDARY_INCOMPLETE:
123                             state = PARTIAL_START_BOUNDARY;
124                             return false;
125                         case BOUNDARY_MATCH:
126                             state = START_PRE_HEADERS;
127                             ptr += consumedBoundaryLen;
128                             break;
129                         default:
130                             throw HttpException.instance("Malformed start boundary");
131                     }
132                     break;
133                 case PRE_HEADERS:
134                     switch (Unsafe.getUnsafe().getByte(ptr)) {
135                         case '\n':
136                             state = HEADERS;
137                             // fall through
138                         case '\r':
139                             ptr++;
140                             break;
141                         case '-':
142                             listener.onPartEnd();
143                             state = DONE;
144                             return true;
145                         default:
146                             listener.onChunk(boundary.getLo(), boundary.getHi());
147                             _lo = ptr;
148                             state = BODY;
149                             break;
150                     }
151                     break;
152                 case START_PRE_HEADERS:
153                     switch (Unsafe.getUnsafe().getByte(ptr)) {
154                         case '\n':
155                             state = START_HEADERS;
156                             // fall through
157                         case '\r':
158                             ptr++;
159                             break;
160                         case '-':
161                             return true;
162                         default:
163                             throw HttpException.instance("Malformed start boundary");
164                     }
165                     break;
166                 case HEADERS:
167                     listener.onPartEnd();
168                     state = HEADERS;
169                     // fall through
170                 case START_HEADERS:
171                     headerParser.clear();
172                     state = START_HEADERS;
173                     // fall through
174                 case PARTIAL_HEADERS:
175                     ptr = headerParser.parse(ptr, hi, false);
176                     if (headerParser.isIncomplete()) {
177                         state = PARTIAL_HEADERS;
178                         return false;
179                     }
180                     _lo = ptr;
181                     listener.onPartBegin(headerParser);
182                     state = BODY;
183                     break;
184                 case BODY:
185                     byte b = Unsafe.getUnsafe().getByte(ptr++);
186                     if (b == boundaryByte) {
187                         boundaryPtr = 1;
188                         switch (matchBoundary(ptr, hi)) {
189                             case BOUNDARY_INCOMPLETE:
190                                 onChunkWithRetryHandle(listener, _lo, ptr - 1, POTENTIAL_BOUNDARY, hi, true);
191                                 return false;
192                             case BOUNDARY_MATCH:
193                                 ptr = onChunkWithRetryHandle(listener, _lo, ptr - 1, PRE_HEADERS, ptr + consumedBoundaryLen, false);
194                                 break;
195                             default:
196                                 break;
197                         }
198                     }
199                     break;
200                 case POTENTIAL_BOUNDARY:
201                     int p = boundaryPtr;
202                     switch (matchBoundary(ptr, hi)) {
203                         case BOUNDARY_INCOMPLETE:
204                             return false;
205                         case BOUNDARY_MATCH:
206                             ptr += consumedBoundaryLen;
207                             state = PRE_HEADERS;
208                             break;
209                         default:
210                             // can only be BOUNDARY_NO_MATCH:
211                             onChunkWithRetryHandle(listener, boundary.getLo(), boundary.getLo() + p, BODY_BROKEN, ptr, true);
212                             break;
213                     }
214                     break;
215                 default:
216                     // DONE
217                     return true;
218             }
219         }
220 
221         if (state == BODY) {
222             onChunkWithRetryHandle(listener, _lo, ptr, BODY_BROKEN, ptr, true);
223         }
224 
225         return false;
226     }

228     private int matchBoundary(long lo, long hi) {
229         long start = lo;
230         int ptr = boundaryPtr;
231 
232         while (lo < hi && ptr < boundaryLen) {
233             if (Unsafe.getUnsafe().getByte(lo++) != boundary.byteAt(ptr++)) {
234                 return BOUNDARY_NO_MATCH;
235             }
236         }
237 
238         this.boundaryPtr = ptr;
239 
240         if (boundaryPtr < boundaryLen) {
241             return BOUNDARY_INCOMPLETE;
242         }
243 
244         this.consumedBoundaryLen = (int) (lo - start);
245         return BOUNDARY_MATCH;
246     }

248     private long onChunkWithRetryHandle(
249             HttpMultipartContentListener listener,
250             long lo,
251             long hi,
252             int state,
253             long resumePtr,
254             boolean handleIncomplete
255     ) throws PeerIsSlowToReadException, PeerDisconnectedException, ServerDisconnectException {
256         RetryOperationException needsRetry = null;
257         try {
258             listener.onChunk(lo, hi);
259         } catch (RetryOperationException e) {
260             // Request re-try.
261             needsRetry = e;
262         } catch (NotEnoughLinesException e) {
263             if (handleIncomplete) {
264                 this.resumePtr = lo;
265                 throw TooFewBytesReceivedException.INSTANCE;
266             } else {
267                 throw e;
268             }
269         }
270 
271         // Roll to the next state.
272         this.state = state;
273 
274         // And save point to continue even if RetryOperationException happened.
275         this.resumePtr = resumePtr;
276 
277         // If retry exception happened, rethrow after setting state and resume point.
278         if (needsRetry != null) throw needsRetry;
279 
280         return resumePtr;
281     }
