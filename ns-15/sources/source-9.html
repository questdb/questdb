124     void append() throws CommitFailedException {
125         TableWriter.Row row = null;
126         try {
127             TableWriterAPI writer = tableUpdateDetails.getWriter();
128             long offset = buffer.getAddress();
129             final long structureVersion = buffer.readLong(offset);
130             offset += Long.BYTES;
131             if (structureVersion > writer.getStructureVersion()) {
132                 // I/O thread has a more recent version of the WAL table metadata than the writer.
133                 // Let the WAL writer commit, so that it refreshes its metadata copy.
134                 writer.commit();
135             }
136             long timestamp = buffer.readLong(offset);
137             offset += Long.BYTES;
138             if (timestamp == LineTcpParser.NULL_TIMESTAMP) {
139                 timestamp = clock.getTicks();
140             }
141             row = writer.newRow(timestamp);
142             final int nEntities = buffer.readInt(offset);
143             offset += Integer.BYTES;
144             final long writerStructureVersion = writer.getStructureVersion();
145             for (int nEntity = 0; nEntity < nEntities; nEntity++) {
146                 int colIndex = buffer.readInt(offset);
147                 offset += Integer.BYTES;
148                 final byte entityType;
149                 if (colIndex > -1) {
150                     entityType = buffer.readByte(offset);
151                     offset += Byte.BYTES;
152                     // Did the I/O thread have the latest structure version when it serialized the row?
153                     if (structureVersion < writerStructureVersion) {
154                         // Nope. For WAL tables, it could mean that the column is already dropped. Let's check it.
155                         if (!writer.getMetadata().hasColumn(colIndex)) {
156                             // The column was dropped, so we skip it.
157                             offset += buffer.columnValueLength(entityType, offset);
158                             continue;
159                         }
160                     }
161                 } else {
162                     // Column is passed by name, it is possible that
163                     // column is new and has to be added. It is also possible that column
164                     // already exist but the publisher is a little out of date and does not yet
165                     // have column index.
166 
167                     // Column name will be UTF16 encoded already
168                     final CharSequence columnName = buffer.readUtf16Chars(offset, -colIndex);
169                     offset += -colIndex * 2L;
170 
171                     entityType = buffer.readByte(offset);
172                     offset += Byte.BYTES;
173                     colIndex = writer.getMetadata().getColumnIndexQuiet(columnName);
174                     if (colIndex < 0) {
175                         // we have to cancel "active" row to avoid writer committing when
176                         // column is added
177                         row.cancel();
178                         row = null;
179                         final int colType = defaultColumnTypes.MAPPED_COLUMN_TYPES[entityType];
180                         // we have to commit before adding a new column as WalWriter doesn't do that automatically
181                         writer.commit();
182                         try {
183                             writer.addColumn(columnName, colType);
184                         } catch (CairoException e) {
185                             colIndex = writer.getMetadata().getColumnIndexQuiet(columnName);
186                             if (colIndex < 0) {
187                                 // the column is still not there, something must be wrong
188                                 throw e;
189                             }
190                             // all good, someone added the column concurrently
191                         }
192 
193                         // Seek to beginning of entities
194                         offset = buffer.getAddressAfterHeader();
195                         nEntity = -1;
196                         row = writer.newRow(timestamp);
197                         continue;
198                     }
199                 }
200 
201                 CharSequence cs;
202                 switch (entityType) {
203                     case LineTcpParser.ENTITY_TYPE_TAG:
204                         cs = buffer.readUtf16Chars(offset);
205                         row.putSym(colIndex, cs);
206                         offset += cs.length() * 2L + Integer.BYTES;
207                         break;
208                     case LineTcpParser.ENTITY_TYPE_CACHED_TAG:
209                         row.putSymIndex(colIndex, buffer.readInt(offset));
210                         offset += Integer.BYTES;
211                         break;
212                     case LineTcpParser.ENTITY_TYPE_LONG:
213                     case LineTcpParser.ENTITY_TYPE_GEOLONG:
214                         row.putLong(colIndex, buffer.readLong(offset));
215                         offset += Long.BYTES;
216                         break;
217                     case LineTcpParser.ENTITY_TYPE_INTEGER:
218                     case LineTcpParser.ENTITY_TYPE_GEOINT:
219                         row.putInt(colIndex, buffer.readInt(offset));
220                         offset += Integer.BYTES;
221                         break;
222                     case LineTcpParser.ENTITY_TYPE_SHORT:
223                     case LineTcpParser.ENTITY_TYPE_GEOSHORT:
224                         row.putShort(colIndex, buffer.readShort(offset));
225                         offset += Short.BYTES;
226                         break;
227                     case LineTcpParser.ENTITY_TYPE_BYTE:
228                     case LineTcpParser.ENTITY_TYPE_GEOBYTE:
229                         row.putByte(colIndex, buffer.readByte(offset));
230                         offset += Byte.BYTES;
231                         break;
232                     case LineTcpParser.ENTITY_TYPE_DATE:
233                         row.putDate(colIndex, buffer.readLong(offset));
234                         offset += Long.BYTES;
235                         break;
236                     case LineTcpParser.ENTITY_TYPE_DOUBLE:
237                         row.putDouble(colIndex, buffer.readDouble(offset));
238                         offset += Double.BYTES;
239                         break;
240                     case LineTcpParser.ENTITY_TYPE_FLOAT:
241                         row.putFloat(colIndex, buffer.readFloat(offset));
242                         offset += Float.BYTES;
243                         break;
244                     case LineTcpParser.ENTITY_TYPE_BOOLEAN:
245                         row.putBool(colIndex, buffer.readByte(offset) == 1);
246                         offset += Byte.BYTES;
247                         break;
248                     case LineTcpParser.ENTITY_TYPE_STRING:
249                         cs = buffer.readUtf16Chars(offset);
250                         row.putStr(colIndex, cs);
251                         offset += cs.length() * 2L + Integer.BYTES;
252                         break;
253                     case LineTcpParser.ENTITY_TYPE_CHAR:
254                         row.putChar(colIndex, buffer.readChar(offset));
255                         offset += Character.BYTES;
256                         break;
257                     case LineTcpParser.ENTITY_TYPE_LONG256:
258                         cs = buffer.readUtf16Chars(offset);
259                         row.putLong256(colIndex, cs);
260                         offset += cs.length() * 2L + Integer.BYTES;
261                         break;
262                     case LineTcpParser.ENTITY_TYPE_TIMESTAMP:
263                         row.putTimestamp(colIndex, buffer.readLong(offset));
264                         offset += Long.BYTES;
265                         break;
266                     case LineTcpParser.ENTITY_TYPE_UUID:
267                         row.putLong128(colIndex, buffer.readLong(offset), buffer.readLong(offset + Long.BYTES));
268                         offset += Long.BYTES * 2;
269                         break;
270                     case ENTITY_TYPE_NULL:
271                         // ignored, default nulls is used
272                         break;
273                     default:
274                         throw new UnsupportedOperationException("entityType " + entityType + " is not implemented!");
275                 }
276             }
277             row.append();
278             tableUpdateDetails.commitIfMaxUncommittedRowsCountReached();
279         } catch (CommitFailedException commitFailedException) {
280             throw commitFailedException;
281         } catch (Throwable th) {
282             LOG.error()
283                     .$("could not write line protocol measurement [tableName=").$(tableUpdateDetails.getTableToken())
284                     .$(", message=").$(th.getMessage())
285                     .$(th)
286                     .I$();
287             if (row != null) {
288                 row.cancel();
289             }
290         }
291     }

293     void createMeasurementEvent(
294             TableUpdateDetails tud,
295             LineTcpParser parser,
296             int workerId
297     ) {
298         writerWorkerId = LineTcpMeasurementEventType.ALL_WRITERS_INCOMPLETE_EVENT;
299         final TableUpdateDetails.ThreadLocalDetails localDetails = tud.getThreadLocalDetails(workerId);
300         localDetails.resetStateIfNecessary();
301         this.tableUpdateDetails = tud;
302         long timestamp = parser.getTimestamp();
303         if (timestamp != LineTcpParser.NULL_TIMESTAMP) {
304             timestamp = timestampAdapter.getMicros(timestamp);
305         }
306         buffer.addStructureVersion(buffer.getAddress(), localDetails.getStructureVersion());
307         // timestamp, entitiesWritten are written to the buffer after saving all fields
308         // because their values are worked out while the columns are processed
309         long offset = buffer.getAddressAfterHeader();
310         int entitiesWritten = 0;
311         for (int nEntity = 0, n = parser.getEntityCount(); nEntity < n; nEntity++) {
312             LineTcpParser.ProtoEntity entity = parser.getEntity(nEntity);
313             byte entityType = entity.getType();
314             int colType;
315             int columnWriterIndex = localDetails.getColumnIndex(entity.getName(), parser.hasNonAsciiChars());
316             if (columnWriterIndex > -1) {
317                 // column index found, processing column by index
318                 if (columnWriterIndex == tud.getTimestampIndex()) {
319                     timestamp = timestampAdapter.getMicros(entity.getLongValue());
320                     continue;
321                 }
322 
323                 offset = buffer.addColumnIndex(offset, columnWriterIndex);
324                 colType = localDetails.getColumnType(columnWriterIndex);
325             } else if (columnWriterIndex == COLUMN_NOT_FOUND) {
326                 // send column by name
327                 final String colNameUtf16 = localDetails.getColNameUtf16();
328                 if (autoCreateNewColumns && TableUtils.isValidColumnName(colNameUtf16, maxColumnNameLength)) {
329                     offset = buffer.addColumnName(offset, colNameUtf16);
330                     colType = localDetails.getColumnType(localDetails.getColNameUtf8(), entityType);
331                 } else if (!autoCreateNewColumns) {
332                     throw newColumnsNotAllowed(tableUpdateDetails, colNameUtf16);
333                 } else {
334                     throw invalidColNameError(tableUpdateDetails, colNameUtf16);
335                 }
336             } else {
337                 // duplicate column, skip
338                 // we could set a boolean in the config if we want to throw exception instead
339                 continue;
340             }
341 
342             entitiesWritten++;
343             switch (entityType) {
344                 case LineTcpParser.ENTITY_TYPE_TAG: {
345                     if (ColumnType.tagOf(colType) == ColumnType.SYMBOL) {
346                         offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
347                     } else {
348                         throw castError("tag", columnWriterIndex, colType, entity.getName());
349                     }
350                     break;
351                 }
352                 case LineTcpParser.ENTITY_TYPE_INTEGER: {
353                     switch (ColumnType.tagOf(colType)) {
354                         case ColumnType.LONG:
355                             offset = buffer.addLong(offset, entity.getLongValue());
356                             break;
357 
358                         case ColumnType.INT: {
359                             final long entityValue = entity.getLongValue();
360                             if (entityValue >= Integer.MIN_VALUE && entityValue <= Integer.MAX_VALUE) {
361                                 offset = buffer.addInt(offset, (int) entityValue);
362                             } else if (entityValue == Numbers.LONG_NaN) {
363                                 offset = buffer.addInt(offset, Numbers.INT_NaN);
364                             } else {
365                                 throw boundsError(entityValue, nEntity, ColumnType.INT);
366                             }
367                             break;
368                         }
369                         case ColumnType.SHORT: {
370                             final long entityValue = entity.getLongValue();
371                             if (entityValue >= Short.MIN_VALUE && entityValue <= Short.MAX_VALUE) {
372                                 offset = buffer.addShort(offset, (short) entityValue);
373                             } else if (entityValue == Numbers.LONG_NaN) {
374                                 offset = buffer.addShort(offset, (short) 0);
375                             } else {
376                                 throw boundsError(entityValue, nEntity, ColumnType.SHORT);
377                             }
378                             break;
379                         }
380                         case ColumnType.BYTE: {
381                             final long entityValue = entity.getLongValue();
382                             if (entityValue >= Byte.MIN_VALUE && entityValue <= Byte.MAX_VALUE) {
383                                 offset = buffer.addByte(offset, (byte) entityValue);
384                             } else if (entityValue == Numbers.LONG_NaN) {
385                                 offset = buffer.addByte(offset, (byte) 0);
386                             } else {
387                                 throw boundsError(entityValue, nEntity, ColumnType.BYTE);
388                             }
389                             break;
390                         }
391                         case ColumnType.TIMESTAMP:
392                             offset = buffer.addTimestamp(offset, entity.getLongValue());
393                             break;
394 
395                         case ColumnType.DATE:
396                             offset = buffer.addDate(offset, entity.getLongValue());
397                             break;
398 
399                         case ColumnType.DOUBLE:
400                             offset = buffer.addDouble(offset, entity.getLongValue());
401                             break;
402 
403                         case ColumnType.FLOAT:
404                             offset = buffer.addFloat(offset, entity.getLongValue());
405                             break;
406 
407                         case ColumnType.SYMBOL:
408                             offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
409                             break;
410 
411                         default:
412                             throw castError("integer", columnWriterIndex, colType, entity.getName());
413                     }
414                     break;
415                 }
416                 case LineTcpParser.ENTITY_TYPE_FLOAT: {
417                     switch (ColumnType.tagOf(colType)) {
418                         case ColumnType.DOUBLE:
419                             offset = buffer.addDouble(offset, entity.getFloatValue());
420                             break;
421 
422                         case ColumnType.FLOAT:
423                             offset = buffer.addFloat(offset, (float) entity.getFloatValue());
424                             break;
425 
426                         case ColumnType.SYMBOL:
427                             offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
428                             break;
429 
430                         default:
431                             throw castError("float", columnWriterIndex, colType, entity.getName());
432                     }
433                     break;
434                 }
435                 case LineTcpParser.ENTITY_TYPE_STRING: {
436                     final int colTypeMeta = localDetails.getColumnTypeMeta(columnWriterIndex);
437                     final DirectByteCharSequence entityValue = entity.getValue();
438                     if (colTypeMeta == 0) { // not geohash
439                         switch (ColumnType.tagOf(colType)) {
440                             case ColumnType.STRING:
441                                 offset = buffer.addString(offset, entityValue, parser.hasNonAsciiChars());
442                                 break;
443 
444                             case ColumnType.CHAR:
445                                 if (entityValue.length() == 1 && entityValue.byteAt(0) > -1) {
446                                     offset = buffer.addChar(offset, entityValue.charAt(0));
447                                 } else if (stringToCharCastAllowed) {
448                                     int encodedResult = Chars.utf8CharDecode(entityValue.getLo(), entityValue.getHi());
449                                     if (Numbers.decodeLowShort(encodedResult) > 0) {
450                                         offset = buffer.addChar(offset, (char) Numbers.decodeHighShort(encodedResult));
451                                     } else {
452                                         throw castError("string", columnWriterIndex, colType, entity.getName());
453                                     }
454                                 } else {
455                                     throw castError("string", columnWriterIndex, colType, entity.getName());
456                                 }
457                                 break;
458 
459                             case ColumnType.SYMBOL:
460                                 offset = buffer.addSymbol(offset, entityValue, parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
461                                 break;
462 
463                             case ColumnType.UUID:
464                                 try {
465                                     offset = buffer.addUuid(offset, entityValue);
466                                 } catch (NumericException e) {
467                                     throw castError("string", columnWriterIndex, colType, entity.getName());
468                                 }
469                                 break;
470 
471                             default:
472                                 throw castError("string", columnWriterIndex, colType, entity.getName());
473                         }
474                     } else {
475                         offset = buffer.addGeoHash(offset, entityValue, colTypeMeta);
476                     }
477                     break;
478                 }
479                 case LineTcpParser.ENTITY_TYPE_LONG256: {
480                     switch (ColumnType.tagOf(colType)) {
481                         case ColumnType.LONG256:
482                             offset = buffer.addLong256(offset, entity.getValue(), parser.hasNonAsciiChars());
483                             break;
484 
485                         case ColumnType.SYMBOL:
486                             offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
487                             break;
488 
489                         default:
490                             throw castError("long256", columnWriterIndex, colType, entity.getName());
491                     }
492                     break;
493                 }
494                 case LineTcpParser.ENTITY_TYPE_BOOLEAN: {
495                     byte entityValue = (byte) (entity.getBooleanValue() ? 1 : 0);
496                     switch (ColumnType.tagOf(colType)) {
497                         case ColumnType.BOOLEAN:
498                             offset = buffer.addBoolean(offset, entityValue);
499                             break;
500 
501                         case ColumnType.BYTE:
502                             offset = buffer.addByte(offset, entityValue);
503                             break;
504 
505                         case ColumnType.SHORT:
506                             offset = buffer.addShort(offset, entityValue);
507                             break;
508 
509                         case ColumnType.INT:
510                             offset = buffer.addInt(offset, entityValue);
511                             break;
512 
513                         case ColumnType.LONG:
514                             offset = buffer.addLong(offset, entityValue);
515                             break;
516 
517                         case ColumnType.FLOAT:
518                             offset = buffer.addFloat(offset, entityValue);
519                             break;
520 
521                         case ColumnType.DOUBLE:
522                             offset = buffer.addDouble(offset, entityValue);
523                             break;
524 
525                         case ColumnType.SYMBOL:
526                             offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
527                             break;
528 
529                         default:
530                             throw castError("boolean", columnWriterIndex, colType, entity.getName());
531                     }
532                     break;
533                 }
534                 case LineTcpParser.ENTITY_TYPE_TIMESTAMP: {
535                     switch (ColumnType.tagOf(colType)) {
536                         case ColumnType.TIMESTAMP:
537                             offset = buffer.addTimestamp(offset, entity.getLongValue());
538                             break;
539 
540                         case ColumnType.DATE:
541                             offset = buffer.addDate(offset, entity.getLongValue() / 1000);
542                             break;
543 
544                         case ColumnType.SYMBOL:
545                             offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
546                             break;
547 
548                         default:
549                             throw castError("timestamp", columnWriterIndex, colType, entity.getName());
550                     }
551                     break;
552                 }
553                 case LineTcpParser.ENTITY_TYPE_SYMBOL: {
554                     if (ColumnType.tagOf(colType) == ColumnType.SYMBOL) {
555                         offset = buffer.addSymbol(offset, entity.getValue(), parser.hasNonAsciiChars(), localDetails.getSymbolLookup(columnWriterIndex));
556                     } else {
557                         throw castError("symbol", columnWriterIndex, colType, entity.getName());
558                     }
559                     break;
560                 }
561                 case ENTITY_TYPE_NULL:
562                     offset = buffer.addNull(offset);
563                     break;
564                 default:
565                     // unsupported types are ignored
566                     break;
567             }
568         }
569         buffer.addDesignatedTimestamp(buffer.getAddress() + Long.BYTES, timestamp);
570         buffer.addNumOfColumns(buffer.getAddress() + 2 * Long.BYTES, entitiesWritten);
571         writerWorkerId = tud.getWriterThreadId();
572     }
