178                 }

183             setWriterInError();

275     long commitIfIntervalElapsed(long wallClockMillis) throws CommitFailedException {
276         if (wallClockMillis < nextCommitTime) {
277             return nextCommitTime;
278         }
279         if (writerAPI != null) {
280             final long commitInterval = getCommitInterval();
281             long start = millisecondClock.getTicks();
282             commit(wallClockMillis - lastMeasurementMillis < commitInterval);
283             // Do not commit row by row if the commit takes longer than commitInterval.
284             // Exclude time to commit from the commit interval.
285             nextCommitTime += commitInterval + millisecondClock.getTicks() - start;
286         }

290     void commitIfMaxUncommittedRowsCountReached() throws CommitFailedException {
291         final long rowsSinceCommit = writerAPI.getUncommittedRowCount();
292         if (rowsSinceCommit < getMetaMaxUncommittedRows()) {
293             if ((rowsSinceCommit & writerTickRowsCountMod) == 0) {
294                 // Tick without commit. Some tick commands may force writer to commit though.
295                 tick();
296             }
297             return;
298         }
299         LOG.debug().$("max-uncommitted-rows commit with lag [").$(tableToken).I$();
300         nextCommitTime = millisecondClock.getTicks() + getCommitInterval();
301 
302         try {
303             commit(true);
304         } catch (Throwable th) {
305             LOG.error()
306                     .$("could not commit line protocol measurement [tableName=").$(writerAPI.getTableToken())
307                     .$(", message=").$(th.getMessage())
308                     .$(th)
309                     .I$();
310             writerAPI.rollback();
311             throw CommitFailedException.instance(th);
312         }
313

401         private DirectByteSymbolLookup addSymbolCache(int colWriterIndex) {
402             try (TableReader reader = engine.getReader(AllowAllCairoSecurityContext.INSTANCE, tableToken)) {
403                 final int symIndex = resolveSymbolIndexAndName(reader.getMetadata(), colWriterIndex);
404                 if (symbolNameTemp == null || symIndex < 0) {
405                     // looks like the column has just been added to the table, and
406                     // the reader is a bit behind and cannot see the column, yet
407                     // we will pass the symbol as string
408                     return NOT_FOUND_LOOKUP;
409                 }
410                 final CairoConfiguration cairoConfiguration = engine.getConfiguration();
411                 path.of(cairoConfiguration.getRoot()).concat(tableToken);
412                 SymbolCache symCache;
413                 final int lastUnusedSymbolCacheIndex = unusedSymbolCaches.size() - 1;
414                 if (lastUnusedSymbolCacheIndex > -1) {
415                     symCache = unusedSymbolCaches.get(lastUnusedSymbolCacheIndex);
416                     unusedSymbolCaches.remove(lastUnusedSymbolCacheIndex);
417                 } else {
418                     symCache = new SymbolCache(configuration);
419                 }
420 
421                 if (this.clean) {
422                     if (this.txReader == null) {
423                         this.txReader = new TxReader(cairoConfiguration.getFilesFacade());
424                     }
425                     int pathLen = path.length();
426                     this.txReader.ofRO(path.concat(TXN_FILE_NAME).$(), reader.getPartitionedBy());
427                     path.trimTo(pathLen);
428                     this.clean = false;
429                 }
430 
431                 long columnNameTxn = reader.getColumnVersionReader().getDefaultColumnNameTxn(colWriterIndex);
432                 assert symIndex <= colWriterIndex;
433                 symCache.of(
434                         cairoConfiguration,
435                         writerAPI,
436                         colWriterIndex,
437                         path,
438                         symbolNameTemp,
439                         symIndex,
440                         txReader,
441                         columnNameTxn
442                 );
443                 symbolCacheByColumnIndex.extendAndSet(colWriterIndex, symCache);
444                 return symCache;
445             }

491         private int resolveSymbolIndexAndName(TableRecordMetadata metadata, int colWriterIndex) {
492             symbolNameTemp = null;
493             int symIndex = -1;
494             for (int i = 0, n = metadata.getColumnCount(); i < n; i++) {
495                 if (metadata.getWriterIndex(i) == colWriterIndex) {
496                     if (!ColumnType.isSymbol(metadata.getColumnType(i))) {
497                         return -1;
498                     }
499                     symIndex++;
500                     symbolNameTemp = metadata.getColumnName(i);
501                     break;
502                 }
503                 if (ColumnType.isSymbol(metadata.getColumnType(i))) {
504                     symIndex++;
505                 }
506             }
507             return symIndex;
508         }
