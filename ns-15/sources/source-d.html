135     public ParseResult parseMeasurement(long bufHi) {
136         assert bufAt != 0 && bufHi >= bufAt;
137         // We can resume from random place of the line message
138         // the class member variables should resume byte by byte parsing from the last place
139         // processing stopped.
140         if (nQuoteCharacters == 1 && tagsComplete && entityHandler == ENTITY_HANDLER_VALUE) {
141             // when nQuoteCharacters it means that parsing of quoted value has started.
142             // continue parsing quoted value
143             if (!prepareQuotedEntity(entityLo, bufHi)) {
144                 // quoted value parsing did not reach the end
145                 if (errorCode == ErrorCode.INVALID_FIELD_VALUE_STR_UNDERFLOW) {
146                     // because buffer exhausted
147                     return ParseResult.BUFFER_UNDERFLOW;
148                 }
149                 // because it reached EOL or another error
150                 return ParseResult.ERROR;
151             }
152             nQuoteCharacters = 0;
153             bufAt++;
154         }
155 
156         // Main parsing loop
157         while (bufAt < bufHi) {
158             // take the byte
159             byte b = Unsafe.getUnsafe().getByte(bufAt);
160 
161             if (nEscapedChars == 0 && b >= 0 && !controlChars[b]) {
162                 // hot path
163                 nextValueCanBeOpenQuote = false;
164                 bufAt++;
165                 continue;
166             }
167 
168             // slow path
169             hasNonAscii |= b < 0;
170             boolean endOfLine = false;
171             boolean appendByte = false;
172             // Important note: don't forget to update controlChars array when changing the following switch.
173             switch (b) {
174                 case '\n':
175                 case '\r':
176                     endOfLine = true;
177                     b = '\n';
178                 case '=':
179                 case ',':
180                 case ' ':
181                     isQuotedFieldValue = false;
182                     if (!completeEntity(b, bufHi)) {
183                         // parse of key or value is unsuccessful
184                         if (errorCode == ErrorCode.EMPTY_LINE) {
185                             // An empty line
186                             bufAt++;
187                             entityLo = bufAt;
188                             break;
189                         }
190                         if (errorCode == ErrorCode.INVALID_FIELD_VALUE_STR_UNDERFLOW) {
191                             return ParseResult.BUFFER_UNDERFLOW;
192                         }
193                         return ParseResult.ERROR;
194                     }
195                     if (endOfLine) {
196                         // EOL reached, time to return
197                         if (nEntities > 0) {
198                             entityHandler = ENTITY_HANDLER_NEW_LINE;
199                             return ParseResult.MEASUREMENT_COMPLETE;
200                         }
201                         errorCode = ErrorCode.NO_FIELDS;
202                         return ParseResult.ERROR;
203                     }
204                     // skip the separator
205                     bufAt++;
206                     if (!isQuotedFieldValue) {
207                         // reset few indicators
208                         nEscapedChars = 0;
209                         // start next value from here
210                         entityLo = bufAt;
211                     }
212                     break;
213 
214                 case '\\':
215                     // escape next character
216                     // look forward, skip the slash
217                     if ((bufAt + 1) >= bufHi) {
218                         return ParseResult.BUFFER_UNDERFLOW;
219                     }
220                     nEscapedChars++;
221                     bufAt++;
222                     b = Unsafe.getUnsafe().getByte(bufAt);
223                     if (b == '\\' && (entityHandler != ENTITY_HANDLER_VALUE)) {
224                         return getError();
225                     }
226                     hasNonAscii |= b < 0;
227                     appendByte = true;
228                     break;
229 
230                 case '"':
231                     if (nextValueCanBeOpenQuote && ++nQuoteCharacters == 1) {
232                         // This means that the processing resumed from "
233                         // and it's allowed to start quoted value at this point
234                         bufAt += 1;
235                         // parse quoted value
236                         if (!prepareQuotedEntity(bufAt - 1, bufHi)) {
237                             // parsing not successful
238                             if (errorCode == ErrorCode.INVALID_FIELD_VALUE_STR_UNDERFLOW) {
239                                 // need more data
240                                 return ParseResult.BUFFER_UNDERFLOW;
241                             }
242                             // invalid character sequence in the quoted value or EOL found
243                             return ParseResult.ERROR;
244                         }
245                         errorCode = ErrorCode.NONE;
246                         nQuoteCharacters = 0;
247                         bufAt += 1;
248                         break;
249                     } else if (isQuotedFieldValue) {
250                         return getError();
251                     } else if (entityLo == bufAt) {
252                         tagStartsWithQuote = true;
253                     }
254 
255                 default:
256                     appendByte = true;
257                     nextValueCanBeOpenQuote = false;
258                     break;
259 
260                 case '\0':
261                     LOG.info().$("could not parse [byte=\\0]").$();
262                     return getError();
263                 case '/':
264                     if (entityHandler != ENTITY_HANDLER_VALUE) {
265                         LOG.info().$("could not parse [byte=/]").$();
266                         return getError();
267                     }
268                     appendByte = true;
269                     nextValueCanBeOpenQuote = false;
270                     break;
271             }
272 
273             if (appendByte) {
274                 // If there is escaped character, like \" or \\ then the escape slash has to be excluded
275                 // from the result key / value.
276                 // shift copy current byte back
277                 if (nEscapedChars > 0) {
278                     Unsafe.getUnsafe().putByte(bufAt - nEscapedChars, b);
279                 }
280                 bufAt++;
281             }
282         }
283         return ParseResult.BUFFER_UNDERFLOW;
284     }

347     private boolean expectEntityName(byte endOfEntityByte, long bufHi) {
348         if (endOfEntityByte == (byte) '=') {
349             if (bufAt - entityLo - nEscapedChars == 0) { // no tag/field name
350                 errorCode = tagsComplete ? ErrorCode.INCOMPLETE_FIELD : ErrorCode.INCOMPLETE_TAG;
351                 return false;
352             }
353 
354             if (entityCache.size() <= nEntities) {
355                 currentEntity = new ProtoEntity();
356                 entityCache.add(currentEntity);
357             } else {
358                 currentEntity = entityCache.get(nEntities);
359                 currentEntity.clear();
360             }
361 
362             nEntities++;
363             currentEntity.setName();
364             entityHandler = ENTITY_HANDLER_VALUE;
365             if (tagsComplete) {
366                 if (bufAt + 3 < bufHi) { // peek oncoming value's 1st byte, only caring for valid strings (2 quotes plus a follow-up byte)
367                     long candidateQuoteIdx = bufAt + 1;
368                     byte b = Unsafe.getUnsafe().getByte(candidateQuoteIdx);
369                     if (b == (byte) '"') {
370                         nEscapedChars = 0;
371                         nQuoteCharacters++;
372                         bufAt += 2;
373                         return prepareQuotedEntity(candidateQuoteIdx, bufHi);// go to first byte of the string, past the '"'
374                     } else {
375                         nextValueCanBeOpenQuote = false;
376                     }
377                 } else {
378                     nextValueCanBeOpenQuote = true;
379                 }
380             }
381             return true;
382         }
383 
384         boolean emptyEntity = bufAt == entityLo;
385         if (emptyEntity) {
386             if (endOfEntityByte == (byte) ' ') {
387                 if (tagsComplete) {
388                     entityHandler = ENTITY_HANDLER_TIMESTAMP;
389                 } else {
390                     tagsComplete = true;
391                 }
392                 return true;
393             }
394 
395             if (endOfEntityByte == (byte) '\n') {
396                 return true;
397             }
398         } else if (tagsComplete && (endOfEntityByte == '\n' || endOfEntityByte == '\r')) {
399             if (currentEntity != null && currentEntity.getType() == ENTITY_TYPE_TAG) {
400                 // One token after last tag, and no fields
401                 // This must be the timestamp
402                 return expectTimestamp(endOfEntityByte);
403             }
404         }
405 
406         if (tagsComplete) {
407             errorCode = ErrorCode.INCOMPLETE_FIELD;
408         } else {
409             errorCode = ErrorCode.INCOMPLETE_TAG;
410         }
411         return false;
412     }

488     private boolean prepareQuotedEntity(long openQuoteIdx, long bufHi) {
489         // the byte at openQuoteIdx (bufAt + 1) is '"', from here it can only be
490         // the start of a string value. Get it ready for immediate consumption by
491         // the next completeEntity call, moving butAt to the next '"'
492         entityLo = openQuoteIdx; // from the quote
493         boolean copyByte;
494         while (bufAt < bufHi) { // consume until the next quote, '\n', or eof
495             byte b = Unsafe.getUnsafe().getByte(bufAt);
496             copyByte = true;
497             hasNonAscii |= b < 0;
498             switch (b) {
499                 case (byte) '\\':
500                     if (!scape) {
501                         nEscapedChars++;
502                         copyByte = false;
503                     }
504                     scape = !scape;
505                     break;
506                 case (byte) '"':
507                     if (!scape) {
508                         isQuotedFieldValue = true;
509                         nQuoteCharacters--;
510                         if (nEscapedChars > 0) {
511                             Unsafe.getUnsafe().putByte(bufAt - nEscapedChars, b);
512                         }
513                         return true;
514                     }
515                     scape = false;
516                     break;
517                 case (byte) '\n':
518                     if (!scape) {
519                         errorCode = ErrorCode.INVALID_FIELD_VALUE;
520                         return false; // missing tail quote
521                     }
522                     scape = false;
523                     break;
524                 default:
525                     scape = false;
526                     break;
527             }
528             nextValueCanBeOpenQuote = false;
529             if (copyByte && nEscapedChars > 0) {
530                 Unsafe.getUnsafe().putByte(bufAt - nEscapedChars, b);
531             }
532             bufAt++;
533         }
534         errorCode = ErrorCode.INVALID_FIELD_VALUE_STR_UNDERFLOW;
535         return false; // missing tail quote as the string extends past the max allowed size
536     }

598         private boolean parse(byte last, int valueLen) {
599             switch (last) {
600                 case 'i':
601                     if (valueLen > 1 && value.charAt(1) != 'x') {
602                         return parseLong(ENTITY_TYPE_INTEGER);
603                     }
604                     if (valueLen > 3 && value.charAt(0) == '0' && (value.charAt(1) | 32) == 'x') {
605                         value.decHi(); // remove 'i'
606                         type = ENTITY_TYPE_LONG256;
607                         return true;
608                     }
609                     type = ENTITY_TYPE_SYMBOL;
610                     return true;
611                 case 't':
612                     if (valueLen > 1) {
613                         return parseLong(ENTITY_TYPE_TIMESTAMP);
614                     }
615                     // fall through
616                 case 'T':
617                 case 'f':
618                 case 'F':
619                 case 'e':
620                 case 'E':
621                     // t
622                     // T
623                     // f
624                     // F
625                     // tru(e)
626                     // fals(e)
627                     if (valueLen == 1) {
628                         if (last != 'e') {
629                             booleanValue = (last | 32) == 't';
630                             type = ENTITY_TYPE_BOOLEAN;
631                         } else {
632                             type = ENTITY_TYPE_SYMBOL;
633                         }
634                     } else {
635                         charSeq.of(value.getLo(), value.getHi());
636                         if (SqlKeywords.isTrueKeyword(charSeq)) {
637                             booleanValue = true;
638                             type = ENTITY_TYPE_BOOLEAN;
639                         } else if (SqlKeywords.isFalseKeyword(charSeq)) {
640                             booleanValue = false;
641                             type = ENTITY_TYPE_BOOLEAN;
642                         } else {
643                             type = ENTITY_TYPE_SYMBOL;
644                         }
645                     }
646                     return true;
647                 case '"': {
648                     byte b = value.byteAt(0);
649                     if (valueLen > 1 && b == '"') {
650                         value.squeeze();
651                         type = ENTITY_TYPE_STRING;
652                         return true;
653                     }
654                     type = ENTITY_TYPE_SYMBOL;
655                     return true;
656                 }
