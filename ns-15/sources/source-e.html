49     public LineTcpReceiver(
50             LineTcpReceiverConfiguration configuration,
51             CairoEngine engine,
52             WorkerPool ioWorkerPool,
53             WorkerPool writerWorkerPool
54     ) {
55         this.scheduler = null;
56         this.metrics = engine.getMetrics();
57         ObjectFactory<LineTcpConnectionContext> factory;
58         if (null == configuration.getAuthDbPath()) {
59             LOG.info().$("using default context").$();
60             factory = () -> new LineTcpConnectionContext(configuration, scheduler, metrics);
61         } else {
62             LOG.info().$("using authenticating context").$();
63             AuthDb authDb = new AuthDb(configuration);
64             factory = () -> new LineTcpAuthConnectionContext(configuration, authDb, scheduler, metrics);
65         }
66 
67         MutableIOContextFactory<LineTcpConnectionContext> contextFactory = new MutableIOContextFactory<>(
68                 factory,
69                 configuration.getConnectionPoolInitialCapacity()
70         );
71         this.dispatcher = IODispatchers.create(
72                 configuration.getDispatcherConfiguration(),
73                 contextFactory
74         );
75         ioWorkerPool.assign(dispatcher);
76         this.scheduler = new LineTcpMeasurementScheduler(configuration, engine, ioWorkerPool, dispatcher, writerWorkerPool);
77 
78         for (int i = 0, n = ioWorkerPool.getWorkerCount(); i < n; i++) {
79             // http context factory has thread local pools
80             // therefore we need each thread to clean their thread locals individually
81             ioWorkerPool.assignThreadLocalCleaner(i, contextFactory::freeThreadLocal);
82         }
83     }
