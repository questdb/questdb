49     AuthDb(LineTcpReceiverConfiguration configuration) {
50         int nLine = 0;
51         String[] tokens = new String[4];
52         try (BufferedReader r = new BufferedReader(new InputStreamReader(new BufferedInputStream(new FileInputStream(configuration.getAuthDbPath()))))) {
53             String line;
54             do {
55                 int nTokens = 0;
56                 line = r.readLine();
57                 nLine++;
58                 while (null != line) {
59                     Matcher m = TOKEN_PATTERN.matcher(line);
60                     if (!m.matches()) {
61                         break;
62                     }
63                     String token = m.group(1);
64                     if (token.startsWith("#")) {
65                         break;
66                     }
67                     if (nTokens == tokens.length) {
68                         throw new IllegalArgumentException("Too many tokens");
69                     }
70                     tokens[nTokens] = m.group(1);
71                     nTokens++;
72                     line = m.group(2);
73                 }
74 
75                 if (nTokens == 0) {
76                     continue;
77                 }
78 
79                 if (nTokens != 4) {
80                     throw new IllegalArgumentException("Was expecting 4 tokens");
81                 }
82 
83                 if (!"ec-p-256-sha256".equals(tokens[1])) {
84                     throw new IllegalArgumentException("Unrecognized type " + tokens[1]);

88                 if (!publicKeyByKeyId.excludes(keyId)) {
89                     throw new IllegalArgumentException("Duplicate keyId " + keyId);
90                 }

95         } catch (Exception ex) {
96             throw new IllegalArgumentException("IO error, failed to read auth db file " + configuration.getAuthDbPath() + " at line " + nLine, ex);

103         try {
