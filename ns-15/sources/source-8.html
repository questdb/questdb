192     public long addSymbol(long address, DirectByteCharSequence value, boolean hasNonAsciiChars, DirectByteSymbolLookup symbolLookup) {
193         final int maxLen = 2 * value.length();
194         checkCapacity(address, Byte.BYTES + Integer.BYTES + maxLen);
195         final long strPos = address + Byte.BYTES + Integer.BYTES; // skip field type and string length
196 
197         // via temp string the utf8 decoder will be writing directly to our buffer
198         tempSink.of(strPos, strPos + maxLen);
199 
200         final int symIndex = symbolLookup.keyOf(value);
201         if (symIndex != SymbolTable.VALUE_NOT_FOUND) {
202             // We know the symbol int value
203             // Encode the int
204             Unsafe.getUnsafe().putByte(address, LineTcpParser.ENTITY_TYPE_CACHED_TAG);
205             Unsafe.getUnsafe().putInt(address + Byte.BYTES, symIndex);
206             return address + Integer.BYTES + Byte.BYTES;
207         } else {
208             // Symbol value cannot be resolved at this point
209             // Encode whole string value into the message
210             if (!hasNonAsciiChars) {
211                 tempSink.put(value);
212             } else {
213                 utf8ToUtf16(value, tempSink, true);
214             }
215             final int length = tempSink.length();
216             Unsafe.getUnsafe().putByte(address, LineTcpParser.ENTITY_TYPE_TAG);
217             Unsafe.getUnsafe().putInt(address + Byte.BYTES, length);
218             return address + length * 2L + Integer.BYTES + Byte.BYTES;
219         }
220     }

229     /**
230      * Add UUID encoded as a string to the buffer.
231      * <p>
232      * Technically, DirectByteCharSequence is UTF-8 encoded, but any non-ASCII character will cause the UUID
233      * to be rejected by the parser. Hence, we do not have to bother with UTF-8 decoding.
234      *
235      * @param offset offset in the buffer to write to
236      * @param value  value to write
237      * @return new offset
238      * @throws NumericException if the value is not a valid UUID string
239      */
240     public long addUuid(long offset, DirectByteCharSequence value) throws NumericException {
241         checkCapacity(offset, Byte.BYTES + 2 * Long.BYTES);
242         Uuid.checkDashesAndLength(value);
243         long hi = Uuid.parseHi(value);
244         long lo = Uuid.parseLo(value);
245         Unsafe.getUnsafe().putByte(offset, LineTcpParser.ENTITY_TYPE_UUID);
246         offset += Byte.BYTES;
247         Unsafe.getUnsafe().putLong(offset, lo);
248         offset += Long.BYTES;
249         Unsafe.getUnsafe().putLong(offset, hi);
250         return offset + Long.BYTES;
251     }

271             case LineTcpParser.ENTITY_TYPE_INTEGER:

338     private long addString(long address, DirectByteCharSequence value, boolean hasNonAsciiChars, byte entityTypeString) {
339         int maxLen = 2 * value.length();
340         checkCapacity(address, Byte.BYTES + Integer.BYTES + maxLen);
341         long strPos = address + Byte.BYTES + Integer.BYTES; // skip field type and string length
342         tempSink.of(strPos, strPos + maxLen);
343         if (hasNonAsciiChars) {
344             utf8ToUtf16Unchecked(value, tempSink);
345         } else {
346             tempSink.put(value);
347         }
348         final int length = tempSink.length();
349         Unsafe.getUnsafe().putByte(address, entityTypeString);
350         Unsafe.getUnsafe().putInt(address + Byte.BYTES, length);
351         return address + length * 2L + Integer.BYTES + Byte.BYTES;
352     }
