119     public void close0(boolean closeDelegate) {
120         int prevState = state;
121         if (prevState == CLOSED) {
122             return;
123         }
124         state = CLOSING;
125         if (prevState == AFTER_HANDSHAKE) {
126             try {
127                 sslEngine.closeOutbound();
128                 wrapLoop(dummyBuffer);
129                 writeToUpstreamAndClear();
130             } catch (Throwable e) {
131                 LOG.error().$("could not send TLS close_notify alert").$(e).$();
132             }
133         }
134         state = CLOSED;
135 
136         if (closeDelegate) {
137             delegate = Misc.free(delegate);
138         }
139 
140         // a bit of ceremony to make sure there is no point that a buffer or a pointer is referencing unallocated memory
141         int capacity = wrapOutputBuffer.capacity();
142         long ptrToFree = wrapOutputBufferPtr;
143         wrapOutputBuffer = null; // if there is an attempt to use a buffer after close() then it's better to throw NPE than segfaulting
144         wrapOutputBufferPtr = 0;
145         Unsafe.free(ptrToFree, capacity, MemoryTag.NATIVE_TLS_RSS);
146 
147         capacity = unwrapInputBuffer.capacity();
148         ptrToFree = unwrapInputBufferPtr;
149         unwrapInputBuffer = null;
150         unwrapInputBufferPtr = 0;
151         Unsafe.free(ptrToFree, capacity, MemoryTag.NATIVE_TLS_RSS);
152 
153         capacity = unwrapOutputBuffer.capacity();
154         ptrToFree = unwrapOutputBufferPtr;
155         unwrapOutputBuffer = null;
156         unwrapOutputBufferPtr = 0;
157         Unsafe.free(ptrToFree, capacity, MemoryTag.NATIVE_TLS_RSS);
158     }

196     private static SSLEngine createSslEngine(String trustStorePath, char[] trustStorePassword, Sender.TlsValidationMode validationMode, String peerHost) {
197         assert trustStorePath == null || validationMode == Sender.TlsValidationMode.DEFAULT;
198         try {
199             SSLContext sslContext;
200             if (trustStorePath != null) {
201                 sslContext = SSLContext.getInstance("TLS");
202                 TrustManagerFactory tmf = TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
203                 KeyStore jks = KeyStore.getInstance("JKS");
204                 try (InputStream trustStoreStream = openTruststoreStream(trustStorePath)) {
205                     jks.load(trustStoreStream, trustStorePassword);
206                 }
207                 tmf.init(jks);
208                 TrustManager[] trustManagers = tmf.getTrustManagers();
209                 sslContext.init(null, trustManagers, new SecureRandom());
210             } else if (validationMode == Sender.TlsValidationMode.INSECURE) {
211                 sslContext = SSLContext.getInstance("TLS");
212                 sslContext.init(null, BLIND_TRUST_MANAGERS, new SecureRandom());
213             } else {
214                 sslContext = SSLContext.getDefault();
215             }
216 
217             // SSLEngine needs to know hostname during TLS handshake to validate a server certificate was issued
218             // for the server we are connecting to. For details see the comment below.
219             // Hostname validation does not use port at all hence we can get away with a dummy value -1
220             SSLEngine sslEngine = sslContext.createSSLEngine(peerHost, -1);
221             if (validationMode != Sender.TlsValidationMode.INSECURE) {
222                 SSLParameters sslParameters = sslEngine.getSSLParameters();
223                 // The https validation algorithm? That looks confusing! After all we are not using any
224                 // https here at so what does it mean?
225                 // It's actually simple: It just instructs the SSLEngine to perform the same hostname validation
226                 // as it does during HTTPS connections. SSLEngine does not do hostname validation by default. Without
227                 // this option SSLEngine would happily accept any certificate as long as it's signed by a trusted CA.
228                 // This option will make sure certificates are accepted only if they were issued for the
229                 // server we are connecting to.
230                 sslParameters.setEndpointIdentificationAlgorithm("https");
231                 sslEngine.setSSLParameters(sslParameters);
232             }
233             sslEngine.setUseClientMode(true);
234             return sslEngine;
235         } catch (Throwable t) {
236             if (t instanceof LineSenderException) {
237                 throw (LineSenderException) t;
238             }
239             throw new LineSenderException("could not create SSL engine", t);
240         }
241     }

284     private void handshakeLoop() throws SSLException {
285         if (state != INITIAL_STATE) {
286             return;
287         }
288 
289         // trigger handshaking - otherwise the initial state is NOT_HANDSHAKING
290         sslEngine.beginHandshake();
291         for (; ; ) {
292             SSLEngineResult.HandshakeStatus status = sslEngine.getHandshakeStatus();
293             switch (status) {
294                 case NOT_HANDSHAKING:
295                     state = AFTER_HANDSHAKE;
296                     return;
297                 case NEED_TASK:
298                     sslEngine.getDelegatedTask().run();
299                     break;
300                 case NEED_WRAP:
301                     wrapLoop(dummyBuffer);
302                     break;
303                 case NEED_UNWRAP:
304                     unwrapLoop();
305                     break;
306                 case FINISHED:
307                     throw new LineSenderException("getHandshakeStatus() returned FINISHED. It should not have been possible.");
308                 default:
309                     throw new LineSenderException(status + "not supported");
310             }
311         }
312     }

314     private void readFromUpstream(boolean force) {
315         if (unwrapInputBuffer.position() != 0 && !force) {
316             // we don't want to block on receive() if there are still data to be processed
317             // unless we are forced to do so
318             return;
319         }
320 
321         assert unwrapInputBuffer.limit() == unwrapInputBuffer.capacity();
322         int remainingLen = unwrapInputBuffer.remaining();
323         if (remainingLen == 0) {
324             growUnwrapInputBuffer();
325             remainingLen = unwrapInputBuffer.remaining();
326         }
327         assert Unsafe.getUnsafe().getLong(unwrapInputBuffer, ADDRESS_FIELD_OFFSET) == unwrapInputBufferPtr;
328         long adjustedPtr = unwrapInputBufferPtr + unwrapInputBuffer.position();
329 
330         int receive = delegate.receive(adjustedPtr, remainingLen);
331         if (receive < 0) {
332             throw new LineSenderException("connection closed");
333         }
334         unwrapInputBuffer.position(unwrapInputBuffer.position() + receive);
335     }

337     private void unwrapLoop() throws SSLException {
338         // we want the loop to return as soon as we have some unwrapped data in the output buffer
339         while (unwrapOutputBuffer.position() == 0) {
340             readFromUpstream(false);
341             unwrapInputBuffer.flip();
342             SSLEngineResult result = sslEngine.unwrap(unwrapInputBuffer, unwrapOutputBuffer);
343             unwrapInputBuffer.compact();
344             switch (result.getStatus()) {
345                 case BUFFER_UNDERFLOW:
346                     // we need more input no matter what. so let's force reading from the upstream channel
347                     readFromUpstream(true);
348                     break;
349                 case BUFFER_OVERFLOW:
350                     if (unwrapOutputBuffer.position() != 0) {
351                         // we have at least something, that's enough
352                         // if it's not enough then it's up to the caller to call us again
353                         return;
354                     }
355 
356                     // there was overflow, and we have nothing
357                     // apparently the output buffer cannot fit even a single TLS record. let's grow it!
358                     growUnwrapOutputBuffer();
359                     break;
360                 case OK:
361                     return;
362                 case CLOSED:
363                     throw new LineSenderException("server closed connection unexpectedly");
364             }
365         }
366     }

368     private int unwrapOutputBufferToPtr(long dstPtr, int dstLen) {
369         int oldPosition = unwrapOutputBuffer.position();
370 
371         assert Unsafe.getUnsafe().getLong(unwrapOutputBufferPtr, ADDRESS_FIELD_OFFSET) == unwrapOutputBufferPtr;
372         long srcPtr = unwrapOutputBufferPtr + oldPosition;
373         int srcLen = unwrapOutputBuffer.remaining();
374         int len = Math.min(dstLen, srcLen);
375         Vect.memcpy(dstPtr, srcPtr, len);
376         unwrapOutputBuffer.position(oldPosition + len);
377         return len;
378     }

380     private void wrapLoop(ByteBuffer src) throws SSLException {
381         do {
382             SSLEngineResult result = sslEngine.wrap(src, wrapOutputBuffer);
383             switch (result.getStatus()) {
384                 case BUFFER_UNDERFLOW:
385                     throw new LineSenderException("should not happen");
386                 case BUFFER_OVERFLOW:
387                     growWrapOutputBuffer();
388                     break;
389                 case OK:
390                     writeToUpstreamAndClear();
391                     break;
392                 case CLOSED:
393                     if (state != CLOSING) {
394                         throw new LineSenderException("server closed connection unexpectedly");
395                     }
396                     return;
397             }
398         } while (src.hasRemaining());
399     }

401     private void writeToUpstreamAndClear() {
402         assert wrapOutputBuffer.limit() == wrapOutputBuffer.capacity();
403 
404         // we don't flip the wrapOutputBuffer before reading from it
405         // hence the writer position is the actual length to be sent to the upstream channel
406         int len = wrapOutputBuffer.position();
407 
408         assert Unsafe.getUnsafe().getLong(wrapOutputBuffer, ADDRESS_FIELD_OFFSET) == wrapOutputBufferPtr;
409         delegate.send(wrapOutputBufferPtr, len);
410 
411         // we know limit == capacity
412         // thus setting the position to 0 is equivalent to clearing
413         wrapOutputBuffer.position(0);
414     }
