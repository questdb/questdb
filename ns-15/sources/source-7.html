92     private void doHandleDisconnectEvent() {
93         if (parser.getBufferAddress() == recvBufEnd) {

131     /**
132      * Moves incompletely received measurement to start of the receive buffer. Also updates the state of the
133      * context and protocol parser such that all pointers that point to the incomplete measurement will remain
134      * valid. This allows protocol parser to resume execution from the point of where measurement ended abruptly
135      *
136      * @param recvBufStartOfMeasurement the address in receive buffer where incomplete measurement starts. Everything from
137      *                                  this address to end of the receive buffer will be copied to the start of the
138      *                                  receive buffer
139      * @return true if there was an incomplete measurement in the first place
140      */
141     protected final boolean compactBuffer(long recvBufStartOfMeasurement) {
142         assert recvBufStartOfMeasurement <= recvBufPos;
143         if (recvBufStartOfMeasurement > recvBufStart) {
144             final long len = recvBufPos - recvBufStartOfMeasurement;
145             if (len > 0) {
146                 Vect.memmove(recvBufStart, recvBufStartOfMeasurement, len); // Use memmove, there may be an overlap
147                 final long shl = recvBufStartOfMeasurement - recvBufStart;
148                 parser.shl(shl);
149                 this.recvBufStartOfMeasurement -= shl;
150             } else {
151                 assert len == 0;
152                 resetParser();
153             }
154             recvBufPos = recvBufStart + len;
155             return true;
156         }
157         return false;
158     }

169     protected final IOContextResult parseMeasurements(NetworkIOJob netIoJob) {
170         while (true) {
171             try {
172                 ParseResult rc = goodMeasurement ? parser.parseMeasurement(recvBufPos) : parser.skipMeasurement(recvBufPos);
173                 switch (rc) {
174                     case MEASUREMENT_COMPLETE: {
175                         if (goodMeasurement) {
176                             if (scheduler.scheduleEvent(netIoJob, parser)) {
177                                 // Waiting for writer threads to drain queue, request callback as soon as possible
178                                 if (checkQueueFullLogHysteresis()) {
179                                     LOG.debug().$('[').$(fd).$("] queue full").$();
180                                 }
181                                 return IOContextResult.QUEUE_FULL;
182                             }
183                         } else {
184                             logParseError();
185                             goodMeasurement = true;
186                         }
187 
188                         startNewMeasurement();
189 
190                         continue;
191                     }
192 
193                     case ERROR: {
194                         if (disconnectOnError) {
195                             logParseError();
196                             return IOContextResult.NEEDS_DISCONNECT;
197                         }
198                         goodMeasurement = false;
199                         continue;
200                     }
201 
202                     case BUFFER_UNDERFLOW: {
203                         if (recvBufPos == recvBufEnd && !compactBuffer(recvBufStartOfMeasurement)) {
204                             doHandleDisconnectEvent();
205                             return IOContextResult.NEEDS_DISCONNECT;
206                         }
207 
208                         if (!read()) {
209                             if (peerDisconnected) {
210                                 return IOContextResult.NEEDS_DISCONNECT;
211                             }
212                             return IOContextResult.NEEDS_READ;
213                         }
214                         break;
215                     }
216                 }
217             } catch (CairoException ex) {
218                 LOG.error()
219                         .$('[').$(fd).$("] could not process line data [table=").$(parser.getMeasurementName())
220                         .$(", msg=").$(ex.getFlyweightMessage())
221                         .$(", errno=").$(ex.getErrno())
222                         .I$();
223                 if (disconnectOnError) {
224                     logParseError();
225                     return IOContextResult.NEEDS_DISCONNECT;
226                 }
227                 goodMeasurement = false;
228             } catch (Throwable ex) {
229                 LOG.critical()
230                         .$('[').$(fd).$("] could not process line data [table=").$(parser.getMeasurementName())
231                         .$(", ex=").$(ex)
232                         .I$();
233                 // This is a critical error, so we treat it as an unhandled one.
234                 metrics.health().incrementUnhandledErrors();
235                 return IOContextResult.NEEDS_DISCONNECT;
236             }
237         }
238     }

240     protected boolean read() {
241         int bufferRemaining = (int) (recvBufEnd - recvBufPos);
242         final int orig = bufferRemaining;
243         if (bufferRemaining > 0 && !peerDisconnected) {
244             int bytesRead = nf.recv(fd, recvBufPos, bufferRemaining);
245             if (bytesRead > 0) {
246                 recvBufPos += bytesRead;
247                 bufferRemaining -= bytesRead;
248             } else {
249                 peerDisconnected = bytesRead < 0;
250             }
251             return bufferRemaining < orig;
252         }
253         return !peerDisconnected;
254     }
