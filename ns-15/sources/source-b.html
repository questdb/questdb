198     public long commitWalTables(ByteCharSequenceObjHashMap<TableUpdateDetails> tableUpdateDetailsUtf8, long wallClockMillis) {
199         long minTableNextCommitTime = Long.MAX_VALUE;
200         for (int n = 0, sz = tableUpdateDetailsUtf8.size(); n < sz; n++) {
201             final ByteCharSequence tableNameUtf8 = tableUpdateDetailsUtf8.keys().get(n);
202             final TableUpdateDetails tud = tableUpdateDetailsUtf8.get(tableNameUtf8);
203 
204             if (tud.isWal()) {
205                 final MillisecondClock millisecondClock = tud.getMillisecondClock();
206                 try {
207                     long tableNextCommitTime = tud.commitIfIntervalElapsed(wallClockMillis);
208                     // get current time again, commit is not instant and take quite some time.
209                     wallClockMillis = millisecondClock.getTicks();
210                     if (tableNextCommitTime < minTableNextCommitTime) {
211                         // taking the earliest commit time
212                         minTableNextCommitTime = tableNextCommitTime;
213                     }
214                 } catch (Throwable ex) {
215                     LOG.critical().$("commit failed [table=").$(tud.getTableNameUtf16()).$(",ex=").$(ex).I$();
216                     engine.getMetrics().health().incrementUnhandledErrors();
217                 }
218             }
219         }
220         // if no tables, just use the default commit interval
221         long commitIntervalDefault = configuration.getCommitIntervalDefault();
222         return minTableNextCommitTime != Long.MAX_VALUE ? minTableNextCommitTime : wallClockMillis + commitIntervalDefault;
223     }

225     public boolean doMaintenance(
226             ByteCharSequenceObjHashMap<TableUpdateDetails> tableUpdateDetailsUtf8,
227             int readerWorkerId,
228             long millis
229     ) {
230         walIdleUpdateDetailsUtf8.closeIdle(millis, writerIdleTimeout, listener);
231 
232         for (int n = 0, sz = tableUpdateDetailsUtf8.size(); n < sz; n++) {
233             final ByteCharSequence tableNameUtf8 = tableUpdateDetailsUtf8.keys().get(n);
234             final TableUpdateDetails tud = tableUpdateDetailsUtf8.get(tableNameUtf8);
235 
236             if (millis - tud.getLastMeasurementMillis() >= writerIdleTimeout) {
237                 tableUpdateDetailsLock.writeLock().lock();
238                 try {
239                     if (tud.getNetworkIOOwnerCount() == 1) {
240                         final int writerWorkerId = tud.getWriterThreadId();
241                         final long seq = getNextPublisherEventSequence(writerWorkerId);
242                         if (seq > -1) {
243                             LineTcpMeasurementEvent event = queue[writerWorkerId].get(seq);
244                             event.createWriterReleaseEvent(tud, true);
245                             tableUpdateDetailsUtf8.remove(tableNameUtf8);
246                             final CharSequence tableNameUtf16 = tud.getTableNameUtf16();
247                             tableUpdateDetailsUtf16.remove(tableNameUtf16);
248                             idleTableUpdateDetailsUtf16.put(tableNameUtf16, tud);
249                             tud.removeReference(readerWorkerId);
250                             pubSeq[writerWorkerId].done(seq);
251                             if (listener != null) {
252                                 // table going idle
253                                 listener.onEvent(tud.getTableToken(), 1);
254                             }
255                             LOG.info().$("active table going idle [tableName=").$(tableNameUtf16).I$();
256                         }
257                         return true;
258                     } else {
259                         tableUpdateDetailsUtf8.remove(tableNameUtf8);
260                         tud.removeReference(readerWorkerId);
261 
262                         if (tud.isWal()) {
263                             if (listener != null) {
264                                 // table going idle
265                                 listener.onEvent(tud.getTableToken(), 1);
266                             }
267                             tud.releaseWriter(true);
268                             tud.close();
269                         }
270                     }
271                     return sz > 1;
272                 } finally {
273                     tableUpdateDetailsLock.writeLock().unlock();
274                 }
275             }
276         }
277         return false;
278     }

280     public void processWriterReleaseEvent(LineTcpMeasurementEvent event, int workerId) {
281         tableUpdateDetailsLock.readLock().lock();
282         try {
283             final TableUpdateDetails tub = event.getTableUpdateDetails();
284             if (tub.getWriterThreadId() != workerId) {
285                 return;
286             }
287             if (!event.getTableUpdateDetails().isWriterInError() && tableUpdateDetailsUtf16.keyIndex(tub.getTableNameUtf16()) < 0) {
288                 // Table must have been re-assigned to an IO thread
289                 return;
290             }
291             LOG.info()
292                     .$("releasing writer, its been idle since ").$ts(tub.getLastMeasurementMillis() * 1_000)
293                     .$("[tableName=").$(tub.getTableNameUtf16())
294                     .I$();
295 
296             event.releaseWriter();
297         } finally {
298             tableUpdateDetailsLock.readLock().unlock();
299         }
300     }

605                         }
606                         break;

610                         break;

617         } catch (Throwable th) {

634         final int writerThreadId = tud.getWriterThreadId();
635         long seq = getNextPublisherEventSequence(writerThreadId);
636         if (seq > -1) {
637             try {
638                 if (tud.isWriterInError()) {
639                     throw CairoException.critical(0).put("writer is in error, aborting ILP pipeline");
640                 }
641                 queue[writerThreadId].get(seq).createMeasurementEvent(tud, parser, netIoJob.getWorkerId());
642             } finally {

647         }
648         return true;

651     private TableUpdateDetails getTableUpdateDetailsFromSharedArea(@NotNull NetworkIOJob netIoJob, @NotNull LineTcpParser parser) {
652         final DirectByteCharSequence tableNameUtf8 = parser.getMeasurementName();
653         TableUpdateDetails walCachedTud = walIdleUpdateDetailsUtf8.get(tableNameUtf8);
654         if (walCachedTud != null) {
655             netIoJob.addTableUpdateDetails(walCachedTud.getTableNameUtf8(), walCachedTud);
656             return walCachedTud;
657         }
658 
659         final StringSink tableNameUtf16 = tableNameSinks[netIoJob.getWorkerId()];
660         tableNameUtf16.clear();
661         Chars.utf8Decode(tableNameUtf8.getLo(), tableNameUtf8.getHi(), tableNameUtf16);
662 
663         tableUpdateDetailsLock.writeLock().lock();
664         try {
665             TableUpdateDetails tud;
666             // check if the global cache has the table
667             final int tudKeyIndex = tableUpdateDetailsUtf16.keyIndex(tableNameUtf16);
668             if (tudKeyIndex < 0) {
669                 // it does, which means that table is non-WAL
670                 // we should not have "shared" WAL tables
671                 tud = tableUpdateDetailsUtf16.valueAt(tudKeyIndex);
672             } else {
673                 TableToken tableToken = engine.getTableTokenIfExists(tableNameUtf16);
674                 int status = engine.getStatus(securityContext, path, tableToken);
675                 if (status != TableUtils.TABLE_EXISTS) {
676                     if (!autoCreateNewTables) {
677                         throw CairoException.nonCritical()
678                                 .put("table does not exist, creating new tables is disabled [table=").put(tableNameUtf16)
679                                 .put(']');
680                     }
681                     if (!autoCreateNewColumns) {
682                         throw CairoException.nonCritical()
683                                 .put("table does not exist, cannot create table, creating new columns is disabled [table=").put(tableNameUtf16)
684                                 .put(']');
685                     }
686                     // validate that parser entities do not contain NULLs
687                     TableStructureAdapter tsa = tableStructureAdapter.of(tableNameUtf16, parser);
688                     for (int i = 0, n = tsa.getColumnCount(); i < n; i++) {
689                         if (tsa.getColumnType(i) == LineTcpParser.ENTITY_TYPE_NULL) {
690                             throw CairoException.nonCritical().put("unknown column type [columnName=").put(tsa.getColumnName(i)).put(']');
691                         }
692                     }
693                     LOG.info().$("creating table [tableName=").$(tableNameUtf16).$(']').$();
694                     tableToken = engine.createTable(securityContext, ddlMem, path, true, tsa, false);
695                 }
696 
697                 // by the time we get here, definitely exists on disk
698                 // check the global idle cache - TUD can be there
699                 final int idleTudKeyIndex = idleTableUpdateDetailsUtf16.keyIndex(tableNameUtf16);
700                 if (idleTudKeyIndex < 0) {
701                     // TUD is found in global idle cache - this meant it is non-WAL
702                     tud = idleTableUpdateDetailsUtf16.valueAt(idleTudKeyIndex);
703                     LOG.info().$("idle table going active [tableName=").$(tud.getTableNameUtf16()).I$();
704                     if (tud.getWriter() == null) {
705                         tud.closeNoLock();
706                         // Use actual table name from the "details" to avoid case mismatches in the
707                         // WriterPool. There was an error in the LineTcpReceiverFuzzTest, which helped
708                         // to identify the cause
709                         tud = unsafeAssignTableToWriterThread(tudKeyIndex, tud.getTableNameUtf16(), tud.getTableNameUtf8());
710                     } else {
711                         idleTableUpdateDetailsUtf16.removeAt(idleTudKeyIndex);
712                         tableUpdateDetailsUtf16.putAt(tudKeyIndex, tud.getTableNameUtf16(), tud);
713                     }
714                 } else {
715                     TelemetryTask.store(telemetry, TelemetryOrigin.ILP_TCP, TelemetrySystemEvent.ILP_RESERVE_WRITER);
716                     // check if table on disk is WAL
717                     path.of(engine.getConfiguration().getRoot());
718                     if (engine.isWalTable(tableToken)) {
719                         // create WAL-oriented TUD and NOT add it to the global cache
720                         tud = new TableUpdateDetails(
721                                 configuration,
722                                 engine,
723                                 engine.getWalWriter(
724                                         securityContext,
725                                         tableToken
726                                 ),
727                                 -1,
728                                 netIoJobs,
729                                 defaultColumnTypes,
730                                 ByteCharSequence.newInstance(tableNameUtf8)
731                         );
732                     } else {
733                         tud = unsafeAssignTableToWriterThread(tudKeyIndex, tableNameUtf16, ByteCharSequence.newInstance(tableNameUtf8));
734                     }
735                 }
736             }
737 
738             // tud.getTableNameUtf8() can be different case from incoming tableNameUtf8
739             ByteCharSequence key = Chars.equals(tud.getTableNameUtf8(), tableNameUtf8) ? tud.getTableNameUtf8() : ByteCharSequence.newInstance(tableNameUtf8);
740             netIoJob.addTableUpdateDetails(key, tud);
741             return tud;
742         } finally {
743             tableUpdateDetailsLock.writeLock().unlock();
744         }
745     }

752     private TableUpdateDetails unsafeAssignTableToWriterThread(int tudKeyIndex, CharSequence tableNameUtf16, ByteCharSequence tableNameUtf8) {
753         unsafeCalcThreadLoad();
754         long leastLoad = Long.MAX_VALUE;
755         int threadId = 0;
756 
757         for (int i = 0, n = loadByWriterThread.length; i < n; i++) {
758             if (loadByWriterThread[i] < leastLoad) {
759                 leastLoad = loadByWriterThread[i];

761             }
762         }
763         TableToken tableToken = engine.getTableToken(tableNameUtf16);
764         final TableUpdateDetails tud = new TableUpdateDetails(
765                 configuration,
766                 engine,
767                 // get writer here to avoid constructing
768                 // object instance and potentially leaking memory if
769                 // writer allocation fails

771                 threadId,
772                 netIoJobs,
773                 defaultColumnTypes,
774                 tableNameUtf8
775         );
776         tableUpdateDetailsUtf16.putAt(tudKeyIndex, tud.getTableNameUtf16(), tud);
777         LOG.info().$("assigned ").$(tableToken).$(" to thread ").$(threadId).$();
778         return tud;

808     boolean scheduleEvent(NetworkIOJob netIoJob, LineTcpParser parser) {
809         DirectByteCharSequence measurementName = parser.getMeasurementName();
810         TableUpdateDetails tud;
811         try {
812             tud = netIoJob.getLocalTableDetails(measurementName);
813             if (tud == null || (tud.isWal() && tud.isWriterInError())) {
814                 if (tud != null) {
815                     TableUpdateDetails removed = netIoJob.removeTableUpdateDetails(measurementName);
816                     assert tud == removed;
817                     removed.releaseWriter(true);
818                     if (listener != null) {
819                         listener.onEvent(tud.getTableToken(), 1);
820                     }
821                     removed.close();
822                 }
823                 tud = getTableUpdateDetailsFromSharedArea(netIoJob, parser);
824             }
825         } catch (EntryUnavailableException ex) {
826             // Table writer is locked
827             LOG.info().$("could not get table writer [tableName=").$(measurementName)
828                     .$(", ex=`")
829                     .$(ex.getFlyweightMessage())
830                     .$("`]")
831                     .$();
832             return true;
833         } catch (CairoException ex) {
834             // Table could not be created
835             LOG.error().$("could not create table [tableName=").$(measurementName)
836                     .$(", errno=").$(ex.getErrno())
837                     .$(", ex=`")
838                     .$(ex.getFlyweightMessage())
839                     .$("`]")
840                     .I$();
841             // More details will be logged by catching thread
842             throw ex;
843         }
844 
845         if (tud.isWal()) {
846             try {
847                 appendToWal(netIoJob, parser, tud);
848             } catch (Throwable ex) {
849                 tud.setWriterInError();
850                 LOG.critical().$("closing writer because of error [table=").$(tud.getTableNameUtf16())
851                         .$(",ex=")
852                         .$(ex)
853                         .I$();
854                 throw CairoException.critical(0).put("could not append to WAL [tableName=").put(measurementName).put(", error=").put(ex.getMessage()).put(']');
855             }
856             return false;
857         }
858         return dispatchEvent(netIoJob, parser, tud);
859     }
