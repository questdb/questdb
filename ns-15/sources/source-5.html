37 class IOTableUpdateDetailsPool implements Closeable {
38     private static final int ABASE;
39     private static final int ASHIFT;
40     private final Log LOG = LogFactory.getLog(IOTableUpdateDetailsPool.class);
41     private final Function<CharSequence, ? extends TableUpdateDetails[]> createNewTableUpdateDetails;
42     private final ConcurrentHashMap<TableUpdateDetails[]> pool = new ConcurrentHashMap<>();
43     private final int pooledSize;
44     private volatile boolean closed = false;

46     public IOTableUpdateDetailsPool(int maxSize) {

48         createNewTableUpdateDetails = k -> new TableUpdateDetails[pooledSize];

51     @Override
52     public void close() throws IOException {
53         closed = true;
54         for (CharSequence key : pool.keySet()) {
55             final TableUpdateDetails[] tudArray = pool.get(key);
56             for (int n = tudArray.length, i = 0; i < n; i++) {
57                 TableUpdateDetails tud = getVolatile(tudArray, i);
58                 if (tud != null) {
59                     releaseTud(tud, null);
60                 }
61             }
62         }
63     }

65     public void closeIdle(long nowMillis, long idleTimeout, LineTcpReceiver.SchedulerListener listener) {
66         for (CharSequence key : pool.keySet()) {
67             final TableUpdateDetails[] tudArray = pool.get(key);
68             if (tudArray != null) {
69                 for (int n = tudArray.length, i = 0; i < n; i++) {
70                     TableUpdateDetails tud = getVolatile(tudArray, i);
71                     if (tud != null && nowMillis - tud.getLastMeasurementMillis() >= idleTimeout) {
72                         if (casTabAt(tudArray, i, tud, null)) {
73                             LOG.info().$("active table going idle [tableName=").$(tud.getTableNameUtf16()).I$();
74                             releaseTud(tud, listener);
75                         }
76                     }
77                 }
78             }
79         }
80     }

82     public TableUpdateDetails get(CharSequence tableNameUtf8) {
83         TableUpdateDetails[] tudArray = pool.get(tableNameUtf8);
84         if (tudArray != null) {
85             for (int n = tudArray.length, i = 0; i < n; i++) {
86                 TableUpdateDetails tud = getVolatile(tudArray, i);
87                 if (tud != null && casTabAt(tudArray, i, tud, null)) {
88                     return tud;
89                 }
90             }
91         }
92         return null;
93     }

95     public void put(CharSequence tableNameUtf8, TableUpdateDetails tud) {
96         TableUpdateDetails[] tudArray = pool.computeIfAbsent(tableNameUtf8, createNewTableUpdateDetails);
97         while (true) {
98             if (closed) {
99                 releaseTud(tud, null);
100                 return;
101             }
102             // There must be a place to return the tud, arrays are pre-allocate to hold enough for all IO threads
103             for (int n = tudArray.length, i = 0; i < n; i++) {
104                 if (casTabAt(tudArray, i, null, tud)) {
105                     return;
106                 }
107             }
108         }
109     }
110 
111     private void releaseTud(TableUpdateDetails tud, LineTcpReceiver.SchedulerListener listener) {
112         tud.releaseWriter(true);
113         Misc.free(tud);
114         if (listener != null) {
115             listener.onEvent(tud.getTableToken(), 1);
116         }
117     }

119     static boolean casTabAt(TableUpdateDetails[] tab, int i, TableUpdateDetails expected, TableUpdateDetails v) {
120         return Unsafe.getUnsafe().compareAndSwapObject(tab, ((long) i << ASHIFT) + ABASE, expected, v);
121     }

123     static TableUpdateDetails getVolatile(TableUpdateDetails[] tab, int i) {
124         return (TableUpdateDetails) Unsafe.getUnsafe().getObjectVolatile(tab, ((long) i << ASHIFT) + ABASE);
125     }
