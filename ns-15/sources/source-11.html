44 /**
45  * Important note:
46  * This cache is optimized for ASCII and UTF8 DirectByteCharSequence lookups. Lookups of UTF16
47  * strings (j.l.String) with non-ASCII chars will not work correctly, so make sure to re-encode
48  * the string in UTF8.
49  */
50 class SymbolCache implements Closeable, DirectByteSymbolLookup {
51     private final MicrosecondClock clock;
52     private final SymbolMapReaderImpl symbolMapReader = new SymbolMapReaderImpl();
53     private final ByteCharSequenceIntHashMap symbolValueToKeyMap = new ByteCharSequenceIntHashMap(
54             256,
55             0.5,
56             SymbolTable.VALUE_NOT_FOUND
57     );
58     private final StringSink tempSink = new StringSink();
59     private final long waitUsBeforeReload;
60     private int columnIndex;
61     private long lastSymbolReaderReloadTimestamp;
62     private int symbolIndexInTxFile;
63     private TxReader txReader;
64     private TableWriterAPI writerAPI;
65 
66     SymbolCache(LineTcpReceiverConfiguration configuration) {
67         this.clock = configuration.getMicrosecondClock();
68         this.waitUsBeforeReload = configuration.getSymbolCacheWaitUsBeforeReload();
69     }
70 
71     @Override
72     public void close() {
73         txReader = null;
74         writerAPI = null;
75         symbolMapReader.close();
76         symbolValueToKeyMap.reset();
77     }
78 
79     @Override
80     public int keyOf(DirectByteCharSequence value) {
81         final int index = symbolValueToKeyMap.keyIndex(value);
82         if (index < 0) {
83             return symbolValueToKeyMap.valueAt(index);
84         }
85 
86         final long ticks = clock.getTicks();
87         int symbolValueCount;
88 
89         if (
90                 ticks - lastSymbolReaderReloadTimestamp > waitUsBeforeReload &&
91                         (symbolValueCount = readSymbolCount(symbolIndexInTxFile, true)) > symbolMapReader.getSymbolCount()
92         ) {
93             symbolMapReader.updateSymbolCount(symbolValueCount);
94             lastSymbolReaderReloadTimestamp = ticks;
95         }
96 
97         utf8ToUtf16Unchecked(value, tempSink);
98         final int symbolKey = symbolMapReader.keyOf(tempSink);
99 
100         if (symbolKey != SymbolTable.VALUE_NOT_FOUND) {
101             symbolValueToKeyMap.putAt(index, ByteCharSequence.newInstance(value), symbolKey);
102         }
103 
104         return symbolKey;
105     }
106 
107     private int readSymbolCount(int symbolIndexInTxFile, boolean initialStateOk) {
108         int watermark = writerAPI.getSymbolCountWatermark(columnIndex);
109         if (watermark != -1) {
110             return watermark;
111         }
112         return safeReadUncommittedSymbolCount(symbolIndexInTxFile, initialStateOk);
113     }
114 
115     private int safeReadUncommittedSymbolCount(int symbolIndexInTxFile, boolean initialStateOk) {
116         // TODO: avoid reading dirty distinct counts from _txn file, add new file instead
117         boolean offsetReloadOk = initialStateOk;
118         while (true) {
119             if (offsetReloadOk) {
120                 int count = txReader.unsafeReadSymbolTransientCount(symbolIndexInTxFile);
121                 Unsafe.getUnsafe().loadFence();
122 
123                 if (txReader.unsafeReadVersion() == txReader.getVersion()) {
124                     return count;
125                 }
126             }
127             offsetReloadOk = txReader.unsafeLoadBaseOffset();
128         }
129     }
130 
131     int getCacheCapacity() {
132         return symbolValueToKeyMap.capacity();
133     }
134 
135     int getCacheValueCount() {
136         return symbolValueToKeyMap.size();
137     }

139     void of(
140             CairoConfiguration configuration,
141             TableWriterAPI writerAPI,
142             int columnIndex,
143             Path path,
144             CharSequence columnName,
145             int symbolIndexInTxFile,
146             TxReader txReader,
147             long columnNameTxn
148     ) {
149         this.writerAPI = writerAPI;
150         this.columnIndex = columnIndex;
151         this.symbolIndexInTxFile = symbolIndexInTxFile;
152         final int plen = path.length();
153         this.txReader = txReader;
154         int symCount = readSymbolCount(symbolIndexInTxFile, false);
155         path.trimTo(plen);
156         symbolMapReader.of(configuration, path, columnName, columnNameTxn, symCount);
157         symbolValueToKeyMap.clear();
158     }
