33 public final class PlainTcpLineChannel implements LineChannel {

39     public PlainTcpLineChannel(NetworkFacade nf, int address, int port, int sndBufferSize) {
40         this.nf = nf;
41         this.fd = nf.socketTcp(true);
42         if (fd < 0) {
43             throw new LineSenderException("could not allocate a file descriptor").errno(nf.errno());
44         }
45         this.sockaddr = nf.sockaddr(address, port);
46         if (nf.connect(fd, sockaddr) != 0) {
47             int errno = nf.errno();
48             nf.close(fd, LOG);
49             nf.freeSockAddr(sockaddr);
50             throw new LineSenderException("could not connect to host ")
51                     .put("[ip=").appendIPv4(address).put("]")
52                     .errno(errno);
53         }
54         configureBuffers(nf, sndBufferSize);
55     }

57     public PlainTcpLineChannel(NetworkFacade nf, CharSequence host, int port, int sndBufferSize) {
58         this.nf = nf;
59         this.sockaddr = -1;
60         this.fd = nf.socketTcp(true);
61         if (fd < 0) {
62             throw new LineSenderException("could not allocate a file descriptor").errno(nf.errno());
63         }
64         long addrInfo = nf.getAddrInfo(host, port);
65         if (addrInfo == -1) {
66             nf.close(fd, LOG);
67             throw new LineSenderException("could not resolve host ")
68                     .put("[host=").put(host).put("]");
69         }
70         if (nf.connectAddrInfo(fd, addrInfo) != 0) {
71             int errno = nf.errno();
72             nf.close(fd, LOG);
73             nf.freeAddrInfo(addrInfo);
74             throw new LineSenderException("could not connect to host ")
75                     .put("[host=").put(host).put("]").errno(errno);
76         }
77         nf.freeAddrInfo(addrInfo);
78         configureBuffers(nf, sndBufferSize);
79     }
