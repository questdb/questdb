76     @Override
77     public void close() {
78         LOG.info().$("line protocol writer closing [threadId=").$(workerId).$(']').$();
79         // Finish all jobs in the queue before stopping
80         for (int n = 0; n < queue.getCycle(); n++) {
81             if (!run(workerId, Job.TERMINATING_STATUS)) {
82                 break;
83             }
84         }
85 
86         Misc.free(path);
87     }

89     @Override
90     public boolean run(int workerId, @NotNull RunStatus runStatus) {
91         assert this.workerId == workerId;
92         boolean busy = drainQueue();
93         // while ILP is hammering the database via multiple connections the writer
94         // is likely to be very busy so commitTables() will run infrequently
95         // commit should run regardless the busy flag but has to finish quickly
96         // idea is to store the tables in a heap data structure being the tables most
97         // desperately need a commit on the top
98         if (!busy) {
99             commitTables();
100             tickWriters();
101         }
102         return busy;
103     }

105     private void commitTables() {
106         long wallClockMillis = millisecondClock.getTicks();
107         if (wallClockMillis > nextCommitTime) {
108             long minTableNextCommitTime = Long.MAX_VALUE;
109             for (int n = 0, sz = assignedTables.size(); n < sz; n++) {
110                 // the heap based solution mentioned above will eliminate the minimum search
111                 // we could just process the min element of the heap until we hit the first commit
112                 // time greater than millis and that will be our nextCommitTime
113                 try {
114                     long tableNextCommitTime = assignedTables.getQuick(n).commitIfIntervalElapsed(wallClockMillis);
115                     // get current time again, commit is not instant and take quite some time.
116                     wallClockMillis = millisecondClock.getTicks();
117                     if (tableNextCommitTime < minTableNextCommitTime) {
118                         // taking the earliest commit time
119                         minTableNextCommitTime = tableNextCommitTime;
120                     }
121                 } catch (Throwable ex) {
122                     LOG.critical()
123                             .$("commit failed [table=").$(assignedTables.getQuick(n).getTableToken())
124                             .$(",ex=").$(ex)
125                             .I$();
126                     metrics.health().incrementUnhandledErrors();
127                 }
128             }
129             // if no tables, just use the default commit interval
130             nextCommitTime = minTableNextCommitTime != Long.MAX_VALUE ? minTableNextCommitTime : wallClockMillis + commitIntervalDefault;
131         }
132     }

134     private boolean drainQueue() {
135         boolean busy = false;
136         while (true) {
137             long cursor;
138             while ((cursor = sequence.next()) < 0) {
139                 if (cursor == -1) {
140                     return busy;
141                 }
142                 Os.pause();
143             }
144             busy = true;
145             final LineTcpMeasurementEvent event = queue.get(cursor);
146 
147             try {
148                 // we check the event's writer thread ID to avoid consuming
149                 // incomplete events
150 
151                 final TableUpdateDetails tud = event.getTableUpdateDetails();
152                 boolean closeWriter = false;
153                 if (event.getWriterWorkerId() == workerId) {
154                     try {
155                         if (tud.isWriterInError()) {
156                             closeWriter = true;
157                         } else {
158                             if (!tud.isAssignedToJob()) {
159                                 assignedTables.add(tud);
160                                 tud.setAssignedToJob(true);
161                                 nextCommitTime = millisecondClock.getTicks();
162                                 LOG.info()
163                                         .$("assigned table to writer thread [tableName=").$(tud.getTableToken())
164                                         .$(", threadId=").$(workerId)
165                                         .I$();
166                             }
167                             event.append();
168                         }
169                     } catch (Throwable ex) {
170                         tud.setWriterInError();
171                         LOG.critical()
172                                 .$("closing writer because of error [table=").$(tud.getTableToken())
173                                 .$(",ex=").$(ex)
174                                 .I$();
175                         metrics.health().incrementUnhandledErrors();
176                         closeWriter = true;
177                         event.createWriterReleaseEvent(tud, false);
178                         // This is a critical error, so we treat it as an unhandled one.
179                     }
180                 } else {
181                     if (event.getWriterWorkerId() == LineTcpMeasurementEventType.ALL_WRITERS_RELEASE_WRITER) {
182                         closeWriter = true;
183                     }
184                 }
185 
186                 if (closeWriter && tud.getWriter() != null) {
187                     scheduler.processWriterReleaseEvent(event, workerId);
188                     assignedTables.remove(tud);
189                     tud.setAssignedToJob(false);
190                     nextCommitTime = millisecondClock.getTicks();
191                 }
192             } catch (Throwable ex) {
193                 LOG.error().$("failed to process ILP event because of exception [ex=").$(ex).I$();
194             }
195 
196             sequence.done(cursor);
197         }
198     }

200     private void tickWriters() {
201         for (int n = 0, sz = assignedTables.size(); n < sz; n++) {
202             assignedTables.getQuick(n).tick();
203         }
204     }
