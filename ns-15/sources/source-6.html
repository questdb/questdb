90     private int findLineEnd() {
91         read();
92         int len = (int) (recvBufPos - recvBufStart);
93         int n = 0;
94         int lineEnd = -1;
95         while (n < len) {
96             byte b = Unsafe.getUnsafe().getByte(recvBufStart + n);
97             if (b == (byte) '\n') {
98                 lineEnd = n;
99                 break;
100             }
101             n++;
102         }
103         if (lineEnd != -1) {
104             return lineEnd;
105         }
106 
107         if (recvBufPos == recvBufEnd) {
108             LOG.info().$('[').$(fd).$("] authentication token is too long").$();
109             authState = AuthState.FAILED;
110             return -1;
111         }
112         if (peerDisconnected) {
113             LOG.info().$('[').$(fd).$("] authentication disconnected by peer when reading token").$();
114             authState = AuthState.FAILED;
115             return -1;
116         }
117         return -1;
118     }

120     private IOContextResult handleAuth(NetworkIOJob netIoJob) {
121         switch (authState) {
122             case WAITING_FOR_KEY_ID:
123                 readKeyId();
124                 break;
125             case SENDING_CHALLENGE:
126                 sendChallenge();
127                 break;
128             case WAITING_FOR_RESPONSE:
129                 waitForResponse();
130                 if (authenticated && recvBufPos > recvBufStart) {
131                     // if authentication is completed and there are still bytes remaining in the buffer
132                     // we have to parse them
133                     return parseMeasurements(netIoJob);
134                 }
135                 break;
136             default:
137                 break;
138         }
139         return authState.ioContextResult;
140     }

142     private void readKeyId() {
143         int lineEnd = findLineEnd();
144         if (lineEnd != -1) {
145             charSeq.of(recvBufStart, recvBufStart + lineEnd);
146             LOG.info().$('[').$(fd).$("] authentication read key id [keyId=").$(charSeq).$(']').$();
147             pubKey = authDb.getPublicKey(charSeq);
148 
149             recvBufPos = recvBufStart;
150             // Generate a challenge with printable ASCII characters 0x20 to 0x7e
151             int n = 0;
152             SecureRandom srand = tlSrand.get();
153             while (n < CHALLENGE_LEN) {
154                 assert recvBufStart + n < recvBufEnd;
155                 int r = (int) (srand.nextDouble() * 0x5f) + 0x20;
156                 Unsafe.getUnsafe().putByte(recvBufStart + n, (byte) r);
157                 challengeBytes[n] = (byte) r;
158                 n++;
159             }
160             Unsafe.getUnsafe().putByte(recvBufStart + n, (byte) '\n');
161             authState = AuthState.SENDING_CHALLENGE;
162         }
163     }

165     private void sendChallenge() {
166         int n = CHALLENGE_LEN + 1 - (int) (recvBufPos - recvBufStart);
167         assert n > 0;
168         while (true) {
169             int nWritten = nf.send(fd, recvBufPos, n);
170             if (nWritten > 0) {
171                 if (n == nWritten) {
172                     recvBufPos = recvBufStart;
173                     authState = AuthState.WAITING_FOR_RESPONSE;
174                     return;
175                 }
176                 recvBufPos += nWritten;
177                 continue;
178             }
179 
180             if (nWritten == 0) {
181                 return;
182             }
183 
184             break;
185         }
186         LOG.info().$('[').$(fd).$("] authentication peer disconnected when challenge was being sent").$();
187         authState = AuthState.FAILED;
188     }

190     private void waitForResponse() {
191         int lineEnd = findLineEnd();
192         if (lineEnd != -1) {
193             // Verify signature
194             if (null == pubKey) {
195                 LOG.info().$('[').$(fd).$("] authentication failed, unknown key id").$();
196                 authState = AuthState.FAILED;
197                 return;
198             }
199 
200             byte[] signature = new byte[lineEnd];
201             for (int n = 0; n < lineEnd; n++) {
202                 signature[n] = Unsafe.getUnsafe().getByte(recvBufStart + n);
203             }
204             authState = AuthState.FAILED;
205 
206             byte[] signatureRaw = Base64.getDecoder().decode(signature);
207             Signature sig = signatureRaw.length == 64 ? tlSigP1363.get() : tlSigDER.get();
208             boolean verified;
209             try {
210                 // On some out of date JDKs zeros can be valid signature because of a bug in the JDK code
211                 // Check that it's not the case.
212                 if (checkAllZeros(signatureRaw)) {
213                     LOG.info().$('[').$(fd).$("] invalid signature, can be cyber attack!").$();
214                     authState = AuthState.FAILED;
215                     return;
216                 }
217                 sig.initVerify(pubKey);
218                 sig.update(challengeBytes);
219                 verified = sig.verify(signatureRaw);
220             } catch (InvalidKeyException | SignatureException ex) {
221                 LOG.info().$('[').$(fd).$("] authentication exception ").$(ex).$();
222                 verified = false;
223             }
224 
225             if (!verified) {
226                 LOG.info().$('[').$(fd).$("] authentication failed, signature was not verified").$();
227                 authState = AuthState.FAILED;
228                 return;
229             }
230 
231             authenticated = true;
232             authState = AuthState.COMPLETE;
233             compactBuffer(recvBufStart + lineEnd + 1);
234             // we must reset start of measurement address
235             resetParser();
236             LOG.info().$('[').$(fd).$("] authentication success").$();
237         }
238     }
