43 class LineTcpNetworkIOJob implements NetworkIOJob {
44     private final static Log LOG = LogFactory.getLog(LineTcpNetworkIOJob.class);
45     private final IODispatcher<LineTcpConnectionContext> dispatcher;
46     private final long maintenanceInterval;

48     private final LineTcpMeasurementScheduler scheduler;
49     private final ByteCharSequenceObjHashMap<TableUpdateDetails> tableUpdateDetailsUtf8 = new ByteCharSequenceObjHashMap<>();
50     private final ObjList<SymbolCache> unusedSymbolCaches = new ObjList<>();

79     @Override
80     public void close() {
81         if (busyContext != null) {
82             busyContext.getDispatcher().disconnect(busyContext, DISCONNECT_REASON_RETRY_FAILED);
83             busyContext = null;
84         }
85         Misc.freeObjList(unusedSymbolCaches);
86 
87         for (int n = 0, sz = tableUpdateDetailsUtf8.size(); n < sz; n++) {
88             final ByteCharSequence tableNameUtf8 = tableUpdateDetailsUtf8.keys().get(n);
89             final TableUpdateDetails tud = tableUpdateDetailsUtf8.get(tableNameUtf8);
90             if (tud.isWal()) {
91                 tud.releaseWriter(true);
92                 tud.close();
93             }
94         }
95     }

128     @Override
129     public boolean run(int workerId, @NotNull RunStatus runStatus) {
130         assert this.workerId == workerId;
131         boolean busy = false;
132         if (busyContext != null) {
133             if (handleIO(busyContext)) {
134                 // queue is still full
135                 return true;
136             }
137             LOG.debug().$("context is no longer waiting on a full queue [fd=").$(busyContext.getFd()).$(']').$();
138             busyContext = null;
139             busy = true;
140         }
141 
142         if (dispatcher.processIOQueue(onRequest)) {
143             busy = true;
144         }
145 
146         if (!busy) {
147             final long wallClockMillis = millisecondClock.getTicks();
148             if (wallClockMillis > nextCommitTime) {
149                 nextCommitTime = scheduler.commitWalTables(tableUpdateDetailsUtf8, wallClockMillis);
150             }
151         }
152 
153         final long millis = millisecondClock.getTicks();
154         if (millis > maintenanceJobDeadline) {
155             busy = scheduler.doMaintenance(tableUpdateDetailsUtf8, workerId, millis);
156             if (!busy) {
157                 maintenanceJobDeadline = millis + maintenanceInterval;
158             }
159         }
160 
161         return busy;
162     }

164     private boolean handleIO(LineTcpConnectionContext context) {
165         if (!context.invalid()) {
166             switch (context.handleIO(this)) {
167                 case NEEDS_READ:
168                     context.getDispatcher().registerChannel(context, IOOperation.READ);
169                     return false;
170                 case NEEDS_WRITE:
171                     context.getDispatcher().registerChannel(context, IOOperation.WRITE);
172                     return false;
173                 case QUEUE_FULL:
174                     return true;
175                 case NEEDS_DISCONNECT:
176                     context.getDispatcher().disconnect(context, DISCONNECT_REASON_UNKNOWN_OPERATION);
177                     return false;
178             }
179         }
180         return false;
181     }
182 
183     private boolean onRequest(int operation, LineTcpConnectionContext context) {
184         if (handleIO(context)) {
185             busyContext = context;
186             LOG.debug().$("context is waiting on a full queue [fd=").$(context.getFd()).$(']').$();
187             return false;
188         }
189         return true;
190     }
