25 package io.questdb.network;
26 
27 /**
28  * pipe(2)-based suspend event object. Used on OS X and BSD in combination with kqueue.
29  */
30 public class PipeSuspendEvent extends SuspendEvent {
31 
32     private final KqueueFacade kqf;
33     private final int readEndFd;
34     private final int writeEndFd;
35 
36     public PipeSuspendEvent(KqueueFacade kqf) {
37         this.kqf = kqf;
38         long fds = kqf.pipe();
39         if (fds < 0) {
40             throw NetworkError.instance(kqf.getNetworkFacade().errno(), "could not create PipeSuspendEvent");
41         }
42         this.readEndFd = (int) (fds >>> 32);
43         this.writeEndFd = (int) fds;
44         kqf.getNetworkFacade().bumpFdCount(readEndFd);
45         kqf.getNetworkFacade().bumpFdCount(writeEndFd);
46     }
47 
48     @Override
49     public void _close() {
50         kqf.getNetworkFacade().close(readEndFd);
51         kqf.getNetworkFacade().close(writeEndFd);
52     }
53 
54     @Override
55     public boolean checkTriggered() {
56         return kqf.readPipe(readEndFd) == 1;
57     }
58 
59     @Override
60     public int getFd() {
61         return readEndFd;
62     }
63 
64     @Override
65     public void trigger() {
66         if (kqf.writePipe(writeEndFd) < 0) {
67             throw NetworkError.instance(kqf.getNetworkFacade().errno())
68                     .put("could not write to pipe [fd=").put(writeEndFd)
69                     .put(']');
70         }
71     }
72 }
