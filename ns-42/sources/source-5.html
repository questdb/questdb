25 package io.questdb.network;

27 import io.questdb.log.Log;
28 import io.questdb.log.LogFactory;

33 import java.io.Closeable;
34 
35 public final class Epoll implements Closeable {
36     private static final Log LOG = LogFactory.getLog(Epoll.class);
37 
38     private final int capacity;
39     private final EpollFacade epf;
40     private final int epollFd;
41     private final long events;
42     private long _rPtr;
43     private boolean closed = false;
44 
45     public Epoll(EpollFacade epf, int capacity) {
46         this.epf = epf;
47         this.capacity = capacity;
48         this.events = this._rPtr = Unsafe.calloc(EpollAccessor.SIZEOF_EVENT * (long) capacity, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
49         this.epollFd = epf.epollCreate();
50         if (epollFd < 0) {
51             throw NetworkError.instance(epf.errno(), "could not create epoll");
52         }
53         Files.bumpFileCount(this.epollFd);
54     }
55 
56     @Override
57     public void close() {
58         if (closed) {
59             return;
60         }
61         epf.getNetworkFacade().close(epollFd, LOG);
62         Unsafe.free(events, EpollAccessor.SIZEOF_EVENT * (long) capacity, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
63         closed = true;
64     }
65 
66     public int control(int fd, long id, int cmd, int event) {
67         Unsafe.getUnsafe().putInt(events + EpollAccessor.EVENTS_OFFSET, event | EpollAccessor.EPOLLET | EpollAccessor.EPOLLONESHOT);
68         Unsafe.getUnsafe().putLong(events + EpollAccessor.DATA_OFFSET, id);
69         return epf.epollCtl(epollFd, cmd, fd, events);
70     }
71 
72     public long getData() {
73         return Unsafe.getUnsafe().getLong(_rPtr + EpollAccessor.DATA_OFFSET);
74     }
75 
76     public int getEvent() {
77         return Unsafe.getUnsafe().getInt(_rPtr + EpollAccessor.EVENTS_OFFSET);
78     }
79 
80     public void listen(int sfd) {
81         Unsafe.getUnsafe().putInt(events + EpollAccessor.EVENTS_OFFSET, EpollAccessor.EPOLLIN | EpollAccessor.EPOLLET);
82         Unsafe.getUnsafe().putLong(events + EpollAccessor.DATA_OFFSET, 0);
83         if (epf.epollCtl(epollFd, EpollAccessor.EPOLL_CTL_ADD, sfd, events) != 0) {
84             throw NetworkError.instance(epf.errno(), "epoll_ctl");
85         }
86     }
87 
88     public int poll() {
89         return epf.epollWait(epollFd, events, capacity, 0);
90     }
91 
92     public void removeListen(int sfd) {
93         Unsafe.getUnsafe().putInt(events + EpollAccessor.EVENTS_OFFSET, EpollAccessor.EPOLLIN | EpollAccessor.EPOLLET);
94         Unsafe.getUnsafe().putLong(events + EpollAccessor.DATA_OFFSET, 0);
95         if (epf.epollCtl(epollFd, EpollAccessor.EPOLL_CTL_DEL, sfd, events) != 0) {
96             throw NetworkError.instance(epf.errno(), "epoll_ctl");
97         }
98     }
99 
100     public void setOffset(int offset) {
101         this._rPtr = this.events + offset;
102     }
