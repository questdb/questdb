273     /**
274      * Return true if a disconnect happened, false otherwise.
275      */
276     @Override
277     public boolean testConnection(int fd, long buffer, int bufferSize) {
278         if (fd == -1) {
279             return true;
280         }
281 
282         final int nRead = Net.peek(fd, buffer, bufferSize);
283 
284         if (nRead == 0) {
285             return false;
286         }
287 
288         if (nRead < 0) {
289             return true;
290         }
291 
292         // Read \r\n from the input stream and discard it since some HTTP clients
293         // send these chars as a keep alive in between requests.
294         int index = 0;
295         while (index < nRead) {
296             byte b = Unsafe.getUnsafe().getByte(buffer + index);
297             if (b != (byte) '\r' && b != (byte) '\n') {
298                 break;
299             }
300             index++;
301         }
302 
303         if (index > 0) {
304             Net.recv(fd, buffer, index);
305         }
306 
307         return false;
308     }
