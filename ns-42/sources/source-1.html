201     private void addPending(int fd, long timestamp) {
202         // append pending connection
203         // all rows below watermark will be registered with epoll (or similar)
204         int r = pending.addRow();
205         LOG.debug().$("pending [row=").$(r).$(", fd=").$(fd).$(']').$();
206         pending.set(r, OPM_TIMESTAMP, timestamp);
207         pending.set(r, OPM_FD, fd);

213     private void createListenFd() throws NetworkError {
214         this.serverFd = nf.socketTcp(false);
215         final int backlog = configuration.getListenBacklog();
216         if (this.port == 0) {
217             // Note that `configuration.getBindPort()` might also be 0.
218             // In such case, we will bind to an ephemeral port.
219             this.port = configuration.getBindPort();
220         }
221         if (nf.bindTcp(this.serverFd, configuration.getBindIPv4Address(), this.port)) {
222             if (this.port == 0) {
223                 // We resolve port 0 only once. In case we close and re-open the
224                 // listening socket, we will reuse the previously resolved
225                 // ephemeral port.
226                 this.port = nf.resolvePort(this.serverFd);
227             }
228             nf.listen(this.serverFd, backlog);
229         } else {
230             throw NetworkError.instance(nf.errno()).couldNotBindSocket(
231                     configuration.getDispatcherLogName(),
232                     configuration.getBindIPv4Address(),
233                     this.port);
234         }
235         LOG.advisory().$("listening on ").$ip(configuration.getBindIPv4Address()).$(':').$(configuration.getBindPort())
236                 .$(" [fd=").$(serverFd)
237                 .$(" backlog=").$(backlog)
238                 .I$();

245     protected void accept(long timestamp) {
246         int tlConCount = this.connectionCount.get();
247         while (tlConCount < activeConnectionLimit) {
248             // this 'accept' is greedy, rather than to rely on epoll (or similar) to
249             // fire accept requests at us one at a time we will be actively accepting
250             // until nothing left.
251 
252             int fd = nf.accept(serverFd);
253 
254             if (fd < 0) {
255                 if (nf.errno() != Net.EWOULDBLOCK) {
256                     LOG.error().$("could not accept [ret=").$(fd).$(", errno=").$(nf.errno()).$(']').$();
257                 }
258                 break;
259             }
260 
261             if (nf.configureNonBlocking(fd) < 0) {
262                 LOG.error().$("could not configure non-blocking [fd=").$(fd).$(", errno=").$(nf.errno()).$(']').$();
263                 nf.close(fd, LOG);
264                 break;
265             }
266 
267             if (nf.setTcpNoDelay(fd, true) < 0) {
268                 // Randomly on OS X, if a client connects and the peer TCP socket has SO_LINGER set to false, then setting the TCP_NODELAY
269                 // option fails!
270                 LOG.info().$("could not turn off Nagle's algorithm [fd=").$(fd).$(", errno=").$(nf.errno()).$(']').$();
271             }
272 
273             if (peerNoLinger) {
274                 nf.configureNoLinger(fd);
275             }
276 
277             if (sndBufSize > 0) {
278                 nf.setSndBuf(fd, sndBufSize);
279             }
280 
281             if (rcvBufSize > 0) {
282                 nf.setRcvBuf(fd, rcvBufSize);
283             }
284 
285             LOG.info().$("connected [ip=").$ip(nf.getPeerIP(fd)).$(", fd=").$(fd).$(']').$();
286             tlConCount = connectionCount.incrementAndGet();
287             addPending(fd, timestamp);
288         }
289 
290         if (tlConCount >= activeConnectionLimit) {
291             if (connectionCount.get() >= activeConnectionLimit) {
292                 unregisterListenerFd();
293                 listening = false;
294                 closeListenFdEpochMs = timestamp + queuedConnectionTimeoutMs;
295                 LOG.info().$("max connection limit reached, unregistered listener [serverFd=").$(serverFd).I$();
296             }
297         }
298     }

300     protected void doDisconnect(C context, int src) {
301         if (context == null || context.invalid()) {
302             return;
303         }
304 
305         final int fd = context.getFd();
306         LOG.info()
307                 .$("disconnected [ip=").$ip(nf.getPeerIP(fd))
308                 .$(", fd=").$(fd)
309                 .$(", src=").$(DISCONNECT_SOURCES[src])
310                 .I$();
311         nf.close(fd, LOG);
312         if (closed) {
313             Misc.free(context);
314         } else {
315             ioContextFactory.done(context);
316         }
317         if (connectionCount.getAndDecrement() >= activeConnectionLimit) {
318             if (connectionCount.get() < activeConnectionLimit) {
319                 if (serverFd < 0) {
320                     createListenFd();
321                 }
322                 registerListenerFd();
323                 listening = true;
324                 LOG.info().$("below maximum connection limit, registered listener [serverFd=").$(serverFd).I$();
325             }
326         }
327     }
