81     private void queryFdSets(long timestamp) {
82         // collect reads into hash map
83         for (int i = 0, n = readFdSet.getCount(); i < n; i++) {
84             final long fd = readFdSet.get(i);

94             final long fd = writeFdSet.get(i);

116         final long timestamp = clock.getTicks();
117         processDisconnects(timestamp);
118

123                 LOG.error().$("select failure [err=").$(nf.errno()).I$();
124                 return false;
125             }
126         } else {
127             count = 0;
128         }
129 
130         boolean useful = false;
131         fds.clear();
132 
133         // collect reads into hash map
134         if (count > 0) {
135             queryFdSets(timestamp);
136             useful = true;
137         }
138 
139         // process returned fds
140         useful |= processRegistrations(timestamp);
141 
142         // re-arm select() fds
143         int readFdCount = 0;

145         readFdSet.reset();
146         writeFdSet.reset();
147         long deadline = timestamp - idleConnectionTimeout;
148         for (int i = 0, n = pending.size(); i < n; ) {
149             final C context = pending.get(i);
150 
151             // check if the context is waiting for a suspend event
152             final SuspendEvent suspendEvent = context.getSuspendEvent();
153             if (suspendEvent != null) {
154                 if (suspendEvent.checkTriggered() || suspendEvent.isDeadlineMet(timestamp)) {
155                     // the event has been triggered or expired already, clear it and proceed
156                     context.clearSuspendEvent();
157                 }
158             }
159 
160             final long ts = pending.get(i, OPM_TIMESTAMP);
161             final int fd = (int) pending.get(i, OPM_FD);
162             final int newOp = fds.get(fd);
163             assert fd != serverFd;
164 
165             if (newOp == -1) {
166                 // new operation case
167 
168                 // check if the connection was idle for too long
169                 if (ts < deadline) {
170                     doDisconnect(context, DISCONNECT_SRC_IDLE);
171                     pending.deleteRow(i);
172                     n--;
173                     useful = true;
174                     continue;
175                 }
176 
177                 int operation = (int) pending.get(i, OPM_OPERATION);
178                 i++;
179                 if (suspendEvent != null) {
180                     // if the operation was suspended, we request a read to be able to detect a client disconnect
181                     operation = IOOperation.READ;
182                 }
183

185                     readFdSet.add(fd);
186                     readFdCount++;
187                 } else {
188                     writeFdSet.add(fd);
189                     writeFdCount++;
190                 }
191             } else {
192                 // select()'ed operation case
193                 if (suspendEvent != null) {
194                     // the event is still pending, check if we have a client disconnect
195                     if (testConnection(context.getFd())) {
196                         doDisconnect(context, DISCONNECT_SRC_PEER_DISCONNECT);
197                         pending.deleteRow(i);
198                         n--;
199                     } else {

201                     }
202                     continue;
203                 }
204 
205                 // publish event and remove from pending
206                 useful = true;
207

209                     publishOperation(IOOperation.READ, context);
210                 }
211                 if ((newOp & SelectAccessor.FD_WRITE) > 0) {
212                     publishOperation(IOOperation.WRITE, context);
213                 }
214

216                 n--;
217             }
218         }
219

221             assert serverFd >= 0;
222             readFdSet.add(serverFd);
223             readFdCount++;
224         }
225 
226         readFdSet.setCount(readFdCount);
227         writeFdSet.setCount(writeFdCount);
228         return useful;
229     }
