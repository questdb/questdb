60     private void doDisconnect(C context, long id, int reason) {
61         final SuspendEvent suspendEvent = context.getSuspendEvent();
62         if (suspendEvent != null) {
63             // yes, we can do a binary search over EVM_OPERATION_ID since
64             // these ref ids are monotonically growing
65             int eventRow = pendingEvents.binarySearch(id, EVM_OPERATION_ID);
66             if (eventRow < 0) {
67                 LOG.critical().$("internal error: suspend event not found [id=").$(id).I$();
68             } else {
69                 final long eventId = pendingEvents.get(eventRow, EVM_ID);
70                 if (epoll.control(suspendEvent.getFd(), eventId, EpollAccessor.EPOLL_CTL_DEL, 0) < 0) {
71                     LOG.critical().$("internal error: epoll_ctl remove suspend event failure [eventId=").$(eventId)
72                             .$(", err=").$(nf.errno()).I$();
73                 }
74                 pendingEvents.deleteRow(eventRow);
75             }
76         }
77         doDisconnect(context, reason);
78     }

80     private void enqueuePending(int watermark) {
81         for (int i = watermark, sz = pending.size(), offset = 0; i < sz; i++, offset += EpollAccessor.SIZEOF_EVENT) {
82             final long id = pending.get(i, OPM_ID);
83             if (
84                     epoll.control(
85                             (int) pending.get(i, OPM_FD),
86                             id,
87                             EpollAccessor.EPOLL_CTL_ADD,
88                             initialBias == IODispatcherConfiguration.BIAS_READ ? EpollAccessor.EPOLLIN : EpollAccessor.EPOLLOUT
89                     ) < 0
90             ) {
91                 LOG.critical().$("internal error: epoll_ctl failure [id=").$(id)
92                         .$(", err=").$(nf.errno()).I$();
93             }
94         }
95     }

97     private boolean handleSocketOperation(long id) {
98         // find row in pending for two reasons:
99         // 1. find payload
100         // 2. remove row from pending, remaining rows will be timed out
101         final int row = pending.binarySearch(id, OPM_ID);
102         if (row < 0) {
103             LOG.critical().$("internal error: epoll returned unexpected id [id=").$(id).I$();
104             return false;
105         }
106 
107         final C context = pending.get(row);
108         final SuspendEvent suspendEvent = context.getSuspendEvent();
109         if (suspendEvent != null) {
110             // the operation is suspended, check if we have a client disconnect
111             if (testConnection(context.getFd())) {
112                 doDisconnect(context, id, DISCONNECT_SRC_PEER_DISCONNECT);
113                 pending.deleteRow(row);
114                 return true;
115             }
116         } else {
117             publishOperation(
118                     (epoll.getEvent() & EpollAccessor.EPOLLIN) > 0 ? IOOperation.READ : IOOperation.WRITE,
119                     context
120             );
121             pending.deleteRow(row);
122             return true;
123         }
124         return false;
125     }

127     private void handleSuspendEvent(long id) {
128         final int eventsRow = pendingEvents.binarySearch(id, EVM_ID);
129         if (eventsRow < 0) {
130             LOG.critical().$("internal error: epoll returned unexpected event id [eventId=").$(id).I$();
131             return;
132         }
133 
134         final long opId = pendingEvents.get(eventsRow, EVM_OPERATION_ID);
135         final int row = pending.binarySearch(opId, OPM_ID);
136         if (row < 0) {
137             LOG.critical().$("internal error: suspended operation not found [id=").$(opId).$(", eventId=").$(id).I$();
138             return;
139         }
140 
141         final int operation = (int) pending.get(row, OPM_OPERATION);
142         final C context = pending.get(row);
143         final SuspendEvent suspendEvent = context.getSuspendEvent();
144         assert suspendEvent != null;
145 
146         resumeOperation(context, opId, operation);
147         pendingEvents.deleteRow(eventsRow);
148     }

158     private void processIdleConnections(long idleTimestamp) {
159         int count = 0;
160         for (int i = 0, n = pending.size(); i < n && pending.get(i, OPM_TIMESTAMP) < idleTimestamp; i++, count++) {
161             doDisconnect(pending.get(i), pending.get(i, OPM_ID), DISCONNECT_SRC_IDLE);
162         }

166     private boolean processRegistrations(long timestamp) {
167         boolean useful = false;
168         long cursor;
169         while ((cursor = interestSubSeq.next()) > -1) {
170             IOEvent<C> event = interestQueue.get(cursor);
171             C context = event.context;
172             int requestedOperation = event.operation;
173             interestSubSeq.done(cursor);
174 
175             useful = true;
176             final long opId = nextOpId();
177             final int fd = context.getFd();
178             int operation = requestedOperation;
179             LOG.debug().$("processing registration [fd=").$(fd)
180                     .$(", op=").$(operation)
181                     .$(", id=").$(opId).I$();
182 
183             final SuspendEvent suspendEvent = context.getSuspendEvent();
184             if (suspendEvent != null) {
185                 // if the operation was suspended, we request a read to be able to detect a client disconnect 
186                 operation = IOOperation.READ;
187             }
188 
189             int opRow = pending.addRow();
190             pending.set(opRow, OPM_TIMESTAMP, timestamp);
191             pending.set(opRow, OPM_FD, fd);
192             pending.set(opRow, OPM_ID, opId);
193             pending.set(opRow, OPM_OPERATION, requestedOperation);
194             pending.set(opRow, context);
195 
196             // we re-arm epoll globally, in that even when we disconnect
197             // because we have to remove FD from epoll
198             final int epollOp = operation == IOOperation.READ ? EpollAccessor.EPOLLIN : EpollAccessor.EPOLLOUT;
199             if (epoll.control(fd, opId, EpollAccessor.EPOLL_CTL_MOD, epollOp) < 0) {
200                 LOG.critical().$("internal error: epoll_ctl modify operation failure [id=").$(opId)
201                         .$(", err=").$(nf.errno()).I$();
202             }
203 
204             if (suspendEvent != null) {
205                 // ok, the operation was suspended, so we need to track the suspend event
206                 final long eventId = nextEventId();
207                 LOG.debug().$("registering suspend event [fd=").$(fd)
208                         .$(", op=").$(operation)
209                         .$(", eventId=").$(eventId)
210                         .$(", suspendedOpId=").$(opId)
211                         .$(", deadline=").$(suspendEvent.getDeadline()).I$();
212 
213                 int eventRow = pendingEvents.addRow();
214                 pendingEvents.set(eventRow, EVM_ID, eventId);
215                 pendingEvents.set(eventRow, EVM_OPERATION_ID, opId);
216                 pendingEvents.set(eventRow, EVM_DEADLINE, suspendEvent.getDeadline());
217 
218                 if (epoll.control(suspendEvent.getFd(), eventId, EpollAccessor.EPOLL_CTL_ADD, EpollAccessor.EPOLLIN) < 0) {
219                     LOG.critical().$("internal error: epoll_ctl add suspend event failure [id=").$(eventId)
220                             .$(", err=").$(nf.errno()).I$();
221                 }
222             }
223         }
224         return useful;
225     }

227     private void processSuspendEventDeadlines(long timestamp) {
228         int count = 0;
229         for (int i = 0, n = pendingEvents.size(); i < n && pendingEvents.get(i, EVM_DEADLINE) < timestamp; i++, count++) {
230             final long eventId = pendingEvents.get(i, EVM_ID);
231             final long opId = pendingEvents.get(i, EVM_OPERATION_ID);
232             final int pendingRow = pending.binarySearch(opId, OPM_ID);
233             if (pendingRow < 0) {
234                 LOG.critical().$("internal error: failed to find operation for expired suspend event [id=").$(opId).I$();
235                 continue;
236             }
237             // First, remove the suspend event from the epoll interest list.
238             final C context = pending.get(pendingRow);
239             final int operation = (int) pending.get(pendingRow, OPM_OPERATION);
240             final SuspendEvent suspendEvent = context.getSuspendEvent();
241             assert suspendEvent != null;
242             if (epoll.control(suspendEvent.getFd(), eventId, EpollAccessor.EPOLL_CTL_DEL, 0) < 0) {
243                 LOG.critical().$("internal error: epoll_ctl remove suspend event failure [eventId=").$(eventId)
244                         .$(", err=").$(nf.errno()).I$();
245             }
246             // Next, resume the original operation and close the event.
247             resumeOperation(context, opId, operation);
248         }
249         pendingEvents.zapTop(count);
250     }

252     private void resumeOperation(C context, long id, int operation) {
253         // to resume a socket operation, we simply re-arm epoll
254         if (
255                 epoll.control(
256                         context.getFd(),
257                         id,
258                         EpollAccessor.EPOLL_CTL_MOD,
259                         operation == IOOperation.READ ? EpollAccessor.EPOLLIN : EpollAccessor.EPOLLOUT
260                 ) < 0
261         ) {
262             LOG.critical().$("internal error: epoll_ctl operation mod failure [id=").$(id)
263                     .$(", err=").$(nf.errno()).I$();
264         }
265         context.clearSuspendEvent();
266     }

278     @Override
279     protected boolean runSerially() {
280         // todo: introduce fairness factor
281         //  current worker impl will still proceed to execute another job even if this one was useful
282         //  we should see if we can stay inside of this method until we have a completely idle iteration
283         //  at the same time we should hog this thread in case we are always 'useful', we can probably
284         //  introduce a loop count after which we always exit
285         boolean useful = false;
286 
287         final long timestamp = clock.getTicks();
288         processDisconnects(timestamp);
289         final int n = epoll.poll();
290         int watermark = pending.size();
291         int offset = 0;
292         if (n > 0) {
293             // check all activated FDs
294             LOG.debug().$("epoll [n=").$(n).$(']').$();
295             for (int i = 0; i < n; i++) {
296                 epoll.setOffset(offset);
297                 offset += EpollAccessor.SIZEOF_EVENT;
298                 final long id = epoll.getData();
299                 // this is server socket, accept if there aren't too many already
300                 if (id == 0) {
301                     accept(timestamp);
302                     useful = true;
303                     continue;
304                 }
305                 if (isEventId(id)) {
306                     handleSuspendEvent(id);
307                     continue;
308                 }
309                 if (handleSocketOperation(id)) {
310                     useful = true;
311                     watermark--;
312                 }
313             }
314         }
315 
316         // process rows over watermark (new connections)
317         if (watermark < pending.size()) {
318             enqueuePending(watermark);
319         }
320 
321         // process timed out suspend events and resume the original operations
322         if (pendingEvents.size() > 0 && pendingEvents.get(0, EVM_DEADLINE) < timestamp) {
323             processSuspendEventDeadlines(timestamp);
324         }
325 
326         // process timed out connections
327         final long idleTimestamp = timestamp - idleConnectionTimeout;
328         if (pending.size() > 0 && pending.get(0, OPM_TIMESTAMP) < idleTimestamp) {
329             processIdleConnections(idleTimestamp);
330             useful = true;
331         }
332 
333         return processRegistrations(timestamp) || useful;
334     }
