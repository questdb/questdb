62     private void doDisconnect(C context, long id, int reason) {
63         final SuspendEvent suspendEvent = context.getSuspendEvent();
64         if (suspendEvent != null) {
65             // yes, we can do a binary search over EVM_OPERATION_ID since
66             // these ref ids are monotonically growing
67             int eventRow = pendingEvents.binarySearch(id, EVM_OPERATION_ID);
68             if (eventRow < 0) {
69                 LOG.critical().$("internal error: suspend event not found [id=").$(id).I$();
70             } else {
71                 kqueue.setWriteOffset(0);
72                 kqueue.removeFD(suspendEvent.getFd());
73                 registerWithKQueue(1);
74                 pendingEvents.deleteRow(eventRow);
75             }
76         }
77         doDisconnect(context, reason);
78     }

80     private void enqueuePending(int watermark) {
81         int index = 0;
82         for (int i = watermark, sz = pending.size(), offset = 0; i < sz; i++, offset += KqueueAccessor.SIZEOF_KEVENT) {
83             kqueue.setWriteOffset(offset);
84 
85             final int fd = (int) pending.get(i, OPM_FD);
86             final int operation = initialBias == IODispatcherConfiguration.BIAS_READ ? IOOperation.READ : IOOperation.WRITE;
87 
88             if (operation == IOOperation.READ) {
89                 kqueue.readFD(fd, pending.get(i, OPM_ID));
90             } else {
91                 kqueue.writeFD(fd, pending.get(i, OPM_ID));
92             }
93             if (++index > capacity - 1) {
94                 registerWithKQueue(index);
95                 index = 0;
96                 offset = 0;
97             }
98         }
99         if (index > 0) {
100             registerWithKQueue(index);
101         }
102     }

104     private boolean handleSocketOperation(long id) {
105         // find row in pending for two reasons:
106         // 1. find payload
107         // 2. remove row from pending, remaining rows will be timed out
108         final int row = pending.binarySearch(id, OPM_ID);
109         if (row < 0) {
110             LOG.critical().$("internal error: kqueue returned unexpected id [id=").$(id).I$();
111             return false;
112         }
113 
114         final C context = pending.get(row);
115         final SuspendEvent suspendEvent = context.getSuspendEvent();
116         if (suspendEvent != null) {
117             // the operation is suspended, check if we have a client disconnect
118             if (testConnection(context.getFd())) {
119                 doDisconnect(context, id, DISCONNECT_SRC_PEER_DISCONNECT);
120                 pending.deleteRow(row);
121                 return true;
122             }
123         } else {
124             publishOperation(
125                     kqueue.getFilter() == KqueueAccessor.EVFILT_READ ? IOOperation.READ : IOOperation.WRITE,
126                     context
127             );
128             pending.deleteRow(row);
129             return true;
130         }
131         return false;
132     }

134     private void handleSuspendEvent(long id) {
135         final int eventsRow = pendingEvents.binarySearch(id, EVM_ID);
136         if (eventsRow < 0) {
137             LOG.critical().$("internal error: kqueue returned unexpected event id [eventId=").$(id).I$();
138             return;
139         }
140 
141         final long opId = pendingEvents.get(eventsRow, EVM_OPERATION_ID);
142         final int row = pending.binarySearch(opId, OPM_ID);
143         if (row < 0) {
144             LOG.critical().$("internal error: suspended operation not found [id=").$(opId).$(", eventId=").$(id).I$();
145             return;
146         }
147 
148         final long eventId = pendingEvents.get(eventsRow, EVM_ID);
149         final int operation = (int) pending.get(row, OPM_OPERATION);
150         final C context = pending.get(row);
151         final SuspendEvent suspendEvent = context.getSuspendEvent();
152         assert suspendEvent != null;
153 
154         LOG.debug().$("handling triggered suspend event and resuming original operation [fd=").$(context.getFd())
155                 .$(", opId=").$(opId)
156                 .$(", eventId=").$(eventId).I$();
157 
158         context.clearSuspendEvent();
159         kqueue.setWriteOffset(0);
160         if (operation == IOOperation.READ) {
161             kqueue.readFD(context.getFd(), opId);
162         } else {
163             kqueue.writeFD(context.getFd(), opId);
164         }
165         registerWithKQueue(1);
166 
167         pendingEvents.deleteRow(eventsRow);
168     }

179         int count = 0;
180         for (int i = 0, n = pending.size(); i < n && pending.get(i, OPM_TIMESTAMP) < deadline; i++, count++) {

261     private void processSuspendEventDeadlines(long timestamp) {
262         int index = 0;
263         int offset = 0;
264         int count = 0;
265         for (int i = 0, n = pendingEvents.size(); i < n && pendingEvents.get(i, EVM_DEADLINE) < timestamp; i++, count++) {
266             final long opId = pendingEvents.get(i, EVM_OPERATION_ID);
267             final int pendingRow = pending.binarySearch(opId, OPM_ID);
268             if (pendingRow < 0) {
269                 LOG.critical().$("internal error: failed to find operation for expired suspend event [id=").$(opId).I$();
270                 continue;
271             }
272 
273             // First, remove the suspend event from kqueue tracking.
274             final C context = pending.get(pendingRow);
275             final int operation = (int) pending.get(pendingRow, OPM_OPERATION);
276             final SuspendEvent suspendEvent = context.getSuspendEvent();
277             assert suspendEvent != null;
278             kqueue.setWriteOffset(offset);
279             kqueue.removeFD(suspendEvent.getFd());
280             offset += KqueueAccessor.SIZEOF_KEVENT;
281             if (++index > capacity - 1) {
282                 registerWithKQueue(index);
283                 index = offset = 0;
284             }
285 
286             // Next, close the event and resume the original operation.
287             context.clearSuspendEvent();
288             kqueue.setWriteOffset(offset);
289             if (operation == IOOperation.READ) {
290                 kqueue.readFD(context.getFd(), opId);
291             } else {
292                 kqueue.writeFD(context.getFd(), opId);
293             }
294             offset += KqueueAccessor.SIZEOF_KEVENT;
295             if (++index > capacity - 1) {
296                 registerWithKQueue(index);
297                 index = offset = 0;
298             }
299         }
300         if (index > 0) {
301             registerWithKQueue(index);
302         }
303         pendingEvents.zapTop(count);
304     }

325     @Override
326     protected boolean runSerially() {
327         boolean useful = false;
328 
329         final long timestamp = clock.getTicks();
330         processDisconnects(timestamp);
331         final int n = kqueue.poll();
332         int watermark = pending.size();
333         int offset = 0;
334         if (n > 0) {
335             LOG.debug().$("poll [n=").$(n).$(']').$();
336             // check all activated FDs
337             for (int i = 0; i < n; i++) {
338                 kqueue.setReadOffset(offset);
339                 offset += KqueueAccessor.SIZEOF_KEVENT;
340                 final int fd = kqueue.getFd();
341                 final long id = kqueue.getData();
342                 // this is server socket, accept if there aren't too many already
343                 if (fd == serverFd) {
344                     accept(timestamp);
345                     useful = true;
346                     continue;
347                 }
348                 if (isEventId(id)) {
349                     handleSuspendEvent(id);
350                     continue;
351                 }
352                 if (handleSocketOperation(id)) {
353                     useful = true;
354                     watermark--;
355                 }
356             }
357         }
358 
359         // process rows over watermark (new connections)
360         if (watermark < pending.size()) {
361             enqueuePending(watermark);
362         }
363 
364         // process timed out suspend events and resume the original operations
365         if (pendingEvents.size() > 0 && pendingEvents.get(0, EVM_DEADLINE) < timestamp) {
366             processSuspendEventDeadlines(timestamp);
367         }
368 
369         // process timed out connections
370         final long deadline = timestamp - idleConnectionTimeout;
371         if (pending.size() > 0 && pending.get(0, OPM_TIMESTAMP) < deadline) {
372             processIdleConnections(deadline);
373             useful = true;
374         }
375 
376         return processRegistrations(timestamp) || useful;
377     }
