198     public static int parseIPv4(CharSequence ipv4Address) {
199         int ip = 0;
200         int count = 0;
201         int lo = 0;
202         int hi;
203         try {
204             while ((hi = Chars.indexOf(ipv4Address, lo, '.')) > -1) {
205                 int n = Numbers.parseInt(ipv4Address, lo, hi);
206                 ip = (ip << 8) | n;
207                 count++;
208                 lo = hi + 1;
209             }
210 
211             if (count != 3) {
212                 throw NetworkError.instance(0, "invalid address [").put(ipv4Address).put(']');
213             }
214 
215             return (ip << 8) | Numbers.parseInt(ipv4Address, lo, ipv4Address.length());
216         } catch (NumericException e) {
217             throw NetworkError.instance(0, "invalid address [").put(ipv4Address).put(']');
218         }
219     }
