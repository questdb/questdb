25 package io.questdb.network;

27 import io.questdb.std.Unsafe;
28 import org.jetbrains.annotations.TestOnly;
29 
30 import java.io.Closeable;
31 
32 /**
33  * Used for generic single-time events that can be used with
34  * {@link IODispatcher}. Such events aren't necessarily network I/O
35  * related, but instead provide an efficient way of suspending a task
36  * done in the IODispatcher's event loop and resume it later when another
37  * thread sends us a notification through the event.
38  * <p>
39  * To be more specific, we use eventfd(2) in Linux and pipes in OS X.
40  */
41 public abstract class SuspendEvent implements Closeable {
42 
43     private static final long REF_COUNT_OFFSET;
44 
45     private long deadline = Long.MAX_VALUE;
46 
47     // set by using Unsafe, see REF_COUNT_OFFSET, close().
48     @SuppressWarnings({"FieldCanBeLocal", "FieldMayBeFinal"})
49     private volatile int refCount = 2;
50 
51     public abstract void _close();
52 
53     /**
54      * Returns true if the event was triggered.
55      *
56      * @return true if the event was triggered
57      */
58     public abstract boolean checkTriggered();
59 
60     /**
61      * Event is assumed to be held and then closed by two parties.
62      * The underlying OS resources are freed on the second close call.
63      */
64     @Override
65     public void close() {
66         final int prevRefCount = Unsafe.getUnsafe().getAndAddInt(this, REF_COUNT_OFFSET, -1);
67         if (prevRefCount == 1) {
68             _close();
69         }
70     }
71 
72     /**
73      * Event deadline.
74      *
75      * @return a deadline (epoch millis) for the event or Long.MAX_VALUE if the deadline is not set
76      */
77     public long getDeadline() {
78         return deadline;
79     }
80 
81     /**
82      * Returns fd to be used to listen/wait for the event.
83      *
84      * @return fd to be used to listen/wait for the event
85      */
86     public abstract int getFd();
87 
88     @TestOnly
89     public boolean isClosedByAtLeastOneSide() {
90         return refCount <= 1;
91     }
92 
93     /**
94      * Returns true if the deadline was set to an older value than the given timestamp,
95      * false - otherwise.
96      *
97      * @param timestamp timestamp to compare against
98      * @return true if the deadline was set to an older value than the given timestamp
99      */
100     public boolean isDeadlineMet(long timestamp) {
101         return deadline > 0 && deadline <= timestamp;
102     }
103 
104     /**
105      * Sets a deadline (epoch millis) for the event.
106      *
107      * @param deadline deadline (epoch millis) for the event
108      */
109     public void setDeadline(long deadline) {
110         this.deadline = deadline;
111     }
112 
113     /**
114      * Sends the event to the receiving side.
115      */
116     public abstract void trigger();
117 
118     static {
119         REF_COUNT_OFFSET = Unsafe.getFieldOffset(SuspendEvent.class, "refCount");
120     }
121 }
