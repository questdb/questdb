25 package io.questdb.network;

27 /**
28  * eventfd(2)-based suspend event object. Used on Linux in combination with epoll.
29  */
30 public class EventFdSuspendEvent extends SuspendEvent {
31 
32     private final EpollFacade epf;
33     private final int fd;
34 
35     public EventFdSuspendEvent(EpollFacade epf) {
36         this.epf = epf;
37         int fd = epf.eventFd();
38         if (fd < 0) {
39             throw NetworkError.instance(epf.errno(), "could not create EventFdSuspendEvent");
40         }
41         this.fd = fd;
42         epf.getNetworkFacade().bumpFdCount(fd);
43     }
44 
45     @Override
46     public void _close() {
47         epf.getNetworkFacade().close(fd);
48     }
49 
50     @Override
51     public boolean checkTriggered() {
52         return epf.readEventFd(fd) == 1;
53     }
54 
55     @Override
56     public int getFd() {
57         return fd;
58     }
59 
60     @Override
61     public void trigger() {
62         if (epf.writeEventFd(fd) < 0) {
63             throw NetworkError.instance(epf.errno())
64                     .put("could not write to eventfd [fd=").put(fd)
65                     .put(']');
66         }
67     }
68 }
