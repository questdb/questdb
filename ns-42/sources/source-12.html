47     public Kqueue(KqueueFacade kqf, int capacity) {
48         this.kqf = kqf;
49         this.capacity = capacity;
50         this.bufferSize = KqueueAccessor.SIZEOF_KEVENT * capacity;
51         this.changeList = this.writeAddress = Unsafe.calloc(bufferSize, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
52         this.eventList = this.readAddress = Unsafe.calloc(bufferSize, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
53         this.kq = kqf.kqueue();
54         if (kq < 0) {
55             throw NetworkError.instance(kqf.getNetworkFacade().errno(), "could not create kqueue");
56         }
57         Files.bumpFileCount(this.kq);
58     }
59 
60     @Override

62         kqf.getNetworkFacade().close(kq, LOG);
63         Unsafe.free(this.changeList, bufferSize, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
64         Unsafe.free(this.eventList, bufferSize, MemoryTag.NATIVE_IO_DISPATCHER_RSS);
65     }
