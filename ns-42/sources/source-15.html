35 public class MutableIOContextFactory<C extends MutableIOContext<C>>
36         implements IOContextFactory<C>, Closeable, EagerThreadSetup {
37 
38     private final ThreadLocal<WeakMutableObjectPool<C>> contextPool;
39     private volatile boolean closed = false;
40 
41     public MutableIOContextFactory(ObjectFactory<C> factory, int poolSize) {
42         // todo: this is very slow, refactor
43         this.contextPool = new ThreadLocal<>(() -> new WeakMutableObjectPool<>(factory, poolSize));
44     }
45 
46     @Override
47     public void close() {
48         closed = true;
49     }
50 
51     @Override
52     public void done(C context) {
53         if (closed) {
54             Misc.free(context);
55         } else {
56             context.of(-1, null);
57             contextPool.get().push(context);
58         }
59     }
60 
61     public void freeThreadLocal() {
62         // helper call, it will free only thread-local instance and not others
63         Misc.free(this.contextPool);
64     }
65 
66     @Override
67     public C newInstance(int fd, IODispatcher<C> dispatcher) {
68         return contextPool.get().pop().of(fd, dispatcher);
69     }
70

72     public void setup() {
