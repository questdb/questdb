56     public Function newInstance(
57             int position, ObjList<Function> args,
58             IntList argPositions, CairoConfiguration configuration,
59             SqlExecutionContext sqlExecutionContext) {
60         // there are optimisation opportunities
61         // 1. when one of args is constant null comparison can boil down to checking
62         //    length of non-constant (must be -1)
63         // 2. when one of arguments is constant, save method call and use a field
64 
65         SymbolFunction symFunc = (SymbolFunction) args.getQuick(0);
66         Function chrFunc = args.getQuick(1);
67 
68         if (chrFunc.isConstant()) {
69             final char constValue = chrFunc.getChar(null);
70             if (symFunc.getStaticSymbolTable() != null) {
71                 return new ConstCheckColumnFunc(symFunc, constValue);
72             } else {
73                 return new ConstCheckFunc(symFunc, constValue);
74             }
75         }
76 
77         return new Func(symFunc, chrFunc);
78     }
