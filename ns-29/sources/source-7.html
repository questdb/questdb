51     @Override
52     public Function newInstance(int position, ObjList<Function> args, IntList argPositions, CairoConfiguration configuration, SqlExecutionContext sqlExecutionContext) {
53         // this is probably a special case factory
54         // NaN is always a double, so this could lead comparisons of all primitive types
55         // to NaN route to this factory. Obviously comparing naively will not work
56         // We have to check arg types and when NaN is present we would generate special case
57         // functions for NaN checks.
58 
59         Function left = args.getQuick(0);
60         Function right = args.getQuick(1);
61 
62         int leftType = left.getType();
63         int rightType = right.getType();
64 
65         if (isNullConstant(left, leftType)) {
66             return dispatchUnaryFunc(right, rightType);
67         }
68         if (isNullConstant(right, rightType)) {
69             return dispatchUnaryFunc(left, leftType);
70         }
71         return new Func(args.getQuick(0), args.getQuick(1));
72     }

74     private static Function dispatchUnaryFunc(Function operand, int operandType) {
75         switch (ColumnType.tagOf(operandType)) {
76             case ColumnType.INT:
77                 return new FuncIntIsNaN(operand);
78             case ColumnType.LONG:
79                 return new FuncLongIsNaN(operand);
80             case ColumnType.DATE:
81                 return new FuncDateIsNaN(operand);
82             case ColumnType.TIMESTAMP:
83                 return new FuncTimestampIsNaN(operand);
84             case ColumnType.FLOAT:
85                 return new FuncFloatIsNaN(operand);
86             default:
87                 // double
88                 return new FuncDoubleIsNaN(operand);
89         }
90     }

92     private static boolean isNullConstant(Function operand, int operandType) {
93         return operand.isConstant() &&
94                 (ColumnType.isDouble(operandType) && Double.isNaN(operand.getDouble(null))
95                         ||
96                         operandType == ColumnType.NULL);
97     }

122     protected static class Func extends AbstractEqBinaryFunction {
123         // This function class uses both subtraction and equality to judge whether two parameters
124         // are equal because subtraction is to prevent the judging mistakes with different
125         // precision, and equality is for the comparison of Infinity. In java,
126         // Infinity - Infinity = NaN, so it won't satisfy the subtraction situation. But
127         // Infinity = Infinity, so use equality could solve this problem.
128 
129         public Func(Function left, Function right) {
130             super(left, right);
131         }
132 
133         @Override
134         public boolean getBool(Record rec) {
135             final double l = left.getDouble(rec);
136             final double r = right.getDouble(rec);
137             return negated != (l != l && r != r || Math.abs(l - r) < 0.0000000001 || l == r);
138         }
139     }
