51     public Function newInstance(int position, ObjList<Function> args, IntList argPositions, CairoConfiguration configuration, SqlExecutionContext sqlExecutionContext) {
52         // there are optimisation opportunities
53         // 1. when one of args is constant null comparison can boil down to checking
54         //    length of non-constant (must be -1)
55         // 2. when one of arguments is constant, save method call and use a field
56 
57         Function a = args.getQuick(0);
58         Function b = args.getQuick(1);
59 
60         if (a.isConstant() && !b.isConstant()) {
61             return createHalfConstantFunc(a, b);
62         }
63 
64         if (!a.isConstant() && b.isConstant()) {
65             return createHalfConstantFunc(b, a);
66         }
67 
68         return new Func(a, b);
69     }

71     private Function createHalfConstantFunc(Function constFunc, Function varFunc) {
72         CharSequence constValue = constFunc.getStr(null);
73 
74         if (constValue == null) {
75             return new NullCheckFunc(varFunc);
76         }
77 
78         return new ConstCheckFunc(varFunc, constValue);
79     }
