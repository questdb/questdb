54         // there are optimisation opportunities
55         // 1. when one of args is constant null comparison can boil down to checking
56         //    length of non-constant (must be -1)
57         // 2. when one of arguments is constant, save method call and use a field

59         Function symFunc = args.getQuick(0);
60         Function strFunc = args.getQuick(1);
61 
62         // SYMBOL cannot be constant

65         }

69     private Function createHalfConstantFunc(Function constFunc, Function varFunc) {
70         CharSequence constValue = constFunc.getStr(null);
71         SymbolFunction func = (SymbolFunction) varFunc;
72         if (func.getStaticSymbolTable() != null) {
73             return new ConstCheckColumnFunc(func, constValue);
74         } else {
75             if (constValue == null) {
76                 return new NullCheckFunc(varFunc);
77             }
78             if (func.isSymbolTableStatic()) {
79                 return new ConstSymIntCheckFunc(func, constValue);
80             }
81             return new ConstCheckFunc(func, constValue);
82         }
83     }
