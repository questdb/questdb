25 package io.questdb.cairo.sql;

27 import io.questdb.cairo.CairoConfiguration;
28 import io.questdb.cairo.ColumnType;
29 import io.questdb.std.*;
30 
31 public class PageAddressCache implements Mutable {
32 
33     private final long cacheSizeThreshold;
34     // Index remapping for variable length columns.
35     private final IntList varLenColumnIndexes = new IntList();
36     private int columnCount;
37     // Index page addresses and page sizes are stored only for variable length columns.
38     private LongList indexPageAddresses = new LongList();
39     private LongList pageAddresses = new LongList();
40     private LongList pageRowIdOffsets = new LongList();
41     private LongList pageSizes = new LongList();
42     private int varLenColumnCount;
43 
44     public PageAddressCache(CairoConfiguration configuration) {
45         cacheSizeThreshold = configuration.getSqlJitPageAddressCacheThreshold() / Long.BYTES;
46     }
47 
48     public void add(int frameIndex, @Transient PageFrame frame) {
49         if (pageAddresses.size() >= columnCount * (frameIndex + 1)) {
50             return; // The page frame is already cached
51         }
52         for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
53             pageAddresses.add(frame.getPageAddress(columnIndex));
54             int varLenColumnIndex = varLenColumnIndexes.getQuick(columnIndex);
55             if (varLenColumnIndex > -1) {
56                 indexPageAddresses.add(frame.getIndexPageAddress(columnIndex));
57                 pageSizes.add(frame.getPageSize(columnIndex));
58             }
59         }
60         pageRowIdOffsets.add(Rows.toRowID(frame.getPartitionIndex(), frame.getPartitionLo()));
61     }
62 
63     @Override
64     public void clear() {
65         varLenColumnIndexes.clear();
66         if (pageAddresses.size() < cacheSizeThreshold) {
67             pageAddresses.clear();
68             indexPageAddresses.clear();
69             pageSizes.clear();
70             pageRowIdOffsets.clear();
71         } else {
72             pageAddresses = new LongList();
73             indexPageAddresses = new LongList();
74             pageSizes = new LongList();
75             pageRowIdOffsets = new LongList();
76         }
77     }
78 
79     public int getColumnCount() {
80         return columnCount;
81     }
82 
83     public long getIndexPageAddress(int frameIndex, int columnIndex) {
84         assert indexPageAddresses.size() >= varLenColumnCount * (frameIndex + 1);
85         int varLenColumnIndex = varLenColumnIndexes.getQuick(columnIndex);
86         assert varLenColumnIndex > -1;
87         return indexPageAddresses.getQuick(varLenColumnCount * frameIndex + varLenColumnIndex);
88     }
89 
90     public long getPageAddress(int frameIndex, int columnIndex) {
91         assert pageAddresses.size() >= columnCount * (frameIndex + 1);
92         return pageAddresses.getQuick(columnCount * frameIndex + columnIndex);
93     }
94 
95     public long getPageSize(int frameIndex, int columnIndex) {
96         assert pageSizes.size() >= varLenColumnCount * (frameIndex + 1);
97         int varLenColumnIndex = varLenColumnIndexes.getQuick(columnIndex);
98         assert varLenColumnIndex > -1;
99         return pageSizes.getQuick(varLenColumnCount * frameIndex + varLenColumnIndex);
100     }
101 
102     public boolean hasColumnTops(int frameIndex) {
103         assert pageAddresses.size() >= columnCount * (frameIndex + 1);
104         for (int columnIndex = 0, baseIndex = columnCount * frameIndex; columnIndex < columnCount; columnIndex++) {
105             if (pageAddresses.getQuick(baseIndex + columnIndex) == 0) {
106                 return true;
107             }
108         }
109         return false;
110     }
111 
112     public void of(@Transient RecordMetadata metadata) {
113         this.columnCount = metadata.getColumnCount();
114         this.varLenColumnIndexes.setAll(columnCount, -1);
115         this.varLenColumnCount = 0;
116         for (int columnIndex = 0; columnIndex < columnCount; columnIndex++) {
117             final int columnType = metadata.getColumnType(columnIndex);
118             if (ColumnType.isVariableLength(columnType)) {
119                 varLenColumnIndexes.setQuick(columnIndex, varLenColumnCount++);
120             }
121         }
122     }
123 
124     public long toTableRowID(int frameIndex, long index) {
125         return pageRowIdOffsets.get(frameIndex) + index;
126     }
127 }
