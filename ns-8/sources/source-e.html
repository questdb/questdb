193     default void toJson(CharSink sink) {
194         sink.put('{');
195         sink.putQuoted("columnCount").put(':').put(getColumnCount());
196         sink.put(',');
197         sink.putQuoted("columns").put(':');
198         sink.put('[');
199         for (int i = 0, n = getColumnCount(); i < n; i++) {
200             final int type = getColumnType(i);
201             if (i > 0) {
202                 sink.put(',');
203             }
204             sink.put('{');
205             sink.putQuoted("index").put(':').put(i).put(',');
206             sink.putQuoted("name").put(':').putQuoted(getColumnName(i)).put(',');
207             sink.putQuoted("type").put(':').putQuoted(ColumnType.nameOf(type));
208             if (isColumnIndexed(i)) {
209                 sink.put(',').putQuoted("indexed").put(":true");
210                 sink.put(',').putQuoted("indexValueBlockCapacity").put(':').put(getIndexValueBlockCapacity(i));
211             }
212             sink.put('}');
213         }
214         sink.put(']');
215         sink.put(',').putQuoted("timestampIndex").put(':').put(getTimestampIndex());
216         sink.put('}');
217     }
