316     private BinarySequence getBin(long base, long offset, long pageLimit, MemoryCR.ByteSequenceView view) {
317         final long address = base + offset;
318         final long len = Unsafe.getUnsafe().getLong(address);
319         if (len != TableUtils.NULL_LEN) {
320             if (len + Long.BYTES + offset <= pageLimit) {
321                 return view.of(address + Long.BYTES, len);
322             }
323             throw CairoException.critical(0)
324                     .put("Bin is outside of file boundary [offset=")
325                     .put(offset)
326                     .put(", len=")
327                     .put(len)
328                     .put(", pageLimit=")
329                     .put(pageLimit)
330                     .put(']');
331         }
332         return null;
333     }

335     private CharSequence getStr(long base, long offset, long size, MemoryCR.CharSequenceView view) {
336         final long address = base + offset;
337         final int len = Unsafe.getUnsafe().getInt(address);
338         if (len != TableUtils.NULL_LEN) {
339             if (len + 4 + offset <= size) {
340                 return view.of(address + Vm.STRING_LENGTH_BYTES, len);
341             }
342             throw CairoException.critical(0)
343                     .put("String is outside of file boundary [offset=")
344                     .put(offset)
345                     .put(", len=")
346                     .put(len)
347                     .put(", size=")
348                     .put(size)
349                     .put(']');
350         }
351         return null;
352     }
