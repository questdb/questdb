41     // Any order means that algorithm is able to work with frames in any order.

52             case ORDER_DESC:
53                 return "backward";
54             default:
55                 return "any";
56         }
57     }

61             case ORDER_ASC:
62                 return ORDER_DESC;
63             case ORDER_DESC:
64                 return ORDER_ASC;
65             default:
66                 return ORDER_ANY;
67         }
68     }
