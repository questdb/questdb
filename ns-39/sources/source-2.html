157         clear();
158 
159         tableToken = event.getTableToken();
160         long readPtr = event.getData();

162         final long hi = readPtr + length;
163 
164         // This is not hot path, do safe deserialization

166             throw CairoException.critical(0).put("invalid alter statement serialized to writer queue [1]");

172         deserializeMem.of(readPtr, length);

177     public void deserializeBody(MemoryCR buffer, long offsetLo, long offsetHi) {
178         clear();
179         long readPtr = offsetLo;
180         if (readPtr + 10 > offsetHi) {
181             throw CairoException
182                     .critical(0)
183                     .put("cannot read alter statement serialized, data is too short to read 10 bytes header [offset=")
184                     .put(offsetLo)
185                     .put(", size=")
186                     .put(offsetHi - offsetLo)
187                     .put(']');
188         }
189         command = buffer.getShort(readPtr);
190         readPtr += 2;
191         tableNamePosition = buffer.getInt(readPtr);
192         readPtr += 4;
193         int longSize = buffer.getInt(readPtr);
194         readPtr += 4;
195         if (longSize < 0 || readPtr + longSize * 8L >= offsetHi) {
196             throw CairoException.critical(0).put("invalid alter statement serialized to writer queue [2]");
197         }
198         extraInfo.clear();
199         for (int i = 0; i < longSize; i++) {
200             extraInfo.add(buffer.getLong(readPtr));
201             readPtr += 8;
202         }
203         directExtraStrInfo.of(buffer, readPtr, offsetHi);
204         activeExtraStrInfo = directExtraStrInfo;
205     }

463         public void of(MemoryCR buffer, long lo, long hi) {
464             if (lo + Integer.BYTES > hi) {
465                 throw CairoException.critical(0).put("invalid alter statement serialized to writer queue [11]");
466             }
467             int size = buffer.getInt(lo);
468             lo += 4;
469             for (int i = 0; i < size; i++) {
470                 if (lo + Integer.BYTES > hi) {
471                     throw CairoException.critical(0).put("invalid alter statement serialized to writer queue [12]");
472                 }
473                 int stringSize = 2 * buffer.getInt(lo);
474                 lo += 4;
475                 if (lo + stringSize > hi) {
476                     throw CairoException.critical(0).put("invalid alter statement serialized to writer queue [13]");
477                 }
478                 long address = buffer.addressOf(lo);
479                 offsets.add(address, address + stringSize);
480                 lo += stringSize;
481             }
482         }
483 
484         @Override
485         public int size() {
486             return offsets.size() / 2;
487         }
