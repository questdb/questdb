127             SqlExecutionContext executionContext,

129             int tableNamePositionInSql,

136         final FanOut writerEventFanOut = engine.getMessageBus().getTableWriterEventFanOut();

143             final CharSequence cmdName = asyncWriterCommand.getCommandName();
144             tableToken = asyncWriterCommand.getTableToken();

146             asyncWriterCommand.setCommandCorrelationId(correlationId);
147 
148             try (TableWriter writer = engine.getWriterOrPublishCommand(
149                     executionContext.getCairoSecurityContext(),
150                     asyncWriterCommand.getTableToken(),

182         if (status == QUERY_COMPLETE) {

184         }
185         status = Math.max(status, awaitWriterEvent(timeout));

187     }

189     private int awaitWriterEvent(long timeout) throws SqlException {
190         assert eventSubSeq != null : "No sequence to wait on";
191         assert correlationId > -1 : "No command id to wait for";
192         assert timeout > 0;
193 
194         final MillisecondClock clock = engine.getConfiguration().getMillisecondClock();
195         final long start = clock.getTicks();
196         final RingQueue<TableWriterTask> tableWriterEventQueue = engine.getMessageBus().getTableWriterEventQueue();
197 
198         int status = this.status;
199         while (true) {
200             long seq = eventSubSeq.next();
201             if (seq < 0) {
202                 if (seq == -1) {
203                     // Queue is empty, check if the execution blocked for too long.
204                     if (clock.getTicks() - start > timeout) {
205                         queryFutureUpdateListener.reportBusyWaitExpired(tableToken, correlationId);
206                         return status;
207                     }
208                 } else {
209                     Os.pause();
210                 }
211                 continue;
212             }
213 
214             try {
215                 TableWriterTask event = tableWriterEventQueue.get(seq);
216                 int type = event.getType();
217                 if (event.getInstance() != correlationId || (type != TSK_BEGIN && type != TSK_COMPLETE)) {
218                     LOG.info()
219                             .$("writer command response received and ignored [instance=").$(event.getInstance())
220                             .$(", type=").$(type)
221                             .$(", expectedInstance=").$(correlationId)
222                             .I$();
223                     Os.pause();
224                 } else if (type == TSK_COMPLETE) {
225                     LOG.info().$("writer command response received [instance=").$(correlationId).I$();
226                     final int code = Unsafe.getUnsafe().getInt(event.getData());
227                     switch (code) {
228                         case OK:
229                             affectedRowsCount = Unsafe.getUnsafe().getInt(event.getData() + Integer.BYTES);
230                             queryFutureUpdateListener.reportProgress(correlationId, QUERY_COMPLETE);
231                             return QUERY_COMPLETE;
232                         case READER_OUT_OF_DATE:
233                             throw TableReferenceOutOfDateException.of(tableToken);
234                         default:
235                             LOG.error().$("error writer command response [instance=").$(correlationId)
236                                     .$(", errorCode=").$(code).I$();
237                             final int strLen = Unsafe.getUnsafe().getInt(event.getData() + Integer.BYTES);
238                             final long strLo = event.getData() + 2L * Integer.BYTES;
239                             if (strLen == 0) {
240                                 throw SqlException.$(tableNamePositionInSql, "statement execution failed");
241                             } else {
242                                 throw SqlException.$(tableNamePositionInSql, strLo, strLo + 2L * strLen);
243                             }
244                     }
245                 } else {
246                     status = QUERY_STARTED;
247                     queryFutureUpdateListener.reportProgress(correlationId, QUERY_STARTED);
248                     LOG.info().$("writer command QUERY_STARTED response received [instance=").$(correlationId).I$();
249                 }
250             } finally {
251                 eventSubSeq.done(seq);
252             }
253         }
254     }
