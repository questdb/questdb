89     @Override
90     public RecordCursor getCursor(SqlExecutionContext executionContext) throws SqlException {
91         final RingQueue<TextImportRequestTask> textImportRequestQueue = messageBus.getTextImportRequestQueue();
92         final MPSequence textImportRequestPubSeq = messageBus.getTextImportRequestPubSeq();
93         final AtomicBooleanCircuitBreaker circuitBreaker = textImportExecutionContext.getCircuitBreaker();
94 
95         long inProgressImportId = textImportExecutionContext.getActiveImportId();
96         if (inProgressImportId == TextImportExecutionContext.INACTIVE) {
97             long processingCursor = textImportRequestPubSeq.next();
98             if (processingCursor > -1) {
99                 final TextImportRequestTask task = textImportRequestQueue.get(processingCursor);
100 
101                 long importId = textImportExecutionContext.assignActiveImportId();
102                 task.of(
103                         importId,
104                         tableName,
105                         fileName,
106                         headerFlag,
107                         timestampColumn,
108                         delimiter,
109                         timestampFormat,
110                         partitionBy,
111                         atomicity
112                 );
113 
114                 circuitBreaker.reset();
115                 textImportRequestPubSeq.done(processingCursor);
116 
117                 importIdSink.clear();
118                 Numbers.appendHex(importIdSink, importId, true);
119                 record.setValue(importIdSink);
120                 cursor.toTop();
121                 return cursor;
122             } else {
123                 throw SqlException.$(0, "Unable to process the import request. Another import request may be in progress.");
124             }
125         }
126 
127         importIdSink.clear();
128         Numbers.appendHex(importIdSink, inProgressImportId, true);
129         throw SqlException.$(0, "Another import request is in progress. ")
130                 .put("[activeImportId=")
131                 .put(importIdSink)
132                 .put(']');
133     }
