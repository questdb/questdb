30 import java.lang.management.GarbageCollectorMXBean;

33 /**
34  * GC metrics don't rely on MetricsRegistry to be able to obtain and write all metrics
35  * to the sink in one go.
36  */
37 public class GCMetrics implements Scrapable {
38 
39     private static final CharSequenceHashSet majorGCNames = new CharSequenceHashSet();
40     private static final CharSequenceHashSet minorGCNames = new CharSequenceHashSet();
41 
42     @Override
43     public void scrapeIntoPrometheus(CharSink sink) {
44         long majorCount = 0;
45         long majorTime = 0;
46         long minorCount = 0;
47         long minorTime = 0;
48         long unknownCount = 0;
49         long unknownTime = 0;
50 
51         for (GarbageCollectorMXBean gc : ManagementFactory.getGarbageCollectorMXBeans()) {
52             long count = gc.getCollectionCount();
53             if (count > -1) {
54                 if (majorGCNames.contains(gc.getName())) {
55                     majorCount += count;
56                     majorTime += gc.getCollectionTime();
57                 } else if (minorGCNames.contains(gc.getName())) {
58                     minorCount += count;
59                     minorTime += gc.getCollectionTime();
60                 } else {
61                     unknownCount += count;
62                     unknownTime += gc.getCollectionTime();
63                 }
64             }
65         }
66 
67         appendCounter(sink, majorCount, "jvm_major_gc_count");
68         appendCounter(sink, majorTime, "jvm_major_gc_time");
69         appendCounter(sink, minorCount, "jvm_minor_gc_count");
70         appendCounter(sink, minorTime, "jvm_minor_gc_time");
71         appendCounter(sink, unknownCount, "jvm_unknown_gc_count");
72         appendCounter(sink, unknownTime, "jvm_unknown_gc_time");
73     }
74 
75     private void appendCounter(CharSink sink, long value, String name) {
76         PrometheusFormatUtils.appendCounterType(name, sink);
77         PrometheusFormatUtils.appendCounterNamePrefix(name, sink);
78         PrometheusFormatUtils.appendSampleLineSuffix(sink, value);
79         PrometheusFormatUtils.appendNewLine(sink);
80     }
81

103         minorGCNames.add("scavenge");
104     }
