43     abstract static class AbstractTimestampFloorFunction extends TimestampFunction implements UnaryFunction {
44         private final Function arg;
45 
46         public AbstractTimestampFloorFunction(Function arg) {
47             this.arg = arg;
48         }
49 
50         @Override
51         public Function getArg() {
52             return arg;
53         }
54 
55         @Override
56         public final long getTimestamp(Record rec) {
57             long micros = arg.getTimestamp(rec);
58             return micros == Numbers.LONG_NaN ? Numbers.LONG_NaN : floor(micros);
59         }
60 
61         @Override
62         public void toPlan(PlanSink sink) {
63             sink.val("timestamp_floor('").val(getUnit()).val("',").val(getArg()).val(')');
64         }
65 
66         abstract protected long floor(long timestamp);
67 
68         abstract CharSequence getUnit();
69     }
