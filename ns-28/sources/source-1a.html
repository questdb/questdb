51             ObjList<Function> args,
52             IntList argPositions,
53             CairoConfiguration configuration,
54             SqlExecutionContext sqlExecutionContext
55     ) {
56         if (args.getQuick(0).isConstant()) {
57             final long start = args.getQuick(0).getTimestamp(null);
58             if (start == Numbers.LONG_NaN) {
59                 return TimestampConstant.NULL;
60             }
61             return new TimestampSequenceFunction(start, args.getQuick(1));
62         } else {
63             return new TimestampSequenceVariableFunction(args.getQuick(0), args.getQuick(1));
64         }
65     }
