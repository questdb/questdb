73     private void rotateLeft(long p) {
74         if (p != -1) {
75             final long r = rightOf(p);
76             final long lr = leftOf(r);
77             setRight(p, lr);
78             if (lr != -1) {
79                 setParent(lr, p);
80             }
81             final long pp = parentOf(p);
82             setParent(r, pp);
83             if (pp == -1) {
84                 root = r;
85             } else if (leftOf(pp) == p) {
86                 setLeft(pp, r);
87             } else {
88                 setRight(pp, r);
89             }
90             setLeft(r, p);
91             setParent(p, r);
92         }
93     }

162     protected static long successor(long current) {
163         long p = rightOf(current);
164         if (p != -1) {
165             long l;
166             while ((l = leftOf(p)) != -1) {

213         }
214 
215         boolean isLeftChild = parent != EMPTY && leftOf(parent) == node;
216

344 
345         long nodeToRemove;
346         if (leftOf(node) == EMPTY || rightOf(node) == EMPTY) {
347             nodeToRemove = node;
348         } else {

351

353         long parent = parentOf(nodeToRemove);
354         if (current != EMPTY) {
355             setParent(current, parent);
356         }
357 
358         if (parent == EMPTY) {
359             root = current;
360         } else {
361             if (leftOf(parent) == nodeToRemove) {
362                 setLeft(parent, current);
363             } else {
364                 setRight(parent, current);
365             }
366         }
367

371             setRef(node, tmp);
