142         public void of(RecordCursorFactory base, SqlExecutionContext executionContext) throws SqlException {
143             //we can't use getCursor() because that could take a lot of time and execute e.g. table hashing
144             //on the other hand until we run it factories may be incomplete
145             if (!isBaseClosed) {
146                 planSink.of(base, executionContext);
147                 base.close();//close base factory and associated cursors, otherwise it may keep holding eagerly allocated memory

152         }
