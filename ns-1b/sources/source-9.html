186         private void countLimit() {
187             if (lo < 0 && hiFunction == null) {
188                 // last N rows
189                 countRows();
190 
191 
192                 // lo is negative, -5 for example
193                 // if we have 12 records, we need to skip 12-5 = 7
194                 // if we have 4 records, return all of them
195                 if (rowCount > -lo) {
196                     skipRows(rowCount + lo);
197                 } else {
198                     base.toTop();
199                 }
200                 // set limit to return remaining rows
201                 limit = Math.min(rowCount, -lo);
202                 size = limit;
203             } else if (lo > -1 && hiFunction == null) {
204                 // first N rows
205                 long baseRowCount = base.size();
206                 if (baseRowCount > -1) { // we don't want to cause a pass-through whole data set
207                     limit = Math.min(baseRowCount, lo);
208                 } else {
209                     limit = lo;
210                 }
211                 size = limit;
212             } else {
213                 // at this stage we have 'hi'
214                 if (lo < 0) {
215                     // right, here we are looking for something like
216                     // -10,-5 five rows away from tail
217 
218                     if (lo < hi) {
219                         countRows();
220                         // when count < -hi we have empty cursor
221                         if (rowCount >= -hi) {
222                             if (rowCount < -lo) {
223                                 base.toTop();
224                                 // if we asked for -9,-4 but there are 7 records in cursor
225                                 // we would first ignore last 4 and return first 3
226                                 limit = rowCount + hi;
227                             } else {
228                                 skipRows(rowCount + lo);
229                                 limit = Math.min(rowCount, -lo + hi);
230                             }
231                             size = limit;
232                         }
233                     } else {
234                         // this is invalid bottom range, for example -3, -10
235                         limit = 0;
236                         size = 0;
237                     }
238                 } else {
239                     if (hi < 0) {
240                         countRows();
241                         limit = Math.max(rowCount - lo + hi, 0);
242                         size = limit;
243 
244                         if (lo > 0 && limit > 0) {
245                             skipRows(lo);
246                         } else {
247                             base.toTop();
248                         }
249                     } else {
250                         long baseRowCount = base.size();
251                         if (baseRowCount > -1L) { // we don't want to cause a pass-through whole data set
252                             limit = Math.max(0, Math.min(baseRowCount, hi) - lo);
253                         } else {
254                             limit = Math.max(0, hi - lo); // doesn't handle hi exceeding number of rows
255                         }
256                         size = limit;
257 
258                         if (lo > 0 && limit > 0) {
259                             skipRows(lo);
260                         }
261                     }
262                 }
263             }
264         }

266         private void countRows() {
267             if (rowCount == -1) {
268                 rowCount = base.size();
269                 if (rowCount > -1) {
270                     areRowsCounted = true;
271                     return;
272                 }
273                 rowCount = 0;
274             }
275 
276             if (!areRowsCounted) {
277                 while (base.hasNext()) {
278                     rowCount++;
279                 }
280                 areRowsCounted = true;
281             }
282         }

284         private void skipRows(long rowCount) {
285             if (skipToRows == -1) {
286                 skipToRows = Math.max(0, rowCount);
287                 base.toTop();
288             }
289             if (skipToRows > 0) {
290                 if (base.skipTo(rowCount)) {
291                     skipToRows = 0;
292                 }
293                 while (skipToRows > 0 && base.hasNext()) {
294                     skipToRows--;
295                 }
296             }
297         }
