98     public long applyWAL(
99             TableToken tableToken,
100             CairoEngine engine,
101             OperationCompiler operationCompiler,
102             Job.RunStatus runStatus
103     ) {
104         long lastSequencerTxn = -1;
105         long lastWriterTxn = -1;
106         Path tempPath = Path.PATH.get();
107 
108         try {
109             do {
110                 // security context is checked on writing to the WAL and can be ignored here
111                 TableToken updatedToken = engine.getUpdatedTableToken(tableToken);
112                 if (updatedToken == null) {
113                     if (engine.isTableDropped(tableToken)) {
114                         return tryDestroyDroppedTable(tableToken, null, engine, tempPath) ? Long.MAX_VALUE : -1;
115                     }
116                     // else: table is dropped and fully cleaned, this is late notification.
117                     return Long.MAX_VALUE;
118                 }
119 
120                 if (!engine.isWalTable(tableToken)) {
121                     LOG.info().$("table '").utf8(tableToken.getDirName()).$("' does not exist, skipping WAL application").$();
122                     return 0;
123                 }
124 
125                 rowsSinceLastCommit = 0;
126                 try (TableWriter writer = engine.getWriterUnsafe(updatedToken, WAL_2_TABLE_WRITE_REASON, false)) {
127                     assert writer.getMetadata().getTableId() == tableToken.getTableId();
128                     applyOutstandingWalTransactions(tableToken, writer, engine, operationCompiler, tempPath, runStatus);
129                     lastWriterTxn = writer.getSeqTxn();
130                 } catch (EntryUnavailableException tableBusy) {
131                     if (!WAL_2_TABLE_WRITE_REASON.equals(tableBusy.getReason()) && !WAL_2_TABLE_RESUME_REASON.equals(tableBusy.getReason())) {
132                         LOG.critical().$("unsolicited table lock [table=").utf8(tableToken.getDirName()).$(", lock_reason=").$(tableBusy.getReason()).I$();
133                         // Don't suspend table. Perhaps writer will be unlocked with no transaction applied.
134                     }
135                     // This is good, someone else will apply the data
136                     break;
137                 }
138 
139                 lastSequencerTxn = engine.getTableSequencerAPI().lastTxn(tableToken);
140             } while (lastWriterTxn < lastSequencerTxn && !runStatus.isTerminating());
141         } catch (CairoException ex) {
142             if (engine.isTableDropped(tableToken)) {
143                 // Table is dropped, and we received cairo exception in the middle of apply
144                 return tryDestroyDroppedTable(tableToken, null, engine, tempPath) ? Long.MAX_VALUE : -1;
145             }
146             telemetryFacade.store(TelemetryOrigin.WAL_APPLY, WAL_APPLY_SUSPEND);
147             LOG.critical().$("WAL apply job failed, table suspended [table=").utf8(tableToken.getDirName())
148                     .$(", error=").$(ex.getFlyweightMessage())
149                     .$(", errno=").$(ex.getErrno())
150                     .I$();
151             return WAL_APPLY_FAILED;
152         }
153         assert lastWriterTxn == lastSequencerTxn || runStatus.isTerminating();
154 
155         return lastWriterTxn;
156     }

164     private static boolean cleanDroppedTableDirectory(CairoEngine engine, Path tempPath, TableToken tableToken) {
165         // Clean all the files inside table folder name except WAL directories and SEQ_DIR directory
166         boolean allClean = true;
167         FilesFacade ff = engine.getConfiguration().getFilesFacade();
168         tempPath.of(engine.getConfiguration().getRoot()).concat(tableToken);
169         int rootLen = tempPath.length();
170 
171         long p = ff.findFirst(tempPath.$());
172         if (p > 0) {
173             try {
174                 do {
175                     long pUtf8NameZ = ff.findName(p);
176                     int type = ff.findType(p);
177                     if (ff.isDirOrSoftLinkDirNoDots(tempPath, rootLen, pUtf8NameZ, type)) {
178                         if (!Chars.endsWith(tempPath, SEQ_DIR) && !Chars.equals(tempPath, rootLen + 1, rootLen + 1 + WAL_NAME_BASE.length(), WAL_NAME_BASE, 0, WAL_NAME_BASE.length())) {
179                             if (ff.unlinkOrRemove(tempPath, LOG) != 0) {
180                                 allClean = false;
181                             }
182                         }
183 
184                     } else if (type == Files.DT_FILE) {
185                         tempPath.trimTo(rootLen);
186                         tempPath.concat(pUtf8NameZ);
187 
188                         if (Chars.endsWith(tempPath, TableUtils.TXN_FILE_NAME) || Chars.endsWith(tempPath, TableUtils.META_FILE_NAME) || matchesWalLock(tempPath)) {
189                             continue;
190                         }
191 
192                         if (!ff.remove(tempPath.$())) {
193                             allClean = false;
194                             LOG.info().$("could not remove [tempPath=").utf8(tempPath).$(", errno=").$(ff.errno()).I$();
195                         }
196                     }
197                 } while (ff.findNext(p) > 0);
198 
199                 if (allClean) {
200                     // Remove _txn and _meta files when all other files are removed
201                     ff.remove(tempPath.trimTo(rootLen).concat(TableUtils.TXN_FILE_NAME).$());
202                     ff.remove(tempPath.trimTo(rootLen).concat(TableUtils.META_FILE_NAME).$());
203                     return true;
204                 }
205             } finally {
206                 ff.findClose(p);
207             }
208         }
209         return false;
210     }

230     private static boolean matchesWalLock(CharSequence name) {
231         if (Chars.endsWith(name, ".lock")) {
232             for (int i = name.length() - ".lock".length() - 1; i > 0; i--) {
233                 char c = name.charAt(i);
234                 if (c < '0' || c > '9') {
235                     return Chars.equals(name, i - WAL_NAME_BASE.length() + 1, i + 1, WAL_NAME_BASE, 0, WAL_NAME_BASE.length());
236                 }
237             }
238         }
239 
240         for (int i = 0, n = name.length(); i < n; i++) {
241             char c = name.charAt(i);
242             if (c < '0' || c > '9') {
243                 return false;
244             }
245         }
246         return true;
247     }

249     private static boolean tryDestroyDroppedTable(TableToken tableToken, TableWriter writer, CairoEngine engine, Path tempPath) {
250         if (engine.lockReadersByTableToken(tableToken)) {
251             TableWriter writerToClose = null;
252             try {
253                 final CairoConfiguration configuration = engine.getConfiguration();
254                 if (writer == null && TableUtils.exists(configuration.getFilesFacade(), tempPath, configuration.getRoot(), tableToken.getDirName()) == TABLE_EXISTS) {
255                     try {
256                         writer = writerToClose = engine.getWriterUnsafe(tableToken, WAL_2_TABLE_WRITE_REASON, false);
257                     } catch (CairoException ex) {
258                         // Ignore it, table can be half deleted.
259                     }
260                 }
261                 if (writer != null) {
262                     // Force writer to close all the files.
263                     writer.destroy();
264                 }
265                 return cleanDroppedTableDirectory(engine, tempPath, tableToken);
266             } finally {
267                 if (writerToClose != null) {
268                     writerToClose.close();
269                 }
270                 engine.releaseReadersByTableToken(tableToken);
271             }
272         } else {
273             LOG.info().$("table '").utf8(tableToken.getDirName())
274                     .$("' is dropped, waiting to acquire Table Readers lock to delete the table files").$();
275         }
276         return false;
277     }

279     private void applyOutstandingWalTransactions(
280             TableToken tableToken,
281             TableWriter writer,
282             CairoEngine engine,
283             OperationCompiler operationCompiler,
284             Path tempPath,
285             Job.RunStatus runStatus
286     ) {
287         boolean isTerminating;
288         try (TransactionLogCursor transactionLogCursor = tableSequencerAPI.getCursor(tableToken, writer.getSeqTxn())) {
289             TableMetadataChangeLog structuralChangeCursor = null;
290 
291             try {
292 
293                 int iTransaction = 0;
294                 int totalTransactionCount = 0;
295                 long rowsAdded = 0;
296                 long physicalRowsAdded = 0;
297                 long insertTimespan = 0;
298 
299                 tempPath.of(engine.getConfiguration().getRoot()).concat(tableToken).slash();
300                 int rootLen = tempPath.length();
301 
302                 // Populate transactionMeta with timestamps of future transactions
303                 // to avoid O3 commits by pre-calculating safe to commit timestamp for every commit.
304                 LongList transactionMeta = readObservableTxnMeta(tempPath, transactionLogCursor, rootLen, writer.getMaxTimestamp());
305                 transactionLogCursor.toTop();
306 
307                 isTerminating = runStatus.isTerminating();
308                 WHILE_TRANSACTION_CURSOR:
309                 while (transactionLogCursor.hasNext() && !isTerminating) {
310                     final int walId = transactionLogCursor.getWalId();
311                     final int segmentId = transactionLogCursor.getSegmentId();
312                     final long segmentTxn = transactionLogCursor.getSegmentTxn();
313                     final long commitTimestamp = transactionLogCursor.getCommitTimestamp();
314                     final long seqTxn = transactionLogCursor.getTxn();
315 
316                     if (seqTxn != writer.getSeqTxn() + 1) {
317                         throw CairoException.critical(0)
318                                 .put("unexpected sequencer transaction, expected ").put(writer.getSeqTxn() + 1)
319                                 .put(" but was ").put(seqTxn);
320                     }
321 
322                     switch (walId) {
323                         case METADATA_WALID:
324                             // This is metadata change
325                             // to be taken from Sequencer directly
326                             final long newStructureVersion = transactionLogCursor.getStructureVersion();
327                             if (writer.getStructureVersion() != newStructureVersion - 1) {
328                                 throw CairoException.critical(0)
329                                         .put("unexpected new WAL structure version [walStructure=").put(newStructureVersion)
330                                         .put(", tableStructureVersion=").put(writer.getStructureVersion())
331                                         .put(']');
332                             }
333 
334                             boolean hasNext;
335                             if (structuralChangeCursor == null || !(hasNext = structuralChangeCursor.hasNext())) {
336                                 Misc.free(structuralChangeCursor);
337                                 structuralChangeCursor = tableSequencerAPI.getMetadataChangeLog(tableToken, newStructureVersion - 1);
338                                 hasNext = structuralChangeCursor.hasNext();
339                             }
340 
341                             if (hasNext) {
342                                 final long start = microClock.getTicks();
343                                 walTelemetryFacade.store(WAL_TXN_APPLY_START, tableToken, walId, seqTxn, -1L, -1L, start - commitTimestamp);
344                                 structuralChangeCursor.next().apply(writer, true);
345                                 writer.setSeqTxn(seqTxn);
346                                 walTelemetryFacade.store(WAL_TXN_STRUCTURE_CHANGE_APPLIED, tableToken, walId, seqTxn, -1L, -1L, microClock.getTicks() - start);
347                             } else {
348                                 // Something messed up in sequencer.
349                                 // There is a transaction in WAL but no structure change record.
350                                 throw CairoException.critical(0)
351                                         .put("could not apply structure change from WAL to table. WAL metadata change does not exist [structureVersion=")
352                                         .put(newStructureVersion)
353                                         .put(']');
354                             }
355                             break;
356 
357                         case DROP_TABLE_WALID:
358                             tryDestroyDroppedTable(tableToken, writer, engine, tempPath);
359                             return;
360 
361                         case 0:
362                             throw CairoException.critical(0)
363                                     .put("broken table transaction record in sequencer log, walId cannot be 0 [table=")
364                                     .put(tableToken.getTableName()).put(", seqTxn=").put(seqTxn).put(']');
365 
366                         default:
367                             // Always set full path when using thread static path
368                             operationCompiler.setNowAndFixClock(commitTimestamp);
369                             tempPath.of(engine.getConfiguration().getRoot()).concat(tableToken).slash().put(WAL_NAME_BASE).put(walId).slash().put(segmentId);
370                             final long start = microClock.getTicks();
371 
372                             if (iTransaction > 0 && transactionMeta.size() < (iTransaction + lookAheadTransactionCount) * TXN_METADATA_LONGS_SIZE) {
373                                 // Last few transactions left to process from the list
374                                 // of observed transactions built upfront in the beginning of the loop.
375                                 // Check if more transaction exist, exit restart the loop to have better picture
376                                 // of the future transactions and optimise the application.
377                                 if (transactionLogCursor.reset()) {
378                                     transactionMeta = readObservableTxnMeta(tempPath, transactionLogCursor, rootLen, writer.getMaxTimestamp());
379                                     transactionLogCursor.toTop();
380                                     totalTransactionCount += iTransaction;
381                                     iTransaction = 0;
382                                     continue;
383                                 }
384                             }
385 
386                             isTerminating = runStatus.isTerminating();
387                             final long added = processWalCommit(
388                                     writer,
389                                     walId,
390                                     tempPath,
391                                     segmentTxn,
392                                     operationCompiler,
393                                     seqTxn,
394                                     commitTimestamp,
395                                     isTerminating,
396                                     transactionMeta,
397                                     iTransaction * TXN_METADATA_LONGS_SIZE
398                             );
399 
400                             if (added > -1L) {
401                                 insertTimespan += microClock.getTicks() - start;
402                                 rowsAdded += added;
403                                 iTransaction++;
404                                 physicalRowsAdded += writer.getPhysicallyWrittenRowsSinceLastCommit();
405                             }
406                             if (added == -2L || isTerminating) {
407                                 // transaction cursor goes beyond prepared transactionMeta or termination requested. Re-run the loop.
408                                 break WHILE_TRANSACTION_CURSOR;
409                             }
410                     }
411                 }
412                 totalTransactionCount += iTransaction;
413                 if (totalTransactionCount > 0) {
414                     LOG.info().$("WAL apply job finished [table=").$(writer.getTableToken())
415                             .$(", transactions=").$(totalTransactionCount)
416                             .$(", rows=").$(rowsAdded)
417                             .$(", time=").$(insertTimespan / 1000)
418                             .$("ms, rate=").$(rowsAdded * 1000000L / Math.max(1, insertTimespan))
419                             .$("rows/s, physicalWrittenRowsMultiplier=").$(Math.round(100.0 * physicalRowsAdded / rowsAdded) / 100.0)
420                             .I$();
421                 }
422             } finally {
423                 Misc.free(structuralChangeCursor);
424             }
425         }
426     }

436     private long processWalCommit(
437             TableWriter writer,
438             int walId,
439             @Transient Path walPath,
440             long segmentTxn,
441             OperationCompiler operationCompiler,
442             long seqTxn,
443             long commitTimestamp,
444             boolean isTerminating,
445             LongList minTimestamps,
446             int minTimestampsIndex
447     ) {
448         try (WalEventReader eventReader = walEventReader) {
449             final WalEventCursor walEventCursor = eventReader.of(walPath, WAL_FORMAT_VERSION, segmentTxn);
450             final byte walTxnType = walEventCursor.getType();
451             switch (walTxnType) {
452                 case DATA:
453                     final WalEventCursor.DataInfo dataInfo = walEventCursor.getDataInfo();
454                     if (minTimestampsIndex < minTimestamps.size()) {
455                         long commitToTimestamp = minTimestamps.getQuick(minTimestampsIndex);
456                         long rowCount = dataInfo.getEndRowID() - dataInfo.getStartRowID();
457                         if (commitToTimestamp < 0 || isTerminating) {
458                             // commit everything, do not store data in memory LAG buffer
459                             commitToTimestamp = Long.MAX_VALUE;
460                             rowsSinceLastCommit = 0;
461                         } else {
462                             rowsSinceLastCommit += rowCount;
463                             if (rowsSinceLastCommit < commitSquashRowLimit) {
464                                 // This is an optimisation to apply small commits.
465                                 // We want to store data in memory LAG buffer and commit it later when the buffer size is reasonably big
466                                 // or when there is no more data available in WAL.
467                                 // Do not commit yet, copy to LAG memory buffer and wait for more rows
468                                 commitToTimestamp = -1;
469                             }
470                         }
471                         final long start = microClock.getTicks();
472                         walTelemetryFacade.store(WAL_TXN_APPLY_START, writer.getTableToken(), walId, seqTxn, -1L, -1L, start - commitTimestamp);
473                         final long rowsAdded = writer.processWalData(
474                                 walPath,
475                                 !dataInfo.isOutOfOrder(),
476                                 dataInfo.getStartRowID(),
477                                 dataInfo.getEndRowID(),
478                                 dataInfo.getMinTimestamp(),
479                                 dataInfo.getMaxTimestamp(),
480                                 dataInfo,
481                                 seqTxn,
482                                 commitToTimestamp
483                         );
484                         rowsSinceLastCommit -= rowsAdded;
485                         final long latency = microClock.getTicks() - start;
486                         long physicalRowCount = writer.getPhysicallyWrittenRowsSinceLastCommit();
487                         metrics.addApplyRowsWritten(rowCount, physicalRowCount, latency);
488                         walTelemetryFacade.store(WAL_TXN_DATA_APPLIED, writer.getTableToken(), walId, seqTxn, rowsAdded, physicalRowCount, latency);
489                         return rowCount;
490                     } else {
491                         return -2L;
492                     }
493 
494                 case SQL:
495                     final WalEventCursor.SqlInfo sqlInfo = walEventCursor.getSqlInfo();
496                     final long start = microClock.getTicks();
497                     walTelemetryFacade.store(WAL_TXN_APPLY_START, writer.getTableToken(), walId, seqTxn, -1L, -1L, start - commitTimestamp);
498                     processWalSql(writer, sqlInfo, operationCompiler, seqTxn);
499                     walTelemetryFacade.store(WAL_TXN_SQL_APPLIED, writer.getTableToken(), walId, seqTxn, -1L, -1L, microClock.getTicks() - start);
500                     return -1L;
501                 case TRUNCATE:
502                     long txn = writer.getTxn();
503                     writer.setSeqTxn(seqTxn);
504                     writer.removeAllPartitions();
505                     if (writer.getTxn() == txn) {
506                         // force mark the transaction as applied
507                         writer.markSeqTxnCommitted(seqTxn);
508                     }
509                     return -1L;
510                 default:
511                     throw new UnsupportedOperationException("Unsupported WAL txn type: " + walTxnType);
512             }
513         }
514     }

516     private void processWalSql(TableWriter tableWriter, WalEventCursor.SqlInfo sqlInfo, OperationCompiler operationCompiler, long seqTxn) {
517         final int cmdType = sqlInfo.getCmdType();
518         final CharSequence sql = sqlInfo.getSql();
519         operationCompiler.resetRnd(sqlInfo.getRndSeed0(), sqlInfo.getRndSeed1());
520         sqlInfo.populateBindVariableService(operationCompiler.getBindVariableService());
521         try {
522             switch (cmdType) {
523                 case CMD_ALTER_TABLE:
524                     AlterOperation alterOperation = compileAlter(tableWriter, operationCompiler, sql, seqTxn);
525                     try {
526                         tableWriter.apply(alterOperation, seqTxn);
527                     } finally {
528                         Misc.free(alterOperation);
529                     }
530                     break;
531                 case CMD_UPDATE_TABLE:
532                     UpdateOperation updateOperation = compileUpdate(tableWriter, operationCompiler, sql, seqTxn);
533                     try {
534                         tableWriter.apply(updateOperation, seqTxn);
535                     } finally {
536                         Misc.free(updateOperation);
537                     }
538                     break;
539                 default:
540                     throw new UnsupportedOperationException("Unsupported command type: " + cmdType);
541             }
542         } catch (SqlException ex) {
543             // This is fine, some syntax error, we should not block WAL processing if SQL is not valid
544             LOG.error().$("error applying SQL to wal table [table=")
545                     .utf8(tableWriter.getTableToken().getTableName()).$(", sql=").$(sql).$(", error=").$(ex.getFlyweightMessage()).I$();

547             if (e.isWALTolerable()) {
548                 // This is fine, some syntax error, we should not block WAL processing if SQL is not valid
549                 LOG.error().$("error applying SQL to wal table [table=")
550                         .utf8(tableWriter.getTableToken().getTableName()).$(", sql=").$(sql).$(", error=").$(e.getFlyweightMessage()).I$();

553             }

557     private LongList readObservableTxnMeta(Path tempPath, TransactionLogCursor transactionLogCursor, int rootLen, long maxCommittedTimestamp) {
558         try (WalEventReader eventReader = walEventReader) {
559             transactionMeta.clear();
560             int prevWalId = Integer.MIN_VALUE;
561             int prevSegmentId = Integer.MIN_VALUE;
562             int prevSegmentTxn = Integer.MIN_VALUE;
563             WalEventCursor walEventCursor = null;
564 
565             while (transactionLogCursor.hasNext()) {
566 
567                 final int walId = transactionLogCursor.getWalId();
568                 final int segmentId = transactionLogCursor.getSegmentId();
569                 final int segmentTxn = transactionLogCursor.getSegmentTxn();
570 
571                 boolean recordAdded = false;
572                 if (walId > 0) {
573                     tempPath.trimTo(rootLen).put(WAL_NAME_BASE).put(walId).slash().put(segmentId);
574 
575                     if (prevWalId != walId || prevSegmentId != segmentId || prevSegmentTxn + 1 != segmentTxn) {
576                         walEventCursor = eventReader.of(tempPath, WAL_FORMAT_VERSION, segmentTxn);
577                     } else {
578                         // This is same WALE file, just read next txn transaction.
579                         if (!walEventCursor.hasNext()) {
580                             walEventCursor = eventReader.of(tempPath, WAL_FORMAT_VERSION, segmentTxn);
581                         }
582                     }
583 
584                     final byte walTxnType = walEventCursor.getType();
585                     if (walTxnType == DATA) {
586                         recordAdded = true;
587                         WalEventCursor.DataInfo commitInfo = walEventCursor.getDataInfo();
588                         transactionMeta.add(-1); // commit to timestamp
589                         transactionMeta.add(commitInfo.getMaxTimestamp());
590                         transactionMeta.add(commitInfo.getMinTimestamp());
591                     }
592                 }
593                 prevWalId = walId;
594                 prevSegmentId = segmentId;
595                 prevSegmentTxn = segmentTxn;
596 
597                 // This is a structural change, UPDATE or non-structural ALTER
598                 // when it happens in between the transactions, we want to commit everything before it.
599                 if (!recordAdded && transactionMeta.size() > 0) {
600                     transactionMeta.setQuick(transactionMeta.size() - 3, FORCE_FULL_COMMIT); // commit to timestamp of prev record
601                 }
602             }
603         }
604 
605         // find min timestamp after every transaction
606         long runningMinTimestamp = Long.MAX_VALUE;
607         long maxLag = 0;
608         for (int n = transactionMeta.size(), i = n - 1; i > -1; i -= TXN_METADATA_LONGS_SIZE) {
609 
610             long currentMinTimestamp = transactionMeta.getQuick(i);
611             long currentMaxTimestamp = transactionMeta.getQuick(i - 1);
612 
613             long nextMinTimestamp = Math.min(currentMaxTimestamp, runningMinTimestamp);
614             long lag = currentMaxTimestamp - nextMinTimestamp;
615             if (lag > maxLag) {
616                 maxLag = lag;
617             }
618 
619             runningMinTimestamp = Math.min(runningMinTimestamp, currentMinTimestamp);
620             // No point to hold data in lag buffer if it's already intersects with committed data
621             transactionMeta.setQuick(i, Math.max(maxCommittedTimestamp, runningMinTimestamp));
622 
623             // Leave last commitToTimestamp as -1 so everything is committed at the end
624             if (i < n - 1) {
625                 long commitToTimestamp = transactionMeta.getQuick(i - 2);
626                 if (commitToTimestamp != FORCE_FULL_COMMIT) {
627                     // set commitToTimestamp to be nextMinTimestamp
628                     // so that O3 does not happen
629                     transactionMeta.setQuick(i - 2, nextMinTimestamp);
630                 } else {
631                     // This is a flag that the commit has to be done in full
632                     // because of following UPDATE or ALTER.
633                     // Everything will be committed at this point, so it's safe to reset runningMinTimestamp
634                     runningMinTimestamp = Long.MAX_VALUE;
635                 }
636             }
637         }
638         return transactionMeta;
639     }

647     @Override
648     protected boolean doRun(int workerId, long cursor, RunStatus runStatus) {
649         final TableToken tableToken;
650         final long seqTxn;
651 
652         try {
653             WalTxnNotificationTask task = queue.get(cursor);
654             tableToken = task.getTableToken();
655             seqTxn = task.getTxn();
656         } finally {
657             // Don't hold the queue until the all the transactions applied to the table
658             subSeq.done(cursor);
659         }
660 
661         final int tableId = tableToken.getTableId();
662         if (lastAppliedSeqTxns.get(tableId) < seqTxn) {
663             // Check, maybe we already processed this table to higher txn.
664             final long lastAppliedSeqTxn = applyWAL(tableToken, engine, operationCompiler, runStatus);
665             if (lastAppliedSeqTxn > -1L) {
666                 lastAppliedSeqTxns.put(tableId, lastAppliedSeqTxn);
667             } else if (lastAppliedSeqTxn == WAL_APPLY_FAILED) {
668                 // Set processed transaction marker as Long.MAX_VALUE - 1
669                 // so that when the table is unsuspended it's notified with transaction Long.MAX_VALUE
670                 // and got picked up for processing in this apply job.
671                 lastAppliedSeqTxns.put(tableId, Long.MAX_VALUE - 1);
672                 engine.getTableSequencerAPI().suspendTable(tableToken);
673             }
674         } else {
675             LOG.debug().$("Skipping WAL processing for table, already processed [table=").$(tableToken).$(", txn=").$(seqTxn).I$();
676         }
677         return true;
678     }
