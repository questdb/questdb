56         long o = offset;
57         while (true) {
58             if (memSize >= o + Integer.BYTES) {
59                 final int value = eventMem.getInt(o);
60                 o += Integer.BYTES;
61                 if (value == SymbolMapDiffImpl.END_OF_SYMBOL_ENTRIES) {
62                     offset = o;

68                 o += storageLength;
69             } else {
70                 throw CairoException.critical(0).put("WAL event file is too small, size=").put(memSize)
71                         .put(", required=").put(o + Integer.BYTES);
72             }
