44 public class OperationCompiler implements Closeable {
45     private final BindVariableService bindVariableService;
46     private final Rnd rnd;
47     private final SqlCompiler sqlCompiler;
48     private final WalSqlExecutionContextImpl sqlExecutionContext;
49 
50     public OperationCompiler(
51             CairoEngine engine,
52             int workerCount,
53             int sharedWorkerCount,
54             @Nullable FunctionFactoryCache functionFactoryCache
55     ) {
56         rnd = new Rnd();
57         bindVariableService = new BindVariableServiceImpl(engine.getConfiguration());
58         sqlExecutionContext = new WalSqlExecutionContextImpl(
59                 engine,
60                 workerCount,
61                 sharedWorkerCount
62         );
63         sqlExecutionContext.with(
64                 AllowAllCairoSecurityContext.INSTANCE,
65                 bindVariableService,
66                 rnd,
67                 -1,
68                 null
69         );
70         this.sqlCompiler = new SqlCompiler(engine, functionFactoryCache, null);
71     }
72 
73     @Override
74     public void close() {
75         Misc.free(sqlCompiler);
76         Misc.free(sqlExecutionContext);
77     }
78 
79     public AlterOperation compileAlterSql(CharSequence alterSql, TableToken tableToken) throws SqlException {
80         sqlExecutionContext.remapTableNameResolutionTo(tableToken);
81         final CompiledQuery compiledQuery = sqlCompiler.compile(alterSql, sqlExecutionContext);
82         final AlterOperation alterOp = compiledQuery.getAlterOperation();
83         alterOp.withContext(sqlExecutionContext);
84         return alterOp;
85     }
86 
87     public UpdateOperation compileUpdateSql(CharSequence updateSql, TableToken tableToken) throws SqlException {
88         sqlExecutionContext.remapTableNameResolutionTo(tableToken);
89         final CompiledQuery compiledQuery = sqlCompiler.compile(updateSql, sqlExecutionContext);
90         final UpdateOperation updateOperation = compiledQuery.getUpdateOperation();
91         updateOperation.withContext(sqlExecutionContext);
92         return updateOperation;
93     }
