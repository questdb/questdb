35 public class CheckWalTransactionsJob extends SynchronizedJob {
36     private final TableSequencerAPI.RegisteredTable checkNotifyOutstandingTxnInWal;
37     private final CharSequence dbRoot;
38     private final CairoEngine engine;
39     private final FilesFacade ff;
40     private final MillisecondClock millisecondClock;
41     private final long spinLockTimeout;
42     private final ObjList<TableToken> tableTokenBucket = new ObjList<>();
43     private final TxReader txReader;
44     private long lastProcessedCount = 0;
45     private Path threadLocalPath;

47     public CheckWalTransactionsJob(CairoEngine engine) {
48         this.engine = engine;
49         this.ff = engine.getConfiguration().getFilesFacade();
50         txReader = new TxReader(engine.getConfiguration().getFilesFacade());
51         dbRoot = engine.getConfiguration().getRoot();
52         millisecondClock = engine.getConfiguration().getMillisecondClock();
53         spinLockTimeout = engine.getConfiguration().getSpinLockTimeout();
54         checkNotifyOutstandingTxnInWal = (tableToken, txn, txn2) -> checkNotifyOutstandingTxnInWal(txn, txn2);
55     }

57     public void checkMissingWalTransactions() {
58         threadLocalPath = Path.PATH.get().of(dbRoot);
59         engine.getTableSequencerAPI().forAllWalTables(tableTokenBucket, true, checkNotifyOutstandingTxnInWal);
60     }

62     public void checkNotifyOutstandingTxnInWal(TableToken tableToken, long txn) {
63         if (
64                 txn < 0 && TableUtils.exists(
65                         ff,
66                         threadLocalPath,
67                         dbRoot,
68                         tableToken.getDirName()
69                 ) == TableUtils.TABLE_EXISTS
70         ) {
71             // Dropped table
72             engine.notifyWalTxnCommitted(tableToken, Long.MAX_VALUE);
73         } else {
74             threadLocalPath.trimTo(dbRoot.length()).concat(tableToken).concat(TableUtils.META_FILE_NAME).$();
75             if (ff.exists(threadLocalPath)) {
76                 threadLocalPath.trimTo(dbRoot.length()).concat(tableToken).concat(TableUtils.TXN_FILE_NAME).$();
77                 try (TxReader txReader2 = txReader.ofRO(threadLocalPath, PartitionBy.NONE)) {
78                     TableUtils.safeReadTxn(txReader, millisecondClock, spinLockTimeout);
79                     if (txReader2.getSeqTxn() < txn) {
80                         engine.notifyWalTxnCommitted(tableToken, txn);
81                     }
82                 }
83             } else {
84                 // table is dropped, notify the JOB to delete the data
85                 engine.notifyWalTxnCommitted(tableToken, Long.MAX_VALUE);
86             }
87         }
88     }
89 
90     @Override
91     protected boolean runSerially() {
92         long unpublishedWalTxnCount = engine.getUnpublishedWalTxnCount();
93         if (unpublishedWalTxnCount == lastProcessedCount) {

96         checkMissingWalTransactions();
97         lastProcessedCount = unpublishedWalTxnCount;
98         return true;
99     }
