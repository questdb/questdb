61     public WalReader(CairoConfiguration configuration, TableToken tableToken, CharSequence walName, int segmentId, long rowCount) {
62         this.tableName = tableToken.getTableName();
63         this.walName = Chars.toString(walName);
64         this.rowCount = rowCount;
65 
66         ff = configuration.getFilesFacade();
67         path = new Path();
68         path.of(configuration.getRoot()).concat(tableToken.getDirName()).concat(walName);
69         rootLen = path.length();
70 
71         try {
72             metadata = new SequencerMetadata(ff, true);
73             metadata.open(path.slash().put(segmentId), rootLen);
74             columnCount = metadata.getColumnCount();
75             events = new WalEventReader(ff);
76             LOG.debug().$("open [table=").$(tableName).I$();
77             int pathLen = path.length();
78             eventCursor = events.of(path.slash().put(segmentId), WAL_FORMAT_VERSION, -1L);
79             path.trimTo(pathLen);
80             openSymbolMaps(eventCursor, configuration);
81             path.slash().put(segmentId);
82             eventCursor.reset();
83

85             columns = new ObjList<>(capacity);
86             columns.setPos(capacity + 2);
87             columns.setQuick(0, NullMemoryMR.INSTANCE);
88             columns.setQuick(1, NullMemoryMR.INSTANCE);

162     private void loadColumnAt(int columnIndex) {
163         final int pathLen = path.length();
164         try {
165             final int columnType = metadata.getColumnType(columnIndex);
166             if (columnType > 0) {
167                 final CharSequence name = metadata.getColumnName(columnIndex);
168                 final int primaryIndex = getPrimaryColumnIndex(columnIndex);
169                 final int secondaryIndex = primaryIndex + 1;
170                 final MemoryMR primaryMem = columns.getQuick(primaryIndex);
171 
172                 if (ColumnType.isVariableLength(columnType)) {
173                     long columnSize = (rowCount + 1) << 3;
174                     TableUtils.iFile(path.trimTo(pathLen), name);
175                     MemoryMR secondaryMem = columns.getQuick(secondaryIndex);
176                     secondaryMem = openOrCreateMemory(path, columns, secondaryIndex, secondaryMem, columnSize);
177                     columnSize = secondaryMem.getLong(rowCount << 3);
178                     TableUtils.dFile(path.trimTo(pathLen), name);
179                     openOrCreateMemory(path, columns, primaryIndex, primaryMem, columnSize);
180                 } else {
181                     long columnSize = rowCount << ColumnType.pow2SizeOf(columnType);
182                     TableUtils.dFile(path.trimTo(pathLen), name);
183                     openOrCreateMemory(path, columns, primaryIndex, primaryMem, columnIndex == getTimestampIndex() ? columnSize << 1 : columnSize);
184                     Misc.free(columns.getAndSetQuick(secondaryIndex, null));
185                 }
186             }
187         } finally {
188             path.trimTo(pathLen);
189         }
190     }

215     private void openSymbolMaps(WalEventCursor eventCursor, CairoConfiguration configuration) {
216         while (eventCursor.hasNext()) {
217             if (eventCursor.getType() == DATA) {
218                 WalEventCursor.DataInfo dataInfo = eventCursor.getDataInfo();
219                 SymbolMapDiff symbolDiff = dataInfo.nextSymbolMapDiff();
220                 while (symbolDiff != null) {
221                     int cleanSymbolCount = symbolDiff.getCleanSymbolCount();
222                     int columnIndex = symbolDiff.getColumnIndex();
223                     final IntObjHashMap<CharSequence> symbolMap;
224 
225                     if (symbolMaps.size() <= columnIndex || symbolMaps.getQuick(columnIndex) == null) {
226                         symbolMap = new IntObjHashMap<>();
227                         if (cleanSymbolCount > 0) {
228                             try (
229                                     SymbolMapReaderImpl symbolMapReader = new SymbolMapReaderImpl(
230                                             configuration,
231                                             path,
232                                             this.metadata.getColumnName(columnIndex),
233                                             COLUMN_NAME_TXN_NONE,
234                                             cleanSymbolCount
235                                     )
236                             ) {
237                                 for (int key = 0; key < cleanSymbolCount; key++) {
238                                     CharSequence symbol = symbolMapReader.valueOf(key);
239                                     symbolMap.put(key, String.valueOf(symbol));
240                                 }
241                             }
242                         }
243                         symbolMaps.extendAndSet(columnIndex, symbolMap);
244                     } else {
245                         symbolMap = symbolMaps.getQuick(columnIndex);
246                     }
247 
248                     SymbolMapDiffEntry entry = symbolDiff.nextEntry();
249                     while (entry != null) {
250                         symbolMap.put(entry.getKey(), String.valueOf(entry.getSymbol()));
251                         entry = symbolDiff.nextEntry();
252                     }
253 
254                     symbolDiff = dataInfo.nextSymbolMapDiff();
255                 }
256             }
257         }
258     }
