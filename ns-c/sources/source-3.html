43     public static void rollColumnToSegment(
44             FilesFacade ff,
45             long options,
46             MemoryMA primaryColumn,
47             MemoryMA secondaryColumn,
48             @Transient Path walPath,
49             int newSegment,
50             CharSequence columnName,
51             int columnType,
52             long rowOffset,
53             long rowCount,
54             LongList newColumnFiles,
55             int columnIndex
56     ) {
57         Path newSegPath = Path.PATH.get().of(walPath).slash().put(newSegment);
58         int setPathRoot = newSegPath.length();
59         dFile(newSegPath, columnName, COLUMN_NAME_TXN_NONE);
60         int primaryFd = openRW(ff, newSegPath, LOG, options);
61         newColumnFiles.setQuick(columnIndex * NEW_COL_RECORD_SIZE, primaryFd);
62 
63         int secondaryFd;
64         if (ColumnType.isVariableLength(columnType)) {
65             iFile(newSegPath.trimTo(setPathRoot), columnName, COLUMN_NAME_TXN_NONE);
66             secondaryFd = openRW(ff, newSegPath, LOG, options);
67             newColumnFiles.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 3, secondaryFd);
68         } else {
69             secondaryFd = -1;
70         }
71 
72         boolean success;
73         if (ColumnType.isVariableLength(columnType)) {
74             success = copyVarLenFile(ff, primaryColumn, secondaryColumn, primaryFd, secondaryFd, rowOffset, rowCount, newColumnFiles, columnIndex);
75         } else if (columnType > 0) {
76             success = copyFixLenFile(ff, primaryColumn, primaryFd, rowOffset, rowCount, columnType, newColumnFiles, columnIndex);
77         } else {
78             success = copyTimestampFile(ff, primaryColumn, primaryFd, rowOffset, rowCount, newColumnFiles, columnIndex);
79         }
80 
81         if (!success) {
82             throw CairoException.critical(ff.errno()).put("failed to copy column file to new segment" +
83                             " [path=").put(newSegPath)
84                     .put(", column=").put(columnName)
85                     .put(", rowOffset=").put(rowOffset)
86                     .put(", rowCount=").put(rowCount)
87                     .put(", columnType=").put(columnType).put("]");
88         }
89     }

91     private static boolean copyFixLenFile(
92             FilesFacade ff,
93             MemoryMA primaryColumn,
94             int primaryFd,
95             long rowOffset,
96             long rowCount,
97             int columnType,
98             LongList newOffsets,
99             int columnIndex
100     ) {
101         int shl = ColumnType.pow2SizeOf(columnType);
102         long offset = rowOffset << shl;
103         long length = rowCount << shl;
104 
105         boolean success = ff.copyData(primaryColumn.getFd(), primaryFd, offset, length) == length;
106         if (success) {
107             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 1, offset);
108             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 2, length);
109         }
110         return success;
111     }

113     private static boolean copyTimestampFile(
114             FilesFacade ff,
115             MemoryMA primaryColumn,
116             int primaryFd,
117             long rowOffset,
118             long rowCount,
119             LongList newOffsets,
120             int columnIndex
121     ) {
122         // Designated timestamp column is written as 2 long values
123         if (!copyFixLenFile(ff, primaryColumn, primaryFd, rowOffset, rowCount, ColumnType.LONG128, newOffsets, columnIndex)) {
124             return false;
125         }
126         long size = rowCount << 4;
127         long srcDataTimestampAddr = TableUtils.mapRW(ff, primaryFd, size, MEMORY_TAG);
128         try {
129             Vect.flattenIndex(srcDataTimestampAddr, rowCount);
130         } finally {
131             ff.munmap(srcDataTimestampAddr, size, MEMORY_TAG);
132         }
133         return true;
134     }

136     private static boolean copyVarLenFile(
137             FilesFacade ff,
138             MemoryMA primaryColumn,
139             MemoryMA secondaryColumn,
140             int primaryFd,
141             int secondaryFd,
142             long rowOffset,
143             long rowCount,
144             LongList newOffsets,
145             int columnIndex
146     ) {
147         long indexMapSize = (rowOffset + rowCount + 1) * Long.BYTES;
148         long srcIndexAddr = TableUtils.mapRW(ff, secondaryColumn.getFd(), indexMapSize, MEMORY_TAG);
149         try {
150             long varStart = Unsafe.getUnsafe().getLong(srcIndexAddr + rowOffset * Long.BYTES);
151             long varEnd = Unsafe.getUnsafe().getLong(srcIndexAddr + (rowOffset + rowCount) * Long.BYTES);
152             long varCopyLen = varEnd - varStart;
153             boolean success = ff.copyData(primaryColumn.getFd(), primaryFd, varStart, varCopyLen) == varCopyLen;
154             if (!success) {
155                 return false;
156             }
157 
158             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 1, varStart);
159             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 2, varCopyLen);
160 
161             long indexLen = (rowCount + 1) * Long.BYTES;
162             long dstIndexAddr = TableUtils.mapRW(ff, secondaryFd, indexLen, MEMORY_TAG);
163             ff.madvise(dstIndexAddr, indexLen, Files.POSIX_MADV_RANDOM);
164 
165             Vect.shiftCopyFixedSizeColumnData(varStart, srcIndexAddr, rowOffset, rowOffset + rowCount, dstIndexAddr);
166             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 4, (rowOffset + 1) * Long.BYTES);
167             newOffsets.setQuick(columnIndex * NEW_COL_RECORD_SIZE + 5, indexLen);
168 
169             // All in memory calls, no need to unmap in finally
170             ff.munmap(dstIndexAddr, indexLen, MEMORY_TAG);
171             return true;
172         } finally {
173             ff.munmap(srcIndexAddr, indexMapSize, MEMORY_TAG);
174         }
175     }
