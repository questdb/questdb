84         eventMem.putInt(type);
85 
86         switch (ColumnType.tagOf(type)) {

89                 break;
90             case ColumnType.BYTE:

94                 eventMem.putByte(function.getGeoByte(null));
95                 break;
96             case ColumnType.SHORT:
97                 eventMem.putShort(function.getShort(null));
98                 break;
99             case ColumnType.GEOSHORT:

152     private void writeNamedVariables(BindVariableService bindVariableService) {
153         final int count = bindVariableService != null ? bindVariableService.getNamedVariables().size() : 0;
154         eventMem.putInt(count);
155 
156         if (count > 0) {
157             final ObjList<CharSequence> namedVariables = bindVariableService.getNamedVariables();
158             for (int i = 0; i < count; i++) {
159                 final CharSequence name = namedVariables.get(i);
160                 eventMem.putStr(name);
161                 sink.clear();
162                 sink.put(':').put(name);
163                 writeFunction(bindVariableService.getFunction(sink));
164             }
165         }

168     private void writeSymbolMapDiffs() {
169         final int columns = txnSymbolMaps.size();
170         for (int i = 0; i < columns; i++) {
171             final CharSequenceIntHashMap symbolMap = txnSymbolMaps.getQuick(i);
172             if (symbolMap != null) {
173                 final int initialCount = initialSymbolCounts.get(i);
174                 if (initialCount > 0 || (initialCount == 0 && symbolMap.size() > 0)) {
175                     eventMem.putInt(i);
176                     eventMem.putBool(symbolMapNullFlags.get(i));
177                     eventMem.putInt(initialCount);
178 
179                     final int size = symbolMap.size();
180                     long appendAddress = eventMem.getAppendOffset();
181                     eventMem.putInt(size);
182 
183                     int symbolCount = 0;
184                     for (int j = 0; j < size; j++) {
185                         final CharSequence symbol = symbolMap.keys().getQuick(j);
186                         final int value = symbolMap.get(symbol);
187                         // Ignore symbols cached from symbolMapReader
188                         if (value >= initialCount) {
189                             eventMem.putInt(value);
190                             eventMem.putStr(symbol);
191                             symbolCount += 1;
192                         }
193                     }
194                     // Update the size with the exact symbolCount
195                     // An empty SymbolMapDiff can be created because symbolCount can be 0
196                     // in case all cached symbols come from symbolMapReader.
197                     // Alternatively, two-pass approach can be used.
198                     eventMem.putInt(appendAddress, symbolCount);
199                     eventMem.putInt(SymbolMapDiffImpl.END_OF_SYMBOL_ENTRIES);
200                 }
201             }
202         }
203         eventMem.putInt(SymbolMapDiffImpl.END_OF_SYMBOL_DIFFS);
204     }
