118     /**
119      * Validate equivalent of "^wal\d+$" regex.
120      */
121     private static boolean matchesWalNamePattern(CharSequence name) {
122         final int len = name.length();
123         if (len < (WalUtils.WAL_NAME_BASE.length() + 1)) {
124             return false;
125         }
126 
127         if (name.charAt(0) != 'w' || name.charAt(1) != 'a' || name.charAt(2) != 'l') {
128             return false;  // Not a "wal" prefix.
129         }
130 
131         for (int i = 3; i < len; ++i) {
132             final char c = name.charAt(i);
133             if (c < '0' || c > '9') {
134                 return false;  // Not a number.
135             }
136         }
137 
138         return true;
139     }

191     private void broadSweep(int tableId, final TableToken tableToken, long lastTxn) {
192         try {
193             this.tableToken = tableToken;
194             discoveredWalSegments.clear();
195             lockedWalIds.clear();
196             discoveredWalIds.clear();
197             nextToApplyWalSegments.clear();
198 
199             discoverWalSegments();
200             if (discoveredWalSegments.size() != 0) {
201 
202                 populateNextToApplyWalSegments();
203                 accumDebugState();
204 
205                 // Any of the calls above may leave outstanding `discoveredWalIds` that are still on the filesystem
206                 // and don't have any active segments. Any unlocked walNNN directories may be deleted if they don't have
207                 // pending segments that are yet to be applied to the table.
208                 // Note that this also handles cases where a wal directory was created shortly before a crash and thus
209                 // never recorded and tracked by the sequencer for that table.
210                 deleteOutstandingWalDirectories();
211             }
212 
213             if (lastTxn < 0 && engine.isTableDropped(tableToken)) {
214                 // Delete sequencer files
215                 deleteTableSequencerFiles(tableToken);
216 
217                 if (
218                         TableUtils.exists(
219                                 ff,
220                                 Path.getThreadLocal(""),
221                                 configuration.getRoot(),
222                                 tableToken.getDirName()
223                         ) != TableUtils.TABLE_EXISTS
224                 ) {
225                     // Fully deregister the table
226                     LOG.info().$("table is fully dropped [tableDir=").$(tableToken.getDirName()).I$();
227                     ff.rmdir(Path.getThreadLocal(configuration.getRoot()).concat(tableToken).slash$());
228                     engine.removeTableToken(tableToken);
229                 } else {
230                     LOG.info().$("table is not fully dropped, pinging WAL Apply job to delete table files [tableDir=").$(tableToken.getDirName()).I$();
231                     // Ping ApplyWal2TableJob to clean up the table files
232                     engine.notifyWalTxnRepublisher();
233                 }
234             }
235         } catch (CairoException ce) {
236             LOG.error().$("broad sweep failed [table=").$(tableToken)
237                     .$(", msg=").$((Throwable) ce)
238                     .$(", errno=").$(ff.errno()).$(']').$();
239         }
240     }

254     private void deleteOutstandingWalDirectories() {
255         // Merge join discoveredWalSegments and nextToApplyWalSegments
256         // Both are sorted by WalId and then SegmentId
257         // Delete those segments which are lower than first committed segment
258         // Keep locked segments and all segments higher than first locked segment
259         int committedWalSegmentIndex = -1;
260         int committedWalSegmentId = -1;
261         int committedWalId = -1;
262         int committedSegmentSize = nextToApplyWalSegments.size();
263 
264         for (int i = 0, n = discoveredWalSegments.size(); i < n; i++) {
265             long walSegment = discoveredWalSegments.get(i);
266             int walId = Numbers.decodeHighInt(walSegment);
267             int segmentIdLocked = Numbers.decodeLowInt(walSegment);
268             int segmentId = segmentIdLocked >> 1;
269             boolean segmentIdIsLocked = segmentIdLocked != Integer.MAX_VALUE && (segmentIdLocked & 1) == 1;
270 
271             // If the current segment is locked, scroll to next wall id.
272             int searchWalId = segmentIdIsLocked ? walId + 1 : walId;
273             while (committedWalId < searchWalId && ++committedWalSegmentIndex < committedSegmentSize) {
274                 long committedWalSegment = nextToApplyWalSegments.get(committedWalSegmentIndex);
275                 committedWalId = Numbers.decodeHighInt(committedWalSegment);
276                 committedWalSegmentId = Numbers.decodeLowInt(committedWalSegment) >> 1;
277             }
278 
279             if (walId == committedWalId) {
280                 // WAL is in use, there are committed transactions in Sequencer
281                 if (committedWalSegmentId > segmentId) {
282                     // Segment already applied.
283                     deleteSegmentDirectory(tableToken, walId, segmentId);
284                 }
285             } else if (!segmentIdIsLocked) {
286                 // Nothing in Sequencer for this WAL and no locked segment means WAL can be deleted if it was not locked.
287                 if (!lockedWalIds.contains(walId)) {
288                     deleteWalDirectory(walId);
289                 } else {
290                     // WAL is locked but the segment in it is not in outstanding commits, delete it.
291                     // Segment with ID of Integer.MAX_VALUE is the WAL directory itself, only delete it if it's unlocked.
292                     if (segmentId != (Integer.MAX_VALUE >> 1)) {
293                         deleteSegmentDirectory(tableToken, walId, segmentId);
294                     }
295                 }
296             }
297         }
298     }

325     private void discoverWalSegments() {
326         Path path = setTablePath(tableToken);
327         long p = ff.findFirst(path);
328         int rootPathLen = path.length();
329         if (p > 0) {
330             try {
331                 do {
332                     int type = ff.findType(p);
333                     long pUtf8NameZ = ff.findName(p);
334 
335                     if (type == Files.DT_DIR && matchesWalNamePattern(walName.of(pUtf8NameZ))) {
336                         // We just record the name for now in a set which we'll remove items to know when we're done.
337                         try {
338                             int walId = Numbers.parseInt(walName, 3, walName.length());
339                             if (walIsInUse(tableToken, walId)) {
340                                 lockedWalIds.add(walId);
341                             }
342                             discoveredWalIds.add(walId);
343                             // SegmentId of Integer.MAX_VALUE is to track WAL directory itself.
344                             discoveredWalSegments.add(Numbers.encodeLowHighInts(Integer.MAX_VALUE, walId));
345 
346                             // Search for segments.
347                             path.trimTo(rootPathLen).concat(pUtf8NameZ);
348                             int walPathLen = path.length();
349                             long sp = ff.findFirst(path.$());
350 
351                             try {
352                                 do {
353                                     type = ff.findType(sp);
354                                     pUtf8NameZ = ff.findName(sp);
355 
356                                     if (type == Files.DT_DIR && matchesSegmentName(walName.of(pUtf8NameZ))) {
357                                         try {
358                                             int segmentId = Numbers.parseInt(walName);
359                                             Path segmentPath = path.trimTo(walPathLen).slash().put(segmentId);
360                                             TableUtils.lockName(segmentPath);
361                                             boolean unlocked = unlocked(segmentPath.$());
362                                             discoveredWalSegments.add(Numbers.encodeLowHighInts((segmentId << 1) + (unlocked ? 0 : 1), walId));
363                                         } catch (NumericException ne) {
364                                             // Non-Segment directory, ignore.
365                                         }
366                                     }
367                                 } while (ff.findNext(sp) > 0);
368                             } finally {
369                                 ff.findClose(sp);
370                             }
371                         } catch (NumericException ne) {
372                             // Non-WAL directory, ignore.
373                         }
374                     }
375                 } while (ff.findNext(p) > 0);
376                 discoveredWalSegments.sort();
377             } finally {
378                 ff.findClose(p);
379             }
380         }
381     }

390     private void populateNextToApplyWalSegments() {
391         setTxnPath(tableToken);
392         if (!engine.isTableDropped(tableToken)) {
393             try {
394                 txReader.ofRO(path, PartitionBy.NONE);
395                 TableUtils.safeReadTxn(txReader, millisecondClock, spinLockTimeout);
396                 final long lastAppliedTxn = txReader.getSeqTxn();
397 
398                 TableSequencerAPI tableSequencerAPI = engine.getTableSequencerAPI();
399                 try (TransactionLogCursor transactionLogCursor = tableSequencerAPI.getCursor(tableToken, lastAppliedTxn)) {
400                     while (discoveredWalIds.size() > 0 && transactionLogCursor.hasNext()) {
401                         int walId = transactionLogCursor.getWalId();
402                         if (discoveredWalIds.contains(walId)) {
403                             int segmentId = transactionLogCursor.getSegmentId();
404                             nextToApplyWalSegments.add(Numbers.encodeLowHighInts(segmentId << 1, walId));
405                             discoveredWalIds.remove(walId);
406                         }
407                     }
408                     nextToApplyWalSegments.sort();
409                 }
410             } finally {
411                 txReader.close();
412             }
413         }
414         // If table is dropped, all wals can be deleted.
415         // No need to do anything, all discovered segments / wals will be deleted
416     }

481         final long t = clock.getTicks();
482         if (last + checkInterval < t) {
483             last = t;
484             if (runLock.tryLock()) {
485                 try {
486                     broadSweep();
487                 } finally {
488                     runLock.unlock();
