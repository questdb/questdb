61     public WalEventCursor of(Path path, int expectedVersion, long segmentTxn) {
62         int trimTo = path.length();
63         try {
64             final int pathLen = path.length();
65 
66             path.concat(EVENT_FILE_NAME).$();
67             eventMem.of(
68                     ff,
69                     path,
70                     ff.getPageSize(),
71                     WALE_HEADER_SIZE + Integer.BYTES,
72                     MemoryTag.MMAP_TABLE_WAL_READER,
73                     CairoConfiguration.O_NONE,
74                     -1
75             );
76 
77             if (segmentTxn > -1) {
78                 // Read record offset and size
79                 int fdi = openRO(ff, path.trimTo(pathLen).concat(EVENT_INDEX_FILE_NAME).$(), LOG);
80                 try {
81                     int maxTxn = eventMem.getInt(WALE_MAX_TXN_OFFSET_32);
82                     long offset = ff.readNonNegativeLong(fdi, segmentTxn << 3);
83                     long size = ff.readNonNegativeLong(fdi, (maxTxn + 1L) << 3);
84                     if (offset < 0 || size < WALE_HEADER_SIZE + Integer.BYTES || offset >= size) {
85                         int errno = offset < 0 || size < 0 ? ff.errno() : 0;
86                         throw CairoException.critical(errno).put("segment ")
87                                 .put(path).put(" does not have txn with id ").put(segmentTxn);
88                     }
89 
90                     // WAL-E file has record indicator for the next record always present 
91                     // so file size is the size read + 4 bytes
92                     eventMem.extend(size + Integer.BYTES);
93                     eventCursor.openOffset(offset);
94                 } finally {
95                     ff.close(fdi);
96                 }
97             } else {
98                 eventCursor.openOffset(-1);
99             }
100 
101             validateMetaVersion(eventMem, WAL_FORMAT_OFFSET_32, expectedVersion);
102             return eventCursor;
103         } catch (Throwable e) {
104             close();
105             throw e;
106         } finally {
107             path.trimTo(trimTo);
108         }
109     }
110 }
