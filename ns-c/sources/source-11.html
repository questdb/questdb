641             } catch (CairoException e) {
642                 if (retry) {
643                     // Table schema (metadata) changed and this Alter is not valid anymore.
644                     // Try to update WAL metadata to latest and repeat one more time.
645                     goActive();
646                     alterOp.apply(metaValidatorSvc, true);
647                 } else {
648                     throw e;
649                 }
650             }
651 
652             try {
653                 txn = sequencer.nextStructureTxn(tableToken, metadata.getStructureVersion(), alterOp);
654                 if (txn == NO_TXN) {
655                     applyMetadataChangeLog(Long.MAX_VALUE);
656                 }
657             } catch (CairoException e) {
658                 distressed = true;
659                 throw e;
660             }
661         } while (txn == NO_TXN);
662 
663         // Apply to itself.
664         try {
665             alterOp.apply(metaWriterSvc, true);
666         } catch (Throwable th) {
667             // Transaction successful, but writing using this WAL writer should not be possible.
668             LOG.error().$("Exception during alter [ex=").$(th).I$();
669             distressed = true;
670         }
671         return txn;
672     }

834     private void configureSymbolTable() {
835         boolean initialized = false;
836         try {
837             int denseSymbolIndex = 0;
838 
839             for (int i = 0; i < columnCount; i++) {
840                 int columnType = metadata.getColumnType(i);
841                 if (!ColumnType.isSymbol(columnType)) {
842                     // Maintain sparse list of symbol writers
843                     // Note: we don't need to set initialSymbolCounts and symbolMapNullFlags values
844                     // here since we already filled it with -1 and false initially
845                     symbolMapReaders.extendAndSet(i, null);
846                     symbolMaps.extendAndSet(i, null);
847                     utf8SymbolMaps.extendAndSet(i, null);
848                 } else {
849                     if (txReader == null) {
850                         txReader = new TxReader(ff);
851                         columnVersionReader = new ColumnVersionReader();
852                     }
853 
854                     if (!initialized) {
855                         MillisecondClock milliClock = configuration.getMillisecondClock();
856                         long spinLockTimeout = configuration.getSpinLockTimeout();
857 
858                         // todo: use own path
859                         Path path = Path.PATH2.get();
860                         path.of(configuration.getRoot()).concat(tableToken).concat(TXN_FILE_NAME).$();
861 
862                         // Does not matter which PartitionBy, as long as it is partitioned
863                         // WAL tables must be partitioned
864                         txReader.ofRO(path, PartitionBy.DAY);
865                         path.of(configuration.getRoot()).concat(tableToken).concat(COLUMN_VERSION_FILE_NAME).$();
866                         columnVersionReader.ofRO(ff, path);
867 
868                         initialized = true;
869                         long structureVersion = getStructureVersion();
870 
871                         do {
872                             TableUtils.safeReadTxn(txReader, milliClock, spinLockTimeout);
873                             if (txReader.getStructureVersion() != structureVersion) {
874                                 initialized = false;
875                                 break;
876                             }
877                             columnVersionReader.readSafe(milliClock, spinLockTimeout);
878                         } while (txReader.getColumnVersion() != columnVersionReader.getVersion());
879                     }
880 
881                     if (initialized) {
882                         int symbolValueCount = txReader.getSymbolValueCount(denseSymbolIndex);
883                         long columnNameTxn = columnVersionReader.getDefaultColumnNameTxn(i);
884                         configureSymbolMapWriter(i, metadata.getColumnName(i), symbolValueCount, columnNameTxn);
885                     } else {
886                         // table on disk structure version does not match the structure version of the WalWriter
887                         // it is not possible to re-use table symbol table because the column name may not match.
888                         // The symbol counts stored as dense in _txn file and removal of symbols
889                         // shifts the counts that's why it's not possible to find out the symbol count if metadata versions
890                         // don't match.
891                         configureSymbolMapWriter(i, metadata.getColumnName(i), 0, COLUMN_NAME_TXN_NONE);
892                     }
893                 }
894 
895                 if (columnType == ColumnType.SYMBOL || columnType == -ColumnType.SYMBOL) {
896                     denseSymbolIndex++;
897                 }
898             }
899         } finally {
900             Misc.free(columnVersionReader);
901             Misc.free(txReader);
902         }
903     }
